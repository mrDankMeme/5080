import SwiftUI

struct AllChatsListView: View {

    @StateObject var vm: UnifiedHistoryListViewModel

    @State var deleteTarget: UnifiedThreadRowViewModel? = nil
    @State var isDeleteConfirmPresented: Bool = false

    @State var renameTarget: UnifiedThreadRowViewModel? = nil
    @State var renameText: String = ""
    @State var isRenamePresented: Bool = false

    @Environment(\.dismiss) var dismiss

    var openAIDestinationBuilder: (String) -> AnyView
    var generationDestinationBuilder: (GenerationKind, UUID) -> AnyView

    init(
        vm: UnifiedHistoryListViewModel,
        openAIDestinationBuilder: @escaping (String) -> AnyView,
        generationDestinationBuilder: @escaping (GenerationKind, UUID) -> AnyView
    ) {
        _vm = StateObject(wrappedValue: vm)
        self.openAIDestinationBuilder = openAIDestinationBuilder
        self.generationDestinationBuilder = generationDestinationBuilder
    }

    var body: some View {
        ZStack {
            backgroundColor.ignoresSafeArea()

            VStack(spacing: 0) {
                header
                    .padding(.horizontal, 16.scale)
                    .padding(.top, 16.scale)

                list
                    .padding(.top, 12.scale)
            }
        }
        .toolbar(.hidden, for: .navigationBar)

        .alert("all_chats.rename.title".localizable, isPresented: $isRenamePresented) {
            TextField("all_chats.rename.placeholder".localizable, text: $renameText)

            Button("all_chats.rename.save".localizable) {
                guard let row = renameTarget else { return }
                let newTitle = renameText.trimmingCharacters(in: .whitespacesAndNewlines)
                guard newTitle.isEmpty == false else { return }
                Task { await vm.rename(row, newTitle: newTitle) }
            }

            Button("all_chats.rename.cancel".localizable, role: .cancel) { }
        } message: {
            Text("all_chats.rename.message".localizable)
        }

        .confirmationDialog(
            "all_chats.delete.title".localizable,
            isPresented: $isDeleteConfirmPresented,
            titleVisibility: .visible
        ) {
            Button("all_chats.delete.confirm".localizable, role: .destructive) {
                guard let row = deleteTarget else { return }
                Task { await vm.delete(row) }
            }
            Button("all_chats.delete.cancel".localizable, role: .cancel) { }
        } message: {
            Text(deleteMessageText())
        }
    }
}

                                },

                                    isDeleteConfirmPresented = true

                        .padding(.horizontal, 16)

        case .openAI(let conversationId, _):

        case .generation(let kind, let threadId):
            return generationDestinationBuilder(kind, threadId)
        }

            return "all_chats.delete.cannot_undo".localizable
        }

            return "all_chats.delete.openai".localizable
        case .generation:
            return "all_chats.delete.generation".localizable
        }

import Foundation
import Combine

@MainActor
final class UnifiedHistoryListViewModel: ObservableObject {

    // MARK: - Output

    @Published private(set) var all: [UnifiedThreadRowViewModel] = []
    @Published private(set) var visible: [UnifiedThreadRowViewModel] = []
    @Published private(set) var errorText: String? = nil

    // MARK: - Input

    @Published var searchQuery: String = ""

    // MARK: - Dependencies

    private let observeUseCase: ObserveUnifiedThreadsUseCase
    private let deleteChatThreadUseCase: DeleteChatThreadUseCase
    private let renameChatThreadUseCase: RenameChatThreadUseCase

    private let getChatThreadUseCase: GetChatThreadUseCase
    private let upsertChatThreadUseCase: UpsertChatThreadUseCase

    private let genRepo: GenerationHistoryRepository

    private let mediaCache: MediaCacheService

    // MARK: - Private

    private var bag = Set<AnyCancellable>()
    private let relativeFormatter = RelativeDateTimeFormatter()

    private var latestThreads: [UnifiedThread] = []

    // MARK: - Init

    init(
        observeUseCase: ObserveUnifiedThreadsUseCase,
        deleteChatThreadUseCase: DeleteChatThreadUseCase,
        renameChatThreadUseCase: RenameChatThreadUseCase,
        getChatThreadUseCase: GetChatThreadUseCase,
        upsertChatThreadUseCase: UpsertChatThreadUseCase,
        genRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService
    ) {
        self.observeUseCase = observeUseCase
        self.deleteChatThreadUseCase = deleteChatThreadUseCase
        self.renameChatThreadUseCase = renameChatThreadUseCase
        self.getChatThreadUseCase = getChatThreadUseCase
        self.upsertChatThreadUseCase = upsertChatThreadUseCase
        self.genRepo = genRepo
        self.mediaCache = mediaCache
        bind()
    }

    // MARK: - Actions

    func delete(_ row: UnifiedThreadRowViewModel) async {
        errorText = nil
        do {
            switch row.source {
            case .openAI(_, let localThreadId):
                try await deleteChatThreadUseCase.execute(threadId: localThreadId)

            case .generation(_, let threadId):
                try await mediaCache.deleteCache(threadId: threadId)
                try await genRepo.delete(threadId: threadId)
            }
        } catch {
            errorText = "Delete failed: \(String(describing: error))"
        }
    }

    func rename(_ row: UnifiedThreadRowViewModel, newTitle: String) async {
        let trimmed = newTitle.trimmingCharacters(in: .whitespacesAndNewlines)
        guard trimmed.isEmpty == false else { return }

        errorText = nil

        do {
            switch row.source {
            case .openAI(let conversationId, _):
                try await renameChatThreadUseCase.execute(conversationId: conversationId, newTitle: trimmed)

            case .generation(_, let threadId):
                try await genRepo.renameThread(threadId: threadId, title: trimmed)
            }
        } catch {
            errorText = "Rename failed: \(String(describing: error))"
        }
    }

    func togglePin(_ row: UnifiedThreadRowViewModel) async {
        errorText = nil

        do {
            switch row.source {
            case .openAI(let conversationId, _):
                guard let thread = try await getChatThreadUseCase.execute(conversationId: conversationId) else {
                    return
                }
                let updated = thread.updatingPinned(!thread.isPinned)
                try await upsertChatThreadUseCase.execute(updated)

            case .generation(let kind, let threadId):
                if let src = latestThreads.first(where: { $0.id == row.id }) {
                    let updated = GenerationThread(
                        id: threadId,
                        kind: kind,
                        title: src.title,
                        createdAt: src.createdAt,
                        lastActivityAt: src.lastActivityAt,
                        preview: src.preview,
                        isPinned: !src.isPinned
                    )
                    try await genRepo.upsert(updated)
                } else {
                    let fallback = GenerationThread(
                        id: threadId,
                        kind: kind,
                        title: row.title,
                        createdAt: Date(),
                        lastActivityAt: Date(),
                        preview: row.preview,
                        isPinned: !row.isPinned
                    )
                    try await genRepo.upsert(fallback)
                }
            }
        } catch {
            errorText = "Pin failed: \(String(describing: error))"
        }
    }

    // MARK: - Bindings

    private func bind() {
            let threadsPublisher = observeUseCase.execute()
                .receive(on: DispatchQueue.main)
                .handleEvents(receiveOutput: { [weak self] threads in
                    self?.latestThreads = threads
                })
                .share()

            threadsPublisher
                .map { [weak self] threads -> [UnifiedThreadRowViewModel] in
                    guard let self else { return [] }
                    return threads.map { self.mapThreadToRow($0) }
                }
                .sink { [weak self] rows in
                    self?.all = rows
                    self?.visible = rows
                }
                .store(in: &bag)

            let queryPublisher = $searchQuery
                .map { $0.trimmingCharacters(in: .whitespacesAndNewlines) }
                .removeDuplicates()
                .debounce(for: .milliseconds(300), scheduler: DispatchQueue.main)

            Publishers.CombineLatest(threadsPublisher, queryPublisher)
                .map { [weak self] threads, query -> [UnifiedThreadRowViewModel] in
                    guard let self else { return [] }

                    guard query.isEmpty == false else {
                        return threads.map { self.mapThreadToRow($0) }
                    }

                    let q = query.lowercased()

                    let filtered = threads.filter { t in
                        let title = t.title.lowercased()
                        let preview = (t.preview ?? "").lowercased()

                        let kindText: String = {
                            switch t.kind {
                            case .openAI(let icon):
                                return "openai \(icon.systemImageName)".lowercased()

                            case .generation(let kind):
                                return "\(kind.rawValue) \(kind.defaultTitle)".lowercased()
                            }
                        }()

                        return title.contains(q) || preview.contains(q) || kindText.contains(q)
                    }

                    return filtered.map { self.mapThreadToRow($0) }
                }
                .sink { [weak self] rows in
                    self?.visible = rows
                }
                .store(in: &bag)
        }

    // MARK: - Mapping

    private func mapThreadToRow(_ thread: UnifiedThread) -> UnifiedThreadRowViewModel {
        let titleTrimmed = thread.title.trimmingCharacters(in: .whitespacesAndNewlines)
        let safeTitle = titleTrimmed.isEmpty ? "Чат" : titleTrimmed

        relativeFormatter.unitsStyle = .short
        let timeAgo = relativeFormatter.localizedString(for: thread.lastActivityAt, relativeTo: Date())

        return UnifiedThreadRowViewModel(
            id: thread.id,
            title: safeTitle,
            preview: thread.preview,
            timeAgo: timeAgo,
            iconSystemName: thread.kind.systemImageName,
            isPinned: thread.isPinned,
            source: thread.source
        )
    }
}

import Foundation

public struct UnifiedThreadRowViewModel: Identifiable, Equatable, Sendable {

    public let id: UUID
    public let title: String
    public let preview: String?
    public let timeAgo: String
    public let iconSystemName: String
    public let isPinned: Bool
    public let source: UnifiedThreadSource

    public init(
        id: UUID,
        title: String,
        preview: String?,
        timeAgo: String,
        iconSystemName: String,
        isPinned: Bool,
        source: UnifiedThreadSource
    ) {
        self.id = id
        self.title = title
        self.preview = preview
        self.timeAgo = timeAgo
        self.iconSystemName = iconSystemName
        self.isPinned = isPinned

    }
