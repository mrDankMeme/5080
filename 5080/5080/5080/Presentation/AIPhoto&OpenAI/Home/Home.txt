import Foundation
import Combine
import UIKit

@MainActor
final class HomeViewModel: ObservableObject {

    // MARK: - Input

    @Published var prompt: String = "cinematic portrait, soft light, 35mm"
    @Published var templateIdText: String = "0"

    // MARK: - Output / State

    @Published private(set) var isLoading: Bool = false
    @Published private(set) var statusText: String?
    @Published private(set) var errorText: String?
    @Published private(set) var resultURL: URL?
    @Published private(set) var resultRawText: String?
    @Published private(set) var logs: [String] = []

    // MARK: - Auth / Debug

    @Published private(set) var apphudId: String = ""
    @Published private(set) var bearerTokenMasked: String = "—"
    @Published private(set) var tokensBalance: Int?
    @Published private(set) var avatarTokensBalance: Int?

    // MARK: - Dependencies

    private let useCase: GenerateImageFromTextUseCase
    private let auth: AuthRepository
    private let tokenStore: AccessTokenStore

    // MARK: - Internals
    
    private let printFullBearerTokenInConsole: Bool = true

    private var cancellables = Set<AnyCancellable>()
    private var task: Task<Void, Never>?

    init(
        useCase: GenerateImageFromTextUseCase,
        auth: AuthRepository,
        tokenStore: AccessTokenStore
    ) {
        self.useCase = useCase
        self.auth = auth
        self.tokenStore = tokenStore
        self.apphudId = Self.resolveApphudId()
        refreshAuthSnapshotLog()
    }

    deinit {
        task?.cancel()
    }

    // MARK: - Actions

    func start() {
        cancel()

        errorText = nil
        statusText = nil
        resultURL = nil
        resultRawText = nil

        let prompt = self.prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let templateId = Int(self.templateIdText.trimmingCharacters(in: .whitespacesAndNewlines)) ?? 0

        guard !prompt.isEmpty else {
            errorText = "Prompt is empty"
            return
        }

        isLoading = true
        statusText = "starting…"
        log("🚀 start txt2img prompt_len=\(prompt.count) template_id=\(templateId)")

        task = Task { [weak self] in
            guard let self else { return }
            defer {
                self.isLoading = false
            }

            do {
                let dto = try await self.useCase.execute(prompt: prompt, templateId: templateId)

                self.statusText = "final: \(dto.status.debugString)"
                self.log("✅ final status=\(dto.status.debugString) error=\(dto.error ?? "—")")

                self.applyResult(from: dto)

                if dto.status == .failed, let e = dto.error, !e.isEmpty {
                    self.errorText = e
                }
            } catch {
                self.statusText = "failed"
                self.errorText = "\(error)"
                self.log("❌ \(error)")
            }
        }
    }

    func cancel() {
        task?.cancel()
        task = nil
        isLoading = false
        statusText = "cancelled"
        log("🛑 cancel")
    }

    func refreshMe() {
        Task {
            do {
                try await auth.ensureAuthorized(apphudId: apphudId)
                let me = try await auth.me()

                tokensBalance = me.tokens
                avatarTokensBalance = me.avatar_tokens

                log("👤 /me tokens=\(me.tokens) avatar_tokens=\(me.avatar_tokens)")
                refreshAuthSnapshotLog()
            } catch {
                errorText = "\(error)"
                log("❌ /me error \(error)")
                refreshAuthSnapshotLog()
            }
        }
    }

    func forceReAuth() {
        tokenStore.clear()
        refreshAuthSnapshotLog()
        log("🧹 token cleared; next call will re-auth")
    }

    // MARK: - Private

    private func refreshAuthSnapshotLog() {
        let rawToken = tokenStore.read() ?? ""
        bearerTokenMasked = rawToken.isEmpty ? "—" : Self.maskToken(rawToken)

        log("🪪 apphud_id=\(apphudId)")
        if !rawToken.isEmpty {
#if DEBUG
            if printFullBearerTokenInConsole {
                print("🧾 BEARER_FULL \(rawToken)")
            }
#endif
        }
        log("🔑 bearer=\(bearerTokenMasked)")
    }

    private func applyResult(from dto: GenerationReadDTO) {
        let raw = dto.result ?? ""
        resultRawText = raw

        if let url = URL(string: raw), url.scheme != nil {
            resultURL = url
        }
    }

    private func log(_ s: String) {
        logs.append(s)
#if DEBUG
        print("🧪 TXT2IMG \(s)")
#endif
    }

    // MARK: - Helpers

    private static func resolveApphudId() -> String {
        return "EDCC9EA3-2658-4E60-9F7F-1EB341E8DA2D"
    }

    private static func maskToken(_ token: String) -> String {
        let trimmed = token.trimmingCharacters(in: .whitespacesAndNewlines)
        guard trimmed.count > 12 else { return trimmed }
        let prefix = trimmed.prefix(6)
        let suffix = trimmed.suffix(6)
        return "\(prefix)…\(suffix) (len=\(trimmed.count))"
    }

    private static func decodeServerErrorBody(_ data: Data?) -> String? {
        guard let data, !data.isEmpty else { return nil }

        if let obj = try? JSONSerialization.jsonObject(with: data, options: []),
           JSONSerialization.isValidJSONObject(obj),
           let pretty = try? JSONSerialization.data(withJSONObject: obj, options: [.prettyPrinted]),
           let prettyString = String(data: pretty, encoding: .utf8) {
            return prettyString
        }

        if let text = String(data: data, encoding: .utf8) {
            return text
        }
        return data.map { String(format: "%02x", $0) }.joined()
    }

    private static func prettyJSONString<T: Encodable>(_ value: T) -> String? {
        let encoder = JSONEncoder()
        encoder.outputFormatting = [.prettyPrinted, .sortedKeys]
        guard let data = try? encoder.encode(value) else { return nil }
        return String(data: data, encoding: .utf8)
    }
}

// MARK: - Helpers

private extension GenerationStatus {
    var debugString: String {
        switch self {
        case .queued: return "queued"
        case .processing: return "processing"
        case .done: return "done"
        case .failed: return "failed"
        case .unknown(let v): return "unknown(\(v))"
        }
    }
}

        case .textToImage:

            PhotoEditChatView(vm: makePhotoEditVM(threadId: threadId))

            PhotoEnhanceChatView(vm: makePhotoEnhanceVM(threadId: threadId))
        case .textToVideo:
            TextToVideoChatView(vm: makeTextToVideoVM(threadId: threadId))

            PhotoToVideoChatView(

    // MARK: - Notifications polling

    private func startNotificationsPollingIfNeeded() {
        guard scenePhase == .active else { return }
        guard notificationsPollTask == nil else { return }

        notificationsPollTask = Task {
            await refreshNotificationsBadge()

            while !Task.isCancelled {
                try? await Task.sleep(nanoseconds: 800_000_000)
                await refreshNotificationsBadge()
            }
        }
    }

    private func stopNotificationsPolling() {
        notificationsPollTask?.cancel()
        notificationsPollTask = nil
    }

    @MainActor
    private func refreshNotificationsBadge() async {
        guard let center = resolver.resolve(InternalNotificationCenter.self) else {
            hasUnreadNotifications = false
            return
        }

        let count = await center.unreadCount()
        hasUnreadNotifications = (count > 0)
    }

    func onNotificationsTap() {
        withAnimation(.easeOut(duration: 0.15)) {
            isNotificationsPresented = true
        }
    }

    // MARK: - Request counter polling

    private func startRequestCounterPollingIfNeeded() {
        guard scenePhase == .active else { return }
        guard requestCounterPollTask == nil else { return }

        requestCounterPollTask = Task {
            await refreshTotalRequests()

            while !Task.isCancelled {
                try? await Task.sleep(nanoseconds: 600_000_000)
                await refreshTotalRequests()
            }
        }
    }

    private func stopRequestCounterPolling() {
        requestCounterPollTask?.cancel()
        requestCounterPollTask = nil
    }

    @MainActor
    private func refreshTotalRequests() async {
        guard let counter = resolver.resolve(RequestCounterService.self) else {
            totalRequests = 0
            return
        }

        let snap = await counter.snapshot()
        totalRequests = snap.total
    }

    // MARK: - Dialogs count (Unified threads)

    private func startDialogsCountObservationIfNeeded() {
        guard scenePhase == .active else { return }
        guard dialogsCountCancellable == nil else { return }

        guard let observe = resolver.resolve(ObserveUnifiedThreadsUseCase.self) else {
            totalDialogs = 0
            return
        }

        dialogsCountCancellable = observe.execute()
            .map { $0.count }
            .removeDuplicates()
            .receive(on: DispatchQueue.main)
            .sink { count in
                totalDialogs = count
            }
    }

    private func stopDialogsCountObservation() {
        dialogsCountCancellable?.cancel()
        dialogsCountCancellable = nil
    }

    // MARK: - Model picker

    func presentModelPicker() {
        withAnimation(.easeOut(duration: 0.22)) {
            isModelPickerPresented = true
        }
    }
}

        }
        .onChange(of: isQuickChatActive) { isActive in
            guard isActive == false else { return }
            quickPromptText = ""
        }
        .onDisappear {
            if isHomeRecordingSpeech { stopHomeSpeech() }
        }
    }

    // MARK: - UI bits

    private var homeSpeechTimerPill: some View {
        Text(homeSpeechTimerText)
            .font(Tokens.Font.semibold13)
            .monospacedDigit()
            .foregroundStyle(Color.white.opacity(0.92))
            .frame(maxWidth: .infinity, maxHeight: .infinity)
            .background(

                    .fill(Color.white.opacity(0.12))
            )
            .overlay(
                Capsule(style: .continuous)
                    .stroke(Color.white.opacity(0.10), lineWidth: 1.scale)
            )
    }

    private var hasQuickPromptText: Bool {
        quickPromptText
            .trimmingCharacters(in: .whitespacesAndNewlines)
            .isEmpty == false
    }

    // MARK: - Speech (Home)

    private func toggleHomeSpeech() async {
        if isHomeRecordingSpeech {
            stopHomeSpeech()
            return
        }

        let service: SpeechToTextService = {
            if let existing = homeSpeechService { return existing }
            let resolved = resolver.resolve(SpeechToTextService.self)!
            homeSpeechService = resolved
            return resolved
        }()

        let status = await service.requestPermissions()
        guard status == .authorized else {
            #if DEBUG
            print("🎙️ HOME: no permissions")
            #endif
            return
        }

        startHomeSpeech(using: service)
    }

    private func startHomeSpeech(using service: SpeechToTextService) {
        isHomeRecordingSpeech = true
        homeSpeechSeconds = 0
        homeSpeechTimerText = formatMMSS(homeSpeechSeconds)
        startHomeSpeechTimer()

        service.start(
            onPartial: { text in
                Task { @MainActor in
                    quickPromptText = text
                }
            },
            onFinal: { text in
                Task { @MainActor in
                    quickPromptText = text
                    stopHomeSpeech()
                }
            },
            onError: { error in
                #if DEBUG
                print("🎙️ HOME: speech error: \(error)")
                #endif
                Task { @MainActor in
                    cancelHomeSpeech()
                }
            }
        )

        #if DEBUG
        print("🎙️ HOME: speech START")
        #endif
    }

    private func stopHomeSpeech() {
        homeSpeechService?.stop()
        isHomeRecordingSpeech = false
        stopHomeSpeechTimer()

        #if DEBUG
        print("🎙️ HOME: speech STOP")
        #endif
    }

    private func cancelHomeSpeech() {
        homeSpeechService?.cancel()
        isHomeRecordingSpeech = false
        stopHomeSpeechTimer()

        #if DEBUG
        print("🎙️ HOME: speech CANCEL")
        #endif
    }

    private func startHomeSpeechTimer() {
        homeSpeechTimerTask?.cancel()
        homeSpeechTimerTask = Task {
            while !Task.isCancelled && isHomeRecordingSpeech {
                try? await Task.sleep(nanoseconds: 1_000_000_000)
                guard isHomeRecordingSpeech else { break }
                await MainActor.run {
                    homeSpeechSeconds += 1
                    homeSpeechTimerText = formatMMSS(homeSpeechSeconds)
                }
            }
        }
    }

    private func stopHomeSpeechTimer() {
        homeSpeechTimerTask?.cancel()
        homeSpeechTimerTask = nil
        homeSpeechSeconds = 0
        homeSpeechTimerText = "00:00"
    }

    private func formatMMSS(_ seconds: Int) -> String {
        let m = seconds / 60
        let s = seconds % 60
        return String(format: "%02d:%02d", m, s)
    }

    // MARK: - Send prompt (navigate to new chat)

    private func sendQuickPromptIfPossible() {
        let text = quickPromptText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard text.isEmpty == false else { return }

        if isHomeRecordingSpeech {
            stopHomeSpeech()
        }

        quickChatNavToken = UUID()
        quickPromptText = text
        isQuickChatActive = true
    }
}

struct ModelPickerSheetOverlay: View {

    @Binding var selected: NeuralModelOption
    let onClose: () -> Void

    private let topInset: CGFloat = 0.scale
    private let dismissThreshold: CGFloat = 60.scale

    private var accent: Color { Tokens.Color.accent }
    private var sheetBG: Color { Color(hex: "FAFAF5") ?? Color.white }
    private var cellBG: Color { Color(hex: "CB704D")?.opacity(0.15) ?? Color.black.opacity(0.06) }

    @GestureState private var dragY: CGFloat = 0
    @State private var settledOffsetY: CGFloat = 0
    @State private var isDismissing: Bool = false

    var body: some View {
        GeometryReader { geo in
            let currentY = max(0, settledOffsetY + dragY)

            ZStack(alignment: .bottom) {

                Color.black.opacity(0.45)
                    .ignoresSafeArea()
                    .onTapGesture { animateDismiss(height: geo.size.height) }

                sheet
                    .frame(maxWidth: .infinity)
                    .frame(height: geo.size.height - topInset, alignment: .top)
                    .background(sheetBG)
                    .clipShape(TopRoundedCorner(radius: 24.scale))
                    .frame(maxHeight: .infinity, alignment: .bottom)
                    .padding(.top, topInset)
                    .ignoresSafeArea(edges: .bottom)

                    // ✅ двигаем лист
                    .offset(y: currentY)

                    // ✅ свайп вниз
                    .gesture(
                        DragGesture(minimumDistance: 8, coordinateSpace: .global)
                            .updating($dragY) { value, state, _ in
                                if value.translation.height > 0, !isDismissing {
                                    state = value.translation.height
                                }
                            }
                            .onEnded { value in
                                guard !isDismissing else { return }

                                let dy = value.translation.height

                                if dy > dismissThreshold {
                                    // ✅ удерживаем позицию (чтобы не было “снэпа” вверх)
                                    settledOffsetY = max(0, dy)
                                    animateDismiss(height: geo.size.height)
                                } else {
                                    withAnimation(.spring(response: 0.28, dampingFraction: 0.92)) {
                                        settledOffsetY = 0
                                    }
                                }
                            }
                    )
            }
            .ignoresSafeArea()
        }
    }

    private func animateDismiss(height: CGFloat) {
        guard !isDismissing else { return }
        isDismissing = true

        // ✅ уезжаем вниз (без “прыжка”)
        withAnimation(.easeOut(duration: 0.22)) {
            settledOffsetY = height + 40.scale
        }

        // ✅ после анимации реально закрываем оверлей
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.22) {
            onClose()
            // сброс на всякий случай (когда снова откроется)
            settledOffsetY = 0
            isDismissing = false
        }
    }

    private var sheet: some View {

import Foundation

@MainActor
final class NotificationsListViewModel: ObservableObject {

    @Published private(set) var items: [InternalNotification] = []
    @Published private(set) var isLoading: Bool = false

    private let center: InternalNotificationCenter

    init(center: InternalNotificationCenter) {
        self.center = center
    }

    func refresh() async {
        isLoading = true
        defer { isLoading = false }

        let list = await center.list()
        items = list
    }

    func markRead(id: UUID) async {
        await center.markRead(id: id)
        await refresh()
    }
}

                .monospacedDigit()
        }
    }

    private var reservationsCard: some View {
        VStack(alignment: .leading, spacing: 10) {
            Text("Active reservations")
                .font(.system(size: 13, weight: .semibold))
                .foregroundStyle(.secondary)

            if vm.snapshot.reservations.isEmpty {
                Text("Нет активных job’ов.")
                    .font(.system(size: 14))
                    .foregroundStyle(.secondary)
                    .padding(.vertical, 6)
            } else {
                VStack(spacing: 10) {
                    ForEach(sorted(vm.snapshot.reservations)) { r in
                        reservationRow(r)
                    }
                }
            }
        }
        .padding(12)
        .background(Color.black.opacity(0.03))
        .clipShape(RoundedRectangle(cornerRadius: 14, style: .continuous))
        .overlay(
            RoundedRectangle(cornerRadius: 14, style: .continuous)
                .stroke(Color.black.opacity(0.08), lineWidth: 1)
        )
    }

    private func sorted(_ items: [RequestReservation]) -> [RequestReservation] {
        items.sorted(by: { $0.createdAt > $1.createdAt })
    }

    private func reservationRow(_ r: RequestReservation) -> some View {
        let age = max(0, Date().timeIntervalSince(r.createdAt))
        let ageText = formatAge(age)

        return VStack(alignment: .leading, spacing: 6) {
            HStack {
                Text(r.category.rawValue)
                    .font(.system(size: 12, weight: .semibold))
                    .padding(.horizontal, 8)
                    .padding(.vertical, 4)
                    .background(Color.black.opacity(0.08))
                    .clipShape(Capsule(style: .continuous))

                Spacer()

                Text(ageText)
                    .font(.system(size: 12, weight: .semibold))
                    .monospacedDigit()
                    .foregroundStyle(.secondary)
            }

            Text(r.id)
                .font(.system(size: 12))
                .foregroundStyle(.secondary)
                .textSelection(.enabled)
        }
        .padding(10)
        .background(Color.white.opacity(0.75))
        .clipShape(RoundedRectangle(cornerRadius: 12, style: .continuous))
        .overlay(
            RoundedRectangle(cornerRadius: 12, style: .continuous)
                .stroke(Color.black.opacity(0.06), lineWidth: 1)
        )
    }

    private func formatAge(_ seconds: TimeInterval) -> String {
        let total = Int(seconds.rounded(.down))
        let m = total / 60
        let s = total % 60
        return String(format: "%02d:%02d", m, s)
    }
}

import Foundation

@MainActor
public final class RequestCounterDebugViewModel: ObservableObject {

    @Published public private(set) var snapshot: RequestCounterSnapshot = .init()
    @Published public private(set) var isRunning: Bool = false

    private let counter: RequestCounterService
    private var task: Task<Void, Never>?

    public init(counter: RequestCounterService) {
        self.counter = counter
    }

    deinit {
        stop()
    }

    // MARK: - Public API (safe from any context)

    public nonisolated func start(pollIntervalSeconds: TimeInterval = 0.6) {
        Task { @MainActor [weak self] in
            self?._startOnMainActor(pollIntervalSeconds: pollIntervalSeconds)
        }
    }

    public nonisolated func stop() {
        Task { @MainActor [weak self] in
            self?._stopOnMainActor()
        }
    }

    public func refreshOnce() async {
        let snap = await counter.snapshot()
        snapshot = snap
    }

    public func debugLog(tag: String) async {
        await counter.debugLogCurrentState(tag: tag)
    }

    // MARK: - MainActor internals

    private func _startOnMainActor(pollIntervalSeconds: TimeInterval) {
        guard task == nil else { return }

        isRunning = true

        task = Task { [weak self] in
            guard let self else { return }

            await self.refreshOnce()

            while !Task.isCancelled {
                try? await Task.sleep(nanoseconds: UInt64(pollIntervalSeconds * 1_000_000_000))
                await self.refreshOnce()
            }
        }
    }

    private func _stopOnMainActor() {
        task?.cancel()
        task = nil
        isRunning = false
    }
}
