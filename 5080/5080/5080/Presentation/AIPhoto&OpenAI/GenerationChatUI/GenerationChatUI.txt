import SwiftUI

struct GenerationBubbleView: View {
    let message: GenerationMessage
    let onOpenMedia: (MediaViewerItem, GenerationMessage) -> Void

    private let avatarSize: CGFloat = 32.scale
    private let avatarToBubbleSpacing: CGFloat = 12.scale

    private let generationLoadingWindowSeconds: TimeInterval = 120

    // MARK: - Bubble style

    private let bubbleCorner: CGFloat = 14.scale
    private let systemCorner: CGFloat = 12.scale

    // Серую обводку можно подкрутить тут
    private var bubbleBorderColor: Color { Color.white.opacity(0.14) }
    private var bubbleBorderWidth: CGFloat { 1 } // если хочешь 1px — скажи, подставим onePixel

    var body: some View {
        switch message.role {
        case .system:
            systemRow
        case .assistant:
            assistantRow
        case .user:
            userRow
        }
    }

    // MARK: - Rows

    private var assistantRow: some View {
        HStack(alignment: .top, spacing: avatarToBubbleSpacing) {
            avatar(.assistant)
            bubble
            Spacer(minLength: 40)
        }
    }

    private var userRow: some View {
        HStack(alignment: .top, spacing: avatarToBubbleSpacing) {
            Spacer(minLength: 40)
            bubble
            avatar(.user)
        }
    }

    private var systemRow: some View {
        HStack {
            Spacer(minLength: 0)
            systemBubble
            Spacer(minLength: 0)
        }
    }

    // MARK: - Avatar

    private enum AvatarKind { case assistant, user }

    @ViewBuilder
    private func avatar(_ kind: AvatarKind) -> some View {
        Image(kind == .assistant ? "Chat.robot" : "Chat.user")
            .resizable()
            .scaledToFit()
            .frame(width: avatarSize, height: avatarSize)
    }

    // MARK: - Bubble

    private var bubble: some View {
        Group {
            switch message.payload {

            case .text(let text):
                bubbleTextContent(textBubbleWithOptionalAttachments(text))

            case .error(let text):
                bubbleTextContent(
                    textBubble(text)
                        .foregroundStyle(.red)
                )

            case .system(let text):
                bubbleTextContent(
                    textBubble(text)
                        .foregroundStyle(.white)
                )

            case .image(let remoteURL, let localURL):
                GenerationImageBubble(
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    generationLoadingWindowSeconds: generationLoadingWindowSeconds
                )
                .contentShape(Rectangle())
                .onTapGesture {
                    guard !isImageGenerating(createdAt: message.createdAt, localURL: localURL) else { return }

                    onOpenMedia(
                        .init(
                            id: message.id,
                            kind: .image,
                            remoteURL: remoteURL,
                            localURL: localURL
                        ),
                        message
                    )
                }

            case .video(let remoteURL, let localURL):
                GenerationVideoMessageBubble(
                    messageId: message.id,
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL
                )
                .id(videoBubbleRenderID(messageId: message.id, remoteURL: remoteURL, localURL: localURL))
                .contentShape(Rectangle())
                .onTapGesture {
                    guard !isVideoGenerating(createdAt: message.createdAt, localURL: localURL) else { return }

                    onOpenMedia(
                        .init(
                            id: message.id,
                            kind: .video,
                            remoteURL: remoteURL,
                            localURL: localURL
                        ),
                        message
                    )
                }
            }
        }
    }

    // MARK: - Text bubble wrapper (shared style)

    @ViewBuilder
    private func bubbleTextContent<Content: View>(_ content: Content) -> some View {
        content
            .padding(.vertical, 10.scale)
            .padding(.horizontal, 12.scale)
            .background(
                RoundedRectangle(cornerRadius: bubbleCorner, style: .continuous)
                    .fill(backgroundColor)
            )
            .overlay(
                RoundedRectangle(cornerRadius: bubbleCorner, style: .continuous)
                    .stroke(bubbleBorderColor, lineWidth: bubbleBorderWidth)
            )
    }

    @ViewBuilder
    private func textBubbleWithOptionalAttachments(_ text: String) -> some View {
        let trimmed = text.trimmingCharacters(in: .whitespacesAndNewlines)
        let hasText = !trimmed.isEmpty

        if message.role == .user, message.attachments.isEmpty == false {
            VStack(alignment: .leading, spacing: 8.scale) {

                GenerationMessageAttachmentsRow(attachments: message.attachments) { attachment in
                    guard attachment.kind == .image else { return }
                    guard let local = attachment.localURL, local.isFileURL else { return }

                    let item = MediaViewerItem(
                        id: attachment.id,
                        kind: .image,
                        remoteURL: local,
                        localURL: local
                    )

                    onOpenMedia(item, message)
                }

                if hasText {
                    textBubble(trimmed)
                }
            }
        } else {
            textBubble(hasText ? trimmed : text)
        }
    }

    // MARK: - Generating rules

    private func isVideoGenerating(createdAt: Date, localURL: URL?) -> Bool {
        guard localURL == nil else { return false }
        let age = Date().timeIntervalSince(createdAt)
        return age >= 0 && age <= generationLoadingWindowSeconds
    }

    private func isImageGenerating(createdAt: Date, localURL: URL?) -> Bool {
        guard localURL == nil else { return false }
        let age = Date().timeIntervalSince(createdAt)
        return age >= 0 && age <= generationLoadingWindowSeconds
    }

    // MARK: - System bubble (centered)

    private var systemBubble: some View {
        Group {
            switch message.payload {

            case .system(let text):
                systemText(text)

            case .text(let text):
                systemText(text)

            case .error(let text):
                systemText(text)

            case .image(let remoteURL, let localURL):
                GenerationImageBubble(
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    generationLoadingWindowSeconds: generationLoadingWindowSeconds
                )

            case .video(let remoteURL, let localURL):
                GenerationVideoMessageBubble(
                    messageId: message.id,
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL
                )
                .id(videoBubbleRenderID(messageId: message.id, remoteURL: remoteURL, localURL: localURL))
            }
        }
    }

    private func systemText(_ text: String) -> some View {
        Text(text)
            .font(.footnote)
            .foregroundStyle(.white)
            .multilineTextAlignment(.center)
            .padding(.vertical, 8.scale)
            .padding(.horizontal, 12.scale)
            .background(
                RoundedRectangle(cornerRadius: systemCorner, style: .continuous)
                    .fill(Color.black.opacity(0.02))
            )
    }

    private func videoBubbleRenderID(messageId: UUID, remoteURL: URL, localURL: URL?) -> String {
        let local = localURL?.absoluteString ?? "nil"
        return "\(messageId.uuidString)|\(remoteURL.absoluteString)|\(local)"
    }

    private func textBubble(_ text: String) -> some View {
        Text(text)
            .font(.body)
            .foregroundStyle(Tokens.Color.textPrimary)
    }

    private var backgroundColor: Color {
        switch message.role {
        case .user: return Color(hex: "#202020")!
        case .assistant: return Color(hex: "#202020")!
        case .system: return Color(hex: "#202020")!
        }
    }
}

import SwiftUI
import Foundation

extension GenerationChatComposerView {

    // MARK: - Speech handlers

    func handleMicToggle() {
        if isRecording {
            stopDictation()
            return
        }

        guard isMicEnabled else { return }

        guard isTextInputEnabled else {
            speechErrorText = "composer.speech.unavailable_in_mode".localizable
            return
        }

        guard let speechService else {
            speechErrorText = "composer.speech.service_not_found".localizable
            onMicTap()
            return
        }

        speechErrorText = nil

        Task { @MainActor in
            let status = await speechService.requestPermissions()

            guard status == .authorized else {
                switch status {
                case .denied:
                    speechErrorText = "composer.speech.permission_denied".localizable
                case .restricted:
                    speechErrorText = "composer.speech.permission_restricted".localizable
                case .notDetermined:
                    speechErrorText = "composer.speech.permission_not_determined".localizable
                default:
                    speechErrorText = "composer.speech.permission_unavailable".localizable
                }
                onMicTap()
                return
            }

            let sessionId = UUID()
            activeSpeechSessionId = sessionId

            isRecording = true
            startRecordTimer()

            speechService.start(
                onPartial: { partial in
                    Task { @MainActor in
                        guard activeSpeechSessionId == sessionId else { return }
                        self.text = partial
                    }
                },
                onFinal: { finalText in
                    Task { @MainActor in
                        guard activeSpeechSessionId == sessionId else { return }
                        self.text = finalText
                        self.isRecording = false
                        stopRecordTimer(reset: false)
                    }
                },
                onError: { err in
                    Task { @MainActor in
                        guard activeSpeechSessionId == sessionId else { return }
                        self.speechErrorText = String(
                            format: "composer.speech.error_format".localizable,
                            err.localizedDescription
                        )
                        self.isRecording = false
                        stopRecordTimer(reset: true)
                    }
                }
            )

            self.isRecording = speechService.isRunning
            if !self.isRecording {
                stopRecordTimer(reset: true)
            }
        }
    }

    func stopDictation() {
        activeSpeechSessionId = UUID()
        isRecording = false
        speechService?.stop()
        stopRecordTimer(reset: false)
    }

    func cancelDictation() {
        activeSpeechSessionId = UUID()
        isRecording = false
        speechService?.cancel()
        stopRecordTimer(reset: true)
    }
}

import SwiftUI
import Foundation

extension GenerationChatComposerView {

    var timerText: String {
        let mm = recordSeconds / 60
        let ss = recordSeconds % 60
        let mmStr = mm < 10 ? "0\(mm)" : "\(mm)"
        let ssStr = ss < 10 ? "0\(ss)" : "\(ss)"
        return "\(mmStr):\(ssStr)"
    }

    func startRecordTimer() {
        stopRecordTimer(reset: true)
        recordSeconds = 0
        recordTimer = Timer.scheduledTimer(withTimeInterval: 1.0, repeats: true) { _ in
            recordSeconds += 1
        }
        if let recordTimer {
            RunLoop.main.add(recordTimer, forMode: .common)
        }
    }

    func stopRecordTimer(reset: Bool) {
        recordTimer?.invalidate()
        recordTimer = nil
        if reset {
            recordSeconds = 0
        }
    }
}

import SwiftUI

extension GenerationChatComposerView {

    var bubbleContainer: some View {
        VStack(alignment: .leading, spacing: 0) {

            if shouldShowAttachmentsRow {
                attachmentsRow
                    .padding(.leading, 10.scale)
                    .padding(.top, 10.scale)
                    .padding(.bottom, 10.scale)
            }

            inputBarContent
        }
        .background(
            RoundedRectangle(cornerRadius: barCornerRadius, style: .continuous)
                .fill(Color(hex:"#202020")!)
        )
        .overlay(
            RoundedRectangle(cornerRadius: barCornerRadius, style: .continuous)
                .stroke(borderColor, lineWidth: 1)
        )
        .apply(Tokens.Shadow.card)
    }

    var inputBarContent: some View {
        HStack(spacing: 0) {

            HStack(spacing: 0) {
                if showsPlus {
                    Button {
                        guard !isLoading else { return }
                        onPickPhotoTap()
                    } label: {
                        Image(systemName: "plus")
                            .font(.system(size: 18.scale, weight: .semibold))
                            .foregroundStyle(placeholderColor)
                            .frame(width: plusSize.width, height: plusSize.height)
                            .contentShape(Rectangle())
                    }
                    .buttonStyle(.plain)

                    Spacer().frame(width: 16.scale)
                }
            }
            .frame(height: barHeight, alignment: .center)
            .padding(.leading, leftInset)

            ZStack(alignment: .leading) {
                if shouldShowPlaceholder {
                    Text(placeholderText)
                        .foregroundStyle(placeholderColor)
                        .font(Tokens.Font.regular17)
                }

                TextField("", text: $text, axis: .vertical)
                    .font(Tokens.Font.regular17)
                    .tint(inputTextColor)
                    .foregroundStyle(inputTextColor)
                    .lineLimit(1...4)
                    .disabled(isLoading || !isTextInputEnabled)
                    .submitLabel(.send)
                    .focused($isTextFieldFocused)
            }
            .frame(maxWidth: .infinity, alignment: .leading)
            .frame(minHeight: barHeight, alignment: .center)

            HStack(spacing: spaceBetweenRightButtons) {

                if isRecording {
                    Text(timerText)
                        .font(Tokens.Font.semibold11)
                        .monospacedDigit()
                        .foregroundStyle(Tokens.Color.textSecondary)
                        .padding(.horizontal, 6.scale)
                        .padding(.vertical, 4.scale)
                        .background(
                            Capsule(style: .continuous)
                                .fill(Color.black.opacity(0.06))
                        )
                }

                if showsVideoSettingsButton {
                }

                if rightIcon == .mic {
                    GenerationChatCircleIconButton(
                        imageName: "mic",
                        size: circleButtonSize,
                        fill: Tokens.Color.accent,
                        tint: .white,
                        isDisabled: isLoading,
                        accessibilityLabel: "Микрофон"
                    ) {
                        guard !isLoading else { return }
                        handleMicToggle()
                    }
                } else {
                    GenerationChatCircleIconButton(
                        systemName: iconName(for: rightIcon),
                        size: circleButtonSize,
                        fill: Tokens.Color.accent,
                        tint: .white,
                        isDisabled: isLoading || (rightIcon == .send && !isSendEnabled),
                        accessibilityLabel: (rightIcon == .send ? "Отправить" : "Остановить")
                    ) {
                        guard !isLoading else { return }

                        switch rightIcon {
                        case .mic:
                            handleMicToggle()
                        case .stop:
                            stopDictation()
                        case .send:
                            guard isSendEnabled else { return }
                            if isRecording { stopDictation() }
                            isTextFieldFocused = false
                            onSendTap()
                        }
                    }
                }
            }
            .frame(height: barHeight, alignment: .center)
            .padding(.trailing, rightInset)
        }
        .frame(height: barHeight, alignment: .center)
        .animation(.easeOut(duration: 0.12), value: isTextEmpty)
        .animation(.easeOut(duration: 0.12), value: isRecording)
        .onChange(of: isLoading) { _, newValue in
            if newValue {
                isTextFieldFocused = false
            }
        }
    }
}

import SwiftUI
import UIKit
import Foundation
import ZIPFoundation

extension GenerationChatContainerView {

    // MARK: - Actions

    func copyChatToPasteboard() {
        let text = GenerationChatExportService.makeShareText(title: title, messages: messages)
        UIPasteboard.general.string = text
    }

    func shareChatAsDocument() {
        exportChat()
    }

    func shareChatText() {
        sharePayload = SharePayload(items: [
            GenerationChatExportService.makeShareText(title: title, messages: messages)
        ])
    }

    func exportChat() {
        guard let threadId, let mediaCache else { return }
        actionErrorText = nil

        guard isExporting == false else { return }

        isExporting = true

        Task { @MainActor in
            defer { isExporting = false }

            do {
                let result = try await GenerationChatExportService.exportDirectory(
                    title: title,
                    threadId: threadId,
                    messages: messages,
                    mediaCache: mediaCache
                )

                let directoryURL = result.directoryURL
                exportedTempDirectoryURL = directoryURL

                let zipURL = try makeZip(from: directoryURL, title: title)
                exportedTempZipURL = zipURL

                sharePayload = SharePayload(items: [zipURL])

            } catch {
                actionErrorText = "Export failed: \(String(describing: error))"
            }
        }
    }

    func cleanupExportTempIfNeeded() {
        if let zipURL = exportedTempZipURL {
            exportedTempZipURL = nil
            try? FileManager.default.removeItem(at: zipURL)
        }

        if let dirURL = exportedTempDirectoryURL {
            exportedTempDirectoryURL = nil
            try? FileManager.default.removeItem(at: dirURL)
        }
    }

    func deleteChat() {
        guard let historyRepo, let threadId else { return }
        actionErrorText = nil
        isDeleting = true

        Task { @MainActor in
            defer { isDeleting = false }
            do {
                try await historyRepo.delete(threadId: threadId)
                dismiss()
            } catch {
                actionErrorText = "Delete failed: \(String(describing: error))"
            }
        }
    }
}

// MARK: - ZIP helper (ZIPFoundation)

private extension GenerationChatContainerView {

    func makeZip(from directoryURL: URL, title: String) throws -> URL {
        let safeTitle = title
            .trimmingCharacters(in: .whitespacesAndNewlines)
            .replacingOccurrences(of: "/", with: "_")
            .replacingOccurrences(of: ":", with: "_")

        let fileName = (safeTitle.isEmpty ? "chat_export" : safeTitle) + ".zip"
        let zipURL = FileManager.default.temporaryDirectory.appendingPathComponent(fileName)

        try? FileManager.default.removeItem(at: zipURL)

        try FileManager.default.zipItem(at: directoryURL, to: zipURL, shouldKeepParent: true)

        return zipURL
    }
}

            message: message,
            kind: kind,
            historyRepo: historyRepo,

import Foundation

enum GenerationChatExportService {

    struct ExportResult: Sendable {
        let directoryURL: URL
    }

    // MARK: - Share / Copy (text-only)

    static func makeShareText(title: String, messages: [GenerationMessage]) -> String {
        var lines: [String] = []
        lines.append(title)
        lines.append("")

        let sorted = stableSortMessages(messages)

        var imageIndex = 0
        var videoIndex = 0

        for msg in sorted {
            let prefix: String
            switch msg.role {
            case .user: prefix = "User"
            case .assistant: prefix = "Assistant"
            case .system: prefix = "System"
            }

            switch msg.payload {
            case .text(let t):
                lines.append("[\(prefix)] \(t)")

            case .system(let t):
                lines.append("[\(prefix)] \(t)")

            case .error(let t):
                lines.append("[\(prefix)] \(t)")

            case .image:
                imageIndex += 1
                lines.append("[\(prefix)] [image #\(imageIndex)]")

            case .video:
                videoIndex += 1
                lines.append("[\(prefix)] [video #\(videoIndex)]")
            }

            lines.append("")
        }

        return lines.joined(separator: "\n").trimmingCharacters(in: .whitespacesAndNewlines)
    }

    // MARK: - Export (Directory with attachments)

    static func exportDirectory(
        title: String,
        threadId: UUID,
        messages: [GenerationMessage],
        mediaCache: MediaCacheService
    ) async throws -> ExportResult {

        let workspace = FileManager.default.temporaryDirectory
            .appendingPathComponent("chat_export_\(UUID().uuidString)", isDirectory: true)

        let mediaDir = workspace.appendingPathComponent("media", isDirectory: true)

        try FileManager.default.createDirectory(at: workspace, withIntermediateDirectories: true)
        try FileManager.default.createDirectory(at: mediaDir, withIntermediateDirectories: true)

        let chatTextURL = workspace.appendingPathComponent("chat.txt", isDirectory: false)

        let sorted = stableSortMessages(messages)

        var lines: [String] = []
        lines.append(title)
        lines.append("")

        var imageIndex = 0
        var videoIndex = 0

        for msg in sorted {
            let prefix: String
            switch msg.role {
            case .user: prefix = "User"
            case .assistant: prefix = "Assistant"
            case .system: prefix = "System"
            }

            switch msg.payload {
            case .text(let t):
                lines.append("[\(prefix)] \(t)")

            case .system(let t):
                lines.append("[\(prefix)] \(t)")

            case .error(let t):
                lines.append("[\(prefix)] \(t)")

            case .image(let remoteURL, let localURL):
                imageIndex += 1
                let exported = try await exportMediaFile(
                    kind: "image",
                    index: imageIndex,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    threadId: threadId,
                    mediaCache: mediaCache,
                    mediaDir: mediaDir
                )
                lines.append("[\(prefix)] [image #\(imageIndex)] \(exported.relativePath)")

            case .video(let remoteURL, let localURL):
                videoIndex += 1
                let exported = try await exportMediaFile(
                    kind: "video",
                    index: videoIndex,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    threadId: threadId,
                    mediaCache: mediaCache,
                    mediaDir: mediaDir
                )
                lines.append("[\(prefix)] [video #\(videoIndex)] \(exported.relativePath)")
            }

            lines.append("")
        }

        let text = lines.joined(separator: "\n").trimmingCharacters(in: .whitespacesAndNewlines)
        try text.data(using: .utf8)?.write(to: chatTextURL, options: [.atomic])

        return ExportResult(directoryURL: workspace)
    }

    // MARK: - Stable sort (same fix as in history load)

    private static func stableSortMessages(_ items: [GenerationMessage]) -> [GenerationMessage] {
        items.sorted { lhs, rhs in
            if lhs.createdAt != rhs.createdAt {
                return lhs.createdAt < rhs.createdAt
            }

            let lp = rolePriority(lhs.role)
            let rp = rolePriority(rhs.role)
            if lp != rp { return lp < rp }

            return lhs.id.uuidString < rhs.id.uuidString
        }
    }

    private static func rolePriority(_ role: GenerationMessageRole) -> Int {
        switch role {
        case .system: return 0
        case .user: return 1
        case .assistant: return 2
        }
    }

    // MARK: - Private

    private struct ExportedMedia {
        let relativePath: String // e.g. "media/image_001.jpg"
    }

    private static func exportMediaFile(
        kind: String, // "image" | "video"
        index: Int,
        remoteURL: URL,
        localURL: URL?,
        threadId: UUID,
        mediaCache: MediaCacheService,
        mediaDir: URL
    ) async throws -> ExportedMedia {

        let sourceURL: URL

        if let localURL, FileManager.default.fileExists(atPath: localURL.path) {
            sourceURL = localURL
        } else {
            sourceURL = try await mediaCache.fetch(remoteURL: remoteURL, threadId: threadId, kind: kind)
        }

        let ext = bestFileExtension(preferLocal: sourceURL, fallbackRemote: remoteURL, kind: kind)
        let fileName = "\(kind)_\(String(format: "%03d", index)).\(ext)"
        let dstURL = mediaDir.appendingPathComponent(fileName, isDirectory: false)

        if FileManager.default.fileExists(atPath: dstURL.path) {
            try? FileManager.default.removeItem(at: dstURL)
        }

        try FileManager.default.copyItem(at: sourceURL, to: dstURL)

        return ExportedMedia(relativePath: "media/\(fileName)")
    }

    private static func bestFileExtension(preferLocal local: URL, fallbackRemote remote: URL, kind: String) -> String {
        let localExt = local.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        if localExt.isEmpty == false { return localExt.lowercased() }

        let remoteExt = remote.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        if remoteExt.isEmpty == false { return remoteExt.lowercased() }

        if kind == "video" { return "mp4" }
        return "jpg"
    }
}

                .overlay(

                        .stroke(bubbleStroke, lineWidth: bubbleStrokeWidth)
                )

            content()
                .frame(width: innerSide, height: innerSide)
                .clipShape(RoundedRectangle(cornerRadius: innerCorner, style: .continuous))
        }
        .frame(width: containerSide, height: containerSide)
        .apply(Tokens.Shadow.card)

import SwiftUI
import AVFoundation
import UIKit

struct GenerationVideoMessageBubble: View {

    let messageId: UUID
    let createdAt: Date
    let remoteURL: URL
    let localURL: URL?

    // MARK: - Tuning

    private let durationWarmupWindowSeconds: TimeInterval = 25
    private let durationWarmupTick: TimeInterval = 0.8
    private let thumbnailWarmupWindowSeconds: TimeInterval = 20
    private let thumbnailWarmupTick: TimeInterval = 0.8

    // MARK: - State

    @State private var durationText: String? = nil

    @State private var durationTask: Task<Void, Never>? = nil

    @State private var warmupTask: Task<Void, Never>? = nil

    @State private var thumbnailTask: Task<Void, Never>? = nil

    @State private var thumbnailWarmupTask: Task<Void, Never>? = nil

    @State private var lastRequestKey: String = ""

    @State private var lastThumbnailRequestKey: String = ""

    @State private var isComputing: Bool = false

    @State private var thumbnailImage: UIImage? = nil

    var body: some View {
        Group {
            if shouldShowLoading {
                GenerationVideoStatusBubble(state: .loading)
            } else {
                GenerationVideoStatusBubble(
                    state: .ready(durationText: durationTextOrPlaceholder),
                    thumbnailImage: thumbnailImage
                )
            }
        }
        .onAppear {
            kickDurationComputation(reason: "onAppear")
            kickThumbnailComputation(reason: "onAppear")
            startWarmupIfNeeded()
            startThumbnailWarmupIfNeeded()
        }
        .onChange(of: localURL?.absoluteString ?? "") { _ in
            kickDurationComputation(reason: "localURL changed")
            kickThumbnailComputation(reason: "localURL changed")
            startWarmupIfNeeded()
            startThumbnailWarmupIfNeeded()
        }
        .onChange(of: remoteURL.absoluteString) { _ in
            kickDurationComputation(reason: "remoteURL changed")
            kickThumbnailComputation(reason: "remoteURL changed")
            startWarmupIfNeeded()
            startThumbnailWarmupIfNeeded()
        }
        .onDisappear {
            cancelAllTasks()
        }
    }

    // MARK: - Loading rules

    private var shouldShowLoading: Bool {
        isPlaceholderRemote(remoteURL)
    }

    private func isPlaceholderRemote(_ url: URL) -> Bool {
        if url.host?.lowercased() == "example.com" { return true }
        let s = url.absoluteString.lowercased()
        return s.contains("example.com") || s.hasSuffix("/video.mp4")
    }

    // MARK: - Remote duration policy

    private func isRemoteDurationAllowed(_ url: URL) -> Bool {
        guard !isPlaceholderRemote(url) else { return false }
        let scheme = (url.scheme ?? "").lowercased()
        return scheme == "http" || scheme == "https"
    }

    // MARK: - Duration compute orchestration

    private func kickDurationComputation(reason: String) {
        guard !shouldShowLoading else {
            durationText = nil
            durationTask?.cancel()
            durationTask = nil
            warmupTask?.cancel()
            warmupTask = nil
            isComputing = false
            lastRequestKey = ""
            return
        }

        let localCandidate: URL? = {
            guard let localURL, localURL.isFileURL, fileExists(localURL) else { return nil }
            return localURL
        }()

        let remoteCandidate: URL? = isRemoteDurationAllowed(remoteURL) ? remoteURL : nil

        guard localCandidate != nil || remoteCandidate != nil else {
            durationText = nil
            return
        }

        let requestKey = makeRequestKey(local: localCandidate, remote: remoteCandidate)

        #if DEBUG
        let age = Date().timeIntervalSince(createdAt)
        print("🎬 [BUBBLE] kick msgId=\(messageId) reason=\(reason)")
        print("🎬 [BUBBLE] age=\(String(format: "%.1f", age))s loading=\(shouldShowLoading)")
        print("🎬 [BUBBLE] key=\(requestKey)")
        print("🎬 [BUBBLE] local=\(localCandidate?.path ?? "nil")")
        print("🎬 [BUBBLE] remote=\(remoteCandidate?.absoluteString ?? "nil")")
        #endif

        if requestKey == lastRequestKey {
            if let t = durationText, !t.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty { return }
            if isComputing { return }
        }

        durationTask?.cancel()
        durationTask = nil

        lastRequestKey = requestKey
        isComputing = true

        durationTask = Task { [messageId, localCandidate, remoteCandidate, requestKey] in
            defer {
                Task { @MainActor in
                    if self.lastRequestKey == requestKey {
                        self.isComputing = false
                    }
                }
            }

            if let localCandidate,
               let text = await VideoDurationTextProvider.durationText(for: localCandidate) {
                if Task.isCancelled { return }
                await MainActor.run {
                    guard self.lastRequestKey == requestKey else { return }
                    self.durationText = text
                }
                return
            }

            if let remoteCandidate,
               let text = await VideoDurationTextProvider.durationText(for: remoteCandidate) {
                if Task.isCancelled { return }
                await MainActor.run {
                    guard self.lastRequestKey == requestKey else { return }
                    self.durationText = text
                }
                return
            }

            if Task.isCancelled { return }
            await MainActor.run {
                guard self.lastRequestKey == requestKey else { return }
                self.durationText = nil
            }

            #if DEBUG
            print("🎬 [BUBBLE] done msgId=\(messageId) key=\(requestKey) duration=nil")
            #endif
        }
    }

    private func kickThumbnailComputation(reason: String) {
        guard !shouldShowLoading else {
            thumbnailImage = nil
            thumbnailTask?.cancel()
            thumbnailTask = nil
            thumbnailWarmupTask?.cancel()
            thumbnailWarmupTask = nil
            lastThumbnailRequestKey = ""
            return
        }

        let localCandidate: URL? = {
            guard let localURL, localURL.isFileURL, fileExists(localURL) else { return nil }
            return localURL
        }()

        let remoteCandidate: URL? = isRemoteDurationAllowed(remoteURL) ? remoteURL : nil

        guard localCandidate != nil || remoteCandidate != nil else {
            thumbnailImage = nil
            return
        }

        let requestKey = makeRequestKey(local: localCandidate, remote: remoteCandidate)
        if requestKey == lastThumbnailRequestKey, thumbnailImage != nil { return }

        thumbnailTask?.cancel()
        thumbnailTask = nil
        lastThumbnailRequestKey = requestKey

        thumbnailTask = Task {
            let resolved: UIImage?
            if let localCandidate {
                resolved = await VideoThumbnailProvider.thumbnail(for: localCandidate)
            } else if let remoteCandidate {
                resolved = await VideoThumbnailProvider.thumbnail(for: remoteCandidate)
            } else {
                resolved = nil
            }

            if Task.isCancelled { return }

            await MainActor.run {
                guard self.lastThumbnailRequestKey == requestKey else { return }
                self.thumbnailImage = resolved
            }

            #if DEBUG
            let hasThumb = (resolved != nil)
            print("🎬 [THUMB] msgId=\(messageId) reason=\(reason) success=\(hasThumb) key=\(requestKey)")
            print("🎬 [THUMB] local=\(localCandidate?.path ?? "nil") remote=\(remoteCandidate?.absoluteString ?? "nil")")
            #endif
        }
    }

    private func startThumbnailWarmupIfNeeded() {
        guard !shouldShowLoading else { return }
        if thumbnailImage != nil { return }
        if thumbnailWarmupTask != nil { return }

        thumbnailWarmupTask = Task {
            let deadline = Date().addingTimeInterval(thumbnailWarmupWindowSeconds)

            while !Task.isCancelled, Date() < deadline {
                let hasThumbnail = await MainActor.run { self.thumbnailImage != nil }
                if hasThumbnail { break }

                await MainActor.run {
                    self.kickThumbnailComputation(reason: "thumbnail warmup")
                }

                try? await Task.sleep(nanoseconds: UInt64(thumbnailWarmupTick * 1_000_000_000))
            }

            await MainActor.run {
                self.thumbnailWarmupTask?.cancel()
                self.thumbnailWarmupTask = nil
            }
        }
    }

    private func startWarmupIfNeeded() {
        guard !shouldShowLoading else { return }
        if let t = durationText, !t.isEmpty { return }
        if warmupTask != nil { return }

        warmupTask = Task {
            let deadline = Date().addingTimeInterval(durationWarmupWindowSeconds)
            var didNudgeOnce = false

            while !Task.isCancelled, Date() < deadline {
                let current = await MainActor.run { self.durationText }
                if let current, !current.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
                    break
                }

                if !didNudgeOnce, Date() > deadline.addingTimeInterval(-durationWarmupWindowSeconds / 2) {
                    didNudgeOnce = true
                    await MainActor.run {
                        self.kickDurationComputation(reason: "warmup nudge")
                    }
                }

                try? await Task.sleep(nanoseconds: UInt64(durationWarmupTick * 1_000_000_000))
            }

            await MainActor.run {
                self.warmupTask?.cancel()
                self.warmupTask = nil
            }
        }
    }

    private func cancelAllTasks() {
        durationTask?.cancel()
        durationTask = nil
        warmupTask?.cancel()
        warmupTask = nil
        thumbnailTask?.cancel()
        thumbnailTask = nil
        thumbnailWarmupTask?.cancel()
        thumbnailWarmupTask = nil
        isComputing = false
    }

    private func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }

    private func makeRequestKey(local: URL?, remote: URL?) -> String {
        let l = local?.path ?? "nil"
        let r = remote?.absoluteString ?? "nil"
        return "L:\(l)|R:\(r)"
    }

    private var durationTextOrPlaceholder: String {
        let t = (durationText ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
        return t.isEmpty ? "—:—" : t
    }
}

private enum VideoThumbnailProvider {

    private static let imageCache = NSCache<NSString, UIImage>()

    static func thumbnail(for url: URL, maxPixelSize: CGFloat = 1000) async -> UIImage? {
        let key = cacheKey(for: url, maxPixelSize: maxPixelSize)
        if let cached = imageCache.object(forKey: key as NSString) {
            return cached
        }

        let resolved = await makeThumbnailSmart(url: url, maxPixelSize: maxPixelSize)

        if let resolved {
            imageCache.setObject(resolved, forKey: key as NSString)
        }

        return resolved
    }

    private static func cacheKey(for url: URL, maxPixelSize: CGFloat) -> String {
        "\(url.absoluteString)|\(Int(maxPixelSize.rounded()))"
    }

    private static func makeThumbnailSmart(url: URL, maxPixelSize: CGFloat) async -> UIImage? {
        let direct = await Task.detached(priority: .utility) {
            return makeThumbnail(url: url, maxPixelSize: maxPixelSize)
        }.value
        if let direct {
            return direct
        }

        guard !url.isFileURL else { return nil }

        guard let tmp = await downloadToTemporaryFile(url) else { return nil }
        defer { try? FileManager.default.removeItem(at: tmp) }

        return await Task.detached(priority: .utility) {
            return makeThumbnail(url: tmp, maxPixelSize: maxPixelSize)
        }.value
    }

    private static func downloadToTemporaryFile(_ url: URL) async -> URL? {
        var req = URLRequest(url: url)
        req.httpMethod = "GET"
        req.timeoutInterval = 60

        do {
            let (tmp, resp) = try await URLSession.shared.download(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }
            guard (200..<300).contains(http.statusCode) else { return nil }

            let ext = url.pathExtension.isEmpty ? "mp4" : url.pathExtension
            let out = FileManager.default.temporaryDirectory
                .appendingPathComponent("thumb-\(UUID().uuidString)")
                .appendingPathExtension(ext)

            try? FileManager.default.removeItem(at: out)
            try FileManager.default.moveItem(at: tmp, to: out)
            return out
        } catch {
            return nil
        }
    }

    private static func makeThumbnail(url: URL, maxPixelSize: CGFloat) -> UIImage? {
        let asset = AVURLAsset(url: url)
        let generator = AVAssetImageGenerator(asset: asset)
        generator.appliesPreferredTrackTransform = true
        generator.maximumSize = CGSize(width: maxPixelSize, height: maxPixelSize)
        generator.requestedTimeToleranceBefore = .positiveInfinity
        generator.requestedTimeToleranceAfter = .positiveInfinity

        let preferredScale: CMTimeScale = 600
        let probeTimes: [CMTime] = [
            CMTime(seconds: 1.0, preferredTimescale: preferredScale),
            CMTime(seconds: 0.2, preferredTimescale: preferredScale),
            CMTime(seconds: 0.6, preferredTimescale: preferredScale),
            .zero
        ]

        for time in probeTimes {
            do {
                let cgImage = try generator.copyCGImage(at: time, actualTime: nil)
                return UIImage(cgImage: cgImage)
            } catch {
                continue
            }
        }

        return nil
    }
}

    let font: Font
    let color: Color

        self.baseText = baseText
        self.font = font
        self.color = color
    }

    var body: some View {
        TimelineView(.periodic(from: .now, by: 0.45)) { context in
            let dotsCount = dotsCount(for: context.date)
            Text(baseText + String(repeating: "•", count: dotsCount))
                .font(font)
                .foregroundStyle(color)
        }
    }

    private func dotsCount(for date: Date) -> Int {

        let step = Int(floor(t / 0.45)) % 3
        return step + 1
    }

    private func startRecoveryIfNeeded() {
        guard recoveryTask == nil else { return }
        guard let ctx = recovery else { return }

        if let local = resolvedLocalURL, fileExists(local) { return }

        isRecoveringCache = true

        recoveryTask = Task { @MainActor in
            defer {
                isRecoveringCache = false
            }

            do {
                let local = try await ctx.mediaCache.fetch(
                    remoteURL: item.remoteURL,
                    threadId: ctx.threadId,
                    kind: ctx.kind
                )
                try Task.checkCancellation()

                let updatedMessage = makeUpdatedMessage(original: ctx.message, localURL: local)

                try await ctx.historyRepo.replaceMessage(
                    threadId: ctx.threadId,
                    messageId: updatedMessage.id,
                    newMessage: updatedMessage
                )

                await ctx.onMessageUpdated(updatedMessage)

                resolvedLocalURL = local

                if item.kind == .video {
                    player?.pause()
                    player = AVPlayer(url: local)
                }

                #if DEBUG
                print("🧾 MEDIA_RECOVERY: success threadId=\(ctx.threadId) msgId=\(updatedMessage.id) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY: cancelled")
                #endif
            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY: failed error=\(error)")
                #endif
            }

            recoveryTask = nil
        }
    }

    private func makeUpdatedMessage(original: GenerationMessage, localURL: URL) -> GenerationMessage {
        switch original.payload {
        case .image(let remote, _):
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remote, localURL: localURL)
            )
        case .video(let remote, _):
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remote, localURL: localURL)
            )
        default:
            return original
        }

        isSaving = true

        Task { @MainActor in
            defer { isSaving = false }

            do {
                switch item.kind {
                case .image:
                    if let local = resolvedLocalURL, fileExists(local) {
                        try await PhotoLibrarySaver.saveLocalImage(at: local)
                    } else {
                        try await PhotoLibrarySaver.saveRemoteImage(from: item.remoteURL)

                case .video:
                    if let local = resolvedLocalURL, fileExists(local) {
                        try await PhotoLibrarySaver.saveLocalVideo(at: local)
                    } else {
                        try await PhotoLibrarySaver.saveRemoteVideo(from: item.remoteURL)
                    }
                }

                alertText = "media_viewer.save.success".localizable
                showAlert = true

            } catch is CancellationError {
            } catch {
                alertText = String(
                    format: "media_viewer.save.failed_format".localizable,
                    error.localizedDescription
                )
                showAlert = true
            }
        }
    }

    // MARK: - Helpers

    private func bestURLForPlayback() -> URL {
        if let local = resolvedLocalURL, fileExists(local) {
            return local
        }
        return item.remoteURL
    }

    private func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }
}

import Foundation
import Photos
import UIKit

enum PhotoLibrarySaver {

    enum SaveError: LocalizedError {
        case permissionDenied
        case invalidLocalImage
        case invalidLocalVideo
        case invalidRemoteImageData
        case downloadFailed(String)
        case saveFailed(String)

        var errorDescription: String? {
            switch self {
            case .permissionDenied:
                return "photo_saver.error.permission_denied".localizable
            case .invalidLocalImage:
                return "photo_saver.error.invalid_local_image".localizable
            case .invalidLocalVideo:
                return "photo_saver.error.invalid_local_video".localizable
            case .invalidRemoteImageData:
                return "photo_saver.error.invalid_remote_image_data".localizable
            case .downloadFailed(let text):
                return text.isEmpty ? "photo_saver.error.download_failed".localizable : text
            case .saveFailed(let text):
                return text.isEmpty ? "photo_saver.error.save_failed".localizable : text
            }
        }
    }

    // MARK: - Public API (Local)

    static func saveLocalImage(at url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        guard FileManager.default.fileExists(atPath: url.path),
              let image = UIImage(contentsOfFile: url.path) else {
            throw SaveError.invalidLocalImage
        }

        try await performChanges {
            PHAssetChangeRequest.creationRequestForAsset(from: image)
        }
    }

    static func saveLocalVideo(at url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        guard FileManager.default.fileExists(atPath: url.path) else {
            throw SaveError.invalidLocalVideo
        }

        try await performChanges {
            PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: url)
        }
    }

    // MARK: - Public API (Remote / Temp)

    static func saveRemoteImage(from url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        do {
            let (data, _) = try await URLSession.shared.data(from: url)
            try Task.checkCancellation()

            guard let image = UIImage(data: data) else {
                throw SaveError.invalidRemoteImageData
            }

            try await performChanges {
                PHAssetChangeRequest.creationRequestForAsset(from: image)
            }
        } catch is CancellationError {
            throw CancellationError()
        } catch {
            throw SaveError.downloadFailed(String(describing: error))
        }
    }

    static func saveRemoteVideo(from url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        do {
            let (tmpLocation, _) = try await URLSession.shared.download(from: url)
            try Task.checkCancellation()

            let finalURL = try copyDownloadToTempWithExtension(from: tmpLocation, remoteURL: url)

            try await performChanges {
                PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: finalURL)
            }

            try? FileManager.default.removeItem(at: finalURL)

        } catch is CancellationError {
            throw CancellationError()
        } catch {
            throw SaveError.downloadFailed(String(describing: error))
        }
    }

    // MARK: - Authorization

    private static func ensureAddOnlyAuthorization() async -> PHAuthorizationStatus {
        let current = PHPhotoLibrary.authorizationStatus(for: .addOnly)
        if current != .notDetermined {
            return current
        }
        return await PHPhotoLibrary.requestAuthorization(for: .addOnly)
    }

    // MARK: - Perform changes

    private static func performChanges(_ changes: @escaping () -> Void) async throws {
        try await withCheckedThrowingContinuation { cont in
            PHPhotoLibrary.shared().performChanges(changes) { success, error in
                if success {
                    cont.resume()
                } else {
                    let msg = (error as NSError?)?.localizedDescription ?? ""
                    cont.resume(throwing: SaveError.saveFailed(msg))
                }
            }
        }
    }

    // MARK: - Temp helpers

    private static func copyDownloadToTempWithExtension(from location: URL, remoteURL: URL) throws -> URL {
        let ext = remoteURL.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        let safeExt = ext.isEmpty ? "mp4" : ext

        let dir = FileManager.default.temporaryDirectory
        let dst = dir.appendingPathComponent("media_\(UUID().uuidString).\(safeExt)")

        if FileManager.default.fileExists(atPath: dst.path) {
            try? FileManager.default.removeItem(at: dst)
        }
        try FileManager.default.copyItem(at: location, to: dst)
        return dst
    }
}

import SwiftUI

struct ZoomableImageView: View {

    let image: Image

    @State private var scale: CGFloat = 1
    @State private var lastScale: CGFloat = 1

    @State private var offset: CGSize = .zero
    @State private var lastOffset: CGSize = .zero

    private let minScale: CGFloat = 1
    private let maxScale: CGFloat = 5

    var body: some View {
        GeometryReader { proxy in
            image
                .resizable()
                .scaledToFit()
                .scaleEffect(scale)
                .offset(offset)
                .frame(width: proxy.size.width, height: proxy.size.height)
                .contentShape(Rectangle())
                .gesture(pinchGesture)
                .simultaneousGesture(panGesture(in: proxy.size))
                .onTapGesture(count: 2) {
                    withAnimation(.spring(response: 0.25, dampingFraction: 0.9)) {
                        if abs(scale - 1) < 0.01 {
                            scale = 2
                            lastScale = 2
                        } else {
                            reset()
                        }
                    }
                }
                .onChange(of: scale) { _ in
                    if scale <= 1.01 {
                        withAnimation(.spring(response: 0.25, dampingFraction: 0.9)) {
                            offset = .zero
                            lastOffset = .zero
                        }
                    }
                }
        }
    }

    private var pinchGesture: some Gesture {
        MagnificationGesture()
            .onChanged { value in
                var next = lastScale * value
                next = clamp(next, minScale, maxScale)
                scale = next
            }
            .onEnded { _ in
                lastScale = scale
                if scale <= 1.01 {
                    reset()
                }
            }
    }

    private func panGesture(in size: CGSize) -> some Gesture {
        DragGesture()
            .onChanged { value in
                guard scale > 1.01 else { return }
                let raw = CGSize(
                    width: lastOffset.width + value.translation.width,
                    height: lastOffset.height + value.translation.height
                )
                offset = clampOffset(raw, in: size)
            }
            .onEnded { _ in
                lastOffset = offset
            }
    }

    private func reset() {
        scale = 1
        lastScale = 1
        offset = .zero
        lastOffset = .zero
    }

    private func clamp(_ value: CGFloat, _ min: CGFloat, _ max: CGFloat) -> CGFloat {
        Swift.max(min, Swift.min(max, value))
    }

    private func clampOffset(_ raw: CGSize, in size: CGSize) -> CGSize {
        let extraW = (size.width * scale - size.width) / 2
        let extraH = (size.height * scale - size.height) / 2

        let clampedX = Swift.max(-extraW, Swift.min(extraW, raw.width))
        let clampedY = Swift.max(-extraH, Swift.min(extraH, raw.height))

        return CGSize(width: clampedX, height: clampedY)
    }
}

import Foundation
import AVFoundation

enum VideoDurationTextProvider {

    struct DebugInfo: Sendable {
        let url: URL
        let isFileURL: Bool
        let fileExists: Bool
        let fileSizeBytes: Int64?
        let ext: String
        let isReachable: Bool
    }

    // MARK: - Public

    static func durationText(for url: URL) async -> String? {
        guard let seconds = await durationSeconds(for: url) else { return nil }
        return format(seconds: seconds)
    }

    static func durationSeconds(for url: URL) async -> Double? {

        let retryDelaysNs: [UInt64] = [
            0,
            250_000_000,  // 0.25s
            700_000_000,  // 0.70s
            1_300_000_000 // 1.30s
        ]

        var lastLocalSize: Int64? = nil

        for attempt in 0..<retryDelaysNs.count {
            if Task.isCancelled { return nil }

            if retryDelaysNs[attempt] > 0 {
                try? await Task.sleep(nanoseconds: retryDelaysNs[attempt])
            }

            let dbg = debugInfo(for: url)

            #if DEBUG
            print("🎬 [DURATION] attempt=\(attempt) url=\(dbg.url.absoluteString)")
            print("🎬 [DURATION] isFile=\(dbg.isFileURL) exists=\(dbg.fileExists) reachable=\(dbg.isReachable) ext=\(dbg.ext) size=\(dbg.fileSizeBytes.map(String.init) ?? "nil")")
            #endif

            if dbg.isFileURL {
                guard dbg.fileExists else { continue }

                if let size = dbg.fileSizeBytes {
                    guard size > 0 else { continue }

                    if let last = lastLocalSize, last == size {
                    } else {
                        lastLocalSize = size
                        if attempt < retryDelaysNs.count - 1 { continue }
                    }
                }

                if let seconds = await loadDurationViaAVURLAsset(url: url) {
                    return seconds
                }

                continue
            } else {
                if let seconds = await remoteDurationSecondsSmart(url: url) {
                    return seconds
                }
            }
        }

        return nil
    }

    static func format(seconds: Double) -> String {
        let total = Int(seconds.rounded(.down))
        let mm = total / 60
        let ss = total % 60
        let ssStr = ss < 10 ? "0\(ss)" : "\(ss)"
        return "\(mm):\(ssStr)"
    }

    // MARK: - Remote smart flow

    private static func remoteDurationSecondsSmart(url: URL) async -> Double? {
        if Task.isCancelled { return nil }

        let probe = await probeURL(url)
        #if DEBUG
        if let probe {
            print("🎬 [DURATION][PROBE] status=\(probe.statusCode) final=\(probe.finalURL.absoluteString)")
            print("🎬 [DURATION][PROBE] contentType=\(probe.contentType ?? "nil") acceptRanges=\(probe.acceptRanges ?? "nil") length=\(probe.contentLength.map(String.init) ?? "nil")")
        } else {
            print("🎬 [DURATION][PROBE] nil (network/proxy/blocked?) url=\(url.absoluteString)")
        }
        #endif

        let finalURL = probe?.finalURL ?? url
        let contentType = probe?.contentType?.lowercased()

        if let ct = contentType, ct.contains("text/html") {
            #if DEBUG
            print("🎬 [DURATION] HTML detected. Trying to extract media URL from page…")
            #endif

            if let mediaURL = await extractMediaURLFromHTMLPage(finalURL) {
                #if DEBUG
                print("🎬 [DURATION] extracted mediaURL=\(mediaURL.absoluteString)")
                #endif
                return await remoteDurationSecondsSmart(url: mediaURL)
            }

            #if DEBUG
            print("🎬 [DURATION] ❌ could not extract media url from HTML")
            #endif
            return nil
        }

        if let seconds = await loadDurationViaAVURLAsset(url: finalURL) {
            return seconds
        }

        #if DEBUG
        print("🎬 [DURATION] direct AVURLAsset failed; trying local download fallback…")
        #endif

        guard let tmpFile = await downloadToTemporaryFile(finalURL) else {
            #if DEBUG
            print("🎬 [DURATION] ❌ download fallback failed")
            #endif
            return nil
        }

        defer {
            try? FileManager.default.removeItem(at: tmpFile)
        }

        return await loadDurationViaAVURLAsset(url: tmpFile)
    }

    // MARK: - AVAsset loader

    private static func loadDurationViaAVURLAsset(url: URL) async -> Double? {
        do {
            let options: [String: Any] = [
                AVURLAssetPreferPreciseDurationAndTimingKey: true
            ]

            let asset = AVURLAsset(url: url, options: options)

            _ = try await asset.load(.isPlayable)
            _ = try await asset.load(.tracks)

            let duration = try await asset.load(.duration)
            let seconds = CMTimeGetSeconds(duration)

            if seconds.isFinite, seconds > 0 {
                #if DEBUG
                print("🎬 [DURATION] ✅ ok seconds=\(seconds) raw=\(duration) url=\(url.absoluteString)")
                #endif
                return seconds
            }

            #if DEBUG
            print("🎬 [DURATION] ⚠️ still-zero raw=\(duration) seconds=\(seconds) url=\(url.absoluteString)")
            #endif
            return nil

        } catch {
            #if DEBUG
            print("🎬 [DURATION] ❌ AVAsset error=\(error) url=\(url.absoluteString)")
            #endif
            return nil
        }
    }

    // MARK: - Networking helpers

    private struct ProbeInfo: Sendable {
        let finalURL: URL
        let statusCode: Int
        let contentType: String?
        let acceptRanges: String?
        let contentLength: Int64?
    }

    private static func probeURL(_ url: URL) async -> ProbeInfo? {
        var req = URLRequest(url: url)
        req.httpMethod = "HEAD"
        req.timeoutInterval = 12

        do {
            let (_, resp) = try await URLSession.shared.data(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }

            let ct = http.value(forHTTPHeaderField: "Content-Type")
            let ar = http.value(forHTTPHeaderField: "Accept-Ranges")
            let cl = http.value(forHTTPHeaderField: "Content-Length").flatMap(Int64.init)

            return ProbeInfo(
                finalURL: http.url ?? url,
                statusCode: http.statusCode,
                contentType: ct,
                acceptRanges: ar,
                contentLength: cl
            )
        } catch {
            return nil
        }
    }

    private static func downloadToTemporaryFile(_ url: URL) async -> URL? {
        if Task.isCancelled { return nil }

        var req = URLRequest(url: url)
        req.httpMethod = "GET"
        req.timeoutInterval = 60

        do {
            let (tmp, resp) = try await URLSession.shared.download(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }
            guard (200..<300).contains(http.statusCode) else { return nil }

            let ext = url.pathExtension.isEmpty ? "mp4" : url.pathExtension
            let out = FileManager.default.temporaryDirectory
                .appendingPathComponent("duration-\(UUID().uuidString)")
                .appendingPathExtension(ext)

            try? FileManager.default.removeItem(at: out)
            try FileManager.default.moveItem(at: tmp, to: out)

            #if DEBUG
            let size = (try? FileManager.default.attributesOfItem(atPath: out.path)[.size] as? NSNumber)?.int64Value
            print("🎬 [DURATION][DL] ✅ saved=\(out.lastPathComponent) size=\(size.map(String.init) ?? "nil") url=\(url.absoluteString)")
            #endif

            return out
        } catch {
            #if DEBUG
            print("🎬 [DURATION][DL] ❌ error=\(error) url=\(url.absoluteString)")
            #endif
            return nil
        }
    }

    private static func extractMediaURLFromHTMLPage(_ pageURL: URL) async -> URL? {
        if Task.isCancelled { return nil }

        var req = URLRequest(url: pageURL)
        req.httpMethod = "GET"
        req.timeoutInterval = 20
        req.setValue("Mozilla/5.0", forHTTPHeaderField: "User-Agent")

        do {
            let (data, resp) = try await URLSession.shared.data(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }
            guard (200..<300).contains(http.statusCode) else { return nil }

            guard data.count <= 2_000_000 else { return nil }

            guard let html = String(data: data, encoding: .utf8) else { return nil }

            let pattern = #"https?:\/\/[^\s"'\\<>]+?\.(mp4|mov|m3u8)(\?[^\s"'\\<>]*)?"#
            let regex = try NSRegularExpression(pattern: pattern, options: [.caseInsensitive])

            let range = NSRange(html.startIndex..<html.endIndex, in: html)
            if let match = regex.firstMatch(in: html, options: [], range: range),
               let r = Range(match.range, in: html) {
                let urlStr = String(html[r])
                return URL(string: urlStr)
            }

            let pattern2 = #"src\s*=\s*["']([^"']+\.(mp4|mov|m3u8)(\?[^"']*)?)["']"#
            let regex2 = try NSRegularExpression(pattern: pattern2, options: [.caseInsensitive])
            if let match2 = regex2.firstMatch(in: html, options: [], range: range),
               match2.numberOfRanges >= 2,
               let r2 = Range(match2.range(at: 1), in: html) {
                let srcStr = String(html[r2])
                if let abs = URL(string: srcStr, relativeTo: pageURL)?.absoluteURL {
                    return abs
                }
                return URL(string: srcStr)
            }

            return nil
        } catch {
            #if DEBUG
            print("🎬 [DURATION][HTML] ❌ error=\(error) page=\(pageURL.absoluteString)")
            #endif
            return nil
        }
    }

    // MARK: - Debug helpers

    private static func debugInfo(for url: URL) -> DebugInfo {
        let isFile = url.isFileURL
        let ext = url.pathExtension.lowercased()

        var exists = false
        var size: Int64? = nil
        var reachable = false

        if isFile {
            exists = FileManager.default.fileExists(atPath: url.path)
            if exists {
                do {
                    let values = try url.resourceValues(forKeys: [.fileSizeKey, .isReadableKey])
                    if let fileSize = values.fileSize {
                        size = Int64(fileSize)
                    }
                    reachable = (values.isReadable == true)
                } catch {
                }
            }
        } else {
            reachable = (URL(string: url.absoluteString) != nil)
        }

        return DebugInfo(
            url: url,
            isFileURL: isFile,
            fileExists: exists,
            fileSizeBytes: size,
            ext: ext,
            isReachable: reachable
        )
    }
