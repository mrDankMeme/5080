import SwiftUI
import AVKit

struct GenerationMediaViewer: View {

    let item: MediaViewerItem
    let recovery: MediaCacheRecoveryContext?

    @Environment(\.dismiss) private var dismiss

    @State private var player: AVPlayer?

    @State private var isSaving: Bool = false
    @State private var alertText: String = ""
    @State private var showAlert: Bool = false

    // ✅ Step 3
    @State private var isRecoveringCache: Bool = false
    @State private var resolvedLocalURL: URL? = nil
    @State private var recoveryTask: Task<Void, Never>? = nil

    init(item: MediaViewerItem, recovery: MediaCacheRecoveryContext? = nil) {
        self.item = item
        self.recovery = recovery
        _resolvedLocalURL = State(initialValue: item.localURL)
    }

    var body: some View {
        ZStack {
            Color.black.ignoresSafeArea()

            content
                .frame(maxWidth: .infinity, maxHeight: .infinity)

            if isRecoveringCache {
                VStack(spacing: 10) {
                    ProgressView()
                        .tint(.white)
                    Text("Восстанавливаю кэш…")
                        .foregroundStyle(.white.opacity(0.9))
                        .font(.footnote)
                }
                .padding(14)
                .background(.black.opacity(0.55))
                .clipShape(RoundedRectangle(cornerRadius: 14, style: .continuous))
                .padding(.top, 64)
            }
        }
        .overlay(alignment: .topLeading) {
            Button {
                dismiss()
            } label: {
                Image(systemName: "xmark.circle.fill")
                    .font(.system(size: 28, weight: .semibold))
                    .foregroundStyle(.white.opacity(0.9))
                    .padding(16)
            }
            .buttonStyle(.plain)
        }
        .overlay(alignment: .topTrailing) {
            HStack(spacing: 10) {
                if isSaving {
                    ProgressView().tint(.white)
                }

                Button { onSaveTap() } label: {
                    Label("Save", systemImage: "square.and.arrow.down")
                        .font(.system(size: 15, weight: .semibold))
                        .foregroundStyle(.white)
                        .padding(.horizontal, 12)
                        .padding(.vertical, 8)
                        .background(.white.opacity(0.18))
                        .clipShape(Capsule())
                }
                .buttonStyle(.plain)
                .disabled(isSaving || isRecoveringCache)
                .opacity((isSaving || isRecoveringCache) ? 0.6 : 1)
            }
            .padding(16)
        }
        .onAppear {
            if item.kind == .video {
                player = AVPlayer(url: bestURLForPlayback())
            }
            print("Start recovery")
            startRecoveryIfNeeded()
        }
        .onDisappear {
            recoveryTask?.cancel()
            recoveryTask = nil

            player?.pause()
            player = nil
        }
        .alert("Save", isPresented: $showAlert) {
            Button("OK", role: .cancel) { }
        } message: {
            Text(alertText)
        }
    }

    // MARK: - Content

    @ViewBuilder
    private var content: some View {
        switch item.kind {
        case .image:
            imageContent
        case .video:
            videoContent
        }
    }

    private var imageContent: some View {
        viewForImage()
            .padding(.horizontal, 0)
    }

    @ViewBuilder
    private func viewForImage() -> some View {
        if let local = resolvedLocalURL, fileExists(local),
           let uiImage = UIImage(contentsOfFile: local.path) {
            ZoomableImageView(image: Image(uiImage: uiImage))
                .padding(.horizontal, 12)

        } else {
            AsyncImage(url: item.remoteURL) { phase in
                switch phase {
                case .empty:
                    ProgressView().tint(.white)

                case .success(let image):
                    ZoomableImageView(image: image)
                        .padding(.horizontal, 12)

                case .failure:
                    VStack(spacing: 12) {
                        Image(systemName: "exclamationmark.triangle.fill")
                            .font(.system(size: 28))
                            .foregroundStyle(.yellow)
                        Text("Не удалось загрузить изображение.")
                            .foregroundStyle(.white.opacity(0.9))
                            .font(.body)
                    }
                    .padding(24)

                @unknown default:
                    EmptyView()
                }
            }
        }
    }

    private var videoContent: some View {
        Group {
            if let player {
                VideoPlayer(player: player)
                    .onTapGesture { }
            } else {
                ProgressView().tint(.white)
            }
        }
        .ignoresSafeArea(edges: .bottom)
    }

    // MARK: - Step 3: Recovery

    private func startRecoveryIfNeeded() {
        guard recoveryTask == nil else { return }
        guard let ctx = recovery else { return }

        // local уже есть и валиден — ничего не делаем
        if let local = resolvedLocalURL, fileExists(local) { return }

        isRecoveringCache = true

        recoveryTask = Task { @MainActor in
            defer {
                isRecoveringCache = false
            }

            do {
                let local = try await ctx.mediaCache.fetch(
                    remoteURL: item.remoteURL,
                    threadId: ctx.threadId,
                    kind: ctx.kind
                )
                try Task.checkCancellation()

                // Same id/role/createdAt, payload updated with localURL
                let updatedMessage = makeUpdatedMessage(original: ctx.message, localURL: local)

                
                try await ctx.historyRepo.replaceMessage(
                    threadId: ctx.threadId,
                    messageId: updatedMessage.id,
                    newMessage: updatedMessage
                )


                // UI update (bubble rebuilds)
                await ctx.onMessageUpdated(updatedMessage)

                // Viewer update
                resolvedLocalURL = local

                if item.kind == .video {
                    player?.pause()
                    player = AVPlayer(url: local)
                }

                #if DEBUG
                print("🧾 MEDIA_RECOVERY: success threadId=\(ctx.threadId) msgId=\(updatedMessage.id) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY: cancelled")
                #endif
            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY: failed error=\(error)")
                #endif
                // не показываем алерты: save может работать и из remote
            }

            recoveryTask = nil
        }
    }

    private func makeUpdatedMessage(original: GenerationMessage, localURL: URL) -> GenerationMessage {
        switch original.payload {
        case .image(let remote, _):
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remote, localURL: localURL)
            )
        case .video(let remote, _):
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remote, localURL: localURL)
            )
        default:
            return original
        }
    }

    // MARK: - Save

    private func onSaveTap() {
        isSaving = true

        Task { @MainActor in
            defer { isSaving = false }

            do {
                switch item.kind {
                case .image:
                    if let local = resolvedLocalURL, fileExists(local) {
                        try await PhotoLibrarySaver.saveLocalImage(at: local)
                    } else {
                        try await PhotoLibrarySaver.saveRemoteImage(from: item.remoteURL)
                    }

                case .video:
                    if let local = resolvedLocalURL, fileExists(local) {
                        try await PhotoLibrarySaver.saveLocalVideo(at: local)
                    } else {
                        try await PhotoLibrarySaver.saveRemoteVideo(from: item.remoteURL)
                    }
                }

                alertText = "Saved ✅"
                showAlert = true

            } catch is CancellationError {
                // ignore
            } catch {
                alertText = "Failed ❌\n\(error.localizedDescription)"
                showAlert = true
            }
        }
    }

    // MARK: - Helpers

    private func bestURLForPlayback() -> URL {
        if let local = resolvedLocalURL, fileExists(local) {
            return local
        }
        return item.remoteURL
    }

    private func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }
}

import Foundation

/// Контекст, который даёт Viewer возможность:
/// 1) восстановить local файл из remote через MediaCacheService
/// 2) обновить payload у сообщения в истории (persist)
/// 3) обновить сообщение в памяти (UI), чтобы bubble перестроился без перезапуска
public struct MediaCacheRecoveryContext: Sendable {

    public let threadId: UUID
    public let message: GenerationMessage
    public let kind: String // "image" | "video"

    public let historyRepo: GenerationHistoryRepository
    public let mediaCache: MediaCacheService

    /// UI callback: заменить сообщение в массиве (same id).
    public let onMessageUpdated: @MainActor (GenerationMessage) -> Void

    public init(
        threadId: UUID,
        message: GenerationMessage,
        kind: String,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService,
        onMessageUpdated: @escaping @MainActor (GenerationMessage) -> Void
    ) {
        self.threadId = threadId
        self.message = message
        self.kind = kind
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache
        self.onMessageUpdated = onMessageUpdated
    }
}

import Foundation

struct MediaViewerItem: Identifiable, Equatable {

    enum Kind: Equatable {
        case image
        case video
    }

    let id: UUID              // ✅ must be stable and tied to GenerationMessage.id
    let kind: Kind
    let remoteURL: URL
    let localURL: URL?

    init(id: UUID, kind: Kind, remoteURL: URL, localURL: URL?) {
        self.id = id
        self.kind = kind
        self.remoteURL = remoteURL
        self.localURL = localURL
    }
}

import Foundation
import Photos
import UIKit

enum PhotoLibrarySaver {

    enum SaveError: LocalizedError {
        case permissionDenied
        case invalidLocalImage
        case invalidLocalVideo
        case invalidRemoteImageData
        case downloadFailed(String)
        case saveFailed(String)

        var errorDescription: String? {
            switch self {
            case .permissionDenied:
                return "Нет доступа к Фото. Разреши доступ в настройках."
            case .invalidLocalImage:
                return "Локальный файл изображения недоступен."
            case .invalidLocalVideo:
                return "Локальный файл видео недоступен."
            case .invalidRemoteImageData:
                return "Не удалось прочитать изображение из сети."
            case .downloadFailed(let text):
                return text.isEmpty ? "Не удалось скачать файл." : text
            case .saveFailed(let text):
                return text.isEmpty ? "Не удалось сохранить." : text
            }
        }
    }

    // MARK: - Public API (Local)

    static func saveLocalImage(at url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        guard FileManager.default.fileExists(atPath: url.path),
              let image = UIImage(contentsOfFile: url.path) else {
            throw SaveError.invalidLocalImage
        }

        try await performChanges {
            PHAssetChangeRequest.creationRequestForAsset(from: image)
        }
    }

    static func saveLocalVideo(at url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        guard FileManager.default.fileExists(atPath: url.path) else {
            throw SaveError.invalidLocalVideo
        }

        try await performChanges {
            PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: url)
        }
    }

    // MARK: - Public API (Remote / Temp)

    static func saveRemoteImage(from url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        do {
            let (data, _) = try await URLSession.shared.data(from: url)
            try Task.checkCancellation()

            guard let image = UIImage(data: data) else {
                throw SaveError.invalidRemoteImageData
            }

            try await performChanges {
                PHAssetChangeRequest.creationRequestForAsset(from: image)
            }
        } catch is CancellationError {
            throw CancellationError()
        } catch {
            throw SaveError.downloadFailed(String(describing: error))
        }
    }

    static func saveRemoteVideo(from url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        do {
            let (tmpLocation, _) = try await URLSession.shared.download(from: url)
            try Task.checkCancellation()

            let finalURL = try copyDownloadToTempWithExtension(from: tmpLocation, remoteURL: url)

            try await performChanges {
                PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: finalURL)
            }

            // cleanup (optional)
            try? FileManager.default.removeItem(at: finalURL)

        } catch is CancellationError {
            throw CancellationError()
        } catch {
            throw SaveError.downloadFailed(String(describing: error))
        }
    }

    // MARK: - Authorization

    private static func ensureAddOnlyAuthorization() async -> PHAuthorizationStatus {
        let current = PHPhotoLibrary.authorizationStatus(for: .addOnly)
        if current != .notDetermined {
            return current
        }
        return await PHPhotoLibrary.requestAuthorization(for: .addOnly)
    }

    // MARK: - Perform changes

    private static func performChanges(_ changes: @escaping () -> Void) async throws {
        try await withCheckedThrowingContinuation { cont in
            PHPhotoLibrary.shared().performChanges(changes) { success, error in
                if success {
                    cont.resume()
                } else {
                    let msg = (error as NSError?)?.localizedDescription ?? ""
                    cont.resume(throwing: SaveError.saveFailed(msg))
                }
            }
        }
    }

    // MARK: - Temp helpers

    private static func copyDownloadToTempWithExtension(from location: URL, remoteURL: URL) throws -> URL {
        let ext = remoteURL.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        let safeExt = ext.isEmpty ? "mp4" : ext

        let dir = FileManager.default.temporaryDirectory
        let dst = dir.appendingPathComponent("media_\(UUID().uuidString).\(safeExt)")

        // location может быть read-only/ephemeral — копируем в управляемый tmp
        if FileManager.default.fileExists(atPath: dst.path) {
            try? FileManager.default.removeItem(at: dst)
        }
        try FileManager.default.copyItem(at: location, to: dst)
        return dst
    }
}

import SwiftUI

/// Простое zoom/pinch + pan для UIImage.
/// - Двойной тап: zoom x2 / reset.
/// - Pinch: плавный scale.
/// - Drag: панорамирование, когда scale > 1.
struct ZoomableImageView: View {

    let image: Image

    @State private var scale: CGFloat = 1
    @State private var lastScale: CGFloat = 1

    @State private var offset: CGSize = .zero
    @State private var lastOffset: CGSize = .zero

    private let minScale: CGFloat = 1
    private let maxScale: CGFloat = 5

    var body: some View {
        GeometryReader { proxy in
            image
                .resizable()
                .scaledToFit()
                .scaleEffect(scale)
                .offset(offset)
                .frame(width: proxy.size.width, height: proxy.size.height)
                .contentShape(Rectangle())
                .gesture(pinchGesture)
                .simultaneousGesture(panGesture(in: proxy.size))
                .onTapGesture(count: 2) {
                    withAnimation(.spring(response: 0.25, dampingFraction: 0.9)) {
                        if abs(scale - 1) < 0.01 {
                            scale = 2
                            lastScale = 2
                        } else {
                            reset()
                        }
                    }
                }
                .onChange(of: scale) { _ in
                    // Если вернулись к 1 — сбросим пан
                    if scale <= 1.01 {
                        withAnimation(.spring(response: 0.25, dampingFraction: 0.9)) {
                            offset = .zero
                            lastOffset = .zero
                        }
                    }
                }
        }
    }

    private var pinchGesture: some Gesture {
        MagnificationGesture()
            .onChanged { value in
                var next = lastScale * value
                next = clamp(next, minScale, maxScale)
                scale = next
            }
            .onEnded { _ in
                lastScale = scale
                if scale <= 1.01 {
                    reset()
                }
            }
    }

    private func panGesture(in size: CGSize) -> some Gesture {
        DragGesture()
            .onChanged { value in
                guard scale > 1.01 else { return }
                let raw = CGSize(
                    width: lastOffset.width + value.translation.width,
                    height: lastOffset.height + value.translation.height
                )
                offset = clampOffset(raw, in: size)
            }
            .onEnded { _ in
                lastOffset = offset
            }
    }

    private func reset() {
        scale = 1
        lastScale = 1
        offset = .zero
        lastOffset = .zero
    }

    private func clamp(_ value: CGFloat, _ min: CGFloat, _ max: CGFloat) -> CGFloat {
        Swift.max(min, Swift.min(max, value))
    }

    private func clampOffset(_ raw: CGSize, in size: CGSize) -> CGSize {
        // Простая оценка допустимого панорамирования:
        // считаем, что scaledToFit занимает примерно весь экран, и ограничиваем пан по "лишнему" масштабу.
        let extraW = (size.width * scale - size.width) / 2
        let extraH = (size.height * scale - size.height) / 2

        let clampedX = Swift.max(-extraW, Swift.min(extraW, raw.width))
        let clampedY = Swift.max(-extraH, Swift.min(extraH, raw.height))

        return CGSize(width: clampedX, height: clampedY)
    }
}
