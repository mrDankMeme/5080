import SwiftUI

struct GenerationChatContainerView<Composer: View>: View {

    let title: String
    let messages: [GenerationMessage]
    let isLoading: Bool
    let errorText: String?
    let composer: Composer

    init(
        title: String,
        messages: [GenerationMessage],
        isLoading: Bool,
        errorText: String?,
        @ViewBuilder composer: () -> Composer
    ) {
        self.title = title
        self.messages = messages
        self.isLoading = isLoading
        self.errorText = errorText
        self.composer = composer()
    }

    var body: some View {
        VStack(spacing: 0) {

            ScrollViewReader { proxy in
                ScrollView {
                    LazyVStack(alignment: .leading, spacing: 10) {
                        ForEach(messages) { msg in
                            GenerationBubbleView(message: msg)
                                .id(msg.id)
                        }

                        if isLoading {
                            HStack(spacing: 8) {
                                ProgressView()
                                Text("Processing…")
                                    .foregroundStyle(.secondary)
                                    .font(.footnote)
                            }
                            .padding(.top, 8)
                        }

                        if let errorText, !errorText.isEmpty {
                            Text(errorText)
                                .foregroundStyle(.red)
                                .font(.footnote)
                                .padding(.top, 8)
                        }
                    }
                    .padding(16)
                }
                .onChange(of: messages.count) { _ in
                    guard let last = messages.last else { return }
                    withAnimation(.easeOut(duration: 0.2)) {
                        proxy.scrollTo(last.id, anchor: .bottom)
                    }
                }
            }

            Divider()

            composer
                .padding(12)
                .background(Color(.systemBackground))
        }
        .navigationTitle(title)
        .navigationBarTitleDisplayMode(.inline)
    }
}

import SwiftUI

struct GenerationBubbleView: View {
    let message: GenerationMessage
    let onOpenMedia: (MediaViewerItem, GenerationMessage) -> Void

    private let avatarSize: CGFloat = 32.scale
    private let avatarToBubbleSpacing: CGFloat = 12.scale

    private let generationLoadingWindowSeconds: TimeInterval = 120

    var body: some View {
        switch message.role {
        case .system:
            systemRow

        case .assistant:
            assistantRow

        case .user:
            userRow
        }
    }

    // MARK: - Rows

    private var assistantRow: some View {
        HStack(alignment: .top, spacing: avatarToBubbleSpacing) {
            avatar(.assistant)
            bubble
            Spacer(minLength: 40)
        }
    }

    private var userRow: some View {
        HStack(alignment: .top, spacing: avatarToBubbleSpacing) {
            Spacer(minLength: 40)
            bubble
            avatar(.user)
        }
    }

    private var systemRow: some View {
        HStack {
            Spacer(minLength: 0)
            systemBubble
            Spacer(minLength: 0)
        }
    }

    // MARK: - Avatar

    private enum AvatarKind { case assistant, user }

    @ViewBuilder
    private func avatar(_ kind: AvatarKind) -> some View {
        Image(kind == .assistant ? "Chat.robot" : "Chat.user")
            .resizable()
            .scaledToFit()
            .frame(width: avatarSize, height: avatarSize)
    }

    // MARK: - Bubble

    private var bubble: some View {
        Group {
            switch message.payload {
            case .text(let text):
                textBubble(text)
                    .padding(.vertical, 10.scale)
                    .padding(.horizontal, 12.scale)
                    .background(backgroundColor)
                    .clipShape(RoundedRectangle(cornerRadius: 14.scale, style: .continuous))

            case .error(let text):
                textBubble(text)
                    .foregroundStyle(.red)
                    .padding(.vertical, 10.scale)
                    .padding(.horizontal, 12.scale)
                    .background(backgroundColor)
                    .clipShape(RoundedRectangle(cornerRadius: 14.scale, style: .continuous))

            case .system(let text):
                textBubble(text)
                    .foregroundStyle(.secondary)
                    .padding(.vertical, 10.scale)
                    .padding(.horizontal, 12.scale)
                    .background(backgroundColor)
                    .clipShape(RoundedRectangle(cornerRadius: 14.scale, style: .continuous))

            case .image(let remoteURL, let localURL):
                GenerationImageBubble(
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    generationLoadingWindowSeconds: generationLoadingWindowSeconds
                )
                .contentShape(Rectangle())
                .onTapGesture {
                    // пока генерируется (локального файла нет и окно генерации активно) — не открываем viewer
                    guard !isImageGenerating(createdAt: message.createdAt, localURL: localURL) else { return }

                    onOpenMedia(
                        .init(
                            id: message.id,
                            kind: .image,
                            remoteURL: remoteURL,
                            localURL: localURL
                        ),
                        message
                    )
                }

            case .video(let remoteURL, let localURL):
                GenerationVideoMessageBubble(
                    messageId: message.id,
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL
                )
                .contentShape(Rectangle())
                .onTapGesture {
                    guard !isVideoGenerating(createdAt: message.createdAt, localURL: localURL) else { return }

                    onOpenMedia(
                        .init(
                            id: message.id,
                            kind: .video,
                            remoteURL: remoteURL,
                            localURL: localURL
                        ),
                        message
                    )
                }
            }
        }
    }

    private func isVideoGenerating(createdAt: Date, localURL: URL?) -> Bool {
        guard localURL == nil else { return false }
        let age = Date().timeIntervalSince(createdAt)
        return age >= 0 && age <= generationLoadingWindowSeconds
    }

    private func isImageGenerating(createdAt: Date, localURL: URL?) -> Bool {
        guard localURL == nil else { return false }
        let age = Date().timeIntervalSince(createdAt)
        return age >= 0 && age <= generationLoadingWindowSeconds
    }

    private var systemBubble: some View {
        Group {
            switch message.payload {
            case .system(let text):
                Text(text)
                    .font(.footnote)
                    .foregroundStyle(.secondary)
                    .multilineTextAlignment(.center)
                    .padding(.vertical, 8.scale)
                    .padding(.horizontal, 12.scale)
                    .background(Color.black.opacity(0.02))
                    .clipShape(RoundedRectangle(cornerRadius: 12.scale, style: .continuous))

            case .text(let text):
                Text(text)
                    .font(.footnote)
                    .foregroundStyle(.secondary)
                    .multilineTextAlignment(.center)
                    .padding(.vertical, 8.scale)
                    .padding(.horizontal, 12.scale)
                    .background(Color.black.opacity(0.02))
                    .clipShape(RoundedRectangle(cornerRadius: 12.scale, style: .continuous))

            case .error(let text):
                Text(text)
                    .font(.footnote)
                    .foregroundStyle(.secondary)
                    .multilineTextAlignment(.center)
                    .padding(.vertical, 8.scale)
                    .padding(.horizontal, 12.scale)
                    .background(Color.black.opacity(0.02))
                    .clipShape(RoundedRectangle(cornerRadius: 12.scale, style: .continuous))

            case .image(let remoteURL, let localURL):
                GenerationImageBubble(
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    generationLoadingWindowSeconds: generationLoadingWindowSeconds
                )

            case .video(let remoteURL, let localURL):
                GenerationVideoMessageBubble(
                    messageId: message.id,
                    createdAt: message.createdAt,
                    remoteURL: remoteURL,
                    localURL: localURL
                )
            }
        }
    }

    private func textBubble(_ text: String) -> some View {
        Text(text)
            .font(.body)
            .foregroundStyle(.primary)
    }

    private var backgroundColor: Color {
        switch message.role {
        case .user:
            return Color.black.opacity(0.08)
        case .assistant:
            return Color.black.opacity(0.04)
        case .system:
            return Color.black.opacity(0.02)
        }
    }
}

import Foundation

extension GenerationChatComposerView {

    var isTextEmpty: Bool {
        text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty
    }

    var shouldShowPlaceholder: Bool {
        isTextEmpty && !isTextFieldFocused
    }

    var rightIcon: RightActionIcon {
        if isRecording { return .stop }
        if forceSendIcon { return .send }
        guard isMicEnabled else { return .send }
        return isTextEmpty ? .mic : .send
    }

    var shouldShowAttachmentsRow: Bool {
        if showsSecondAttachmentSlot {
            return attachmentImage != nil || secondAttachmentImage != nil
        } else {
            return attachmentImage != nil
        }
    }
}

import SwiftUI
import Foundation

extension GenerationChatComposerView {

    // MARK: - Speech handlers

    func handleMicToggle() {
        if isRecording {
            stopDictation()
            return
        }

        guard isMicEnabled else { return }

        guard isTextInputEnabled else {
            speechErrorText = "В этом режиме диктовка недоступна."
            return
        }

        guard let speechService else {
            speechErrorText = "Speech service не найден."
            onMicTap()
            return
        }

        speechErrorText = nil

        Task { @MainActor in
            let status = await speechService.requestPermissions()

            guard status == .authorized else {
                switch status {
                case .denied:
                    speechErrorText = "Нет доступа к микрофону/распознаванию речи. Разреши в Settings."
                case .restricted:
                    speechErrorText = "Распознавание речи ограничено (Restricted)."
                case .notDetermined:
                    speechErrorText = "Доступ к распознаванию речи не получен."
                default:
                    speechErrorText = "Распознавание речи недоступно."
                }
                onMicTap()
                return
            }

            let sessionId = UUID()
            activeSpeechSessionId = sessionId

            isRecording = true
            startRecordTimer()

            speechService.start(
                onPartial: { partial in
                    Task { @MainActor in
                        guard activeSpeechSessionId == sessionId else { return }
                        self.text = partial
                    }
                },
                onFinal: { finalText in
                    Task { @MainActor in
                        guard activeSpeechSessionId == sessionId else { return }
                        self.text = finalText
                        self.isRecording = false
                        stopRecordTimer(reset: false)
                    }
                },
                onError: { err in
                    Task { @MainActor in
                        guard activeSpeechSessionId == sessionId else { return }
                        self.speechErrorText = "Speech error: \(err.localizedDescription)"
                        self.isRecording = false
                        stopRecordTimer(reset: true)
                    }
                }
            )

            self.isRecording = speechService.isRunning
            if !self.isRecording {
                stopRecordTimer(reset: true)
            }
        }
    }

    func stopDictation() {
        activeSpeechSessionId = UUID()
        isRecording = false
        speechService?.stop()
        stopRecordTimer(reset: false)
    }

    func cancelDictation() {
        activeSpeechSessionId = UUID()
        isRecording = false
        speechService?.cancel()
        stopRecordTimer(reset: true)
    }
}

import SwiftUI
import Foundation

extension GenerationChatComposerView {

    var timerText: String {
        let mm = recordSeconds / 60
        let ss = recordSeconds % 60
        let mmStr = mm < 10 ? "0\(mm)" : "\(mm)"
        let ssStr = ss < 10 ? "0\(ss)" : "\(ss)"
        return "\(mmStr):\(ssStr)"
    }

    func startRecordTimer() {
        stopRecordTimer(reset: true)
        recordSeconds = 0
        recordTimer = Timer.scheduledTimer(withTimeInterval: 1.0, repeats: true) { _ in
            recordSeconds += 1
        }
        if let recordTimer {
            RunLoop.main.add(recordTimer, forMode: .common)
        }
    }

    func stopRecordTimer(reset: Bool) {
        recordTimer?.invalidate()
        recordTimer = nil
        if reset {
            recordSeconds = 0
        }
    }
}

import SwiftUI

extension GenerationChatComposerView {

    var bubbleContainer: some View {
        VStack(alignment: .leading, spacing: 0) {

            if shouldShowAttachmentsRow {
                attachmentsRow
                    .padding(.leading, 10.scale)
                    .padding(.top, 10.scale)
                    .padding(.bottom, 10.scale)
            }

            inputBarContent
        }
        .background(
            RoundedRectangle(cornerRadius: barCornerRadius, style: .continuous)
                .fill(Color.white)
        )
        .overlay(
            RoundedRectangle(cornerRadius: barCornerRadius, style: .continuous)
                .stroke(borderColor, lineWidth: 1)
        )
        .apply(Tokens.Shadow.card)
    }

    var inputBarContent: some View {
        HStack(spacing: 0) {

            HStack(spacing: 0) {
                if showsPlus {
                    Button {
                        guard !isLoading else { return }
                        onPickPhotoTap()
                    } label: {
                        Image(systemName: "plus")
                            .font(.system(size: 18.scale, weight: .semibold))
                            .foregroundStyle(placeholderColor)
                            .frame(width: plusSize.width, height: plusSize.height)
                            .contentShape(Rectangle())
                    }
                    .buttonStyle(.plain)

                    Spacer().frame(width: 16.scale)
                }
            }
            .frame(height: barHeight, alignment: .center)
            .padding(.leading, leftInset)

            ZStack(alignment: .leading) {
                if shouldShowPlaceholder {
                    Text(placeholderText)
                        .foregroundStyle(placeholderColor)
                        .font(Tokens.Font.regular17)
                }

                TextField("", text: $text, axis: .vertical)
                    .font(Tokens.Font.regular17)
                    .foregroundStyle(inputTextColor)
                    .lineLimit(1...4)
                    .disabled(isLoading || !isTextInputEnabled)
                    .submitLabel(.send)
                    .focused($isTextFieldFocused)
            }
            .frame(maxWidth: .infinity, alignment: .leading)
            .frame(minHeight: barHeight, alignment: .center)

            HStack(spacing: spaceBetweenRightButtons) {

                if isRecording {
                    Text(timerText)
                        .font(.system(size: 13.scale, weight: .semibold, design: .monospaced))
                        .foregroundStyle(.secondary)
                        .padding(.horizontal, 8.scale)
                        .padding(.vertical, 6.scale)
                        .background(
                            Capsule(style: .continuous)
                                .fill(Color.black.opacity(0.06))
                        )
                }

                if showsVideoSettingsButton {
                    GenerationChatCircleIconButton(
                        imageName: "aspects",
                        size: circleButtonSize,
                        fill: Tokens.Color.accent,
                        tint: .white,
                        isDisabled: isLoading,
                        accessibilityLabel: "Параметры видео"
                    ) {
                        guard !isLoading else { return }
                        onVideoSettingsTap()
                    }
                }

                if rightIcon == .mic {
                    GenerationChatCircleIconButton(
                        imageName: "mic",
                        size: circleButtonSize,
                        fill: Tokens.Color.accent,
                        tint: .white,
                        isDisabled: isLoading,
                        accessibilityLabel: "Микрофон"
                    ) {
                        guard !isLoading else { return }
                        handleMicToggle()
                    }
                } else {
                    GenerationChatCircleIconButton(
                        systemName: iconName(for: rightIcon),
                        size: circleButtonSize,
                        fill: Tokens.Color.accent,
                        tint: .white,
                        isDisabled: isLoading || (rightIcon == .send && !isSendEnabled),
                        accessibilityLabel: (rightIcon == .send ? "Отправить" : "Остановить")
                    ) {
                        guard !isLoading else { return }

                        switch rightIcon {
                        case .mic:
                            handleMicToggle()
                        case .stop:
                            stopDictation()
                        case .send:
                            guard isSendEnabled else { return }
                            if isRecording { stopDictation() }
                            onSendTap()
                        }
                    }
                }
            }
            .frame(height: barHeight, alignment: .center)
            .padding(.trailing, rightInset)
        }
        .frame(height: barHeight, alignment: .center)
        .animation(.easeOut(duration: 0.12), value: isTextEmpty)
        .animation(.easeOut(duration: 0.12), value: isRecording)
    }
}

import SwiftUI
import UIKit

extension GenerationChatContainerView {

    // MARK: - Actions (unchanged)

     func copyChatToPasteboard() {
        let text = GenerationChatExportService.makeShareText(title: title, messages: messages)
        UIPasteboard.general.string = text
    }

     func shareChatText() {
        sharePayload = SharePayload(items: [
            GenerationChatExportService.makeShareText(title: title, messages: messages)
        ])
    }

     func exportChat() {
        guard let threadId, let mediaCache else { return }
        actionErrorText = nil
        isExporting = true

        Task { @MainActor in
            defer { isExporting = false }
            do {
                let result = try await GenerationChatExportService.exportDirectory(
                    title: title,
                    threadId: threadId,
                    messages: messages,
                    mediaCache: mediaCache
                )
                exportedTempDirectoryURL = result.directoryURL
                sharePayload = SharePayload(items: [result.directoryURL])
            } catch {
                actionErrorText = "Export failed: \(String(describing: error))"
            }
        }
    }

     func cleanupExportTempIfNeeded() {
        guard let url = exportedTempDirectoryURL else { return }
        exportedTempDirectoryURL = nil
        try? FileManager.default.removeItem(at: url)
    }

     func deleteChat() {
        guard let historyRepo, let threadId else { return }
        actionErrorText = nil
        isDeleting = true

        Task { @MainActor in
            defer { isDeleting = false }
            do {
                try await historyRepo.delete(threadId: threadId)
                dismiss()
            } catch {
                actionErrorText = "Delete failed: \(String(describing: error))"
            }
        }
    }
}

import SwiftUI
import UIKit

extension GenerationChatContainerView {

    // MARK: - Support types

     struct SharePayload: Identifiable {
        let id = UUID()
        let items: [Any]
    }

     struct ActivityViewController: UIViewControllerRepresentable {
        let activityItems: [Any]

        func makeUIViewController(context: Context) -> UIActivityViewController {
            UIActivityViewController(activityItems: activityItems, applicationActivities: nil)
        }

        func updateUIViewController(_ uiViewController: UIActivityViewController, context: Context) {}
    }

    // MARK: - Recovery

     func makeRecoveryContextIfPossible(for item: MediaViewerItem) -> MediaCacheRecoveryContext? {
        guard
            let threadId,
            let historyRepo,
            let mediaCache,
            let onMessageUpdated
        else { return nil }

        guard let message = messages.first(where: { $0.id == item.id }) else {
            #if DEBUG
            print("🚨 RECOVERY_CTX: message not found in current messages array. itemId=\(item.id)")
            #endif
            return nil
        }

        let kind: String = (item.kind == .image) ? "image" : "video"

        return MediaCacheRecoveryContext(
            threadId: threadId,
            message: message,
            kind: kind,
            historyRepo: historyRepo,
            mediaCache: mediaCache,
            onMessageUpdated: { updated in
                onMessageUpdated(updated)
            }
        )
    }
}

import Foundation

enum GenerationChatExportService {

    struct ExportResult: Sendable {
        let directoryURL: URL
    }

    // MARK: - Share / Copy (text-only)

    static func makeShareText(title: String, messages: [GenerationMessage]) -> String {
        var lines: [String] = []
        lines.append(title)
        lines.append("")

        let sorted = messages.sorted { $0.createdAt < $1.createdAt }

        var imageIndex = 0
        var videoIndex = 0

        for msg in sorted {
            let prefix: String
            switch msg.role {
            case .user: prefix = "User"
            case .assistant: prefix = "Assistant"
            case .system: prefix = "System"
            }

            switch msg.payload {
            case .text(let t):
                lines.append("[\(prefix)] \(t)")

            case .system(let t):
                lines.append("[\(prefix)] \(t)")

            case .error(let t):
                lines.append("[\(prefix)] \(t)")

            case .image:
                imageIndex += 1
                lines.append("[\(prefix)] [image #\(imageIndex)]")

            case .video:
                videoIndex += 1
                lines.append("[\(prefix)] [video #\(videoIndex)]")
            }

            lines.append("")
        }

        return lines.joined(separator: "\n").trimmingCharacters(in: .whitespacesAndNewlines)
    }

    // MARK: - Export (Directory with attachments)

    static func exportDirectory(
        title: String,
        threadId: UUID,
        messages: [GenerationMessage],
        mediaCache: MediaCacheService
    ) async throws -> ExportResult {

        let workspace = FileManager.default.temporaryDirectory
            .appendingPathComponent("chat_export_\(UUID().uuidString)", isDirectory: true)

        let mediaDir = workspace.appendingPathComponent("media", isDirectory: true)

        try FileManager.default.createDirectory(at: workspace, withIntermediateDirectories: true)
        try FileManager.default.createDirectory(at: mediaDir, withIntermediateDirectories: true)

        let chatTextURL = workspace.appendingPathComponent("chat.txt", isDirectory: false)

        let sorted = messages.sorted { $0.createdAt < $1.createdAt }

        var lines: [String] = []
        lines.append(title)
        lines.append("")

        var imageIndex = 0
        var videoIndex = 0

        for msg in sorted {
            let prefix: String
            switch msg.role {
            case .user: prefix = "User"
            case .assistant: prefix = "Assistant"
            case .system: prefix = "System"
            }

            switch msg.payload {
            case .text(let t):
                lines.append("[\(prefix)] \(t)")

            case .system(let t):
                lines.append("[\(prefix)] \(t)")

            case .error(let t):
                lines.append("[\(prefix)] \(t)")

            case .image(let remoteURL, let localURL):
                imageIndex += 1
                let exported = try await exportMediaFile(
                    kind: "image",
                    index: imageIndex,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    threadId: threadId,
                    mediaCache: mediaCache,
                    mediaDir: mediaDir
                )
                lines.append("[\(prefix)] [image #\(imageIndex)] \(exported.relativePath)")

            case .video(let remoteURL, let localURL):
                videoIndex += 1
                let exported = try await exportMediaFile(
                    kind: "video",
                    index: videoIndex,
                    remoteURL: remoteURL,
                    localURL: localURL,
                    threadId: threadId,
                    mediaCache: mediaCache,
                    mediaDir: mediaDir
                )
                lines.append("[\(prefix)] [video #\(videoIndex)] \(exported.relativePath)")
            }

            lines.append("")
        }

        let text = lines.joined(separator: "\n").trimmingCharacters(in: .whitespacesAndNewlines)
        try text.data(using: .utf8)?.write(to: chatTextURL, options: [.atomic])

        return ExportResult(directoryURL: workspace)
    }

    // MARK: - Private

    private struct ExportedMedia {
        let relativePath: String // e.g. "media/image_001.jpg"
    }

    private static func exportMediaFile(
        kind: String, // "image" | "video"
        index: Int,
        remoteURL: URL,
        localURL: URL?,
        threadId: UUID,
        mediaCache: MediaCacheService,
        mediaDir: URL
    ) async throws -> ExportedMedia {

        let sourceURL: URL

        if let localURL, FileManager.default.fileExists(atPath: localURL.path) {
            sourceURL = localURL
        } else {
            sourceURL = try await mediaCache.fetch(remoteURL: remoteURL, threadId: threadId, kind: kind)
        }

        let ext = bestFileExtension(preferLocal: sourceURL, fallbackRemote: remoteURL, kind: kind)
        let fileName = "\(kind)_\(String(format: "%03d", index)).\(ext)"
        let dstURL = mediaDir.appendingPathComponent(fileName, isDirectory: false)

        if FileManager.default.fileExists(atPath: dstURL.path) {
            try? FileManager.default.removeItem(at: dstURL)
        }

        try FileManager.default.copyItem(at: sourceURL, to: dstURL)

        return ExportedMedia(relativePath: "media/\(fileName)")
    }

    private static func bestFileExtension(preferLocal local: URL, fallbackRemote remote: URL, kind: String) -> String {
        let localExt = local.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        if localExt.isEmpty == false { return localExt.lowercased() }

        let remoteExt = remote.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        if remoteExt.isEmpty == false { return remoteExt.lowercased() }

        if kind == "video" { return "mp4" }
        return "jpg"
    }
}

import SwiftUI
import UIKit

struct GenerationImageBubble: View {

    let createdAt: Date
    let remoteURL: URL
    let localURL: URL?
    let generationLoadingWindowSeconds: TimeInterval

    @Environment(\.openURL) private var openURL

    // MARK: - Layout (one source of truth)

    private let containerSide: CGFloat = 260.scale
    private let innerInset: CGFloat = 12.scale

    private let containerCorner: CGFloat = 20.scale
    private let innerCorner: CGFloat = 20.scale

    private let generatingBG: Color = Color(hex: "CB704D") ?? Color.brown

    var body: some View {
        whiteContainer {
            if shouldShowGenerating {
                generatingContent
            } else {
                imageContent
            }
        }
        .fixedSize() // чтобы bubble не растягивался на всю ширину
    }

    // MARK: - Rules

    /// Генерация (коричневый квадрат) показывается:
    /// - пока нет localURL (или файла),
    /// - и сообщение ещё "свежее" (<= window), чтобы история с очищенным кэшем не выглядела как "вечно генерируется".
    private var shouldShowGenerating: Bool {
        guard !hasValidLocalFile else { return false }
        let age = Date().timeIntervalSince(createdAt)
        return age >= 0 && age <= generationLoadingWindowSeconds
    }

    private var hasValidLocalFile: Bool {
        guard let localURL, localURL.isFileURL else { return false }
        return FileManager.default.fileExists(atPath: localURL.path)
    }

    // MARK: - Container

    private func whiteContainer<Content: View>(@ViewBuilder _ content: () -> Content) -> some View {
        let innerSide = containerSide - (innerInset * 2)

        return ZStack {
            RoundedRectangle(cornerRadius: containerCorner, style: .continuous)
                .fill(Color.white)

            content()
                .frame(width: innerSide, height: innerSide)
                .clipShape(RoundedRectangle(cornerRadius: innerCorner, style: .continuous))
        }
        .frame(width: containerSide, height: containerSide)
        .apply(Tokens.Shadow.card)
    }

    // MARK: - Generating UI

    private var generatingContent: some View {
        ZStack {
            RoundedRectangle(cornerRadius: innerCorner, style: .continuous)
                .fill(generatingBG)

            VStack(spacing: 10.scale) {
                Text("Генерирую изображение")
                    .font(Tokens.Font.medium13)
                    .foregroundStyle(Color.white.opacity(0.92))
                    .multilineTextAlignment(.center)

                AnimatedDotsText(
                    font: Tokens.Font.medium13,
                    color: Color.white.opacity(0.92)
                )
                .accessibilityLabel("Генерирую изображение")
            }
            .padding(.horizontal, 12.scale)
        }
    }

    // MARK: - Image UI

    @ViewBuilder
    private var imageContent: some View {
        // 1) local приоритет
        if let localURL, fileExists(localURL),
           let uiImage = UIImage(contentsOfFile: localURL.path) {
            Image(uiImage: uiImage)
                .resizable()
                .scaledToFill()

        } else {
            // 2) fallback remote (и для кейса “кэш очищен”, и если localURL ещё не приехал, но окно генерации уже вышло)
            AsyncImage(url: remoteURL) { phase in
                switch phase {
                case .empty:
                    ZStack {
                        RoundedRectangle(cornerRadius: innerCorner, style: .continuous)
                            .fill(Color.black.opacity(0.05))
                        ProgressView()
                    }

                case .success(let image):
                    image
                        .resizable()
                        .scaledToFill()

                case .failure:
                    failureView

                @unknown default:
                    EmptyView()
                }
            }
        }
    }

    private var failureView: some View {
        VStack(alignment: .leading, spacing: 6) {
            Text("Не удалось загрузить изображение.")
                .font(.footnote)
                .foregroundStyle(.secondary)

            if FeatureFlags.showOpenRemoteButton {
                Text(remoteURL.absoluteString)
                    .font(.caption2)
                    .foregroundStyle(.secondary)
                    .textSelection(.enabled)

                Button {
                    openURL(remoteURL)
                } label: {
                    Label("Open remote", systemImage: "arrow.up.right.square")
                        .font(.footnote)
                }
                .buttonStyle(.bordered)
            }
        }
        .padding(12.scale)
        .frame(maxWidth: .infinity, maxHeight: .infinity, alignment: .leading)
        .background(Color.black.opacity(0.04))
    }

    private func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }
}

// MARK: - Animated dots (no timers, no leaks)

private struct AnimatedDotsText: View {

    let font: Font
    let color: Color

    init(font: Font, color: Color) {
        self.font = font
        self.color = color
    }

    var body: some View {
        TimelineView(.periodic(from: .now, by: 0.45)) { context in
            let dotsCount = dotsCount(for: context.date)
            Text(String(repeating: "•", count: dotsCount))
                .font(font)
                .foregroundStyle(color)
        }
    }

    private func dotsCount(for date: Date) -> Int {
        let t = date.timeIntervalSinceReferenceDate
        let step = Int(floor(t / 0.45)) % 3
        return step + 1
    }
}

import SwiftUI

struct GenerationVideoMessageBubble: View {

    let messageId: UUID
    let createdAt: Date
    let remoteURL: URL
    let localURL: URL?

    // MARK: - Tuning

    private let durationWarmupWindowSeconds: TimeInterval = 25
    private let durationWarmupTick: TimeInterval = 0.8

    // MARK: - State

    @State private var durationText: String? = nil

    /// Единственный in-flight таск, который реально считает duration.
    @State private var durationTask: Task<Void, Never>? = nil

    /// Таск “наблюдатель”: ждёт пока duration появится, без спама перезапусков.
    @State private var warmupTask: Task<Void, Never>? = nil

    /// Ключ последнего “запроса” (local/remote). Нужен, чтобы не перезапускать одно и то же.
    @State private var lastRequestKey: String = ""

    /// Чтобы не стартовать второй расчёт, пока первый по этому же key ещё работает.
    @State private var isComputing: Bool = false

    var body: some View {
        Group {
            if shouldShowLoading {
                GenerationVideoStatusBubble(state: .loading)
            } else {
                GenerationVideoStatusBubble(state: .ready(durationText: durationTextOrPlaceholder))
            }
        }
        .onAppear {
            kickDurationComputation(reason: "onAppear")
            startWarmupIfNeeded()
        }
        .onChange(of: localURL?.absoluteString ?? "") { _ in
            kickDurationComputation(reason: "localURL changed")
            startWarmupIfNeeded()
        }
        .onChange(of: remoteURL.absoluteString) { _ in
            kickDurationComputation(reason: "remoteURL changed")
            startWarmupIfNeeded()
        }
        .onDisappear {
            cancelAllTasks()
        }
    }

    // MARK: - Loading rules

    private var shouldShowLoading: Bool {
        isPlaceholderRemote(remoteURL)
    }

    private func isPlaceholderRemote(_ url: URL) -> Bool {
        if url.host?.lowercased() == "example.com" { return true }
        let s = url.absoluteString.lowercased()
        return s.contains("example.com") || s.hasSuffix("/video.mp4")
    }

    // MARK: - Remote duration policy

    private func isRemoteDurationAllowed(_ url: URL) -> Bool {
        guard !isPlaceholderRemote(url) else { return false }
        let scheme = (url.scheme ?? "").lowercased()
        return scheme == "http" || scheme == "https"
    }

    // MARK: - Duration compute orchestration

    /// Главная точка входа: запускает расчёт duration только если изменились входные данные.
    private func kickDurationComputation(reason: String) {
        guard !shouldShowLoading else {
            durationText = nil
            durationTask?.cancel()
            durationTask = nil
            warmupTask?.cancel()
            warmupTask = nil
            isComputing = false
            lastRequestKey = ""
            return
        }

        let localCandidate: URL? = {
            guard let localURL, localURL.isFileURL, fileExists(localURL) else { return nil }
            return localURL
        }()

        let remoteCandidate: URL? = isRemoteDurationAllowed(remoteURL) ? remoteURL : nil

        // Нечего считать
        guard localCandidate != nil || remoteCandidate != nil else {
            durationText = nil
            return
        }

        let requestKey = makeRequestKey(local: localCandidate, remote: remoteCandidate)

        #if DEBUG
        let age = Date().timeIntervalSince(createdAt)
        print("🎬 [BUBBLE] kick msgId=\(messageId) reason=\(reason)")
        print("🎬 [BUBBLE] age=\(String(format: "%.1f", age))s loading=\(shouldShowLoading)")
        print("🎬 [BUBBLE] key=\(requestKey)")
        print("🎬 [BUBBLE] local=\(localCandidate?.path ?? "nil")")
        print("🎬 [BUBBLE] remote=\(remoteCandidate?.absoluteString ?? "nil")")
        #endif

        // ✅ Если это тот же самый requestKey:
        // - если уже посчитали durationText → ничего не делаем
        // - если прямо сейчас считаем → ничего не делаем
        if requestKey == lastRequestKey {
            if let t = durationText, !t.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty { return }
            if isComputing { return }
        }

        // ✅ Вход поменялся: отменяем старый расчёт и запускаем новый
        durationTask?.cancel()
        durationTask = nil

        lastRequestKey = requestKey
        isComputing = true

        durationTask = Task { [messageId, localCandidate, remoteCandidate, requestKey] in
            defer {
                Task { @MainActor in
                    // Сбрасываем isComputing только если это всё ещё актуальный key
                    if self.lastRequestKey == requestKey {
                        self.isComputing = false
                    }
                }
            }

            // 1) local приоритетнее
            if let localCandidate,
               let text = await VideoDurationTextProvider.durationText(for: localCandidate) {
                if Task.isCancelled { return }
                await MainActor.run {
                    // Если за это время key поменялся — не перезатираем актуальные данные
                    guard self.lastRequestKey == requestKey else { return }
                    self.durationText = text
                }
                return
            }

            // 2) затем remote
            if let remoteCandidate,
               let text = await VideoDurationTextProvider.durationText(for: remoteCandidate) {
                if Task.isCancelled { return }
                await MainActor.run {
                    guard self.lastRequestKey == requestKey else { return }
                    self.durationText = text
                }
                return
            }

            if Task.isCancelled { return }
            await MainActor.run {
                guard self.lastRequestKey == requestKey else { return }
                self.durationText = nil
            }

            #if DEBUG
            print("🎬 [BUBBLE] done msgId=\(messageId) key=\(requestKey) duration=nil")
            #endif
        }
    }

    /// Warmup теперь НЕ перезапускает расчёт каждую секунду.
    /// Он просто ждёт, пока появится durationText, и если в течение окна не появилось —
    /// один раз “пинает” расчёт (на случай редких глитчей), но без спама.
    private func startWarmupIfNeeded() {
        guard !shouldShowLoading else { return }
        if let t = durationText, !t.isEmpty { return }
        if warmupTask != nil { return }

        warmupTask = Task {
            let deadline = Date().addingTimeInterval(durationWarmupWindowSeconds)
            var didNudgeOnce = false

            while !Task.isCancelled, Date() < deadline {
                let current = await MainActor.run { self.durationText }
                if let current, !current.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
                    break
                }

                // Через половину окна можно один раз “пнуть”, но НЕ каждую секунду.
                if !didNudgeOnce, Date() > deadline.addingTimeInterval(-durationWarmupWindowSeconds / 2) {
                    didNudgeOnce = true
                    await MainActor.run {
                        self.kickDurationComputation(reason: "warmup nudge")
                    }
                }

                try? await Task.sleep(nanoseconds: UInt64(durationWarmupTick * 1_000_000_000))
            }

            await MainActor.run {
                self.warmupTask?.cancel()
                self.warmupTask = nil
            }
        }
    }

    private func cancelAllTasks() {
        durationTask?.cancel()
        durationTask = nil
        warmupTask?.cancel()
        warmupTask = nil
        isComputing = false
    }

    private func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }

    private func makeRequestKey(local: URL?, remote: URL?) -> String {
        let l = local?.path ?? "nil"
        let r = remote?.absoluteString ?? "nil"
        return "L:\(l)|R:\(r)"
    }

    private var durationTextOrPlaceholder: String {
        let t = (durationText ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
        return t.isEmpty ? "—:—" : t
    }
}

    @State private var showAlert: Bool = false

    @State private var recoveryTask: Task<Void, Never>? = nil

    init(item: MediaViewerItem, recovery: MediaCacheRecoveryContext? = nil) {

        self.recovery = recovery
        _resolvedLocalURL = State(initialValue: item.localURL)
    }

    var body: some View {
        ZStack {
            Color.black.ignoresSafeArea()

            content
                .frame(maxWidth: .infinity, maxHeight: .infinity)

            if isRecoveringCache {
                VStack(spacing: 10) {

            HStack(spacing: 10) {
                if isSaving {
                    ProgressView().tint(.white)
                }

                Button { onSaveTap() } label: {
                    Label("Save", systemImage: "square.and.arrow.down")
                        .font(.system(size: 15, weight: .semibold))
                        .foregroundStyle(.white)
                        .padding(.horizontal, 12)
                        .padding(.vertical, 8)
                        .background(.white.opacity(0.18))
                        .clipShape(Capsule())
                }
                .buttonStyle(.plain)
                .disabled(isSaving || isRecoveringCache)
                .opacity((isSaving || isRecoveringCache) ? 0.6 : 1)
            }
            .padding(16)
        }
        .onAppear {
            if item.kind == .video {
                player = AVPlayer(url: bestURLForPlayback())
            }
            print("Start recovery")
            startRecoveryIfNeeded()
        }
        .onDisappear {
            recoveryTask?.cancel()
            recoveryTask = nil

            player?.pause()
            player = nil
        }
        .alert("Save", isPresented: $showAlert) {
            Button("OK", role: .cancel) { }
        } message: {
            Text(alertText)
        }
    }

    // MARK: - Content

    @ViewBuilder
    private var content: some View {
        switch item.kind {
        case .image:
            imageContent
        case .video:
            videoContent
        }
    }

            .padding(.horizontal, 0)
    }

    @ViewBuilder
    private func viewForImage() -> some View {

           let uiImage = UIImage(contentsOfFile: local.path) {
            ZoomableImageView(image: Image(uiImage: uiImage))
                .padding(.horizontal, 12)

        } else {
            AsyncImage(url: item.remoteURL) { phase in

                case .empty:
                    ProgressView().tint(.white)

                case .success(let image):
                    ZoomableImageView(image: image)
                        .padding(.horizontal, 12)

                case .failure:
                    VStack(spacing: 12) {
                        Image(systemName: "exclamationmark.triangle.fill")
                            .font(.system(size: 28))
                            .foregroundStyle(.yellow)
                        Text("Не удалось загрузить изображение.")
                            .foregroundStyle(.white.opacity(0.9))
                            .font(.body)

                @unknown default:
                    EmptyView()
                }
            }
        }
    }

    private var videoContent: some View {
        Group {
            if let player {
                VideoPlayer(player: player)
                    .onTapGesture { }
            } else {
                ProgressView().tint(.white)
            }
        }
        .ignoresSafeArea(edges: .bottom)
    }


    private func startRecoveryIfNeeded() {
        guard recoveryTask == nil else { return }
        guard let ctx = recovery else { return }

        if let local = resolvedLocalURL, fileExists(local) { return }

        isRecoveringCache = true

        recoveryTask = Task { @MainActor in
            defer {
                isRecoveringCache = false
            }

            do {
                let local = try await ctx.mediaCache.fetch(
                    remoteURL: item.remoteURL,
                    threadId: ctx.threadId,
                    kind: ctx.kind
                )
                try Task.checkCancellation()

                let updatedMessage = makeUpdatedMessage(original: ctx.message, localURL: local)

                
                try await ctx.historyRepo.replaceMessage(
                    threadId: ctx.threadId,
                    messageId: updatedMessage.id,
                    newMessage: updatedMessage
                )


                await ctx.onMessageUpdated(updatedMessage)

                resolvedLocalURL = local

                if item.kind == .video {
                    player?.pause()
                    player = AVPlayer(url: local)
                }

                #if DEBUG
                print("🧾 MEDIA_RECOVERY: success threadId=\(ctx.threadId) msgId=\(updatedMessage.id) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY: cancelled")
                #endif
            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY: failed error=\(error)")
                #endif
            }

            recoveryTask = nil
        }
    }

    private func makeUpdatedMessage(original: GenerationMessage, localURL: URL) -> GenerationMessage {
        switch original.payload {
        case .image(let remote, _):
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remote, localURL: localURL)
            )
        case .video(let remote, _):
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remote, localURL: localURL)
            )
        default:
            return original
        }
    }

    // MARK: - Save

    private func onSaveTap() {
        isSaving = true

        Task { @MainActor in
            defer { isSaving = false }

            do {
                switch item.kind {
                case .image:
                    if let local = resolvedLocalURL, fileExists(local) {
                        try await PhotoLibrarySaver.saveLocalImage(at: local)
                    } else {
                        try await PhotoLibrarySaver.saveRemoteImage(from: item.remoteURL)
                    }

                case .video:
                    if let local = resolvedLocalURL, fileExists(local) {
                        try await PhotoLibrarySaver.saveLocalVideo(at: local)
                    } else {
                        try await PhotoLibrarySaver.saveRemoteVideo(from: item.remoteURL)
                    }
                }

                alertText = "Saved ✅"
                showAlert = true

            } catch is CancellationError {
            } catch {
                alertText = "Failed ❌\n\(error.localizedDescription)"
                showAlert = true
            }
        }
    }

    // MARK: - Helpers

    private func bestURLForPlayback() -> URL {
        if let local = resolvedLocalURL, fileExists(local) {
            return local
        }
        return item.remoteURL
    }

    private func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }
}

enum PhotoLibrarySaver {

    enum SaveError: LocalizedError {
        case permissionDenied
        case invalidLocalImage
        case invalidLocalVideo
        case invalidRemoteImageData
        case downloadFailed(String)
        case saveFailed(String)

        var errorDescription: String? {
            switch self {
            case .permissionDenied:
                return "Нет доступа к Фото. Разреши доступ в настройках."
            case .invalidLocalImage:
                return "Локальный файл изображения недоступен."
            case .invalidLocalVideo:
                return "Локальный файл видео недоступен."
            case .invalidRemoteImageData:

            case .downloadFailed(let text):
                return text.isEmpty ? "Не удалось скачать файл." : text
            case .saveFailed(let text):
                return text.isEmpty ? "Не удалось сохранить." : text
            }
        }
    }

    // MARK: - Public API (Local)

    static func saveLocalImage(at url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        guard FileManager.default.fileExists(atPath: url.path),
              let image = UIImage(contentsOfFile: url.path) else {
            throw SaveError.invalidLocalImage
        }

        try await performChanges {
            PHAssetChangeRequest.creationRequestForAsset(from: image)
        }
    }

    static func saveLocalVideo(at url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        guard FileManager.default.fileExists(atPath: url.path) else {
            throw SaveError.invalidLocalVideo
        }

        try await performChanges {
            PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: url)
        }
    }

    // MARK: - Public API (Remote / Temp)

    static func saveRemoteImage(from url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        do {
            let (data, _) = try await URLSession.shared.data(from: url)
            try Task.checkCancellation()

            guard let image = UIImage(data: data) else {
                throw SaveError.invalidRemoteImageData
            }

            try await performChanges {
                PHAssetChangeRequest.creationRequestForAsset(from: image)
            }
        } catch is CancellationError {
            throw CancellationError()
        } catch {
            throw SaveError.downloadFailed(String(describing: error))
        }
    }

    static func saveRemoteVideo(from url: URL) async throws {
        let status = await ensureAddOnlyAuthorization()
        guard status == .authorized || status == .limited else {
            throw SaveError.permissionDenied
        }

        do {
            let (tmpLocation, _) = try await URLSession.shared.download(from: url)
            try Task.checkCancellation()

            let finalURL = try copyDownloadToTempWithExtension(from: tmpLocation, remoteURL: url)

            try await performChanges {
                PHAssetChangeRequest.creationRequestForAssetFromVideo(atFileURL: finalURL)
            }

            try? FileManager.default.removeItem(at: finalURL)

        } catch is CancellationError {
            throw CancellationError()
        } catch {
            throw SaveError.downloadFailed(String(describing: error))
        }
    }

    // MARK: - Authorization

    private static func ensureAddOnlyAuthorization() async -> PHAuthorizationStatus {
        let current = PHPhotoLibrary.authorizationStatus(for: .addOnly)
        if current != .notDetermined {
            return current
        }
        return await PHPhotoLibrary.requestAuthorization(for: .addOnly)
    }

    // MARK: - Perform changes

    private static func performChanges(_ changes: @escaping () -> Void) async throws {
        try await withCheckedThrowingContinuation { cont in
            PHPhotoLibrary.shared().performChanges(changes) { success, error in
                if success {
                    cont.resume()
                } else {
                    let msg = (error as NSError?)?.localizedDescription ?? ""
                    cont.resume(throwing: SaveError.saveFailed(msg))
                }
            }
        }
    }

    // MARK: - Temp helpers

    private static func copyDownloadToTempWithExtension(from location: URL, remoteURL: URL) throws -> URL {
        let ext = remoteURL.pathExtension.trimmingCharacters(in: .whitespacesAndNewlines)
        let safeExt = ext.isEmpty ? "mp4" : ext

        let dir = FileManager.default.temporaryDirectory
        let dst = dir.appendingPathComponent("media_\(UUID().uuidString).\(safeExt)")

        if FileManager.default.fileExists(atPath: dst.path) {
            try? FileManager.default.removeItem(at: dst)
        }
        try FileManager.default.copyItem(at: location, to: dst)
        return dst
    }

import SwiftUI

struct ZoomableImageView: View {

    let image: Image

    @State private var scale: CGFloat = 1
    @State private var lastScale: CGFloat = 1

    @State private var offset: CGSize = .zero
    @State private var lastOffset: CGSize = .zero

    private let minScale: CGFloat = 1
    private let maxScale: CGFloat = 5

    var body: some View {
        GeometryReader { proxy in
            image
                .resizable()
                .scaledToFit()
                .scaleEffect(scale)
                .offset(offset)
                .frame(width: proxy.size.width, height: proxy.size.height)
                .contentShape(Rectangle())
                .gesture(pinchGesture)
                .simultaneousGesture(panGesture(in: proxy.size))
                .onTapGesture(count: 2) {
                    withAnimation(.spring(response: 0.25, dampingFraction: 0.9)) {
                        if abs(scale - 1) < 0.01 {
                            scale = 2
                            lastScale = 2
                        } else {
                            reset()
                        }
                    }
                }
                .onChange(of: scale) { _ in
                    if scale <= 1.01 {
                        withAnimation(.spring(response: 0.25, dampingFraction: 0.9)) {
                            offset = .zero
                            lastOffset = .zero
                        }
                    }
                }
        }
    }

    private var pinchGesture: some Gesture {
        MagnificationGesture()
            .onChanged { value in
                var next = lastScale * value
                next = clamp(next, minScale, maxScale)
                scale = next
            }
            .onEnded { _ in
                lastScale = scale
                if scale <= 1.01 {
                    reset()
                }
            }
    }

    private func panGesture(in size: CGSize) -> some Gesture {
        DragGesture()
            .onChanged { value in
                guard scale > 1.01 else { return }
                let raw = CGSize(
                    width: lastOffset.width + value.translation.width,
                    height: lastOffset.height + value.translation.height
                )
                offset = clampOffset(raw, in: size)
            }
            .onEnded { _ in
                lastOffset = offset
            }
    }

    private func reset() {
        scale = 1
        lastScale = 1
        offset = .zero
        lastOffset = .zero
    }

    private func clamp(_ value: CGFloat, _ min: CGFloat, _ max: CGFloat) -> CGFloat {
        Swift.max(min, Swift.min(max, value))
    }

    private func clampOffset(_ raw: CGSize, in size: CGSize) -> CGSize {
        let extraW = (size.width * scale - size.width) / 2
        let extraH = (size.height * scale - size.height) / 2

        let clampedX = Swift.max(-extraW, Swift.min(extraW, raw.width))
        let clampedY = Swift.max(-extraH, Swift.min(extraH, raw.height))

        return CGSize(width: clampedX, height: clampedY)
    }
}

import Foundation
import AVFoundation

enum VideoDurationTextProvider {

    struct DebugInfo: Sendable {
        let url: URL
        let isFileURL: Bool
        let fileExists: Bool
        let fileSizeBytes: Int64?
        let ext: String
        let isReachable: Bool
    }

    // MARK: - Public

    static func durationText(for url: URL) async -> String? {
        guard let seconds = await durationSeconds(for: url) else { return nil }
        return format(seconds: seconds)
    }

    /// Main entry: works for local files AND remote urls.
    static func durationSeconds(for url: URL) async -> Double? {

        let retryDelaysNs: [UInt64] = [
            0,
            250_000_000,  // 0.25s
            700_000_000,  // 0.70s
            1_300_000_000 // 1.30s
        ]

        var lastLocalSize: Int64? = nil

        for attempt in 0..<retryDelaysNs.count {
            if Task.isCancelled { return nil }

            if retryDelaysNs[attempt] > 0 {
                try? await Task.sleep(nanoseconds: retryDelaysNs[attempt])
            }

            let dbg = debugInfo(for: url)

            #if DEBUG
            print("🎬 [DURATION] attempt=\(attempt) url=\(dbg.url.absoluteString)")
            print("🎬 [DURATION] isFile=\(dbg.isFileURL) exists=\(dbg.fileExists) reachable=\(dbg.isReachable) ext=\(dbg.ext) size=\(dbg.fileSizeBytes.map(String.init) ?? "nil")")
            #endif

            if dbg.isFileURL {
                // ---- Local file flow (your original logic, kept) ----
                guard dbg.fileExists else { continue }

                if let size = dbg.fileSizeBytes {
                    guard size > 0 else { continue }

                    if let last = lastLocalSize, last == size {
                        // size stable => ok
                    } else {
                        lastLocalSize = size
                        if attempt < retryDelaysNs.count - 1 { continue }
                    }
                }

                if let seconds = await loadDurationViaAVURLAsset(url: url) {
                    return seconds
                }

                continue
            } else {
                // ---- Remote flow with HTML detection + local-download fallback ----
                if let seconds = await remoteDurationSecondsSmart(url: url) {
                    return seconds
                }
            }
        }

        return nil
    }

    static func format(seconds: Double) -> String {
        let total = Int(seconds.rounded(.down))
        let mm = total / 60
        let ss = total % 60
        let ssStr = ss < 10 ? "0\(ss)" : "\(ss)"
        return "\(mm):\(ssStr)"
    }

    // MARK: - Remote smart flow

    /// 1) Probe headers (Content-Type, redirects)
    /// 2) If HTML -> try extract real media url from HTML and recurse
    /// 3) If media -> try AVURLAsset directly
    /// 4) If still fails -> download locally and read duration by fileURL
    private static func remoteDurationSecondsSmart(url: URL) async -> Double? {
        if Task.isCancelled { return nil }

        // Step 1: quick probe
        let probe = await probeURL(url)
        #if DEBUG
        if let probe {
            print("🎬 [DURATION][PROBE] status=\(probe.statusCode) final=\(probe.finalURL.absoluteString)")
            print("🎬 [DURATION][PROBE] contentType=\(probe.contentType ?? "nil") acceptRanges=\(probe.acceptRanges ?? "nil") length=\(probe.contentLength.map(String.init) ?? "nil")")
        } else {
            print("🎬 [DURATION][PROBE] nil (network/proxy/blocked?) url=\(url.absoluteString)")
        }
        #endif

        let finalURL = probe?.finalURL ?? url
        let contentType = probe?.contentType?.lowercased()

        // Step 2: HTML page => extract mp4/m3u8 and recurse
        if let ct = contentType, ct.contains("text/html") {
            #if DEBUG
            print("🎬 [DURATION] HTML detected. Trying to extract media URL from page…")
            #endif

            if let mediaURL = await extractMediaURLFromHTMLPage(finalURL) {
                #if DEBUG
                print("🎬 [DURATION] extracted mediaURL=\(mediaURL.absoluteString)")
                #endif
                return await remoteDurationSecondsSmart(url: mediaURL)
            }

            #if DEBUG
            print("🎬 [DURATION] ❌ could not extract media url from HTML")
            #endif
            return nil
        }

        // Step 3: try direct AVURLAsset first (fast path)
        if let seconds = await loadDurationViaAVURLAsset(url: finalURL) {
            return seconds
        }

        // Step 4: fallback - download to temp and read local duration
        // (works great for mp4/mov; for m3u8 this will likely not work as a single-file download)
        #if DEBUG
        print("🎬 [DURATION] direct AVURLAsset failed; trying local download fallback…")
        #endif

        guard let tmpFile = await downloadToTemporaryFile(finalURL) else {
            #if DEBUG
            print("🎬 [DURATION] ❌ download fallback failed")
            #endif
            return nil
        }

        defer {
            try? FileManager.default.removeItem(at: tmpFile)
        }

        return await loadDurationViaAVURLAsset(url: tmpFile)
    }

    // MARK: - AVAsset loader

    private static func loadDurationViaAVURLAsset(url: URL) async -> Double? {
        do {
            let options: [String: Any] = [
                AVURLAssetPreferPreciseDurationAndTimingKey: true
            ]

            let asset = AVURLAsset(url: url, options: options)

            _ = try await asset.load(.isPlayable)
            _ = try await asset.load(.tracks)

            let duration = try await asset.load(.duration)
            let seconds = CMTimeGetSeconds(duration)

            if seconds.isFinite, seconds > 0 {
                #if DEBUG
                print("🎬 [DURATION] ✅ ok seconds=\(seconds) raw=\(duration) url=\(url.absoluteString)")
                #endif
                return seconds
            }

            #if DEBUG
            print("🎬 [DURATION] ⚠️ still-zero raw=\(duration) seconds=\(seconds) url=\(url.absoluteString)")
            #endif
            return nil

        } catch {
            #if DEBUG
            print("🎬 [DURATION] ❌ AVAsset error=\(error) url=\(url.absoluteString)")
            #endif
            return nil
        }
    }

    // MARK: - Networking helpers

    private struct ProbeInfo: Sendable {
        let finalURL: URL
        let statusCode: Int
        let contentType: String?
        let acceptRanges: String?
        let contentLength: Int64?
    }

    private static func probeURL(_ url: URL) async -> ProbeInfo? {
        var req = URLRequest(url: url)
        req.httpMethod = "HEAD"
        req.timeoutInterval = 12

        do {
            let (_, resp) = try await URLSession.shared.data(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }

            let ct = http.value(forHTTPHeaderField: "Content-Type")
            let ar = http.value(forHTTPHeaderField: "Accept-Ranges")
            let cl = http.value(forHTTPHeaderField: "Content-Length").flatMap(Int64.init)

            return ProbeInfo(
                finalURL: http.url ?? url,
                statusCode: http.statusCode,
                contentType: ct,
                acceptRanges: ar,
                contentLength: cl
            )
        } catch {
            return nil
        }
    }

    /// Downloads url to a unique temp file.
    /// Note: for large files this can be heavy; but for small ~5–15s clips it's fine.
    private static func downloadToTemporaryFile(_ url: URL) async -> URL? {
        if Task.isCancelled { return nil }

        var req = URLRequest(url: url)
        req.httpMethod = "GET"
        req.timeoutInterval = 60

        do {
            let (tmp, resp) = try await URLSession.shared.download(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }
            guard (200..<300).contains(http.statusCode) else { return nil }

            let ext = url.pathExtension.isEmpty ? "mp4" : url.pathExtension
            let out = FileManager.default.temporaryDirectory
                .appendingPathComponent("duration-\(UUID().uuidString)")
                .appendingPathExtension(ext)

            try? FileManager.default.removeItem(at: out)
            try FileManager.default.moveItem(at: tmp, to: out)

            #if DEBUG
            let size = (try? FileManager.default.attributesOfItem(atPath: out.path)[.size] as? NSNumber)?.int64Value
            print("🎬 [DURATION][DL] ✅ saved=\(out.lastPathComponent) size=\(size.map(String.init) ?? "nil") url=\(url.absoluteString)")
            #endif

            return out
        } catch {
            #if DEBUG
            print("🎬 [DURATION][DL] ❌ error=\(error) url=\(url.absoluteString)")
            #endif
            return nil
        }
    }

    /// Gets HTML and tries to find first mp4/mov/m3u8 url inside.
    /// This is a pragmatic fallback for `/result/{id}` pages.
    private static func extractMediaURLFromHTMLPage(_ pageURL: URL) async -> URL? {
        if Task.isCancelled { return nil }

        var req = URLRequest(url: pageURL)
        req.httpMethod = "GET"
        req.timeoutInterval = 20
        // Sometimes servers behave differently without UA.
        req.setValue("Mozilla/5.0", forHTTPHeaderField: "User-Agent")

        do {
            let (data, resp) = try await URLSession.shared.data(for: req)
            guard let http = resp as? HTTPURLResponse else { return nil }
            guard (200..<300).contains(http.statusCode) else { return nil }

            // Avoid huge memory for big pages (usually tiny anyway)
            guard data.count <= 2_000_000 else { return nil }

            guard let html = String(data: data, encoding: .utf8) else { return nil }

            // Very simple regex: find first http(s)://...(.mp4|.mov|.m3u8)(?...optional)
            // Works for many "result pages" where media url is embedded in HTML/JSON.
            let pattern = #"https?:\/\/[^\s"'\\<>]+?\.(mp4|mov|m3u8)(\?[^\s"'\\<>]*)?"#
            let regex = try NSRegularExpression(pattern: pattern, options: [.caseInsensitive])

            let range = NSRange(html.startIndex..<html.endIndex, in: html)
            if let match = regex.firstMatch(in: html, options: [], range: range),
               let r = Range(match.range, in: html) {
                let urlStr = String(html[r])
                return URL(string: urlStr)
            }

            // Fallback: try src="...mp4"
            let pattern2 = #"src\s*=\s*["']([^"']+\.(mp4|mov|m3u8)(\?[^"']*)?)["']"#
            let regex2 = try NSRegularExpression(pattern: pattern2, options: [.caseInsensitive])
            if let match2 = regex2.firstMatch(in: html, options: [], range: range),
               match2.numberOfRanges >= 2,
               let r2 = Range(match2.range(at: 1), in: html) {
                let srcStr = String(html[r2])
                // Make absolute if needed
                if let abs = URL(string: srcStr, relativeTo: pageURL)?.absoluteURL {
                    return abs
                }
                return URL(string: srcStr)
            }

            return nil
        } catch {
            #if DEBUG
            print("🎬 [DURATION][HTML] ❌ error=\(error) page=\(pageURL.absoluteString)")
            #endif
            return nil
        }
    }

    // MARK: - Debug helpers

    private static func debugInfo(for url: URL) -> DebugInfo {
        let isFile = url.isFileURL
        let ext = url.pathExtension.lowercased()

        var exists = false
        var size: Int64? = nil
        var reachable = false

        if isFile {
            exists = FileManager.default.fileExists(atPath: url.path)
            if exists {
                do {
                    let values = try url.resourceValues(forKeys: [.fileSizeKey, .isReadableKey])
                    if let fileSize = values.fileSize {
                        size = Int64(fileSize)
                    }
                    reachable = (values.isReadable == true)
                } catch {
                    // ok
                }
            }
        } else {
            reachable = (URL(string: url.absoluteString) != nil)
        }

        return DebugInfo(
            url: url,
            isFileURL: isFile,
            fileExists: exists,
            fileSizeBytes: size,
            ext: ext,
            isReachable: reachable
        )
    }
}

        self.apphudId = Self.resolveApphudId()
        refreshAuthSnapshotLog()
    }

    deinit {
        task?.cancel()
    }

    // MARK: - Actions

    func start() {
        cancel()

        errorText = nil
        statusText = nil
        resultURL = nil
        resultRawText = nil

        let prompt = self.prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let templateId = Int(self.templateIdText.trimmingCharacters(in: .whitespacesAndNewlines)) ?? 0

        guard !prompt.isEmpty else {
            errorText = "Prompt is empty"
            return
        }

        isLoading = true
        statusText = "starting…"
        log("🚀 start txt2img prompt_len=\(prompt.count) template_id=\(templateId)")

        task = Task { [weak self] in
            guard let self else { return }
            defer {
                self.isLoading = false
            }

            do {
                let dto = try await self.useCase.execute(prompt: prompt, templateId: templateId)

                self.statusText = "final: \(dto.status.debugString)"
                self.log("✅ final status=\(dto.status.debugString) error=\(dto.error ?? "—")")

                self.applyResult(from: dto)

                if dto.status == .failed, let e = dto.error, !e.isEmpty {
                    self.errorText = e
                }
            } catch {
                self.statusText = "failed"
                self.errorText = "\(error)"
                self.log("❌ \(error)")
            }
        }
    }

    func cancel() {
        task?.cancel()
        task = nil
        isLoading = false
        statusText = "cancelled"
        log("🛑 cancel")
    }

    func refreshMe() {
        Task {
            do {
                try await auth.ensureAuthorized(apphudId: apphudId)
                let me = try await auth.me()

                tokensBalance = me.tokens
                avatarTokensBalance = me.avatar_tokens

                log("👤 /me tokens=\(me.tokens) avatar_tokens=\(me.avatar_tokens)")
                refreshAuthSnapshotLog()
            } catch {
                errorText = "\(error)"
                log("❌ /me error \(error)")
                refreshAuthSnapshotLog()
            }
        }
    }

    func forceReAuth() {
        tokenStore.clear()
        refreshAuthSnapshotLog()
        log("🧹 token cleared; next call will re-auth")
    }

    // MARK: - Private

    private func refreshAuthSnapshotLog() {
        let rawToken = tokenStore.read() ?? ""
        bearerTokenMasked = rawToken.isEmpty ? "—" : Self.maskToken(rawToken)

        log("🪪 apphud_id=\(apphudId)")
        if !rawToken.isEmpty {
#if DEBUG
            if printFullBearerTokenInConsole {
                print("🧾 BEARER_FULL \(rawToken)")
            }
#endif
        }
        log("🔑 bearer=\(bearerTokenMasked)")
    }

    private func applyResult(from dto: GenerationReadDTO) {
        let raw = dto.result ?? ""
        resultRawText = raw

        if let url = URL(string: raw), url.scheme != nil {
            resultURL = url
        }
    }

    private func log(_ s: String) {
        logs.append(s)
#if DEBUG
        print("🧪 TXT2IMG \(s)")
#endif
    }

    // MARK: - Helpers

    private static func resolveApphudId() -> String {
        return "EDCC9EA3-2658-4E60-9F7F-1EB341E8DA2D"
    }

    private static func maskToken(_ token: String) -> String {
        let trimmed = token.trimmingCharacters(in: .whitespacesAndNewlines)
        guard trimmed.count > 12 else { return trimmed }
        let prefix = trimmed.prefix(6)
        let suffix = trimmed.suffix(6)
        return "\(prefix)…\(suffix) (len=\(trimmed.count))"
    }

    private static func decodeServerErrorBody(_ data: Data?) -> String? {
        guard let data, !data.isEmpty else { return nil }

        if let obj = try? JSONSerialization.jsonObject(with: data, options: []),
           JSONSerialization.isValidJSONObject(obj),
           let pretty = try? JSONSerialization.data(withJSONObject: obj, options: [.prettyPrinted]),
           let prettyString = String(data: pretty, encoding: .utf8) {
            return prettyString
        }

        if let text = String(data: data, encoding: .utf8) {
            return text
        }
        return data.map { String(format: "%02x", $0) }.joined()
    }

    private static func prettyJSONString<T: Encodable>(_ value: T) -> String? {
        let encoder = JSONEncoder()
        encoder.outputFormatting = [.prettyPrinted, .sortedKeys]
        guard let data = try? encoder.encode(value) else { return nil }
        return String(data: data, encoding: .utf8)
    }
}

    var debugString: String {
        switch self {
        case .queued: return "queued"
        case .processing: return "processing"
        case .done: return "done"
        case .failed: return "failed"
        case .unknown(let v): return "unknown(\(v))"
        }
    }

import SwiftUI

struct UnifiedHistoryListView: View {

    @StateObject private var vm: UnifiedHistoryListViewModel

    @State private var renameTarget: UnifiedThreadRowViewModel? = nil
    @State private var renameText: String = ""
    @State private var isRenamePresented: Bool = false

    @State private var deleteTarget: UnifiedThreadRowViewModel? = nil
    @State private var isDeleteConfirmPresented: Bool = false

    private let openAIDestinationBuilder: (String) -> AnyView

    private let generationDestinationBuilder: (GenerationKind, UUID) -> AnyView

    init(
        vm: UnifiedHistoryListViewModel,
        openAIDestinationBuilder: @escaping (String) -> AnyView,
        generationDestinationBuilder: @escaping (GenerationKind, UUID) -> AnyView
    ) {
        _vm = StateObject(wrappedValue: vm)
        self.openAIDestinationBuilder = openAIDestinationBuilder
        self.generationDestinationBuilder = generationDestinationBuilder
    }

    var body: some View {
        List {
            if let err = vm.errorText, err.isEmpty == false {
                Text(err)
                    .font(.footnote)
                    .foregroundStyle(.red)
                    .listRowSeparator(.hidden)
                    .listRowBackground(Color.clear)
            }

            if vm.visible.isEmpty {
                emptyState
                    .listRowSeparator(.hidden)
                    .listRowBackground(Color.clear)
            } else {
                ForEach(vm.visible) { rowVM in
                    NavigationLink {
                        destination(for: rowVM)
                    } label: {
                        UnifiedThreadCardView(row: rowVM)
                    }
                    .buttonStyle(.plain)
                    .listRowSeparator(.hidden)
                    .listRowInsets(EdgeInsets(top: 6, leading: 16, bottom: 6, trailing: 16))
                    .listRowBackground(Color.clear)
                    .swipeActions(edge: .trailing, allowsFullSwipe: true) {
                        Button(role: .destructive) {
                            deleteTarget = rowVM
                            isDeleteConfirmPresented = true
                        } label: {
                            Label("Удалить", systemImage: "trash")
                        }
                    }
                    .swipeActions(edge: .leading, allowsFullSwipe: false) {

                        Button {
                            Task { await vm.togglePin(rowVM) }
                        } label: {
                            Label(rowVM.isPinned ? "Unpin" : "Pin",
                                  systemImage: rowVM.isPinned ? "pin.slash" : "pin")
                        }
                        .tint(.orange)

                        Button {
                            renameTarget = rowVM
                            renameText = rowVM.title
                            isRenamePresented = true
                        } label: {
                            Label("Переименовать", systemImage: "pencil")
                        }
                        .tint(.blue)
                    }
                }
            }
        }
        .listStyle(.plain)
        .scrollContentBackground(.hidden)
        .navigationTitle("История")
        .navigationBarTitleDisplayMode(.inline)
        .searchable(
            text: $vm.searchQuery,
            placement: .navigationBarDrawer(displayMode: .always)
        )
        .alert("Переименовать", isPresented: $isRenamePresented) {
            TextField("Название", text: $renameText)

            Button("Сохранить") {
                guard let row = renameTarget else { return }
                let newTitle = renameText.trimmingCharacters(in: .whitespacesAndNewlines)
                guard newTitle.isEmpty == false else { return }

                Task { await vm.rename(row, newTitle: newTitle) }
            }

            Button("Отмена", role: .cancel) { }
        } message: {
            Text("Название хранится локально (на устройстве).")
        }
        .confirmationDialog(
            "Удалить из истории?",
            isPresented: $isDeleteConfirmPresented,
            titleVisibility: .visible
        ) {
            Button("Удалить", role: .destructive) {
                guard let row = deleteTarget else { return }
                Task { await vm.delete(row) }
            }
            Button("Отмена", role: .cancel) { }
        } message: {
            Text(deleteMessageText())
        }
    }

    private func destination(for row: UnifiedThreadRowViewModel) -> AnyView {
        switch row.source {
        case .openAI(let conversationId, _):
            return openAIDestinationBuilder(conversationId)

        case .generation(let kind, let threadId):
            return generationDestinationBuilder(kind, threadId)
        }
    }

    private func deleteMessageText() -> String {
        guard let row = deleteTarget else {
            return "Это действие нельзя отменить."
        }
        switch row.source {
        case .openAI:
            return "Это удалит чат только локально. Серверный conversation не трогаем."
        case .generation:
            return "Это удалит тред генерации и сохранённые сообщения локально."
        }
    }

    private var emptyState: some View {
        VStack(alignment: .leading, spacing: 10) {
            Text(vm.searchQuery.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty
                 ? "Пока нет истории.\nСоздай новый чат или сделай генерацию — и они появятся здесь."

import Foundation
import Combine

@MainActor
final class UnifiedHistoryListViewModel: ObservableObject {

    // MARK: - Output

    @Published private(set) var all: [UnifiedThreadRowViewModel] = []
    @Published private(set) var visible: [UnifiedThreadRowViewModel] = []
    @Published private(set) var errorText: String? = nil

    // MARK: - Input

    @Published var searchQuery: String = ""

    // MARK: - Dependencies

    private let observeUseCase: ObserveUnifiedThreadsUseCase
    private let deleteChatThreadUseCase: DeleteChatThreadUseCase
    private let renameChatThreadUseCase: RenameChatThreadUseCase

    private let getChatThreadUseCase: GetChatThreadUseCase
    private let upsertChatThreadUseCase: UpsertChatThreadUseCase

    private let genRepo: GenerationHistoryRepository

    private let mediaCache: MediaCacheService

    // MARK: - Private

    private var bag = Set<AnyCancellable>()
    private let relativeFormatter = RelativeDateTimeFormatter()

    private var latestThreads: [UnifiedThread] = []

    // MARK: - Init

    init(
        observeUseCase: ObserveUnifiedThreadsUseCase,
        deleteChatThreadUseCase: DeleteChatThreadUseCase,
        renameChatThreadUseCase: RenameChatThreadUseCase,
        getChatThreadUseCase: GetChatThreadUseCase,
        upsertChatThreadUseCase: UpsertChatThreadUseCase,
        genRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService
    ) {
        self.observeUseCase = observeUseCase
        self.deleteChatThreadUseCase = deleteChatThreadUseCase
        self.renameChatThreadUseCase = renameChatThreadUseCase
        self.getChatThreadUseCase = getChatThreadUseCase
        self.upsertChatThreadUseCase = upsertChatThreadUseCase
        self.genRepo = genRepo
        self.mediaCache = mediaCache
        bind()
    }

    // MARK: - Actions

    func delete(_ row: UnifiedThreadRowViewModel) async {
        errorText = nil
        do {
            switch row.source {
            case .openAI(_, let localThreadId):
                try await deleteChatThreadUseCase.execute(threadId: localThreadId)

            case .generation(_, let threadId):
                try await mediaCache.deleteCache(threadId: threadId)
                try await genRepo.delete(threadId: threadId)
            }
        } catch {
            errorText = "Delete failed: \(String(describing: error))"
        }
    }

    func rename(_ row: UnifiedThreadRowViewModel, newTitle: String) async {
        let trimmed = newTitle.trimmingCharacters(in: .whitespacesAndNewlines)
        guard trimmed.isEmpty == false else { return }

        errorText = nil

        do {
            switch row.source {
            case .openAI(let conversationId, _):
                try await renameChatThreadUseCase.execute(conversationId: conversationId, newTitle: trimmed)

            case .generation(_, let threadId):
                try await genRepo.renameThread(threadId: threadId, title: trimmed)
            }
        } catch {
            errorText = "Rename failed: \(String(describing: error))"
        }
    }

    func togglePin(_ row: UnifiedThreadRowViewModel) async {
        errorText = nil

        do {
            switch row.source {
            case .openAI(let conversationId, _):
                guard let thread = try await getChatThreadUseCase.execute(conversationId: conversationId) else {
                    return
                }
                let updated = thread.updatingPinned(!thread.isPinned)
                try await upsertChatThreadUseCase.execute(updated)

            case .generation(let kind, let threadId):
                if let src = latestThreads.first(where: { $0.id == row.id }) {
                    let updated = GenerationThread(
                        id: threadId,
                        kind: kind,
                        title: src.title,
                        createdAt: src.createdAt,
                        lastActivityAt: src.lastActivityAt,
                        preview: src.preview,
                        isPinned: !src.isPinned
                    )
                    try await genRepo.upsert(updated)
                } else {
                    let fallback = GenerationThread(
                        id: threadId,
                        kind: kind,
                        title: row.title,
                        createdAt: Date(),
                        lastActivityAt: Date(),
                        preview: row.preview,
                        isPinned: !row.isPinned
                    )
                    try await genRepo.upsert(fallback)
                }
            }
        } catch {
            errorText = "Pin failed: \(String(describing: error))"
        }
    }

    // MARK: - Bindings

    private func bind() {
            let threadsPublisher = observeUseCase.execute()
                .receive(on: DispatchQueue.main)
                .handleEvents(receiveOutput: { [weak self] threads in
                    self?.latestThreads = threads
                })
                .share()

            threadsPublisher
                .map { [weak self] threads -> [UnifiedThreadRowViewModel] in
                    guard let self else { return [] }
                    return threads.map { self.mapThreadToRow($0) }
                }
                .sink { [weak self] rows in
                    self?.all = rows
                    self?.visible = rows
                }
                .store(in: &bag)

            let queryPublisher = $searchQuery
                .map { $0.trimmingCharacters(in: .whitespacesAndNewlines) }
                .removeDuplicates()
                .debounce(for: .milliseconds(300), scheduler: DispatchQueue.main)

            Publishers.CombineLatest(threadsPublisher, queryPublisher)
                .map { [weak self] threads, query -> [UnifiedThreadRowViewModel] in
                    guard let self else { return [] }

                    guard query.isEmpty == false else {
                        return threads.map { self.mapThreadToRow($0) }
                    }

                    let q = query.lowercased()

                    let filtered = threads.filter { t in
                        let title = t.title.lowercased()
                        let preview = (t.preview ?? "").lowercased()

                        let kindText: String = {
                            switch t.kind {
                            case .openAI(let icon):
                                return "openai \(icon.systemImageName)".lowercased()

                            case .generation(let kind):
                                return "\(kind.rawValue) \(kind.defaultTitle)".lowercased()
                            }
                        }()

                        return title.contains(q) || preview.contains(q) || kindText.contains(q)
                    }

                    return filtered.map { self.mapThreadToRow($0) }
                }
                .sink { [weak self] rows in
                    self?.visible = rows
                }
                .store(in: &bag)
        }

    // MARK: - Mapping

    private func mapThreadToRow(_ thread: UnifiedThread) -> UnifiedThreadRowViewModel {
        let titleTrimmed = thread.title.trimmingCharacters(in: .whitespacesAndNewlines)
        let safeTitle = titleTrimmed.isEmpty ? "Чат" : titleTrimmed

        relativeFormatter.unitsStyle = .short
        let timeAgo = relativeFormatter.localizedString(for: thread.lastActivityAt, relativeTo: Date())

        return UnifiedThreadRowViewModel(
            id: thread.id,
            title: safeTitle,
            preview: thread.preview,
            timeAgo: timeAgo,
            iconSystemName: thread.kind.systemImageName,
            isPinned: thread.isPinned,
            source: thread.source
        )
    }
}
