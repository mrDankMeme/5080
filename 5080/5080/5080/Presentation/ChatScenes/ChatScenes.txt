import SwiftUI
import PhotosUI

struct PhotoEditChatView: View {

    @StateObject private var vm: PhotoEditChatViewModel

    @State private var pickedItem: PhotosPickerItem?
    @State private var pickedPreview: UIImage?
    @State private var isPickerPresented: Bool = false

    init(vm: PhotoEditChatViewModel) {
        _vm = StateObject(wrappedValue: vm)
    }

    var body: some View {
        GenerationChatContainerView(
            title: "Edit Photo (NanoBanana)",
            modeTitle: "Редактировать фото",
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText,
            threadId: vm.threadId,
            historyRepo: vm.historyRepoForViewer,
            mediaCache: vm.mediaCacheForViewer,
            onMessageUpdated: { updated in
                vm.applyMessageUpdateFromViewer(updated)
            }
        ) {
            GenerationChatComposerView(
                showsPlus: true,
                isLoading: vm.isLoading,
                text: $vm.promptText,
                attachmentImage: pickedPreview,
                onPickPhotoTap: {
                    guard !vm.isLoading else { return }
                    isPickerPresented = true
                },
                onMicTap: {
                    // Step 3
                },
                onSendTap: {
                    onApplyPressed()
                }
            )
            // ✅ Открываем picker именно по нажатию "+"
            .photosPicker(
                isPresented: $isPickerPresented,
                selection: $pickedItem,
                matching: .images
            )
            .onChange(of: pickedItem) { _ in
                Task { @MainActor in
                    await loadPreviewIfNeeded()
                }
            }
        }
    }

    private func onApplyPressed() {
        guard let pickedItem else {
            vm.showUserError("Выбери фото.")
            return
        }

        let clean = vm.promptText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !clean.isEmpty else {
            vm.showUserError("Напиши, что нужно сделать с фото.")
            return
        }

        Task { @MainActor in
            do {
                guard let data = try await pickedItem.loadTransferable(type: Data.self) else {
                    vm.showUserError("Не удалось прочитать фото (Data == nil).")
                    return
                }

                vm.onApplyTap(photoData: data)

                // ✅ UX: после отправки очищаем текст + убираем выбранное фото (как ты просил)
                vm.promptText = ""
                self.pickedItem = nil
                self.pickedPreview = nil

            } catch {
                vm.showUserError("Не удалось прочитать фото: \(String(describing: error))")
            }
        }
    }

    @MainActor
    private func loadPreviewIfNeeded() async {
        guard let pickedItem else {
            pickedPreview = nil
            return
        }

        do {
            if let data = try await pickedItem.loadTransferable(type: Data.self),
               let ui = UIImage(data: data) {
                pickedPreview = ui
            } else {
                pickedPreview = nil
            }
        } catch {
            pickedPreview = nil
        }
    }
}

import Foundation

@MainActor
final class PhotoEditChatViewModel: ObservableObject {

    // MARK: - Thread

    let threadId: UUID
    let kind: GenerationKind = .photoEdit
    let shouldLoadHistory: Bool

    // MARK: - UI

    @Published var messages: [GenerationMessage] = [
        GenerationMessage(
            role: .system,
            createdAt: Date(),
            payload: .system("Выбери фото, введи промпт и нажми отправить.")
        )
    ]

    @Published var promptText: String = ""

    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    // MARK: - Dependencies

    let useCase: PromptEditPhotoUseCase
    let historyRepo: GenerationHistoryRepository
    let mediaCache: MediaCacheService

    // Step 3: expose deps read-only for viewer wiring (no mutation)
    var historyRepoForViewer: GenerationHistoryRepository { historyRepo }
    var mediaCacheForViewer: MediaCacheService { mediaCache }

    // Step 3: allow viewer to update a message in UI after cache recovery
    func applyMessageUpdateFromViewer(_ updated: GenerationMessage) {
        replaceGenerationMessageInUI(updated: updated)
    }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    // Step 3E: eager recovery on open-from-history
    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    // Step 5D: защита от гонок (поздний download от старой генерации)
    var activeGenerationMessageId: UUID?

    // MARK: - Init

    init(
        threadId: UUID? = nil,
        useCase: PromptEditPhotoUseCase,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService
    ) {
        self.threadId = threadId ?? UUID()
        self.shouldLoadHistory = (threadId != nil)
        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
        }
    }

    deinit {
        task?.cancel()
        historyTask?.cancel()
        eagerRecoveryTask?.cancel()
    }

    // MARK: - Public

    func showUserError(_ text: String) {
        errorText = text
    }

    func cancel() {
        task?.cancel()
        task = nil
        isLoading = false

        // Step 5D: отмена должна “обнулить” активную генерацию
        activeGenerationMessageId = nil
    }
}

import Foundation

extension PhotoEditChatViewModel {

    func onApplyTap(photoData: Data) {
        errorText = nil
        isLoading = true

        // Step 5D: новая генерация — “перевыбираем” активный id
        activeGenerationMessageId = nil

        let prompt = promptText.trimmingCharacters(in: .whitespacesAndNewlines)

        // UI: доменный user message
        let userText = "\(prompt.isEmpty ? "" : prompt)."
        messages.append(
            GenerationMessage(
                role: .user,
                createdAt: Date(),
                payload: .text(userText)
            )
        )

        // Step 4.2: сохраняем user prompt + обновляем preview/lastActivity
        if !prompt.isEmpty {
            persistUserPromptToHistory(prompt)
        }

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let p = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
                guard !p.isEmpty else {
                    throw GenerationPollingError.invalidState
                }

                let (mime, filename) = Self.detectMimeAndFilename(from: photoData)
                let binary = PhotoBinary(data: photoData, filename: filename, mimeType: mime)

                let request = PhotoPromptEditRequest(
                    prompt: p,
                    images: [binary],
                    aspectRatio: nil
                )

                let result = try await useCase.execute(request)

                let raw = result.resultURL.trimmingCharacters(in: .whitespacesAndNewlines)
                if let url = URL(string: raw), url.scheme != nil {
                    self.lastResultURL = url

                    // Step 5B + Step 5D (guard)
                    await self.appendImagePlaceholderAndDownload(remoteURL: url)

                } else {
                    self.lastResultURL = nil

                    let printable = raw.isEmpty ? "—" : raw
                    let assistantText = "✅ Done, но result не похож на URL:\n\(printable)"
                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .text(assistantText)
                        )
                    )
                    self.persistAssistantTextToHistory(assistantText)
                }

            } catch is CancellationError {
                self.messages.append(
                    GenerationMessage(
                        role: .system,
                        createdAt: Date(),
                        payload: .system("⛔️ Cancelled")
                    )
                )

            } catch {
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                self.messages.append(
                    GenerationMessage(
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                )
                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // MARK: - Binary helpers

    static func detectMimeAndFilename(from data: Data) -> (mime: String, filename: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "photo.png") }

        let jpeg: [UInt8] = [0xFF, 0xD8]
        if data.starts(with: jpeg) { return ("image/jpeg", "photo.jpg") }

        return ("application/octet-stream", "photo.bin")
    }
}

import Foundation

extension PhotoEditChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open PhotoEdit): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open PhotoEdit): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEdit): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEdit): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open PhotoEdit): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // Step 4.4 (Load history) + Step 3E (Eager recovery)
    func loadHistoryIfExists() {
        let repo = historyRepo
        let id = threadId

        historyTask?.cancel()
        historyTask = Task { [weak self] in
            guard let self else { return }

            do {
                let stored = try await repo.loadMessages(threadId: id)
                guard stored.isEmpty == false else { return }

                self.messages = stored.sorted(by: { $0.createdAt < $1.createdAt })

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: loaded \(self.messages.count) messages for thread id=\(id)")
                #endif

                // Step 3E: auto-recover local files
                self.startEagerRecoveryIfNeeded()

            } catch is CancellationError {
                // ignore
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: load history failed: \(error)")
                #endif
            }
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // Step 4.2 (History)
    func persistUserPromptToHistory(_ prompt: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let clean = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let preview = Self.makePreviewText(clean)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .user,
                    createdAt: now,
                    payload: .text(clean)
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: saved user prompt + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: persist user prompt failed: \(error)")
                #endif
            }
        }
    }

    // Step 4.3 (History)
    func persistAssistantTextToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .text(text), previewSeed: text)
    }

    func persistAssistantErrorToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .error(text), previewSeed: text)
    }

    func persistAssistantPayloadToHistory(payload: GenerationPayload, previewSeed: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let preview = Self.makePreviewText(previewSeed)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .assistant,
                    createdAt: now,
                    payload: payload
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: saved assistant message + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: persist assistant message failed: \(error)")
                #endif
            }
        }
    }

    static func makePreviewText(_ text: String, limit: Int = 80) -> String {
        let s = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard s.count > limit else { return s }
        let idx = s.index(s.startIndex, offsetBy: limit)
        return String(s[..<idx]) + "…"
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // Step 5B + Step 5D: Media placeholder → download → replace (guard by messageId)
    func appendImagePlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        // Step 5D: фиксируем id результата как “активный”
        let messageId = UUID()
        activeGenerationMessageId = messageId

        let placeholder = GenerationMessage(
            id: messageId,
            role: .assistant,
            createdAt: now,
            payload: .image(remoteURL: remoteURL, localURL: nil)
        )

        do {
            await ensureThreadRegisteredAsync()

            // 1) Storage FIRST
            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(preview: "🖼️ Edit", at: now)

            // 2) UI placeholder (same id)
            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEdit): appended placeholder msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            // 3) Download
            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "image"
            )
            try Task.checkCancellation()

            // Step 5D: поздний результат от старой генерации — игнорируем
            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(PhotoEdit): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            // 4) Storage replace
            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            // 5) UI replace
            replaceGenerationMessageInUI(updated: updated)

            // 6) touch done
            try await touchThread(preview: "✅ Image", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEdit): replaced with local msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEdit): cancelled msgId=\(messageId)")
            #endif

        } catch {
            let pretty = String(describing: error)

            let failed = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .error("🖼️ Image (remote)\n\(remoteURL.absoluteString)\n\n⚠️ Download failed: \(pretty)")
            )
            replaceGenerationMessageInUI(updated: failed)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEdit): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // Thread helpers

    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

import SwiftUI
import PhotosUI

struct PhotoEnhanceChatView: View {

    @StateObject private var vm: PhotoEnhanceChatViewModel

    @State private var pickedItem: PhotosPickerItem?
    @State private var pickedPreview: UIImage?

    /// В дизайне placeholder есть — текст нужен для UX (но бизнес-логика не зависит)
    @State private var inputText: String = ""

    @State private var isPickerPresented: Bool = false

    init(vm: PhotoEnhanceChatViewModel) {
        _vm = StateObject(wrappedValue: vm)
    }

    var body: some View {
        GenerationChatContainerView(
            title: "Enhance Photo",
            modeTitle: "Улучшить фото",
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText,
            threadId: vm.threadId,
            historyRepo: vm.historyRepoForViewer,
            mediaCache: vm.mediaCacheForViewer,
            onMessageUpdated: { updated in
                vm.applyMessageUpdateFromViewer(updated)
            }
        ) {
            GenerationChatComposerView(
                showsPlus: true,
                isLoading: vm.isLoading,
                text: $inputText,
                attachmentImage: pickedPreview,
                onPickPhotoTap: {
                    guard !vm.isLoading else { return }
                    isPickerPresented = true
                },
                onMicTap: {
                    // mic выключен
                },
                onSendTap: {
                    onUpscalePressed()
                },
                isSendEnabled: pickedItem != nil && !vm.isLoading,
                placeholderText: "Выберите фото",
                isTextInputEnabled: false,
                forceSendIcon: true,
                isMicEnabled: false
            )
            
            .photosPicker(
                isPresented: $isPickerPresented,
                selection: $pickedItem,
                matching: .images
            )
            .onChange(of: pickedItem) { _ in
                Task { @MainActor in
                    await loadPreviewIfNeeded()
                }
            }
        }
    }

    private func onUpscalePressed() {
        guard let pickedItem else {
            vm.showUserError("Выбери фото.")
            return
        }

        Task { @MainActor in
            do {
                guard let data = try await pickedItem.loadTransferable(type: Data.self) else {
                    vm.showUserError("Не удалось прочитать фото (Data == nil).")
                    return
                }

                self.pickedPreview = nil

            } catch {

            }
        }
    }

               let ui = UIImage(data: data) {

            } else {
                pickedPreview = nil

import Foundation

extension PhotoEnhanceChatViewModel {

    func onUpscaleTap(imageData: Data) {
        errorText = nil
        isLoading = true

        // Step 5D: новая генерация — сбрасываем активный id
        activeGenerationMessageId = nil

        let job = jobId.trimmingCharacters(in: .whitespacesAndNewlines)
        let userText = Self.makeUserActionText(jobId: job.isEmpty ? nil : job)

        // UI: фиксируем действие как доменный payload
        messages.append(
            GenerationMessage(
                role: .user,
                createdAt: Date(),
                payload: .text(userText)
            )
        )

        // Step 4.2
        persistUserActionToHistory(userText)

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let dto = try await useCase.execute(jobId: job.isEmpty ? nil : job, imageData: imageData)

                let raw = (dto.result ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
                if let url = URL(string: raw), url.scheme != nil {
                    self.lastResultURL = url

                    // Step 5B + Step 5D (guard)
                    await self.appendImagePlaceholderAndDownload(remoteURL: url)

                } else {
                    self.lastResultURL = nil

                    let printable = raw.isEmpty ? "—" : raw
                    let assistantText = "✅ Done, но result не похож на URL:\n\(printable)"

                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .text(assistantText)
                        )
                    )
                    self.persistAssistantTextToHistory(assistantText)
                }

                if dto.status == .failed, let e = dto.error, !e.isEmpty {
                    let warnText = "⚠️ Backend status=failed, error:\n\(e)"
                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .error(warnText)
                        )
                    )
                    self.persistAssistantErrorToHistory(warnText)
                }

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil
                self.messages.append(
                    GenerationMessage(
                        role: .system,
                        createdAt: Date(),
                        payload: .system("⛔️ Cancelled")
                    )
                )

            } catch {
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                self.messages.append(
                    GenerationMessage(
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                )
                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }
}

import Foundation

extension PhotoEnhanceChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open PhotoEnhance): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open PhotoEnhance): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)

                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEnhance): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEnhance): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open PhotoEnhance): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

    func appendImagePlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        // Step 5D: фиксируем id результата как “активный”
        let messageId = UUID()
        activeGenerationMessageId = messageId

        let placeholder = GenerationMessage(
            id: messageId,

            createdAt: now,
            payload: .image(remoteURL: remoteURL, localURL: nil)
        )

        do {
            await ensureThreadRegisteredAsync()

            // 1) Storage FIRST
            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(preview: "🖼️ Enhance", at: now)

            // 2) UI placeholder (same id)
            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): appended placeholder msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            // 3) Download (await)
            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "image"
            )
            try Task.checkCancellation()

            // Step 5D: поздний результат от старой генерации — игнорируем
            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(PhotoEnhance): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            // 4) Storage replace (same id)
            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            // 5) UI replace (same id)
            replaceGenerationMessageInUI(updated: updated)

            // 6) optional: touch "done" after replace
            try await touchThread(preview: "✅ Image", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): replaced with local msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): cancelled msgId=\(messageId)")
            #endif

                id: messageId,
                role: .assistant,
                createdAt: now,

        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

        let base = threads.sorted { $0.lastActivityAt > $1.lastActivityAt }

        let q = searchText.trimmingCharacters(in: .whitespacesAndNewlines).lowercased()
        guard !q.isEmpty else { return base }

        return base.filter { t in
            let title = t.title.lowercased()
            let preview = (t.preview ?? "").lowercased()
            let kind = t.kind.rawValue.lowercased()
            return title.contains(q) || preview.contains(q) || kind.contains(q)
        }
    }

    func refresh() {
        isLoading = true
        errorText = nil

        Task { [weak self] in
            guard let self else { return }
            do {
                let threads = try await repo.fetchThreads()
                self.threads = threads.sorted { $0.lastActivityAt > $1.lastActivityAt }
                self.isLoading = false
            } catch {
                self.errorText = String(describing: error)
                self.isLoading = false
            }
        }
    }

    func deleteThread(_ thread: GenerationThread) {
        errorText = nil
        Task { [weak self] in
            guard let self else { return }
            do {
                // ✅ Step 5A: сначала чистим MediaCache/<threadId>/, потом удаляем тред и messages.
                try await mediaCache.deleteCache(threadId: thread.id)
                try await repo.delete(threadId: thread.id)
            } catch {
                self.errorText = "Delete failed: \(String(describing: error))"
            }
        }
    }

    func togglePin(_ thread: GenerationThread) {
        errorText = nil
        Task { [weak self] in
            guard let self else { return }
            do {
                var updated = thread
                updated.isPinned.toggle()
                // touch НЕ делаем специально: пин не должен менять lastActivity
                try await repo.upsert(updated)
            } catch {
                self.errorText = "Pin failed: \(String(describing: error))"
            }
        }
    }

    // MARK: - Private

    private func bind() {
        repo.observeThreads()
            .receive(on: DispatchQueue.main)
            .sink { [weak self] list in
                guard let self else { return }
                self.threads = list.sorted { $0.lastActivityAt > $1.lastActivityAt }
                self.isLoading = false
            }
            .store(in: &bag)

        // Первый refresh — чтобы быстрее показать данные и не ждать async init в repo
        refresh()
    }
}

import Foundation

extension TextToImageChatViewModel {

    // MARK: - Actions

    func onGenerateTap() {
        let prompt = self.prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let templateId = Int(self.templateIdText.trimmingCharacters(in: .whitespacesAndNewlines)) ?? 0

        guard !prompt.isEmpty else {
            errorText = "Prompt пустой."
            return
        }

        errorText = nil
        isLoading = true

        // Step 5D: новая генерация — сбрасываем активный id
        activeGenerationMessageId = nil

        // UI: сразу доменное сообщение
        messages.append(
            GenerationMessage(
                role: .user,
                createdAt: Date(),
                payload: .text(prompt)
            )
        )

        // Step 4.2
        persistUserPromptToHistory(prompt)

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let dto = try await useCase.execute(prompt: prompt, templateId: templateId)
                let raw = (dto.result ?? "").trimmingCharacters(in: .whitespacesAndNewlines)

                if let url = URL(string: raw), url.scheme != nil {
                    self.lastResultURL = url

                    // Step 5B + 5D: placeholder → download → replace
                    await self.appendImagePlaceholderAndDownload(remoteURL: url)

                } else {
                    self.lastResultURL = nil

                    let printable = raw.isEmpty ? "—" : raw
                    let assistantText = "✅ Done, но result не похож на URL:\n\(printable)"

                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .text(assistantText)
                        )
                    )

                    self.persistAssistantTextToHistory(assistantText)
                }

                if dto.status == .failed, let e = dto.error, !e.isEmpty {
                    let warnText = "⚠️ Backend status=failed, error:\n\(e)"
                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .error(warnText)
                        )
                    )
                    self.persistAssistantErrorToHistory(warnText)
                }

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil
                self.messages.append(
                    GenerationMessage(
                        role: .system,
                        createdAt: Date(),
                        payload: .system("⛔️ Cancelled")
                    )
                )

            } catch {
                self.lastResultURL = nil
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                self.messages.append(
                    GenerationMessage(
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                )

                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }
}

import Foundation

extension TextToImageChatViewModel {

    // MARK: - Step 3 (eager): recovery on chat open

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            await self.recoverMissingMediaIfNeeded()
            self.eagerRecoveryTask = nil
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

extension TextToImageChatViewModel {

    // MARK: - Step 5B + Step 5D: Media placeholder → download → replace (guard by messageId)

    func appendImagePlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        // Step 5D: фиксируем id результата как “активный”
        let messageId = UUID()
        activeGenerationMessageId = messageId

        let placeholder = GenerationMessage(
            id: messageId,
            role: .assistant,
            createdAt: now,
            payload: .image(remoteURL: remoteURL, localURL: nil)
        )

        do {
            await ensureThreadRegisteredAsync()

            // 1) Storage FIRST
            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(preview: "🖼️ Image", at: now)

            // 2) UI placeholder (same id)
            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): appended placeholder msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            // 3) Download
            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "image"
            )
            try Task.checkCancellation()

            // Step 5D: поздний результат от старой генерации — игнорируем
            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(TextToImage): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            // 4) Storage replace
            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            // 5) UI replace
            replaceGenerationMessageInUI(updated: updated)

            // 6) touch done
            try await touchThread(preview: "✅ Image", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): replaced with local msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): cancelled msgId=\(messageId)")
            #endif

        } catch {
            let pretty = String(describing: error)

            let failed = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .error("🖼️ Image (remote)\n\(remoteURL.absoluteString)\n\n⚠️ Download failed: \(pretty)")
            )
            replaceGenerationMessageInUI(updated: failed)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

import Foundation

extension TextToImageChatViewModel {

    // MARK: - Thread helpers

    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

import Foundation

extension PhotoToVideoChatViewModel {

    // MARK: - Actions (generate)

    func onGenerateTap(photoData: Data) {
        errorText = nil
        lastResultURL = nil

        task?.cancel()
        task = nil

        isLoading = true
        activeGenerationMessageId = nil

        let templateId = Self.parseTemplateId(from: templateIdText)

        let userText = "Генерация видео по шаблону \(templateId)."
        messages.append(
            GenerationMessage(role: .user, createdAt: Date(), payload: .text(userText))
        )

        persistUserTextToHistory(userText)

        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let prepared = Self.preparePhotoForUpload(originalData: photoData)

                let binary = PhotoBinary(
                    data: prepared.data,
                    filename: prepared.filename,
                    mimeType: prepared.mimeType
                )

                let request = PhotoToVideoRequest(
                    templateId: templateId,
                    photo: binary
                )

                let result = try await useCase.execute(request)
                try Task.checkCancellation()

                self.lastResultURL = result.resultURL
                await self.appendVideoPlaceholderAndDownload(remoteURL: result.resultURL)

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil
                self.messages.append(
                    GenerationMessage(role: .system, createdAt: Date(), payload: .system("⛔️ Cancelled"))
                )

            } catch {
                self.lastResultURL = nil
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                self.messages.append(
                    GenerationMessage(role: .assistant, createdAt: Date(), payload: .error(assistantText))
                )

                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }

    // MARK: - Media append → download → replace (sequential + guard)

    fileprivate func appendVideoPlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let messageId = UUID()
        activeGenerationMessageId = messageId

        do {
            await ensureThreadRegisteredAsync()

            let placeholder = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .video(remoteURL: remoteURL, localURL: nil)
            )

            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(preview: "🎞️ Video", at: now)

            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA(PhotoToVideo): appended placeholder threadId=\(id) msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "video"
            )
            try Task.checkCancellation()

            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA(PhotoToVideo): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            replaceGenerationMessageInUI(updated: updated)
            try await touchThread(preview: "✅ Video", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA(PhotoToVideo): replaced with local threadId=\(id) msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

        } catch {
            let pretty = String(describing: error)

            let warning = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .error("🎞️ Video (remote)\n\(remoteURL.absoluteString)\n\n⚠️ Не удалось скачать: \(pretty)")
            )
            replaceGenerationMessageInUI(updated: warning)

            #if DEBUG
            print("🧾 GEN_MEDIA(PhotoToVideo): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }

    // MARK: - Small helpers

    fileprivate static func parseTemplateId(from text: String) -> Int {
        let trimmed = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !trimmed.isEmpty else { return 0 }
        let digits = trimmed.filter(\.isNumber)
        return Int(digits) ?? 0
    }
}

import Foundation

extension PhotoToVideoChatViewModel {

    // MARK: - Load history + eager recovery

     func loadHistoryIfExists() {
        let repo = historyRepo
        let id = threadId

        historyTask?.cancel()
        historyTask = Task { [weak self] in
            guard let self else { return }

            do {
                let stored = try await repo.loadMessages(threadId: id)
                guard stored.isEmpty == false else { return }

                self.messages = stored.sorted(by: { $0.createdAt < $1.createdAt })

                #if DEBUG
                print("🧾 GEN_HISTORY: loaded \(self.messages.count) messages for thread id=\(id)")
                #endif

                self.startEagerRecoveryIfNeeded()

            } catch is CancellationError {
                // ignore
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY: load history failed: \(error)")
                #endif
            }
        }
    }

    fileprivate func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    fileprivate func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open PhotoToVideo): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open PhotoToVideo): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoToVideo): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoToVideo): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open PhotoToVideo): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

     func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }

    // MARK: - History persists

    func persistUserTextToHistory(_ text: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let clean = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !clean.isEmpty else { return }

        let preview = Self.makePreviewText(clean)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .user,
                    createdAt: now,
                    payload: .text(clean)
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY: saved user text + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY: persist user text failed: \(error)")
                #endif
            }
        }
    }

     func persistAssistantErrorToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .error(text), previewSeed: text)
    }

     func persistAssistantPayloadToHistory(payload: GenerationPayload, previewSeed: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let preview = Self.makePreviewText(previewSeed)

        Task {

                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(

                    createdAt: now,

                )
                try await repo.appendMessage(threadId: id, message: msg)

                #if DEBUG
                print("🧾 GEN_HISTORY: saved assistant message + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY: persist assistant message failed: \(error)")
                #endif
            }
        }
    }

     static func makePreviewText(_ text: String, limit: Int = 80) -> String {
        let s = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard s.count > limit else { return s }
        let idx = s.index(s.startIndex, offsetBy: limit)
        return String(s[..<idx]) + "…"
    }

import Foundation

extension PhotoToVideoChatViewModel {

    // MARK: - Step 4: On-demand cache for any template (tap-driven)

    func ensurePreviewCachedForSelectedTemplate() {
        guard let id = selectedTemplateId else { return }
        ensurePreviewCachedForTemplate(id: id)
    }

    func ensurePreviewCachedForTemplate(id: Int) {
        if let local = cachedPreviewLocalURLByTemplateId[id], fileExists(local) {
            return
        }

        if let local = cachedPreviewLocalURLByTemplateId[id], !fileExists(local) {
            cachedPreviewLocalURLByTemplateId[id] = nil
        }

        if let t = onDemandTasks[id] {
            if !t.isCancelled { return }
            onDemandTasks[id] = nil
        }

        guard let template = templates.first(where: { $0.id == id }) else { return }
        guard let remote = bestPreviewRemoteURL(for: template) else { return }

        #if DEBUG
        print("🎞️ ONDEMAND start templateId=\(id)")
        #endif

        let t = Task { [weak self] in
            guard let self else { return }
            defer {
                Task { @MainActor [weak self] in
                    self?.onDemandTasks[id] = nil
                }
            }

            do {
                let local = try await self.templateVideoCache.fetch(remoteURL: remote)
                try Task.checkCancellation()

                await MainActor.run {
                    self.cachedPreviewLocalURLByTemplateId[id] = local
                }

                #if DEBUG
                print("🎞️ ONDEMAND ok templateId=\(id) local=\(local.lastPathComponent)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🎞️ ONDEMAND cancelled templateId=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🎞️ ONDEMAND failed templateId=\(id) error=\(error)")
                #endif
            }
        }

        onDemandTasks[id] = t
    }
}

import Foundation

extension PhotoToVideoChatViewModel {

    // MARK: - Step 3: Prewarm (first 10, parallelism=3)

     func startPrewarmIfNeeded() {
        prewarmTask?.cancel()
        prewarmTask = nil

        guard !templates.isEmpty else { return }

        let slice = Array(templates.prefix(prewarmCount))

        var queue: [(id: Int, remote: URL)] = []
        queue.reserveCapacity(slice.count)

        for t in slice {
            guard let best = bestPreviewRemoteURL(for: t) else { continue }

            if let local = cachedPreviewLocalURLByTemplateId[t.id], fileExists(local) {
                continue
            }
            if let local = cachedPreviewLocalURLByTemplateId[t.id], !fileExists(local) {
                cachedPreviewLocalURLByTemplateId[t.id] = nil
            }

            queue.append((t.id, best))
        }

        if queue.isEmpty {
            #if DEBUG
            print("🎞️ PREWARM: nothing to do (already cached or no urls)")
            #endif
            return
        }

        #if DEBUG
        print("🎞️ PREWARM start count=\(min(prewarmCount, templates.count)) tasks=\(queue.count) parallel=\(prewarmParallelism)")
        #endif

        prewarmTask = Task { [weak self] in
            guard let self else { return }

            var index = 0

            func nextJob() -> (id: Int, remote: URL)? {
                if index >= queue.count { return nil }
                let job = queue[index]
                index += 1
                return job
            }

            await withTaskGroup(of: Void.self) { group in
                let workers = max(1, self.prewarmParallelism)

                for _ in 0..<workers {
                    group.addTask { [weak self] in
                        guard let self else { return }

                        while true {
                            let job: (id: Int, remote: URL)? = await MainActor.run { nextJob() }
                            guard let job else { break }
                            if Task.isCancelled { break }

                            do {
                                let local = try await self.templateVideoCache.fetch(remoteURL: job.remote)
                                try Task.checkCancellation()

                                await MainActor.run {
                                    if let existing = self.cachedPreviewLocalURLByTemplateId[job.id],
                                       self.fileExists(existing) {
                                        return
                                    }
                                    self.cachedPreviewLocalURLByTemplateId[job.id] = local
                                }

                                #if DEBUG
                                print("🎞️ PREWARM ok templateId=\(job.id) local=\(local.lastPathComponent)")
                                #endif

                            } catch is CancellationError {
                                break
                            } catch {
                                #if DEBUG
                                print("🎞️ PREWARM failed templateId=\(job.id) error=\(error)")
                                #endif
                            }
                        }
                    }
                }

                await group.waitForAll()
            }

            #if DEBUG
            print("🎞️ PREWARM done")
            #endif
        }
    }

     func bestPreviewRemoteURL(for t: VideoTemplateDTO) -> URL? {
        let short = (t.videoPreviewShort ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
        if !short.isEmpty, let u = URL(string: short) { return u }

        let full = t.videoPreview.trimmingCharacters(in: .whitespacesAndNewlines)
        if !full.isEmpty, let u = URL(string: full) { return u }

        return nil
    }
}

    func ensureTemplatesLoadedOnAppear(lang: String = "en") {
        guard templateGroups.isEmpty else { return }
        guard !isTemplatesLoading else { return }
        loadVideoTemplates(lang: lang)
    }

    func loadVideoTemplates(userId: String? = nil, tag: String? = nil, lang: String = "en") {
        templatesErrorText = nil
        isTemplatesLoading = true

        templatesTask?.cancel()
        templatesTask = Task { [weak self] in
            guard let self else { return }
            defer { self.isTemplatesLoading = false }

            do {
                let groups = try await templatesService.getVideoTemplates(
                    userId: userId,
                    tag: tag,
                    lang: lang
                )

                self.templateGroups = groups
                self.templates = Self.flattenTemplates(groups: groups)

                #if DEBUG
                print("🎞️ TEMPLATES: templates.count = \(self.templates.count)")
                #endif

                // Step 3: start prewarm after templates loaded
                self.startPrewarmIfNeeded()

            } catch is CancellationError {
                // ignore
            } catch {
                self.templatesErrorText = String(describing: error)
            }
        }
    }

    fileprivate static func flattenTemplates(groups: [VideoTemplateGroupDTO]) -> [VideoTemplateDTO] {
        var seen: Set<Int> = []
        var out: [VideoTemplateDTO] = []
        out.reserveCapacity(groups.reduce(0) { $0 + $1.videos.count })

        for g in groups {
            for v in g.videos {
                let preview = v.photoPreview.trimmingCharacters(in: .whitespacesAndNewlines)
                guard !preview.isEmpty else { continue }

                guard !seen.contains(v.id) else { continue }
                seen.insert(v.id)

                out.append(v)
            }
        }

        return out
    }
}

     func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

     func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

import Foundation
import UIKit

extension PhotoToVideoChatViewModel {

    // MARK: - Upload preparation (JPEG)

     static func preparePhotoForUpload(originalData: Data) -> (data: Data, mimeType: String, filename: String) {
        if let image = UIImage(data: originalData),
           let jpeg = image.jpegData(compressionQuality: 0.92),
           !jpeg.isEmpty {
            return (jpeg, "image/jpeg", "photo.jpg")
        }

        let detected = detectMimeAndFilename(from: originalData)
        return (originalData, detected.mime, detected.filename)
    }

     static func detectMimeAndFilename(from data: Data) -> (mime: String, filename: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "photo.png") }

        let jpg: [UInt8] = [0xFF, 0xD8, 0xFF]
        if data.starts(with: jpg) { return ("image/jpeg", "photo.jpg") }

        return ("application/octet-stream", "photo.bin")
    }
}

import SwiftUI
import AVFoundation
import UIKit

/// Встраиваемое видео-превью без контролов.
/// - Автостарт при изменении playToken (или url)
/// - Проигрывает ОДИН раз и вызывает onFinished()
/// - НЕ считает `.unknown` ошибкой через 1.5s (это нормальная стадия загрузки)
struct InlineAutoplayVideoPreview: View {

    let url: URL?
    let playToken: UUID
    let onFinished: () -> Void

    @State private var player: AVPlayer?
    @State private var item: AVPlayerItem?

    @State private var didFinishObserver: NSObjectProtocol?
    @State private var statusObserver: NSKeyValueObservation?

    @State private var isLoading: Bool = false
    @State private var didStartPlaying: Bool = false

    /// Если хочешь “не зависать серым” — оставь таймаут.
    /// Но он должен быть разумным, иначе ты сам себе ломаешь UX.
    private let loadingTimeoutSec: UInt64 = 8

    var body: some View {
        ZStack {
            Color.black.opacity(0.06)

            if let player {
                PlayerLayerView(player: player)
                    .onDisappear { cleanup() }
            }

            if isLoading {
                ProgressView()
            }
        }
        .onAppear { restart() }
        .onChange(of: playToken) { _ in restart() }
        .onChange(of: url?.absoluteString) { _ in restart() }
    }

    @MainActor
    private func restart() {
        cleanup()

        guard let url else { return }

        isLoading = true
        didStartPlaying = false

        // Новый item каждый запуск
        let newItem = AVPlayerItem(url: url)

        // Быстрее стартуем, меньше "подвисаний"
        let newPlayer = AVPlayer(playerItem: newItem)
        newPlayer.isMuted = true
        newPlayer.actionAtItemEnd = .pause
        newPlayer.automaticallyWaitsToMinimizeStalling = false

        self.item = newItem
        self.player = newPlayer

        // 1) конец воспроизведения (1 раз)
        didFinishObserver = NotificationCenter.default.addObserver(
            forName: .AVPlayerItemDidPlayToEndTime,
            object: newItem,
            queue: .main
        ) { _ in
            onFinished()
        }

        // 2) наблюдаем статус: unknown -> ready/failed
        statusObserver = newItem.observe(\.status, options: [.initial, .new]) { [weak newPlayer] item, _ in
            DispatchQueue.main.async {
                switch item.status {
                case .readyToPlay:
                    self.isLoading = false
                    self.didStartPlaying = true
                    newPlayer?.seek(to: .zero)
                    newPlayer?.play()

                case .failed:
                    self.isLoading = false
                    #if DEBUG
                    print("🎞️ InlinePreview failed: \(item.error?.localizedDescription ?? "unknown") url=\(url.absoluteString)")
                    #endif
                    self.onFinished()

                case .unknown:
                    // Нормальная стадия загрузки — просто ждём.
                    break

                @unknown default:
                    break
                }
            }
        }

        // 3) (опционально) таймаут загрузки — но НЕ 1.5s.
        Task { @MainActor in
            try? await Task.sleep(nanoseconds: loadingTimeoutSec * 1_000_000_000)

            // если к этому моменту мы всё ещё грузимся и item тот же — считаем зависанием
            guard self.item === newItem else { return }
            guard self.didStartPlaying == false else { return }

            #if DEBUG
            print("🎞️ InlinePreview timeout (\(loadingTimeoutSec)s): still not ready url=\(url.absoluteString)")
            #endif

            self.isLoading = false
            self.onFinished()
        }
    }

    @MainActor
    private func cleanup() {
        if let didFinishObserver {
            NotificationCenter.default.removeObserver(didFinishObserver)
            self.didFinishObserver = nil
        }
        statusObserver?.invalidate()
        statusObserver = nil

        player?.pause()
        player = nil
        item = nil

        isLoading = false
        didStartPlaying = false
    }
}

private struct PlayerLayerView: UIViewRepresentable {

    let player: AVPlayer

    func makeUIView(context: Context) -> PlayerUIView {
        let v = PlayerUIView()
        v.setPlayer(player)
        return v
    }

    func updateUIView(_ uiView: PlayerUIView, context: Context) {
        uiView.setPlayer(player)
    }
}

private final class PlayerUIView: UIView {

    override static var layerClass: AnyClass { AVPlayerLayer.self }
    private var playerLayer: AVPlayerLayer { layer as! AVPlayerLayer }

    func setPlayer(_ player: AVPlayer) {
        playerLayer.player = player
        playerLayer.videoGravity = .resizeAspectFill
    }
}

import Foundation

extension TextToVideoChatViewModel {

    func onGenerateTap() {
        let cleanPrompt = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !cleanPrompt.isEmpty else {
            showUserError("Prompt пустой.")
            return
        }

        errorText = nil
        lastResultURL = nil
        isLoading = true

        // новая генерация — сбрасываем “активный id”
        activeGenerationMessageId = nil

        messages.append(
            GenerationMessage(
                role: .user,
                createdAt: Date(),
                payload: .text(cleanPrompt)
            )
        )

        // Step 4.2
        persistUserPromptToHistory(cleanPrompt)

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let req = TextToVideoRequest(
                    prompt: cleanPrompt,
                    aspectRatio: self.selectedAspectRatio.rawValue
                )

                let result = try await self.useCase.execute(req)
                try Task.checkCancellation()

                self.lastResultURL = result.resultURL

                // Step 5B + Step 5D (guard)
                await self.appendVideoPlaceholderAndDownload(remoteURL: result.resultURL)

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil
                self.messages.append(
                    GenerationMessage(
                        role: .system,
                        createdAt: Date(),
                        payload: .system("⛔️ Cancelled")
                    )
                )

            } catch {
                self.lastResultURL = nil
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                self.messages.append(
                    GenerationMessage(
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                )

                // Step 4.3
                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }
}

import Foundation

extension TextToVideoChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open TextToVideo): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open TextToVideo): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open TextToVideo): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open TextToVideo): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open TextToVideo): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

    // Step 4.2 (History)
    func persistUserPromptToHistory(_ prompt: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let clean = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !clean.isEmpty else { return }

        let preview = Self.makePreviewText(clean)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .user,
                    createdAt: now,
                    payload: .text(clean)
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: saved user prompt + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: persist user prompt failed: \(error)")
                #endif
            }
        }
    }

    // Step 4.3 (History)
    func persistAssistantErrorToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .error(text), previewSeed: text)
    }

    func persistAssistantPayloadToHistory(payload: GenerationPayload, previewSeed: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let preview = Self.makePreviewText(previewSeed)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .assistant,
                    createdAt: now,
                    payload: payload
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: saved assistant message + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: persist assistant message failed: \(error)")
                #endif
            }
        }
    }

    static func makePreviewText(_ text: String, limit: Int = 80) -> String {
        let s = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard s.count > limit else { return s }
        let idx = s.index(s.startIndex, offsetBy: limit)
        return String(s[..<idx]) + "…"
    }
}

extension TextToVideoChatViewModel {

    // Step 5B + Step 5D: append → download → replace (строго последовательно + guard)
    func appendVideoPlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        // Step 5D: messageId = “активный”
        let messageId = UUID()
        activeGenerationMessageId = messageId

        do {
            await ensureThreadRegisteredAsync()

            // 1) Storage FIRST
            let placeholder = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .video(remoteURL: remoteURL, localURL: nil)
            )

            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(preview: "🎞️ Video", at: now)

            // 2) UI placeholder (same id)
            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToVideo): appended placeholder threadId=\(id) msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            // 3) Download (await)
            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "video"
            )
            try Task.checkCancellation()

            // Step 5D: поздний результат от старой генерации — игнорируем
            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(TextToVideo): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            // 4) Storage replace (same id)
            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            // 5) UI replace (same id)
            replaceGenerationMessageInUI(updated: updated)

            // 6) preview после успешного replace
            try await touchThread(preview: "✅ Video", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToVideo): replaced with local threadId=\(id) msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToVideo): cancelled msgId=\(messageId)")
            #endif

        } catch {
            let pretty = String(describing: error)

            let failed = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .error("🎞️ Video (remote)\n\(remoteURL.absoluteString)\n\n⚠️ Не удалось скачать: \(pretty)")
            )
            replaceGenerationMessageInUI(updated: failed)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToVideo): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

import Foundation

extension TextToVideoChatViewModel {

    // Thread helpers (Step 1)

    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

        )
    ]

    @Published var prompt: String = ""

    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    var mediaCacheForViewer: MediaCacheService { mediaCache }

        replaceGenerationMessageInUI(updated: updated)
    }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    // Step 3E: eager recovery on open-from-history
    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    /// Нужен, чтобы cancel() мог корректно заменить текущий статус/плейсхолдер,
    /// а не добавлять отдельное сообщение.

    /// Step 5D: защита от гонок “старый download перетёр новый”
    var activeGenerationMessageId: UUID?

        self.shouldLoadHistory = (threadId != nil)
        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
        }
    }

        historyTask?.cancel()
        eagerRecoveryTask?.cancel()

    }

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func onGenerateTap(startPhotoData: Data, endPhotoData: Data) {
        errorText = nil
        lastResultURL = nil

        task?.cancel()
        task = nil

        activeStatusMessageId = nil
        activeGenerationMessageId = nil

        isLoading = true

        let promptTrimmed = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let normalizedPrompt: String? = promptTrimmed.isEmpty ? nil : promptTrimmed

        let userText: String = (normalizedPrompt == nil)
        ? ""
        : "\(promptTrimmed)"

        messages.append(
            GenerationMessage(role: .user, createdAt: Date(), payload: .text(userText))
        )

        // Step 4.2
        persistUserTextToHistory(userText)

        // статус — реальное сообщение с ID (единый id для UI+storage)
        let statusId = UUID()
        activeStatusMessageId = statusId
        messages.append(
            GenerationMessage(id: statusId, role: .assistant, createdAt: Date(), payload: .text("⏳ Генерация: старт…"))
        )

        task = Task { [weak self] in
            guard let self else { return }
            defer {
                self.isLoading = false
                self.activeStatusMessageId = nil
            }

            do {
                self.replaceGenerationMessage(id: statusId, payload: .text("⏳ Генерация: загружаю кадры…"))

                let startBinary = Self.makePhotoBinary(from: startPhotoData, baseName: "start_frame")
                let endBinary = Self.makePhotoBinary(from: endPhotoData, baseName: "end_frame")

                let request = TwoPhotosToVideoRequest(
                    startFrame: startBinary,
                    endFrame: endBinary,
                    prompt: normalizedPrompt
                )

                self.replaceGenerationMessage(id: statusId, payload: .text("⏳ Генерация: ожидаю результат…"))

                let result = try await self.useCase.execute(request)
                try Task.checkCancellation()

                self.lastResultURL = result.resultURL

                // Step 5B + Step 5D (guard внутри)
                await self.appendVideoPlaceholderAndDownload(
                    remoteURL: result.resultURL,
                    replacingMessageId: statusId
                )

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil
                self.replaceGenerationMessage(id: statusId, payload: .system("⛔️ Cancelled"))

            } catch {
                self.lastResultURL = nil
                let pretty = self.prettyErrorText(error)
                self.errorText = pretty

                let assistantText = "❌ Ошибка генерации:\n\(pretty)"
                self.replaceGenerationMessage(id: statusId, payload: .error(assistantText))

                // Step 4.3
                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }
}

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open TwoPhotosToVideo): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

        }
    }

    func replaceGenerationMessageInUI(updated: GenerationMessage) {
        if let idx = messages.firstIndex(where: { $0.id == updated.id }) {
            messages[idx] = updated
        } else {
            messages.append(updated)
        }
    }

    func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }

    // Errors

    func prettyErrorText(_ error: Error) -> String {
        if let e = error as? GenerationPollingError {
            switch e {
            case .timeout:
                return "Таймаут ожидания результата. Попробуй ещё раз."
            case .invalidState:
                return "Некорректное состояние задачи на сервере."
            case .failed(let msg):
                return msg
            }
        }

        let raw = String(describing: error)
        if raw.lowercased().contains("cancel") {
            return "Операция отменена."
        }
        return raw
    }

    // Binary helpers

    static func makePhotoBinary(from data: Data, baseName: String) -> PhotoBinary {
        let (mime, ext) = detectMimeAndExt(from: data)
        let filename = "\(baseName).\(ext)"
        return PhotoBinary(data: data, filename: filename, mimeType: mime)
    }

    static func detectMimeAndExt(from data: Data) -> (mime: String, ext: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "png") }

        let jpg: [UInt8] = [0xFF, 0xD8, 0xFF]
        if data.starts(with: jpg) { return ("image/jpeg", "jpg") }

        return ("application/octet-stream", "bin")
    }

    func appendVideoPlaceholderAndDownload(
        remoteURL: URL,
        replacingMessageId messageId: UUID
    ) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        // Step 5D: этот messageId считаем “активным” (важно, т.к. тут statusId переиспользуется)
        activeGenerationMessageId = messageId

        do {
            await ensureThreadRegisteredAsync()

            // IMPORTANT: statusId уже существует в messages (и не обязательно в repo).
            // Поэтому используем replace-or-append (repo уже “B”), чтобы НЕ ловить дубликаты.
            let placeholder = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .video(remoteURL: remoteURL, localURL: nil)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: placeholder)
            try await touchThread(preview: "🎞️ Video", at: now)

            // UI: placeholder (same id)
            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TwoPhotosToVideo): placeholder(replace-or-append) threadId=\(id) msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            // Download
            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "video"
            )
            try Task.checkCancellation()

            // Step 5D: поздний результат от старой генерации — игнорируем
            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(TwoPhotosToVideo): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            // Storage: replace
            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            // UI: replace (same id)
            replaceGenerationMessageInUI(updated: updated)

            // preview после успешного replace
            try await touchThread(preview: "✅ Video", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TwoPhotosToVideo): replaced with local threadId=\(id) msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            replaceGenerationMessage(id: messageId, payload: .system("⛔️ Cancelled"))

        } catch {
            let pretty = String(describing: error)

            replaceGenerationMessage(
                id: messageId,
                payload: .error("🎞️ Video (remote)\n\(remoteURL.absoluteString)\n\n⚠️ Не удалось скачать: \(pretty)")
            )

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TwoPhotosToVideo): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

extension TwoPhotosToVideoChatViewModel {

    // Thread helpers (Step 1)

    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

import Foundation

@MainActor
public final class OpenAITextChatViewModel: ObservableObject {

    // MARK: - UI State

    @Published public var messages: [ChatMessage] = []
    @Published public var inputText: String = ""
    @Published public var isLoading: Bool = false
    @Published public var errorText: String? = nil

    public struct PendingAttachmentUI: Identifiable, Equatable, Sendable {
        public let id: String
        public let prepared: PreparedAttachment

        public init(id: String = UUID().uuidString, prepared: PreparedAttachment) {
            self.id = id
            self.prepared = prepared
        }
    }

    @Published public var pendingAttachments: [PendingAttachmentUI] = []

    public var model: String = "gpt-5-mini"

    // MARK: - Dependencies

    let useCase: SendOpenAIChatMessageUseCase
    let imageBuilder: ImageAttachmentBuilder
    let fileBuilder: FileAttachmentBuilder

    let chatService: OpenAIChatService?

    let upsertThreadUseCase: UpsertChatThreadUseCase?
    let getThreadUseCase: GetChatThreadUseCase?

    // MARK: - Conversation Context

    var conversationId: String? = nil
    var lastKnownItemId: String? = nil
    var task: Task<Void, Never>? = nil

    var pendingAssistantMessageId: UUID? = nil

    static let initialSystemMessage =
        "Напиши сообщение — я отправлю его в OpenAI-чат и покажу ответ."

    // MARK: - Init

    public init(
        useCase: SendOpenAIChatMessageUseCase,
        imageBuilder: ImageAttachmentBuilder = ImageAttachmentBuilder(),
        fileBuilder: FileAttachmentBuilder = FileAttachmentBuilder(),

    ) {
        self.useCase = useCase
        self.imageBuilder = imageBuilder
        self.fileBuilder = fileBuilder
        self.chatService = chatService
        self.upsertThreadUseCase = upsertThreadUseCase
        self.getThreadUseCase = getThreadUseCase

        self.messages = [
            ChatMessage(role: .system, text: Self.initialSystemMessage)
        ]
    }

    // MARK: - Navigation helpers

    public func resetConversation() {
        cancel()

        conversationId = nil
        lastKnownItemId = nil

        inputText = ""
        errorText = nil
        pendingAssistantMessageId = nil
        clearPendingAttachments()

        messages = [
            ChatMessage(role: .system, text: Self.initialSystemMessage)
        ]
    }
    
    public func openConversation(conversationId: String, lastKnownItemId: String? = nil) {
        cancel()

        self.conversationId = conversationId
        self.lastKnownItemId = lastKnownItemId

        inputText = ""
        errorText = nil
        pendingAssistantMessageId = nil
        clearPendingAttachments()

        messages = [
            ChatMessage(role: .system, text: Self.initialSystemMessage),
            ChatMessage(role: .assistant, text: "⏳ Загружаю историю…")
        ]
    }

    // MARK: - Public

    public func cancel() {
        task?.cancel()
        task = nil
        isLoading = false

        if let id = pendingAssistantMessageId {
            replaceMessage(id: id, role: .assistant, text: "⛔️ Cancelled")
            pendingAssistantMessageId = nil
        }
    }

    public func send() {
        let trimmedText = inputText.trimmingCharacters(in: .whitespacesAndNewlines)

        guard !trimmedText.isEmpty || !pendingAttachments.isEmpty else { return }
        guard !isLoading else { return }

        let userTextForTitle = trimmedText
        let hadAttachments = !pendingAttachments.isEmpty

        errorText = nil
        inputText = ""

        if !trimmedText.isEmpty {
            messages.append(ChatMessage(role: .user, text: trimmedText))
        } else {
            messages.append(ChatMessage(role: .user, text: "📎 Attachment(s)"))
        }

        let pendingId = UUID()
        pendingAssistantMessageId = pendingId
        messages.append(ChatMessage(id: pendingId, role: .assistant, text: "⏳ …"))

        isLoading = true

        var parts: [OpenAIChatMessagePart] = []

        if !trimmedText.isEmpty {
            parts.append(.text(trimmedText))
        }

        let attachmentParts: [OpenAIChatMessagePart] = pendingAttachments.map { ui in
            let prepared = ui.prepared

            switch prepared.kind {
            case .image:
                let att = OpenAIChatAttachment(
                    filename: prepared.filename,
                    mimeType: prepared.mimeType,
                    byteSize: prepared.sizeBytes,
                    source: .photoLibrary,
                    dataURL: prepared.dataURL,
                    base64: prepared.base64
                )
                return .image(att)

            case .file:
                let att = OpenAIChatAttachment(
                    filename: prepared.filename,
                    mimeType: prepared.mimeType,
                    byteSize: prepared.sizeBytes,
                    source: .files,
                    dataURL: nil,
                    base64: prepared.base64
                )
                return .file(att)
            }
        }

        parts.append(contentsOf: attachmentParts)

        let draft = OpenAIChatMessageDraft(
            conversationId: conversationId,
            lastKnownItemId: lastKnownItemId,
            model: model,
            parts: parts,
            instructions: nil,
            maxOutputTokens: nil,
            temperature: nil
        )

        let pendingBubbleId = pendingId

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }

            do {
                let result: OpenAIChatMessageResult

                if let withEvents = self.useCase as? SendOpenAIChatMessageUseCaseWithEvents {
                    result = try await withEvents.execute(draft, onEvent: { [weak self] event in
                        switch event {
                        case .jobAccepted(let id, let status):
                            let short = String(id.uuidString.prefix(6))
                            let text = "⏳ \(status.rawValue) • job \(short)…"

                            Task { @MainActor [weak self] in
                                self?.replaceMessage(id: pendingBubbleId, role: .assistant, text: text)
                            }
                        }
                    })
                } else {
                    result = try await self.useCase.execute(draft)
                }

                try await self.upsertHistoryIfNeeded(
                    result: result,
                    userTextForTitle: userTextForTitle,
                    hadAttachments: hadAttachments
                )

                await MainActor.run {
                    self.conversationId = result.conversationId
                    self.lastKnownItemId = result.lastItemId ?? self.lastKnownItemId

                    self.replaceMessage(id: pendingBubbleId, role: .assistant, text: result.assistantText)

                    self.pendingAssistantMessageId = nil
                    self.clearPendingAttachments()
                    self.isLoading = false
                }

            } catch is CancellationError {
                await MainActor.run {
                    self.isLoading = false
                    self.replaceMessage(id: pendingBubbleId, role: .assistant, text: "⛔️ Cancelled")
                    self.pendingAssistantMessageId = nil
                }

            } catch {
                await MainActor.run {
                    self.isLoading = false
                    self.errorText = String(describing: error)
                    self.replaceMessage(id: pendingBubbleId, role: .assistant, text: "❌ Error:\n\(String(describing: error))")
                    self.pendingAssistantMessageId = nil
                }
            }
        }
    }
}

import Foundation

extension OpenAITextChatViewModel {

    public func loadInitialMessages(limit: Int = 50) async {
        guard let conversationId else { return }
        guard let chatService else {
            await MainActor.run {
                self.replaceLoadingStubIfPresent(with: "⚠️ chatService не внедрён (DI). Историю загрузить нельзя.")
            }
            return
        }

        do {
            let resp = try await chatService.listConversationItems(
                conversationId: conversationId,
                after: nil,
                limit: limit,
                order: .asc,
                include: nil
            )

            let mapped: [ChatMessage] = resp.data.compactMap { Self.mapItemToChatMessage($0) }

            await MainActor.run {
                let system = ChatMessage(role: .system, text: Self.initialSystemMessage)

                if mapped.isEmpty {
                    self.messages = [
                        system,
                        ChatMessage(role: .assistant, text: "История пуста (items нет). Можно продолжить переписку.")
                    ]
                } else {
                    
                    self.messages = [system] + mapped
                }

                
                self.lastKnownItemId = resp.lastId ?? self.lastKnownItemId
            }

            
            try await self.upsertHistoryAfterLoadIfNeeded(conversationId: conversationId)

        } catch {
            await MainActor.run {
                self.replaceLoadingStubIfPresent(with: "❌ Не удалось загрузить историю: \(String(describing: error))")
            }
        }
    }

    fileprivate func replaceLoadingStubIfPresent(with text: String) {
        if let idx = messages.firstIndex(where: { $0.role == .assistant && $0.text.contains("Загружаю историю") }) {
            let old = messages[idx]
            messages[idx] = ChatMessage(id: old.id, role: .assistant, text: text, createdAt: old.createdAt)
        } else {
            messages.append(ChatMessage(role: .assistant, text: text))
        }
    }

    fileprivate static func mapItemToChatMessage(_ item: OpenAIConversationItemDTO) -> ChatMessage? {
        let text: String = extractText(from: item.content)
        let collapsed = collapseSpaces(text)

        guard collapsed.isEmpty == false else { return nil }

        let role: ChatMessage.Role = {
            switch item.role {
            case .user:
                return .user
            case .assistant:
                return .assistant
            case .system, .developer:
                return .system
            default:
                return .system
            }
        }()

        return ChatMessage(role: role, text: collapsed)
    }

    fileprivate static func extractText(from content: OpenAIItemContentDTO?) -> String {
        guard let content else { return "" }

        if let raw = content.rawString {
            return raw
        }

        if let parts = content.parts, parts.isEmpty == false {
            let texts = parts.compactMap { $0.text }.filter { $0.isEmpty == false }
            return texts.joined(separator: "\n")
        }

        return ""
    }

    fileprivate func upsertHistoryAfterLoadIfNeeded(conversationId: String) async throws {
        guard let upsertThreadUseCase else { return }

        let existing = try await getThreadUseCase?.execute(conversationId: conversationId)

        let now = Date()
        let preview: String? = {
            if let last = messages.last(where: { $0.role == .assistant || $0.role == .user }) {
                return Self.makePreview(from: last.text)
            }
            return nil
        }()

        guard let existing else {
            
            return
        }

        let updated = ChatThread(
            id: existing.id,
            conversationId: existing.conversationId,
            title: existing.title,
            createdAt: existing.createdAt,
            lastActivityAt: now,
            preview: preview ?? existing.preview,
            icon: existing.icon,
            isPinned: existing.isPinned
        )

        try await upsertThreadUseCase.execute(updated)
    }
}

import Foundation

extension OpenAITextChatViewModel {

    // MARK: - Step 4: History Upsert

    func upsertHistoryIfNeeded(
        result: OpenAIChatMessageResult,
        userTextForTitle: String,
        hadAttachments: Bool
    ) async throws {
        guard let upsert = upsertThreadUseCase else { return }

        let now = Date()
        let conversationId = result.conversationId

        // если можем — читаем существующий тред, чтобы НЕ перетереть title/pinned/icon/id/createdAt
        let existing = try await getThreadUseCase?.execute(conversationId: conversationId)

        let title: String = {
            if let existingTitle = existing?.title,
               existingTitle.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty == false {
                return existingTitle
            }
            return Self.makeTitle(userText: userTextForTitle, hadAttachments: hadAttachments)
        }()

        let icon: ChatThreadIcon = {
            if let existingIcon = existing?.icon { return existingIcon }
            return Self.inferIcon(from: title)
        }()

        let preview = Self.makePreview(from: result.assistantText)

        let thread = ChatThread(
            id: existing?.id ?? UUID(),
            conversationId: conversationId,
            title: title,
            createdAt: existing?.createdAt ?? now,
            lastActivityAt: now,
            preview: preview,
            icon: icon,
            isPinned: existing?.isPinned ?? false
        )

        try await upsert.execute(thread)
    }

    static func makeTitle(userText: String, hadAttachments: Bool) -> String {
        let collapsed = collapseSpaces(userText)
        if collapsed.isEmpty {
            return hadAttachments ? "Вложения" : "Новый чат"
        }

        let maxLen = 48
        if collapsed.count <= maxLen { return collapsed }

        let idx = collapsed.index(collapsed.startIndex, offsetBy: maxLen)
        return String(collapsed[..<idx]) + "…"
    }

    static func makePreview(from assistantText: String) -> String? {
        let t = collapseSpaces(assistantText)
        guard t.isEmpty == false else { return nil }

        let maxLen = 90
        if t.count <= maxLen { return t }

        let idx = t.index(t.startIndex, offsetBy: maxLen)
        return String(t[..<idx]) + "…"
    }

    static func inferIcon(from title: String) -> ChatThreadIcon {
        let s = title.lowercased()

        if s.contains("python") || s.contains("swift") || s.contains("код") || s.contains("code") || s.contains("ошибка") {
            return .code
        }

        if s.contains("рецепт") || s.contains("еда") || s.contains("паста") || s.contains("карбонара") || s.contains("food") {
            return .food
        }

        if s.contains("квант") || s.contains("физ") || s.contains("наук") || s.contains("science") {
            return .science
        }

        if s.contains("путеше") || s.contains("travel") || s.contains("поезд") || s.contains("flight") {
            return .travel
        }

        return .default
    }
}

import Foundation
import Combine

@MainActor
public final class ChatHistoryListViewModel: ObservableObject {

    // MARK: - Output

    @Published public private(set) var recent: [ChatThreadRowViewModel] = []

    @Published public private(set) var all: [ChatThreadRowViewModel] = []

    @Published public private(set) var filteredAll: [ChatThreadRowViewModel] = []

    // MARK: - Input (Search)

    @Published public var searchQuery: String = ""

    // MARK: - Dependencies

    private let observeUseCase: ObserveChatThreadsUseCase

    // MARK: - Private

    private var bag = Set<AnyCancellable>()
    private let relativeFormatter = RelativeDateTimeFormatter()

    private var latestThreads: [ChatThread] = []

    // MARK: - Init

    public init(observeUseCase: ObserveChatThreadsUseCase) {
        self.observeUseCase = observeUseCase
        bind()
    }

    // MARK: - Bindings

    private func bind() {
        let threadsPublisher = observeUseCase.execute()
            .receive(on: DispatchQueue.main)
            .handleEvents(receiveOutput: { [weak self] threads in
                self?.latestThreads = threads
            })
            .share()

        threadsPublisher
            .map { [weak self] threads -> [ChatThreadRowViewModel] in
                guard let self else { return [] }
                let sorted = self.sortThreadsForUI(threads)
                return sorted.map { self.mapThreadToRow($0) }
            }
            .sink { [weak self] rows in
                guard let self else { return }
                self.all = rows
                self.recent = Array(rows.prefix(5))
            }
            .store(in: &bag)

        let queryPublisher = $searchQuery
            .map { $0.trimmingCharacters(in: .whitespacesAndNewlines) }
            .removeDuplicates()
            .debounce(for: .milliseconds(300), scheduler: DispatchQueue.main)

        Publishers.CombineLatest(threadsPublisher, queryPublisher)
            .map { [weak self] threads, query -> [ChatThreadRowViewModel] in
                guard let self else { return [] }

                let sorted = self.sortThreadsForUI(threads)

                let filtered: [ChatThread]
                if query.isEmpty {
                    filtered = sorted
                } else {
                    let q = query.lowercased()
                    filtered = sorted.filter { thread in
                        let title = thread.title.lowercased()
                        let preview = (thread.preview ?? "").lowercased()
                        return title.contains(q) || preview.contains(q)
                    }
                }

                return filtered.map { self.mapThreadToRow($0) }
            }
            .sink { [weak self] rows in
                self?.filteredAll = rows
            }
            .store(in: &bag)
    }

    // MARK: - Sorting

    private func sortThreadsForUI(_ threads: [ChatThread]) -> [ChatThread] {
        threads.sorted { a, b in
            if a.isPinned != b.isPinned {
                return a.isPinned && !b.isPinned
            }
            return a.lastActivityAt > b.lastActivityAt
        }
    }

    // MARK: - Mapping

    private func mapThreadToRow(_ thread: ChatThread) -> ChatThreadRowViewModel {
        let title = thread.title.trimmingCharacters(in: .whitespacesAndNewlines)
        let safeTitle = title.isEmpty ? "Чат" : title

        let timeAgo = relativeFormatter.localizedString(for: thread.lastActivityAt, relativeTo: Date())
        let icon = thread.icon.systemImageName

        return ChatThreadRowViewModel(
            id: thread.id,
            conversationId: thread.conversationId,
            title: safeTitle,
            timeAgo: timeAgo,
            preview: thread.preview,
            iconSystemName: icon
        )
    }
}

            .task {
                guard didLoad == false else { return }
                didLoad = true

                vm.openConversation(conversationId: conversationId, lastKnownItemId: lastKnownItemId)
                await vm.loadInitialMessages(limit: 80)

                let failed = vm.messages.contains { msg in
                    msg.role == .assistant && msg.text.contains("Не удалось загрузить историю")
                }
                isUnavailable = failed
            }
    }

    private var unavailableBanner: some View {
        VStack(alignment: .leading, spacing: 8) {
            Text("⚠️ Чат недоступен")
                .font(.headline)

            Text("Сервер не отдал историю (404/403/422 и т.п.). Можно удалить тред локально, чтобы не висел в списке.")
                .font(.footnote)
                .foregroundStyle(.secondary)

            Button(role: .destructive) {
                Task { [resolver] in
                    let get = resolver.resolve(GetChatThreadUseCase.self)!
                    let del = resolver.resolve(DeleteChatThreadUseCase.self)!

                    if let thread = try? await get.execute(conversationId: conversationId) {
                        try? await del.execute(threadId: thread.id)
                    }

                    await MainActor.run {
                        dismiss()
                    }
                }
            } label: {
                Text("Удалить локально")
                    .font(.body)
                    .bold()
                    .frame(maxWidth: .infinity)
                    .padding(.vertical, 12)
                    .background(Color.red.opacity(0.10))
                    .clipShape(RoundedRectangle(cornerRadius: 14, style: .continuous))
            }
            .buttonStyle(.plain)
        }
        .padding(12)
        .background(.ultraThinMaterial)
        .clipShape(RoundedRectangle(cornerRadius: 16, style: .continuous))
        .overlay(
            RoundedRectangle(cornerRadius: 16, style: .continuous)
                .stroke(Color.black.opacity(0.10), lineWidth: 1)
        )
    }
