import SwiftUI

// MARK: - Bubble row (avatar + message)

struct ChatBubble: View {
    let message: ChatMessage

    private let avatarSize: CGFloat = 32.scale
    private let bubbleCorner: CGFloat = 20.scale

    private let systemHintMaxWidth: CGFloat = 320.scale
    private let systemHintCorner: CGFloat = 16.scale

    var body: some View {
        switch message.role {
        case .assistant:
            HStack(alignment: .top, spacing: 10.scale) {
                avatar(imageName: "Chat.robot")
                bubble
                Spacer(minLength: 40.scale)
            }

        case .user:
            HStack(alignment: .top, spacing: 10.scale) {
                Spacer(minLength: 40.scale)
                bubble
                avatar(imageName: "Chat.user")
            }

        case .system:
            systemHintRow
        }
    }

    // MARK: - Avatars

    private func avatar(imageName: String) -> some View {
        Image(imageName)
            .resizable()
            .scaledToFit()
            .frame(width: avatarSize, height: avatarSize)
    }

    // MARK: - System hint (compact)

    private var systemHintRow: some View {
        HStack {
            Spacer(minLength: 0)

            systemHint
                .frame(maxWidth: systemHintMaxWidth, alignment: .center)

            Spacer(minLength: 0)
        }
        .padding(.top, 6.scale)
    }

    private var systemHint: some View {
        Text(message.text)
            .font(Tokens.Font.regular11)
            .foregroundStyle(Tokens.Color.textPrimary)
            .multilineTextAlignment(.leading)
            .padding(.horizontal, 14.scale)
            .padding(.vertical, 10.scale)
            .background(
                RoundedRectangle(cornerRadius: systemHintCorner, style: .continuous)
                    .fill(Color.black.opacity(0.04))
            )
    }

    // MARK: - Regular bubble (user/assistant)

    private var bubble: some View {
        VStack(alignment: .leading, spacing: 10.scale) {

            if shouldShowInlineProcessing {
                inlineProcessing
            } else {

                if !message.text.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty {
                    Text(message.text)
                        .font(.system(size: 16.scale, weight: .regular))
                        .foregroundStyle(Tokens.Color.textPrimary)
                }

                if !message.attachments.isEmpty {
                    VStack(alignment: .leading, spacing: 10.scale) {
                        ForEach(message.attachments) { att in
                            MessageAttachmentChip(
                                kind: att.kind,
                                title: att.filename,
                                subtitle: formatBytes(att.sizeBytes)
                            )
                        }
                    }
                }
            }
        }
        .padding(.vertical, 12.scale)
        .padding(.horizontal, 14.scale)
        .background(bubbleBackground)
        .clipShape(RoundedRectangle(cornerRadius: bubbleCorner, style: .continuous))
        .shadow(color: bubbleShadowColor, radius: bubbleShadowRadius, x: 0, y: bubbleShadowYOffset)
    }

    private var shouldShowInlineProcessing: Bool {
        message.role == .assistant && message.text.trimmingCharacters(in: .whitespacesAndNewlines).hasPrefix("⏳")
    }

    private var inlineProcessing: some View {
        HStack(spacing: 8.scale) {
            ProgressView()
            Text("Processing…")
                .font(Tokens.Font.regular11)
                .foregroundStyle(Tokens.Color.textSecondary)
        }
    }

    private var bubbleBackground: Color {
        switch message.role {
        case .assistant:
            return Color(hex:"#202020")!
        case .user:
            return Color(hex:"#202020")!
        case .system:
            return Color(hex:"#202020")!
        }
    }

    private var bubbleShadowColor: Color {
        switch message.role {
        case .assistant:
            return Color.black.opacity(0.08)
        case .user, .system:
            return Color.clear
        }
    }

    private var bubbleShadowRadius: CGFloat {
        message.role == .assistant ? 10.scale : 0
    }

    private var bubbleShadowYOffset: CGFloat {
        message.role == .assistant ? 4.scale : 0
    }

    // MARK: - Helpers

    private func formatBytes(_ bytes: Int) -> String {
        let kb = Double(bytes) / 1024.0
        if kb < 1024.0 { return String(format: "%.0f KB", kb) }
        let mb = kb / 1024.0
        return String(format: "%.1f MB", mb)
    }
}

import SwiftUI

struct ChatContainerView<Composer: View>: View {

    let title: String
    let messages: [ChatMessage]
    let isLoading: Bool
    let errorText: String?
    let onTapMore: (() -> Void)?
    let composer: Composer

    @Environment(\.dismiss) private var dismiss

    @State private var didInitialScroll: Bool = false
    @State private var scrollTask: Task<Void, Never>? = nil

    // “Пользователь внизу?” — если нет, автоскролл не делаем (иначе будет дёргать, когда читают историю)
    @State private var isNearBottom: Bool = true

    // Настройка “насколько близко к низу” считаем “внизу”
    private let nearBottomThreshold: CGFloat = 120.scale

    private let bottomAnchorId: String = "CHAT_BOTTOM_ANCHOR"
    private let scrollSpaceName: String = "CHAT_SCROLL_SPACE"

    init(
        title: String,
        messages: [ChatMessage],
        isLoading: Bool,
        errorText: String?,
        onTapMore: (() -> Void)? = nil,
        @ViewBuilder composer: () -> Composer
    ) {
        self.title = title
        self.messages = messages
        self.isLoading = isLoading
        self.errorText = errorText
        self.onTapMore = onTapMore
        self.composer = composer()
    }

    var body: some View {
        VStack(spacing: 0) {

            header

            ScrollViewReader { proxy in
                ScrollView {
                    LazyVStack(alignment: .leading, spacing: 12.scale) {

                        ForEach(messages) { msg in
                            ChatBubble(message: msg)
                                .id(msg.id)
                        }

                        if let errorText, !errorText.isEmpty {
                            Text(errorText)
                                .foregroundStyle(.red)
                                .font(.footnote)
                                .padding(.top, 8.scale)
                        }

                        // Нижний якорь + геометрия для “я внизу?”
                        BottomAnchorProbe(
                            anchorId: bottomAnchorId,
                            threshold: nearBottomThreshold
                        )
                    }
                    .padding(.horizontal, 16.scale)
                    .padding(.vertical, 16.scale)
                }
                .coordinateSpace(name: scrollSpaceName)
                .contentShape(Rectangle())
                .onTapGesture {
                    UIApplication.dismissKeyboard()
                }
                .onPreferenceChange(IsNearBottomPreferenceKey.self) { newValue in
                    // Старайся не спамить state, но это ок — значение меняется только при реальном скролле/изменении высот.
                    isNearBottom = newValue
                }
                .onAppear {
                    attemptInitialScrollIfNeeded(proxy)
                }
                .onChange(of: scrollSignal) { _ in
                    // Автоскролл:
                    // - всегда при первом появлении/первой загрузке (без анимации)
                    // - дальше только если пользователь был “внизу”
                    if didInitialScroll == false {
                        didInitialScroll = true
                        scheduleScrollToBottom(proxy, animated: false)
                        return
                    }

                    guard isNearBottom else { return }
                    scheduleScrollToBottom(proxy, animated: true)
                }
            }

            composer
                .padding(.bottom, 8.scale)
                .background(TokensColor.chatBackground)
        }
        .background(TokensColor.chatBackground.ignoresSafeArea())
        .navigationBarHidden(true)
        .onDisappear {
            scrollTask?.cancel()
            scrollTask = nil
        }
    }

    // MARK: - Header

    private var header: some View {
        ZStack {
            Text(title)
                .font(Tokens.Font.medium17)
                .foregroundStyle(Tokens.Color.textPrimary)

            HStack {
                Button { dismiss() } label: {
                    Image("ic_nav_back_40")
                        .resizable()
                        .scaledToFit()
                        .frame(width: 40.scale, height: 40.scale)
                }

                Spacer()

                Button { onTapMore?() } label: {
                    Image("ic_nav_more_40")
                        .resizable()
                        .scaledToFit()
                        .frame(width: 40.scale, height: 40.scale)
                }
            }
            .padding(.horizontal, 12.scale)
        }
        .frame(height: 56.scale)
        .padding(.top, 6.scale)
        .background(TokensColor.chatBackground)
    }

    // MARK: - Scroll signal (важно!)

    /// Это ключевой момент.
    /// Мы триггерим автоскролл не только по count / last.id,
    /// но и по изменениям содержимого последнего сообщения (текст / attachments),
    /// чтобы сработало при replaceMessage(...) (id тот же).
    private var scrollSignal: ScrollSignal {
        let last = messages.last
        return ScrollSignal(
            count: messages.count,
            lastId: last?.id,
            lastTextHash: last?.text.hashValue ?? 0,
            lastAttachmentsCount: last?.attachments.count ?? 0,
            lastAttachmentsBytesSum: (last?.attachments.reduce(0, { $0 + $1.sizeBytes }) ?? 0)
        )
    }

    private struct ScrollSignal: Equatable {
        let count: Int
        let lastId: UUID?
        let lastTextHash: Int
        let lastAttachmentsCount: Int
        let lastAttachmentsBytesSum: Int
    }

    // MARK: - Initial scroll

    private func attemptInitialScrollIfNeeded(_ proxy: ScrollViewProxy) {
        guard didInitialScroll == false else { return }
        guard messages.isEmpty == false else { return }

        didInitialScroll = true
        scheduleScrollToBottom(proxy, animated: false)
    }

    // MARK: - Robust scroll (with layout-safe retries)

    private func scheduleScrollToBottom(_ proxy: ScrollViewProxy, animated: Bool) {
        scrollTask?.cancel()

        scrollTask = Task { @MainActor in
            // Для длинных сообщений высота “доезжает” не за один проход.
            // Несколько попыток — нормально и стабильно.
            let delaysMs: [UInt64] = animated
                ? [0, 30, 80, 140, 220, 320]
                : [0, 60, 140]

            for ms in delaysMs {
                if Task.isCancelled { return }

                if ms == 0 {
                    await Task.yield()
                } else {
                    try? await Task.sleep(nanoseconds: ms * 1_000_000)
                }

                if Task.isCancelled { return }

                if animated {
                    withAnimation(.easeOut(duration: 0.20)) {
                        proxy.scrollTo(bottomAnchorId, anchor: .bottom)
                    }
                } else {
                    proxy.scrollTo(bottomAnchorId, anchor: .bottom)
                }
            }
        }
    }
}

// MARK: - Bottom “near-bottom” probe via PreferenceKey

private struct IsNearBottomPreferenceKey: PreferenceKey {
    static var defaultValue: Bool = true
    static func reduce(value: inout Bool, nextValue: () -> Bool) {
        // Если в дереве вдруг будет несколько значений — берём последнее.
        value = nextValue()
    }
}

private struct BottomAnchorProbe: View {
    let anchorId: String
    let threshold: CGFloat

    var body: some View {
        GeometryReader { geo in
            // В coordinateSpace ScrollView’а:
            // когда bottomAnchor "почти на экране", значит мы около низа.
            let frame = geo.frame(in: .named("CHAT_SCROLL_SPACE"))
            let isNear = frame.maxY <= (geo.size.height + threshold)

            Color.clear
                .preference(key: IsNearBottomPreferenceKey.self, value: isNear)
        }
        .frame(height: 1)
        .id(anchorId)
    }
}

import Foundation

public struct ChatMessage: Identifiable, Equatable, Sendable {

    public enum Role: String, Sendable {
        case user
        case assistant
        case system
    }

    public enum AttachmentKind: String, Sendable {
        case image
        case file
    }

    public struct AttachmentPreview: Identifiable, Equatable, Sendable {
        public let id: String
        public let kind: AttachmentKind
        public let filename: String
        public let sizeBytes: Int

        public init(
            id: String = UUID().uuidString,
            kind: AttachmentKind,
            filename: String,
            sizeBytes: Int
        ) {
            self.id = id
            self.kind = kind
            self.filename = filename
            self.sizeBytes = sizeBytes
        }
    }

    public let id: UUID
    public let role: Role
    public let text: String
    public let createdAt: Date
    public let attachments: [AttachmentPreview]

    public init(
        id: UUID = UUID(),
        role: Role,
        text: String,
        createdAt: Date = Date(),
        attachments: [AttachmentPreview] = []
    ) {
        self.id = id
        self.role = role
        self.text = text
        self.createdAt = createdAt
        self.attachments = attachments
    }

    public init(
        id: UUID = UUID(),
        role: Role,
        text: String,
        createdAt: Date = Date()
    ) {
        self.init(id: id, role: role, text: text, createdAt: createdAt, attachments: [])
    }
}

import SwiftUI
import PhotosUI
import UniformTypeIdentifiers
import Swinject
import UIKit

public struct OpenAITextChatView: View {

    @Environment(\.resolver) public var resolver
    @Environment(\.dismiss) public var dismiss

    @EnvironmentObject private var purchaseManager: PurchaseManager
    @StateObject public var vm: OpenAITextChatViewModel

    @State public var photoPickerItems: [PhotosPickerItem] = []
    @State public var isPhotoPickerPresented: Bool = false
    @State public var showFileImporter: Bool = false

    // MARK: - Prefill (Quick Actions)

    public let initialText: String?
    public let autoSendInitialText: Bool
    @State public var didApplyInitialText: Bool = false

    // MARK: - Chat options / share

    @State public var isOptionsPresented: Bool = false

    @State public var shareItems: [Any] = []
    @State public var isSharePresented: Bool = false

    @State public var exportedTempDirectoryURL: URL?
    @State public var exportedTempZipURL: URL?

    // MARK: - RateUs -> Support Email (present AFTER RateUs dismiss)

    @State private var shouldOpenRateUsEmailAfterDismiss: Bool = false
    @State private var isRateUsSupportMailPresented: Bool = false
    @State private var rateUsSupportMailPayload: MailComposerPayload? = nil

    public init(
        vm: OpenAITextChatViewModel,
        initialText: String? = nil,
        autoSendInitialText: Bool = false
    ) {
        _vm = StateObject(wrappedValue: vm)
        self.initialText = initialText
        self.autoSendInitialText = autoSendInitialText
    }

    public var body: some View {
        ChatContainerView(
            title: "Chat",
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText,
            onTapMore: { isOptionsPresented = true }
        ) {
            inputBar
        }
        .task {
            guard didApplyInitialText == false else { return }
            didApplyInitialText = true

            let text = (initialText ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
            guard text.isEmpty == false else { return }

            vm.inputText = text

            if autoSendInitialText {
                vm.send()
            }
        }
        .onChange(of: vm.didHitDailyLimit) { hit in
            guard hit else { return }

            Task { @MainActor in
                try? await Task.sleep(nanoseconds: 3_000_000_000) // 3 секунды
                guard purchaseManager.isShowedPaywall == false else { return }
                purchaseManager.isShowedPaywall = true
            }
        }
        .onChange(of: vm.shouldShowRateUs) { should in
            guard should else { return }
            hideKeyboard()
            RateAskScheduler.shared.markAskShown()

            #if DEBUG
            print("⭐️ RateUs: presenting from OpenAITextChat")
            #endif
        }
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .topBarTrailing) {
                Button {
                    vm.cancel()
                } label: {
                    Image(systemName: "xmark.circle.fill")
                        .font(.system(size: 18, weight: .semibold))
                        .foregroundStyle(Tokens.Color.textSecondary)
                }
                .opacity(vm.isLoading ? 1 : 0)
                .disabled(!vm.isLoading)
            }
        }
        .fileImporter(
            isPresented: $showFileImporter,
            allowedContentTypes: [.pdf],
            allowsMultipleSelection: true
        ) { result in
            switch result {
            case .success(let urls):
                urls.forEach { vm.addFileURL($0) }
            case .failure(let error):
                vm.cancel()
                print("File importer error: \(error)")
            }
        }
        .photosPicker(
            isPresented: $isPhotoPickerPresented,
            selection: $photoPickerItems,
            maxSelectionCount: 10,
            matching: .images
        )
        .onChange(of: photoPickerItems) { newItems in
            guard !newItems.isEmpty else { return }
            Task {
                for item in newItems {
                    do {
                        if let data = try await item.loadTransferable(type: Data.self) {
                            vm.addImageData(data)
                        }
                    } catch {
                        print("PhotosPicker load error: \(error)")
                    }
                }
                photoPickerItems.removeAll()
            }
        }
        .overlay {
            if isOptionsPresented {
                ChatOptionsOverlay(
                    isPresented: $isOptionsPresented,
                    onCopyAll: {
                        copyWholeChat()
                        isOptionsPresented = false
                    },
                    onShare: {
                        shareChatAsZip()
                        isOptionsPresented = false
                    },
                    onExport: {
                        exportChatToFile()
                        isOptionsPresented = false
                    },
                    onDelete: {
                        isOptionsPresented = false
                        deleteChatLocal()
                    },
                    onClose: { isOptionsPresented = false }
                )
                .transition(.opacity)
                .zIndex(999)
            }
        }
        .sheet(isPresented: $isSharePresented, onDismiss: cleanupShareTempIfNeeded) {
            ActivityView(activityItems: shareItems)
        }
        .fullScreenCover(
            isPresented: $vm.shouldShowRateUs,
            onDismiss: {
                guard shouldOpenRateUsEmailAfterDismiss else { return }
                shouldOpenRateUsEmailAfterDismiss = false
                prepareAndPresentRateUsEmail()
            },
            content: {
                RateUsView { action in
                    hideKeyboard()

                    switch action {
                    case .liked:
                        RateAskScheduler.shared.markRated()
                        vm.onRateUsClosed(liked: true)

                    case .laterOpenEmail:
                        vm.onRateUsClosed(liked: false)
                        shouldOpenRateUsEmailAfterDismiss = true

                    case .laterJustClose:
                        vm.onRateUsClosed(liked: false)
                    }
                }
                .environment(\.dynamicTypeSize, .medium)
                .ignoresSafeArea()
            }
        )
        .sheet(isPresented: $isRateUsSupportMailPresented) {
            if let payload = rateUsSupportMailPayload {
                MailComposerView(payload: payload)
            } else {
                VStack(spacing: 12) {
                    Text("rateus.support.mail_prepare_failed".localizable)
                    Button("common.ok".localizable) {
                        isRateUsSupportMailPresented = false
                    }
                }
                .padding()
            }
        }
    }

            exportedTempDirectoryURL = nil
            try? FileManager.default.removeItem(at: dirURL)

    }

// MARK: - ZIP build

private extension OpenAITextChatView {

    func buildZipExport() throws -> (directoryURL: URL, zipURL: URL) {
        let baseName = makeZipBaseName()

        let dirURL = FileManager.default.temporaryDirectory
            .appendingPathComponent("openai_chat_export_\(UUID().uuidString)", isDirectory: true)

        try FileManager.default.createDirectory(at: dirURL, withIntermediateDirectories: true)

        let transcriptURL = dirURL.appendingPathComponent("transcript.txt")
        let text = buildTranscriptText()
        try text.write(to: transcriptURL, atomically: true, encoding: .utf8)

        let zipURL = FileManager.default.temporaryDirectory.appendingPathComponent("\(baseName).zip")
        try? FileManager.default.removeItem(at: zipURL)

        try FileManager.default.zipItem(at: dirURL, to: zipURL, shouldKeepParent: true)

        return (dirURL, zipURL)
    }

    func makeZipBaseName() -> String {
        let df = DateFormatter()
        df.dateFormat = "yyyy-MM-dd_HH-mm-ss"
        let stamp = df.string(from: Date())

        let cid = vm.currentConversationId ?? "local"
        let short = String(cid.prefix(8))

        return "openai_chat_\(short)_\(stamp)"
    }
}

import SwiftUI

extension OpenAITextChatView {

    // MARK: - Transcript helpers

    func buildTranscriptText() -> String {
        let blocks: [String] = vm.messages.map { msg in
            let role: String = {
                switch msg.role {
                case .system: return "SYSTEM"
                case .user: return "USER"
                case .assistant: return "ASSISTANT"
                }
            }()

            var text = "[\(role)] \(msg.text)".trimmingCharacters(in: .whitespacesAndNewlines)

            if msg.attachments.isEmpty == false {
                let attLines = msg.attachments.map { att in
                    "• \(att.filename) (\(formatBytes(att.sizeBytes)))"
                }
                .joined(separator: "\n")

                if text.isEmpty { text = "[\(role)]" }
                text += "\n" + attLines
            }

            return text.trimmingCharacters(in: .whitespacesAndNewlines)
        }

        return blocks
            .filter { !$0.isEmpty }
            .joined(separator: "\n\n")
    }

    func makeExportFilename() -> String {
        let df = DateFormatter()
        df.dateFormat = "yyyy-MM-dd_HH-mm-ss"
        let stamp = df.string(from: Date())

        let cid = vm.currentConversationId ?? "local"
        let short = String(cid.prefix(8))

        return "chat_\(short)_\(stamp).txt"
    }

    func formatBytes(_ bytes: Int) -> String {
        let kb = Double(bytes) / 1024.0
        if kb < 1024.0 { return String(format: "%.0f KB", kb) }
        let mb = kb / 1024.0
        return String(format: "%.1f MB", mb)
    }
}

            let thread = try await get.execute(conversationId: conversationId)
            let anchor = (thread?.lastKnownItemId ?? "").trimmingCharacters(in: .whitespacesAndNewlines)

        } catch {

        }
    }

    @MainActor
    func persistLastKnownItemIdToHistoryIfPossible(
        conversationId: String,
        lastItemId: String?,
        preview: String?
    ) async {
        guard let get = getThreadUseCase, let upsert = upsertThreadUseCase else { return }

        let trimmed = (lastItemId ?? "").trimmingCharacters(in: .whitespacesAndNewlines)

            thread = thread
                .mergingLastKnownItemId(trimmed)
                .updatingActivity(lastActivityAt: Date(), preview: preview)

            try await upsert.execute(thread)

            self.lastKnownItemId = trimmed

            #if DEBUG
            print("🧾 OPENAI_THREAD anchor persisted conversationId=\(conversationId) lastKnownItemId=\(trimmed)")

import Foundation

extension OpenAITextChatViewModel {

    public func loadInitialMessages(limit: Int = 50) async {
        guard let conversationId else { return }
        guard let chatService else {
            await MainActor.run {
                self.replaceLoadingStubIfPresent(with: "openai_chat.history.di_missing".localizable)
            }
            return
        }

        do {
            let resp = try await chatService.listConversationItems(
                conversationId: conversationId,
                after: nil,
                limit: limit,
                order: .asc,
                include: nil
            )

            let mapped: [ChatMessage] = resp.data.compactMap { Self.mapItemToChatMessage($0) }

            await MainActor.run {
                let system = ChatMessage(role: .system, text: Self.initialSystemMessage)

                if mapped.isEmpty {
                    self.messages = [
                        system,
                        ChatMessage(
                            role: .assistant,
                            text: "openai_chat.history.empty_wait".localizable
                        )
                    ]
                } else {
                    self.messages = [system] + mapped
                }

                self.lastKnownItemId = resp.lastId ?? self.lastKnownItemId
            }

            await markNotificationsAsReadForOpenedConversationIfNeeded()

            try await self.upsertHistoryAfterLoadIfNeeded(conversationId: conversationId)

        } catch {
            await MainActor.run {
                let text = String(
                    format: "openai_chat.history.load_failed_with_error".localizable,
                    String(describing: error)
                )
                self.replaceLoadingStubIfPresent(with: text)
            }
        }
    }

    fileprivate func replaceLoadingStubIfPresent(with text: String) {
        if let idx = messages.firstIndex(where: {
            $0.role == .assistant &&
            $0.text.contains("openai_thread.loading.title".localizable)
        }) {
            let old = messages[idx]
            messages[idx] = ChatMessage(id: old.id, role: .assistant, text: text, createdAt: old.createdAt)
        } else {
            messages.append(ChatMessage(role: .assistant, text: text))
        }
    }

    fileprivate static func mapItemToChatMessage(_ item: OpenAIConversationItemDTO) -> ChatMessage? {
        let text: String = extractText(from: item.content)
        let collapsed = collapseSpaces(text)

        guard collapsed.isEmpty == false else { return nil }

        let role: ChatMessage.Role = {
            switch item.role {
            case .user:
                return .user
            case .assistant:
                return .assistant
            case .system, .developer:
                return .system
            default:
                return .system
            }
        }()

        return ChatMessage(role: role, text: collapsed)
    }

    fileprivate static func extractText(from content: OpenAIItemContentDTO?) -> String {
        guard let content else { return "" }

        if let raw = content.rawString {
            return raw
        }

        if let parts = content.parts, parts.isEmpty == false {
            let texts = parts.compactMap { $0.text }.filter { $0.isEmpty == false }
            return texts.joined(separator: "\n")
        }

        return ""
    }

    fileprivate func upsertHistoryAfterLoadIfNeeded(conversationId: String) async throws {
        guard let upsertThreadUseCase else { return }

        let existing = try await getThreadUseCase?.execute(conversationId: conversationId)

        let now = Date()
        let preview: String? = {
            if let last = messages.last(where: { $0.role == .assistant || $0.role == .user }) {
                return Self.makePreview(from: last.text)
            }
            return nil
        }()

        guard let existing else {
            return
        }

        let updated = ChatThread(
            id: existing.id,
            conversationId: existing.conversationId,
            title: existing.title,
            createdAt: existing.createdAt,
            lastActivityAt: now,
            preview: preview ?? existing.preview,
            icon: existing.icon,
            isPinned: existing.isPinned
        )

        try await upsertThreadUseCase.execute(updated)
    }
}

    func upsertHistoryEarlyIfNeeded(
        conversationId: String,
        userTextForTitle: String,
        hadAttachments: Bool
    ) async throws {
        guard let upsert = upsertThreadUseCase else { return }

        let now = Date()
        let existing = try await getThreadUseCase?.execute(conversationId: conversationId)

        let title: String = {
            if let existingTitle = existing?.title,
               existingTitle.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty == false {
                return existingTitle
            }
            return Self.makeTitle(userText: userTextForTitle, hadAttachments: hadAttachments)
        }()

        let icon: ChatThreadIcon = {
            if let existingIcon = existing?.icon { return existingIcon }
            return Self.inferIcon(from: title)
        }()

        let preview: String? = {
            let collapsedUserText = Self.collapseSpaces(userTextForTitle)
            if collapsedUserText.isEmpty == false {
                return Self.makePreview(from: collapsedUserText)
            }

            if hadAttachments {
                return "openai_chat.thread.attachments".localizable
            }

            return "openai_chat.thread.processing".localizable
        }()

        let thread = ChatThread(
            id: existing?.id ?? UUID(),
            conversationId: conversationId,
            title: title,
            createdAt: existing?.createdAt ?? now,
            lastActivityAt: now,
            preview: preview,
            icon: icon,
            isPinned: existing?.isPinned ?? false
        )

        try await upsert.execute(thread)
    }

    func upsertHistoryIfNeeded(
        result: OpenAIChatMessageResult,
        userTextForTitle: String,
        hadAttachments: Bool
    ) async throws {
        guard let upsert = upsertThreadUseCase else { return }

        let now = Date()
        let conversationId = result.conversationId

        let existing = try await getThreadUseCase?.execute(conversationId: conversationId)

        let title: String = {
            if let existingTitle = existing?.title,
               existingTitle.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty == false {
                return existingTitle
            }
            return Self.makeTitle(userText: userTextForTitle, hadAttachments: hadAttachments)
        }()

        let icon: ChatThreadIcon = {
            if let existingIcon = existing?.icon { return existingIcon }
            return Self.inferIcon(from: title)
        }()

        let preview = Self.makePreview(from: result.assistantText)

        let thread = ChatThread(
            id: existing?.id ?? UUID(),
            conversationId: conversationId,
            title: title,
            createdAt: existing?.createdAt ?? now,
            lastActivityAt: now,
            preview: preview,
            icon: icon,
            isPinned: existing?.isPinned ?? false
        )

        try await upsert.execute(thread)
    }

    static func makeTitle(userText: String, hadAttachments: Bool) -> String {
        let collapsed = collapseSpaces(userText)
        if collapsed.isEmpty {
            return hadAttachments
                ? "openai_chat.thread.attachments".localizable
                : "openai_chat.thread.new_chat".localizable
        }

        let maxLen = 48
        if collapsed.count <= maxLen { return collapsed }

        let idx = collapsed.index(collapsed.startIndex, offsetBy: maxLen)
        return String(collapsed[..<idx]) + "…"
    }

    static func makePreview(from assistantText: String) -> String? {
        let t = collapseSpaces(assistantText)
        guard t.isEmpty == false else { return nil }

        let maxLen = 90
        if t.count <= maxLen { return t }

        let idx = t.index(t.startIndex, offsetBy: maxLen)
        return String(t[..<idx]) + "…"
    }

    static func inferIcon(from title: String) -> ChatThreadIcon {
        let s = title.lowercased()

        if s.contains("python") || s.contains("swift") || s.contains("код") || s.contains("code") || s.contains("ошибка") {
            return .code
        }

        if s.contains("рецепт") || s.contains("еда") || s.contains("паста") || s.contains("карбонара") || s.contains("food") {
            return .food
        }

        if s.contains("квант") || s.contains("физ") || s.contains("наук") || s.contains("science") {
            return .science
        }

        if s.contains("путеше") || s.contains("travel") || s.contains("поезд") || s.contains("flight") {
            return .travel
        }

        return .default
    }
}

import Foundation

extension OpenAITextChatViewModel {

    public func send() {
        didHitDailyLimit = false

        if isRecordingSpeech {
            stopSpeech()
        }

        let trimmedText = inputText.trimmingCharacters(in: .whitespacesAndNewlines)

        guard !trimmedText.isEmpty || !pendingAttachments.isEmpty else { return }
        guard !isLoading else { return }

        let attachmentsSnapshot = pendingAttachments

        let userTextForTitle = trimmedText
        let hadAttachments = !attachmentsSnapshot.isEmpty

        errorText = nil
        inputText = ""

        let messageAttachments: [ChatMessage.AttachmentPreview] = attachmentsSnapshot.map { ui in
            let prepared = ui.prepared
            let kind: ChatMessage.AttachmentKind = (prepared.kind == .image) ? .image : .file

            return ChatMessage.AttachmentPreview(
                kind: kind,
                filename: prepared.filename,
                sizeBytes: prepared.sizeBytes
            )
        }

        let userBubbleText: String = {
            if !trimmedText.isEmpty { return trimmedText }
            return ""
        }()

        messages.append(
            ChatMessage(role: .user, text: userBubbleText, attachments: messageAttachments)
        )

        clearPendingAttachments()

        let pendingId = UUID()
        pendingAssistantMessageId = pendingId
        messages.append(
            ChatMessage(
                id: pendingId,
                role: .assistant,
                text: "openai_chat.send.pending_placeholder".localizable
            )
        )

        isLoading = true

        var parts: [OpenAIChatMessagePart] = []

        if !trimmedText.isEmpty {
            parts.append(.text(trimmedText))
        }

        let attachmentParts: [OpenAIChatMessagePart] = attachmentsSnapshot.map { ui in
            let prepared = ui.prepared

            switch prepared.kind {
            case .image:
                let att = OpenAIChatAttachment(
                    filename: prepared.filename,
                    mimeType: prepared.mimeType,
                    byteSize: prepared.sizeBytes,
                    source: .photoLibrary,
                    dataURL: prepared.dataURL,
                    base64: prepared.base64
                )
                return .image(att)

            case .file:
                let att = OpenAIChatAttachment(
                    filename: prepared.filename,
                    mimeType: prepared.mimeType,
                    byteSize: prepared.sizeBytes,
                    source: .files,
                    dataURL: nil,
                    base64: prepared.base64
                )
                return .file(att)
            }
        }

        parts.append(contentsOf: attachmentParts)

        let profile = botProfileStore.load()
        let builtInstructions = BotInstructionsBuilder.build(profile: profile)

        #if DEBUG
        let len = builtInstructions?.count ?? 0
        let isNil = (builtInstructions == nil)
        print("🧠 BOT_INSTRUCTIONS: isNil=\(isNil) len=\(len) conversationId=\(conversationId ?? "nil")")
        #endif

        let draft = OpenAIChatMessageDraft(
            conversationId: conversationId,
            lastKnownItemId: lastKnownItemId,
            model: model,
            parts: parts,
            instructions: builtInstructions,
            maxOutputTokens: nil,
            temperature: nil
        )

        let pendingBubbleId = pendingId

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }

            do {
                let result: OpenAIChatMessageResult

                if let withEvents = self.useCase as? SendOpenAIChatMessageUseCaseWithEvents {
                    result = try await withEvents.execute(draft, onEvent: { [weak self] event in
                        switch event {

                        case .conversationReady(let id):
                            Task { @MainActor [weak self] in
                                guard let self else { return }

                                self.conversationId = id

                                try? await self.upsertHistoryEarlyIfNeeded(
                                    conversationId: id,
                                    userTextForTitle: userTextForTitle,
                                    hadAttachments: hadAttachments
                                )

                                #if DEBUG
                                print("💬 OPENAI_CHAT: conversationReady id=\(id)")
                                #endif
                            }

                        case .jobAccepted(let id, let status):
                            let short = String(id.uuidString.prefix(6))

                            let text = String(
                                format: "openai_chat.send.job_status_format".localizable,
                                status.rawValue,
                                short
                            )

                            Task { @MainActor [weak self] in
                                self?.replaceMessage(id: pendingBubbleId, role: .assistant, text: text)
                            }
                        }
                    })
                } else {
                    result = try await self.useCase.execute(draft)
                }

                try await self.upsertHistoryIfNeeded(
                    result: result,
                    userTextForTitle: userTextForTitle,
                    hadAttachments: hadAttachments
                )

                await MainActor.run {
                    self.conversationId = result.conversationId
                    self.lastKnownItemId = result.lastItemId ?? self.lastKnownItemId

                    self.replaceMessage(id: pendingBubbleId, role: .assistant, text: result.assistantText)

                    self.pendingAssistantMessageId = nil
                    self.isLoading = false

                    self.maybeShowRateUsAfterSuccessfulReply()
                }

                let cid = result.conversationId
                let last = result.lastItemId

                let didCreate = await internalNotifications.postOpenAIChatReplyReady(
                    conversationId: cid,
                    lastItemId: last,
                    title: "openai_chat.thread_title".localizable,
                    createdAt: Date()
                )

                #if DEBUG
                print("🔔 OPENAI_CHAT: post ready didCreate=\(didCreate) conversationId=\(cid) lastItemId=\(last ?? "nil")")
                #endif

            } catch is CancellationError {
                let cid = self.conversationId

                await self.updateHistoryStatusIfPossible(
                    conversationId: cid,
                    preview: "openai_chat.send.cancelled".localizable
                )

                await MainActor.run {
                    self.isLoading = false
                    self.replaceMessage(
                        id: pendingBubbleId,
                        role: .assistant,
                        text: "openai_chat.send.cancelled".localizable
                    )
                    self.pendingAssistantMessageId = nil
                }

            } catch {
                if let limitError = error as? DailyChatLimitError, limitError == .limitReached {
                    await MainActor.run {
                        self.didHitDailyLimit = true

                        self.isLoading = false
                        self.errorText = "daily_chat.limit_reached".localizable

                        self.replaceMessage(
                            id: pendingBubbleId,
                            role: .assistant,
                            text: "daily_chat.limit_reached".localizable
                        )
                        self.pendingAssistantMessageId = nil
                    }

                    #if DEBUG
                    print("🧱 DAILY_CHAT: limit reached → didHitDailyLimit=true (paywall should open)")
                    #endif

                    return
                }

                let cid = self.conversationId

                let statusPreview: String = {
                    if let polling = error as? OpenAIChatPollingError, polling == .timeout {
                        return "openai_chat.send.timeout_try_again".localizable
                    }
                    return "openai_chat.send.error_sending".localizable
                }()

                await self.updateHistoryStatusIfPossible(
                    conversationId: cid,
                    preview: statusPreview
                )

                await MainActor.run {
                    self.isLoading = false
                    self.errorText = String(describing: error)

                    let assistantText = String(
                        format: "openai_chat.send.error_with_details".localizable,
                        String(describing: error)
                    )

                    self.replaceMessage(id: pendingBubbleId, role: .assistant, text: assistantText)
                    self.pendingAssistantMessageId = nil
                }
            }
        }
    }
}

import Foundation

extension OpenAITextChatViewModel {

    // MARK: - Speech

    public func toggleSpeech() async {
        guard !isLoading else { return }

        if isRecordingSpeech {
            stopSpeech()
            return
        }

        let status = await speechService.requestPermissions()
        guard status == .authorized else {
            errorText = "Нет доступа к микрофону/распознаванию речи. Разреши в настройках."
            return
        }

        startSpeech()
    }

     func startSpeech() {
        isRecordingSpeech = true
        speechSeconds = 0
        speechTimerText = formatMMSS(speechSeconds)
        startSpeechTimer()

        speechService.start(
            onPartial: { [weak self] text in
                guard let self else { return }
                self.inputText = text
            },
            onFinal: { [weak self] text in
                guard let self else { return }
                self.inputText = text
                self.stopSpeech()
            },
            onError: { [weak self] error in
                guard let self else { return }
                self.errorText = String(describing: error)
                self.cancelSpeech()
            }
        )

        #if DEBUG
        print("🎙️ OPENAI_CHAT: speech START")
        #endif
    }

     func stopSpeech() {
        speechService.stop()
        isRecordingSpeech = false
        stopSpeechTimer()

        #if DEBUG
        print("🎙️ OPENAI_CHAT: speech STOP")
        #endif
    }

     func cancelSpeech() {
        speechService.cancel()
        isRecordingSpeech = false
        stopSpeechTimer()

        #if DEBUG
        print("🎙️ OPENAI_CHAT: speech CANCEL")
        #endif
    }

     func startSpeechTimer() {
        speechTimerTask?.cancel()
        speechTimerTask = Task { [weak self] in
            guard let self else { return }
            while !Task.isCancelled && self.isRecordingSpeech {
                try? await Task.sleep(nanoseconds: 1_000_000_000)
                guard self.isRecordingSpeech else { break }
                self.speechSeconds += 1
                self.speechTimerText = self.formatMMSS(self.speechSeconds)
            }
        }
    }

     func stopSpeechTimer() {
        speechTimerTask?.cancel()
        speechTimerTask = nil
        speechSeconds = 0
        speechTimerText = "00:00"
    }

     func formatMMSS(_ seconds: Int) -> String {
        let m = seconds / 60
        let s = seconds % 60
        return String(format: "%02d:%02d", m, s)
    }
}
