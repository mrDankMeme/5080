import SwiftUI
import PhotosUI

struct PhotoEnhanceChatView: View {

    @EnvironmentObject private var purchaseManager: PurchaseManager
    @StateObject private var vm: PhotoEnhanceChatViewModel

    @State private var pickedItem: PhotosPickerItem?
    @State private var pickedPreview: UIImage?

    @State private var inputText: String = ""

    @State private var isPickerPresented: Bool = false

    init(vm: PhotoEnhanceChatViewModel) {
        _vm = StateObject(wrappedValue: vm)
    }

    var body: some View {
        GenerationChatContainerView(
            title: "photo_enhance.title".localizable,
            modeTitle: "photo_enhance.mode_title".localizable,
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText,
            threadId: vm.threadId,
            historyRepo: vm.historyRepoForViewer,
            mediaCache: vm.mediaCacheForViewer,
            onMessageUpdated: { updated in
                vm.applyMessageUpdateFromViewer(updated)
            }
        ) {
            GenerationChatComposerView(
                showsPlus: true,
                isLoading: vm.isLoading,
                text: $inputText,
                attachmentImage: pickedPreview,
                onPickPhotoTap: {
                    guard !vm.isLoading else { return }
                    isPickerPresented = true
                },
                onMicTap: {
                },
                onSendTap: {
                    onUpscalePressed()
                },
                isSendEnabled: pickedItem != nil && !vm.isLoading,
                placeholderText: "photo_enhance.composer.placeholder",
                isTextInputEnabled: false,
                forceSendIcon: true,
                isMicEnabled: false
            )
            .photosPicker(
                isPresented: $isPickerPresented,
                selection: $pickedItem,
                matching: .images
            )
            .onChange(of: pickedItem) { _ in
                Task { @MainActor in
                    await loadPreviewIfNeeded()
                }
            }
        }
        .onChange(of: vm.shouldShowPaywall) { should in
            guard should else { return }

            purchaseManager.isShowedPaywall = true
            vm.consumePaywallSignal()

            #if DEBUG
            print("💎 GEN(PhotoEnhance): show paywall from gate (kind=\(purchaseManager.currentSubKind))")
            #endif
        }
        .alert(item: $vm.limitAlert) { item in
            Alert(
                title: Text(item.title),
                message: Text(item.message),
                dismissButton: .default(Text(item.buttonTitle))
            )
        }
    }

    private func onUpscalePressed() {
        guard let pickedItem else {
            vm.showUserError("photo_enhance.error.pick_photo".localizable)
            return
        }

        Task { @MainActor in
            do {
                guard let data = try await pickedItem.loadTransferable(type: Data.self) else {
                    vm.showUserError("photo_enhance.error.read_photo_nil".localizable)
                    return
                }

                vm.onUpscaleTap(imageData: data)

                inputText = ""
                self.pickedItem = nil
                self.pickedPreview = nil

            } catch {
                vm.showUserError(
                    String(
                        format: "photo_enhance.error.read_photo".localizable,
                        String(describing: error)
                    )
                )
            }
        }
    }

    @MainActor
    private func loadPreviewIfNeeded() async {
        guard let pickedItem else {
            pickedPreview = nil
            return
        }

        do {
            if let data = try await pickedItem.loadTransferable(type: Data.self),
               let ui = UIImage(data: data) {
                pickedPreview = ui
            } else {
                pickedPreview = nil
            }
        } catch {
            pickedPreview = nil
        }
    }
}

import Foundation

@MainActor
final class PhotoEnhanceChatViewModel: ObservableObject {

    let threadId: UUID
    let kind: GenerationKind = .photoEnhance
    let shouldLoadHistory: Bool

    // MARK: - UI

    @Published var messages: [GenerationMessage] = [
        GenerationMessage(
            role: .system,
            createdAt: Date(),
            payload: .system("photo_enhance.system.pick_photo_and_send".localizable)
        )
    ]

    @Published var jobId: String = ""
    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    // ✅ Gate outputs (UI reacts)
    @Published var shouldShowPaywall: Bool = false
    @Published var limitAlert: LimitAlertItem? = nil

    // MARK: - Dependencies

    let useCase: UpscalePhotoUseCase
    let historyRepo: GenerationHistoryRepository
    let mediaCache: MediaCacheService
    let internalNotifications: InternalNotificationCenter
    let requestCounter: RequestCounterService
    let pendingService: PendingGenerationService
    let updateCenter: GenerationThreadUpdateCenter

    // MARK: - Viewer

    var historyRepoForViewer: GenerationHistoryRepository { historyRepo }
    var mediaCacheForViewer: MediaCacheService { mediaCache }

    func applyMessageUpdateFromViewer(_ updated: GenerationMessage) {
        replaceGenerationMessageInUI(updated: updated)
    }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    var activeGenerationMessageId: UUID?
    var activeGenerationId: UUID?

    private var updatesTask: Task<Void, Never>?
    private var lastReloadAt: Date = .distantPast

    // MARK: - Subscription snapshot provider (injected from factory)

    typealias SubscriptionSnapshot = (isSubscribed: Bool, kind: SubscriptionKind)

    private var subscriptionSnapshotProvider: (@Sendable () -> SubscriptionSnapshot)?

    func setSubscriptionSnapshotProvider(_ provider: @escaping @Sendable () -> SubscriptionSnapshot) {
        self.subscriptionSnapshotProvider = provider
    }

    // MARK: - Alert item

    struct LimitAlertItem: Identifiable, Equatable {
        let id: String = UUID().uuidString
        let title: String
        let message: String
        let buttonTitle: String
    }

    // MARK: - Anti-spam (DEBUG)

    private static let debugImageParallelLimit: Int = .max

    // MARK: - Init

    init(
        threadId: UUID? = nil,
        useCase: UpscalePhotoUseCase,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService,
        internalNotifications: InternalNotificationCenter,
        requestCounter: RequestCounterService,
        pendingService: PendingGenerationService,
        updateCenter: GenerationThreadUpdateCenter
    ) {
        self.threadId = threadId ?? UUID()
        self.shouldLoadHistory = (threadId != nil)

        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache
        self.internalNotifications = internalNotifications
        self.requestCounter = requestCounter
        self.pendingService = pendingService
        self.updateCenter = updateCenter

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
            subscribeToThreadUpdates()
        }
    }

    deinit {
        task?.cancel()
        historyTask?.cancel()
        eagerRecoveryTask?.cancel()
        updatesTask?.cancel()
    }

    // MARK: - Public

    func showUserError(_ text: String) {
        errorText = text
    }

    func consumePaywallSignal() {
        shouldShowPaywall = false
    }

    func cancel() {
        task?.cancel()
        task = nil

        historyTask?.cancel()
        historyTask = nil

        eagerRecoveryTask?.cancel()
        eagerRecoveryTask = nil

        if let gid = activeGenerationId {
            Task { [pendingService] in
                await pendingService.remove(generationId: gid)
            }
        }

        isLoading = false
        activeGenerationMessageId = nil
        activeGenerationId = nil
    }

    // MARK: - Updates subscription

    private func subscribeToThreadUpdates() {
        updatesTask?.cancel()

        updatesTask = Task { [weak self] in
            guard let self else { return }

            let stream = await self.updateCenter.stream()

            for await updatedThreadId in stream {
                if Task.isCancelled { break }
                guard updatedThreadId == self.threadId else { continue }

                let now = Date()
                if now.timeIntervalSince(self.lastReloadAt) < 0.35 { continue }
                self.lastReloadAt = now

                await MainActor.run {
                    self.loadHistoryIfExists()
                    self.startEagerRecoveryIfNeeded()
                }
            }
        }
    }

    // MARK: - Notifications (Variant A)

    func markNotificationsAsReadForThisThreadIfNeeded() {
        guard shouldLoadHistory else { return }

        Task {
            let list = await internalNotifications.list()

            let targets = list.filter { n in
                n.isRead == false
                && n.payload.threadId == self.threadId
                && n.payload.generationKind == self.kind
            }

            guard targets.isEmpty == false else { return }

            for n in targets {
                await internalNotifications.markRead(id: n.id)
            }

            #if DEBUG
            print("🔔 GEN(PhotoEnhance): marked read \(targets.count) notifications for threadId=\(self.threadId)")
            #endif
        }
    }

    // MARK: - Gate (shared helper)

    func evaluateGate(snapshot: RequestCounterSnapshot) -> GenerationGateDecision {
        let sub = subscriptionSnapshotProvider?() ?? (false, .unknown)
        return GenerationLimitGate.evaluate(
            isSubscribed: sub.isSubscribed,
            kind: sub.kind,
            snapshot: snapshot
        )
    }

    // MARK: - Debug limit helper

    static func validateParallelLimit(snapshot: RequestCounterSnapshot) -> Bool {
        snapshot.image < debugImageParallelLimit
    }
}

import Foundation
import UIKit

extension PhotoEnhanceChatViewModel {

    func onUpscaleTap(imageData: Data) {
        Task { @MainActor [weak self] in
            guard let self else { return }

            let snap = await self.requestCounter.snapshot()

            // ✅ STEP: gate before starting generation
            let gate = self.evaluateGate(snapshot: snap)
            switch gate {
            case .allow:
                break
            case .showPaywall:
                self.shouldShowPaywall = true
                return
            case .showAlert(let title, let message):
                self.limitAlert = LimitAlertItem(
                    title: title,
                    message: message,
                    buttonTitle: "OK"
                )
                return
            }

//            if Self.validateParallelLimit(snapshot: snap) == false {
//                self.showUserError("photo_enhance.error.parallel_limit".localizable)
//                return
//            }

            self.errorText = nil
            self.isLoading = true

            let job = self.jobId.trimmingCharacters(in: .whitespacesAndNewlines)
            let jobValue: String? = job.isEmpty ? nil : job

            let now = Date()
            let userText = Self.makeUserActionText(jobId: jobValue)

            let attachment: GenerationAttachment? = {
                let prepared = Self.prepareImageForAttachment(originalData: imageData)
                guard let localURL = saveInputAttachmentToTemp(
                    data: prepared.data,
                    filenameHint: prepared.filename
                ) else {
                    return nil
                }

                return GenerationAttachment(
                    kind: .image,
                    localURL: localURL,
                    previewLocalURL: localURL,
                    filename: prepared.filename,
                    mimeType: prepared.mimeType
                )
            }()

            let userMessage = GenerationMessage(
                role: .user,
                createdAt: now,
                payload: .text(userText),
                attachments: attachment.map { [$0] } ?? []
            )

            self.messages.append(userMessage)
            self.persistUserMessageToHistory(userMessage, previewSeed: userText)

            let placeholderId = UUID()
            self.activeGenerationMessageId = placeholderId

            let placeholderRemote = URL(string: "https://example.com/image.jpg")!
            let placeholderMsg = GenerationMessage(
                id: placeholderId,
                role: .assistant,
                createdAt: Date(),
                payload: .image(remoteURL: placeholderRemote, localURL: nil)
            )

            self.replaceGenerationMessageInUI(updated: placeholderMsg)

            self.persistAssistantMediaPlaceholderToHistory(
                message: placeholderMsg,
                preview: "photo_enhance.history.preview_enhancing".localizable
            )

            self.task?.cancel()
            self.task = Task { [weak self] in
                guard let self else { return }
                defer { self.isLoading = false }

                do {
                    let started = try await self.useCase.start(jobId: jobValue, imageData: imageData)
                    try Task.checkCancellation()

                    self.activeGenerationId = started.id
                    let counterJobId = started.id.uuidString

                    let pending = PendingGenerationRecord(
                        generationId: started.id,
                        counterJobId: counterJobId,
                        threadId: self.threadId,
                        placeholderMessageId: placeholderId,
                        generationKind: self.kind,
                        category: .image,
                        createdAt: Date()
                    )

                    await self.pendingService.upsert(pending)

                    #if DEBUG
                    print("🧾 PENDING +1 kind=\(self.kind.rawValue) threadId=\(self.threadId) generationId=\(started.id) placeholderId=\(placeholderId)")
                    #endif

                    let dto = try await self.useCase.poll(started: started)
                    try Task.checkCancellation()

                    let raw = (dto.result ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
                    guard self.activeGenerationMessageId == placeholderId else { return }

                    // terminal reason (for releaseOnTerminalFail)
                    let reason = (dto.error ?? "failed").trimmingCharacters(in: .whitespacesAndNewlines)
                    let reasonText = reason.isEmpty ? "failed" : reason

                    if let url = URL(string: raw), url.scheme != nil {
                        self.lastResultURL = url

                        // ✅ count success (even if backend marks failed but gives URL)
                        await self.requestCounter.commitSuccess(category: .image, jobId: counterJobId)

                        await self.updateImagePlaceholderRemote(
                            messageId: placeholderId,
                            remoteURL: url
                        )

                        await self.downloadAndFinalizeImage(
                            messageId: placeholderId,
                            remoteURL: url
                        )

                    } else {
                        self.lastResultURL = nil

                        if dto.status == .failed {
                            // ✅ count release on terminal fail (no URL)
                            await self.requestCounter.releaseOnTerminalFail(
                                category: .image,
                                jobId: counterJobId,
                                reason: reasonText
                            )

                            let assistantText = String(
                                format: "photo_enhance.error.generic".localizable,
                                reasonText
                            )

                            let failed = GenerationMessage(
                                id: placeholderId,
                                role: .assistant,
                                createdAt: placeholderMsg.createdAt,
                                payload: .error(assistantText)
                            )

                            self.replaceGenerationMessageInUI(updated: failed)
                            try? await self.historyRepo.replaceMessage(
                                threadId: self.threadId,
                                messageId: placeholderId,
                                newMessage: failed
                            )
                            self.persistAssistantErrorToHistory(assistantText)

                        } else {
                            // ✅ count success (done but not url)
                            await self.requestCounter.commitSuccess(category: .image, jobId: counterJobId)

                            let printable = raw.isEmpty ? "—" : raw
                            let assistantText = String(
                                format: "photo_enhance.result.not_url".localizable,
                                printable
                            )

                            let final = GenerationMessage(
                                id: placeholderId,
                                role: .assistant,
                                createdAt: placeholderMsg.createdAt,
                                payload: .text(assistantText)
                            )

                            self.replaceGenerationMessageInUI(updated: final)
                            try? await self.historyRepo.replaceMessage(
                                threadId: self.threadId,
                                messageId: placeholderId,
                                newMessage: final
                            )
                            self.persistAssistantTextToHistory(assistantText)
                        }
                    }

                    if dto.status == .failed, let e = dto.error, !e.isEmpty {
                        let warnText = String(
                            format: "photo_enhance.backend_failed".localizable,
                            e
                        )
                        let warn = GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .error(warnText)
                        )
                        self.messages.append(warn)
                        self.persistAssistantErrorToHistory(warnText)
                    }

                    await self.pendingService.remove(generationId: started.id)
                    self.activeGenerationId = nil

                    await self.updateCenter.post(threadId: self.threadId)

                    #if DEBUG
                    print("🧾 PENDING -1 (finished) generationId=\(started.id)")
                    #endif

                } catch is CancellationError {
                    self.lastResultURL = nil
                    self.errorText = nil

                    if let gid = self.activeGenerationId {
                        await self.pendingService.remove(generationId: gid)
                        self.activeGenerationId = nil
                    }

                    let cancelled = GenerationMessage(
                        id: placeholderId,
                        role: .system,
                        createdAt: Date(),
                        payload: .system("photo_enhance.cancelled".localizable)
                    )
                    self.replaceGenerationMessageInUI(updated: cancelled)
                    try? await self.historyRepo.replaceMessage(
                        threadId: self.threadId,
                        messageId: placeholderId,
                        newMessage: cancelled
                    )

                } catch {
                    self.lastResultURL = nil
                    let pretty = String(describing: error)
                    self.errorText = pretty

                    if let gid = self.activeGenerationId {
                        await self.pendingService.remove(generationId: gid)
                        self.activeGenerationId = nil
                    }

                    let assistantText = String(
                        format: "photo_enhance.error.generic".localizable,
                        pretty
                    )

                    let failed = GenerationMessage(
                        id: placeholderId,
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                    self.replaceGenerationMessageInUI(updated: failed)
                    try? await self.historyRepo.replaceMessage(
                        threadId: self.threadId,
                        messageId: placeholderId,
                        newMessage: failed
                    )

                    self.persistAssistantErrorToHistory(assistantText)
                }
            }
        }
    }

    // MARK: - Input attachment helpers (local)

    private static func prepareImageForAttachment(originalData: Data) -> (data: Data, mimeType: String, filename: String) {
        if let image = UIImage(data: originalData),
           let jpeg = image.jpegData(compressionQuality: 0.92),
           !jpeg.isEmpty {
            return (jpeg, "image/jpeg", "input.jpg")
        }

        let detected = detectMimeAndFilename(from: originalData)
        return (originalData, detected.mimeType, detected.filename)
    }

    private static func detectMimeAndFilename(from data: Data) -> (mimeType: String, filename: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "input.png") }

        let jpg: [UInt8] = [0xFF, 0xD8, 0xFF]
        if data.starts(with: jpg) { return ("image/jpeg", "input.jpg") }

        return ("application/octet-stream", "input.bin")
    }

    private func saveInputAttachmentToTemp(data: Data, filenameHint: String) -> URL? {
        let clean = filenameHint.trimmingCharacters(in: .whitespacesAndNewlines)
        let ext = URL(fileURLWithPath: clean).pathExtension.isEmpty
        ? "jpg"
        : URL(fileURLWithPath: clean).pathExtension

        let dir = FileManager.default.temporaryDirectory
            .appendingPathComponent("gen_input_attachments", isDirectory: true)

        do {
            try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        } catch {
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEnhance): create dir failed error=\(error)")
            #endif
            return nil
        }

        let fileURL = dir
            .appendingPathComponent("input_\(UUID().uuidString)")
            .appendingPathExtension(ext)

        do {
            try data.write(to: fileURL, options: [.atomic])
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEnhance): saved \(fileURL.lastPathComponent) bytes=\(data.count)")
            #endif
            return fileURL
        } catch {
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEnhance): save failed error=\(error)")
            #endif
            return nil
        }
    }
}

import Foundation

extension PhotoEnhanceChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open PhotoEnhance): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open PhotoEnhance): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)

                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEnhance): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEnhance): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open PhotoEnhance): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

import Foundation

extension PhotoEnhanceChatViewModel {

    func loadHistoryIfExists() {
        let repo = historyRepo
        let id = threadId

        historyTask?.cancel()
        historyTask = Task { [weak self] in
            guard let self else { return }

            do {
                let stored = try await repo.loadMessages(threadId: id)
                guard stored.isEmpty == false else { return }

                self.messages = stored.sorted(by: { $0.createdAt < $1.createdAt })

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: loaded \(self.messages.count) messages for thread id=\(id)")
                #endif

                self.markNotificationsAsReadForThisThreadIfNeeded()

                self.startEagerRecoveryIfNeeded()

            } catch is CancellationError {
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: load history failed: \(error)")
                #endif
            }
        }
    }
}

import Foundation

extension PhotoEnhanceChatViewModel {

    // MARK: - NEW: persist whole message (with attachments)

    func persistUserMessageToHistory(_ message: GenerationMessage, previewSeed: String) {
        let repo = historyRepo
        let id = threadId

        let preview = Self.makePreviewText(previewSeed)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                try await repo.appendMessage(threadId: id, message: message)
                try await self.touchThread(preview: preview, at: message.createdAt)

                #if DEBUG
                print("🧾 GEN_HISTORY(PhotoEnhance): saved user message (attachments=\(message.attachments.count)) threadId=\(id) msgId=\(message.id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY(PhotoEnhance): persist user message failed: \(error)")
                #endif
            }
        }
    }

    // MARK: - Existing persists (unchanged)

    func persistUserActionToHistory(_ text: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let clean = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !clean.isEmpty else { return }

        let preview = Self.makePreviewText(clean)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .user,
                    createdAt: now,
                    payload: .text(clean)
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: saved user action + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: persist user action failed: \(error)")
                #endif
            }
        }
    }

    func persistAssistantTextToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .text(text), previewSeed: text)
    }

    func persistAssistantErrorToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .error(text), previewSeed: text)
    }

    func persistAssistantPayloadToHistory(payload: GenerationPayload, previewSeed: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let preview = Self.makePreviewText(previewSeed)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .assistant,
                    createdAt: now,
                    payload: payload
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: saved assistant message + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: persist assistant message failed: \(error)")
                #endif
            }
        }
    }

    static func makeUserActionText(jobId: String?) -> String {
        if let jobId, !jobId.isEmpty {
            return "🖼️ Upscale requested (jobId=\(jobId))"
        } else {
            return "Отправлен запрос на улучшение фото."
        }
    }

    static func makePreviewText(_ text: String, limit: Int = 80) -> String {
        let clean = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard clean.count > limit else { return clean }
        let idx = clean.index(clean.startIndex, offsetBy: limit)
        return String(clean[..<idx]) + "…"
    }

import Foundation

extension PhotoEnhanceChatViewModel {

    // MARK: - Placeholder → download local → replace

    func appendImagePlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let messageId = UUID()
        activeGenerationMessageId = messageId

        let placeholder = GenerationMessage(
            id: messageId,
            role: .assistant,
            createdAt: now,
            payload: .image(remoteURL: remoteURL, localURL: nil)
        )

        do {
            await ensureThreadRegisteredAsync()

            // 1) add placeholder to history + touch thread
            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(
                preview: "photo_enhance_vm.history.preview.placeholder".localizable,
                at: now
            )

            // 2) show in UI
            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): appended placeholder msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            // 3) download/cache local
            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "image" // технический параметр кэша, НЕ UI-строка
            )
            try Task.checkCancellation()

            // 4) ignore if another generation became active
            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(PhotoEnhance): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            // 5) replace placeholder with local
            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)
            replaceGenerationMessageInUI(updated: updated)

            try await touchThread(
                preview: "photo_enhance_vm.history.preview.success".localizable,
                at: Date()
            )

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): replaced with local msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("common.cancelled".localizable)
            )
            replaceGenerationMessageInUI(updated: cancelled)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): cancelled msgId=\(messageId)")
            #endif

        } catch {
            let pretty = String(describing: error)

            let failed = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .error(
                    String(
                        format: "photo_enhance_vm.media.download_failed_format".localizable,
                        remoteURL.absoluteString,
                        pretty
                    )
                )
            )
            replaceGenerationMessageInUI(updated: failed)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(PhotoEnhance): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

import Foundation

extension PhotoEnhanceChatViewModel {


    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}
