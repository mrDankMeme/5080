import Foundation

@MainActor
final class PhotoEditChatViewModel: ObservableObject {

    // MARK: - Thread

    let threadId: UUID
    let kind: GenerationKind = .photoEdit
    let shouldLoadHistory: Bool

    // MARK: - UI

    @Published var messages: [GenerationMessage] = [
        GenerationMessage(
            role: .system,
            createdAt: Date(),
            payload: .system("Выбери фото, введи промпт и нажми отправить.")
        )
    ]

    @Published var promptText: String = ""

    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    // MARK: - Dependencies

    let useCase: PromptEditPhotoUseCase
    let historyRepo: GenerationHistoryRepository
    let mediaCache: MediaCacheService

    /// ✅ NEW: persistent store for user input attachments
    let inputAttachmentStore: GenerationInputAttachmentStore

    var historyRepoForViewer: GenerationHistoryRepository { historyRepo }
    var mediaCacheForViewer: MediaCacheService { mediaCache }

    func applyMessageUpdateFromViewer(_ updated: GenerationMessage) {
        replaceGenerationMessageInUI(updated: updated)
    }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    var activeGenerationMessageId: UUID?

    // MARK: - Init

    init(
        threadId: UUID? = nil,
        useCase: PromptEditPhotoUseCase,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService,
        inputAttachmentStore: GenerationInputAttachmentStore = .init()
    ) {
        self.threadId = threadId ?? UUID()
        self.shouldLoadHistory = (threadId != nil)
        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache
        self.inputAttachmentStore = inputAttachmentStore

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
        }
    }

    deinit {
        task?.cancel()
        historyTask?.cancel()
        eagerRecoveryTask?.cancel()
    }

    // MARK: - Public

    func showUserError(_ text: String) {
        errorText = text
    }

    func cancel() {
        task?.cancel()
        task = nil
        isLoading = false

        activeGenerationMessageId = nil
    }
}

import Foundation
import UIKit

extension PhotoEditChatViewModel {

    func onApplyTap(photoData: Data) {
        errorText = nil
        isLoading = true

        let prompt = promptText.trimmingCharacters(in: .whitespacesAndNewlines)
        let userText = "\(prompt)."

        // ✅ 1) ONE user-message: text + image attachment (как в других чатах)
        let now = Date()

        // ✅ Step 1 FIX: сохраняем input через Application Support (GenerationInputAttachmentStore)
        // вместо temporaryDirectory
        let attachment: GenerationAttachment? = {
            let prepared = Self.prepareImageForAttachment(originalData: photoData)

            do {
                let att = try inputAttachmentStore.saveImage(
                    threadId: threadId,
                    data: prepared.data
                )

                #if DEBUG
                // ⚠️ Не трогаем lastPathComponent у previewLocalURL напрямую,
                // потому что у тебя может быть URL?? (double optional) -> ошибка компиляции.
                let previewName: String = {
                    // если previewLocalURL == URL?
                    if let p = att.previewLocalURL as? URL {
                        return p.lastPathComponent
                    }
                    // если previewLocalURL == URL?? или другая optional-обёртка — просто печатаем как строку
                    return String(describing: att.previewLocalURL)
                }()

                print("🧾 INPUT_ATTACH(PhotoEdit): saved persistent full=\(att.localURL?.lastPathComponent) preview=\(previewName)")
                #endif

                return att
            } catch {
                #if DEBUG
                print("🧾 INPUT_ATTACH(PhotoEdit): save persistent failed error=\(error)")
                #endif
                return nil
            }
        }()

        let userMessage = GenerationMessage(
            role: .user,
            createdAt: now,
            payload: .text(userText),
            attachments: attachment.map { [$0] } ?? []
        )

        messages.append(userMessage)
        persistUserMessageToHistory(userMessage, previewSeed: prompt.isEmpty ? userText : prompt)

        // 2) ✅ assistant placeholder СРАЗУ (как было)
        let placeholderId = UUID()
        activeGenerationMessageId = placeholderId

        let placeholderRemote = URL(string: "https://example.com/image.jpg")! // маркер плейсхолдера
        let placeholderMsg = GenerationMessage(
            id: placeholderId,
            role: .assistant,
            createdAt: Date(),
            payload: .image(remoteURL: placeholderRemote, localURL: nil)
        )

        replaceGenerationMessageInUI(updated: placeholderMsg)
        persistAssistantMediaPlaceholderToHistory(message: placeholderMsg, preview: "🖼️ Editing")

        // 3) run pipeline (как было)
        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let p = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
                guard !p.isEmpty else { throw GenerationPollingError.invalidState }

                let (mime, filename) = Self.detectMimeAndFilename(from: photoData)
                let binary = PhotoBinary(data: photoData, filename: filename, mimeType: mime)

                let request = PhotoPromptEditRequest(
                    prompt: p,
                    images: [binary],
                    aspectRatio: nil
                )

                let result = try await useCase.execute(request)

                guard self.activeGenerationMessageId == placeholderId else { return }

                let raw = result.resultURL.trimmingCharacters(in: .whitespacesAndNewlines)
                if let url = URL(string: raw), url.scheme != nil {
                    self.lastResultURL = url

                    // ✅ обновляем тот же message: реальный remote, local=nil
                    await self.updateImagePlaceholderRemote(
                        messageId: placeholderId,
                        remoteURL: url
                    )

                    // ✅ качаем в кэш и финализируем local в тот же messageId
                    await self.downloadAndFinalizeImage(
                        messageId: placeholderId,
                        remoteURL: url
                    )

                } else {
                    self.lastResultURL = nil

                    let printable = raw.isEmpty ? "—" : raw
                    let assistantText = "✅ Done, но result не похож на URL:\n\(printable)"

                    let final = GenerationMessage(
                        id: placeholderId,
                        role: .assistant,
                        createdAt: placeholderMsg.createdAt,
                        payload: .text(assistantText)
                    )

                    self.replaceGenerationMessageInUI(updated: final)
                    try? await self.historyRepo.replaceMessage(threadId: self.threadId, messageId: placeholderId, newMessage: final)
                    self.persistAssistantTextToHistory(assistantText)
                }

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil

                let cancelled = GenerationMessage(
                    id: placeholderId,
                    role: .system,
                    createdAt: Date(),
                    payload: .system("⛔️ Cancelled")
                )
                self.replaceGenerationMessageInUI(updated: cancelled)
                try? await self.historyRepo.replaceMessage(threadId: self.threadId, messageId: placeholderId, newMessage: cancelled)

            } catch {
                self.lastResultURL = nil
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                let failed = GenerationMessage(
                    id: placeholderId,
                    role: .assistant,
                    createdAt: Date(),
                    payload: .error(assistantText)
                )
                self.replaceGenerationMessageInUI(updated: failed)
                try? await self.historyRepo.replaceMessage(threadId: self.threadId, messageId: placeholderId, newMessage: failed)
                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }

    // MARK: - Input attachment helpers (local)

    private static func prepareImageForAttachment(originalData: Data) -> (data: Data, mimeType: String, filename: String) {
        if let image = UIImage(data: originalData),
           let jpeg = image.jpegData(compressionQuality: 0.92),
           !jpeg.isEmpty {
            return (jpeg, "image/jpeg", "input.jpg")
        }

        let detected = detectMimeAndFilename(from: originalData)
        return (originalData, detected.mime, detected.filename)
    }

    /// Сохраняем input-картинку локально, чтобы отрисовать превью внутри user-bubble.
    /// (оставляем как было — просто больше НЕ используем в Step 1)
    private func saveInputAttachmentToTemp(data: Data, filenameHint: String) -> URL? {
        let clean = filenameHint.trimmingCharacters(in: .whitespacesAndNewlines)
        let ext = URL(fileURLWithPath: clean).pathExtension.isEmpty
        ? "jpg"
        : URL(fileURLWithPath: clean).pathExtension

        let dir = FileManager.default.temporaryDirectory
            .appendingPathComponent("gen_input_attachments", isDirectory: true)

        do {
            try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        } catch {
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEdit): create dir failed error=\(error)")
            #endif
            return nil
        }

        let fileURL = dir
            .appendingPathComponent("input_\(UUID().uuidString)")
            .appendingPathExtension(ext)

        do {
            try data.write(to: fileURL, options: [.atomic])
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEdit): saved \(fileURL.lastPathComponent) bytes=\(data.count)")
            #endif
            return fileURL
        } catch {
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEdit): save failed error=\(error)")
            #endif
            return nil
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // MARK: - Binary helpers

    static func detectMimeAndFilename(from data: Data) -> (mime: String, filename: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "photo.png") }

        let jpeg: [UInt8] = [0xFF, 0xD8]
        if data.starts(with: jpeg) { return ("image/jpeg", "photo.jpg") }

        return ("application/octet-stream", "photo.bin")
    }
}

import Foundation

extension PhotoEditChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open PhotoEdit): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open PhotoEdit): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEdit): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEdit): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open PhotoEdit): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

extension PhotoEditChatViewModel {

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: loaded \(self.messages.count) messages for thread id=\(id)")
                #endif

                self.startEagerRecoveryIfNeeded()

            } catch is CancellationError {
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: load history failed: \(error)")
                #endif
            }
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {


    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    func replaceGenerationMessageInUI(updated: GenerationMessage) {
        if let idx = messages.firstIndex(where: { $0.id == updated.id }) {
            messages[idx] = updated
        } else {
            messages.append(updated)
        }
    }

    func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }
}
