import SwiftUI
import PhotosUI

struct PhotoEditChatView: View {

    @EnvironmentObject private var purchaseManager: PurchaseManager
    @StateObject private var vm: PhotoEditChatViewModel

    @State private var pickedItem: PhotosPickerItem?
    @State private var pickedPreview: UIImage?
    @State private var isPickerPresented: Bool = false

    // MARK: - RateUs -> Support Email (present AFTER RateUs dismiss)

    @State private var isShowingRateUs: Bool = false
    @State private var shouldOpenRateUsEmailAfterDismiss: Bool = false
    @State private var isRateUsSupportMailPresented: Bool = false
    @State private var rateUsSupportMailPayload: MailComposerPayload? = nil

    init(vm: PhotoEditChatViewModel) {
        _vm = StateObject(wrappedValue: vm)
    }

    var body: some View {
        GenerationChatContainerView(
            title: "photo_edit.title".localizable,
            modeTitle: "photo_edit.mode_title".localizable,
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText,
            threadId: vm.threadId,
            historyRepo: vm.historyRepoForViewer,
            mediaCache: vm.mediaCacheForViewer,
            onMessageUpdated: { updated in
                vm.applyMessageUpdateFromViewer(updated)
            }
        ) {
            GenerationChatComposerView(
                showsPlus: true,
                isLoading: vm.isLoading,
                text: $vm.promptText,
                attachmentImage: pickedPreview,
                onPickPhotoTap: {
                    guard !vm.isLoading else { return }
                    isPickerPresented = true
                },
                onMicTap: { },
                onSendTap: { onApplyPressed() }
            )
            .photosPicker(
                isPresented: $isPickerPresented,
                selection: $pickedItem,
                matching: .images
            )
            .onChange(of: pickedItem) { _ in
                Task { @MainActor in
                    await loadPreviewIfNeeded()
                }
            }
        }
        .onChange(of: vm.shouldShowPaywall) { should in
            guard should else { return }

            purchaseManager.isShowedPaywall = true
            vm.consumePaywallSignal()

            #if DEBUG
            print("💎 GEN(PhotoEdit): show paywall from gate (kind=\(purchaseManager.currentSubKind))")
            #endif
        }
        .onChange(of: vm.shouldShowRateUs) { should in
            guard should else { return }
            hideKeyboard()
            // ВАЖНО: consume + показываем RateUs локальным стейтом (как и в остальных экранах)
            RateAskScheduler.shared.markAskShown()
            vm.consumeRateUsSignal()
            isShowingRateUs = true

            #if DEBUG
            print("⭐️ RateUs: presenting from PhotoEdit success")
            #endif
        }
        .fullScreenCover(isPresented: $isShowingRateUs) {
            RateUsView { action in
                hideKeyboard()

                switch action {
                case .liked:
                    RateAskScheduler.shared.markRated()
                    isShowingRateUs = false

                case .laterOpenEmail:
                    shouldOpenRateUsEmailAfterDismiss = true
                    isShowingRateUs = false

                case .laterJustClose:
                    isShowingRateUs = false
                }
            }
            .environment(\.dynamicTypeSize, .medium)
            .ignoresSafeArea()
        }
        .onChange(of: isShowingRateUs) { _, newValue in
            guard newValue == false, shouldOpenRateUsEmailAfterDismiss else { return }
            shouldOpenRateUsEmailAfterDismiss = false
            prepareAndPresentRateUsEmail()
        }
        .sheet(isPresented: $isRateUsSupportMailPresented) {
            if let payload = rateUsSupportMailPayload {
                MailComposerView(payload: payload)
            } else {
                VStack(spacing: 12) {
                    Text("rateus.support.mail_prepare_failed".localizable)
                    Button("common.ok".localizable) {
                        isRateUsSupportMailPresented = false
                    }
                }
                .padding()
            }
        }
        .alert(item: $vm.limitAlert) { item in
            Alert(
                title: Text(item.title),
                message: Text(item.message),
                dismissButton: .default(Text(item.buttonTitle))
            )
        }
    }

    // MARK: - Apply

    private func onApplyPressed() {
        guard let pickedItem else {
            vm.showUserError("photo_edit.error.pick_photo".localizable)
            return
        }

        let clean = vm.promptText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !clean.isEmpty else {
            vm.showUserError("photo_edit.error.enter_instruction".localizable)
            return
        }

        Task { @MainActor in
            do {
                guard let data = try await pickedItem.loadTransferable(type: Data.self) else {
                    vm.showUserError("photo_edit.error.read_photo_nil".localizable)
                    return
                }

                vm.onApplyTap(photoData: data)

                vm.promptText = ""
                self.pickedItem = nil
                self.pickedPreview = nil

            } catch {
                vm.showUserError(
                    String(
                        format: "photo_edit.error.read_photo".localizable,
                        String(describing: error)
                    )
                )
            }
        }
    }

    // MARK: - RateUs Email

    private func prepareAndPresentRateUsEmail() {
        let diagnostics = AppDiagnostics.makeSupportDiagnosticsText()
        let body = String(
            format: "rateus.support.email_body_template".localizable,
            diagnostics
        )

        rateUsSupportMailPayload = MailComposerPayload(
            to: "dete8489@gmail.com",
            subject: "rateus.support.email_subject".localizable,
            body: body,
            isHTML: false,
            fallbackMailToURL: MailComposerPayload.makeMailToURL(
                to: "dete8489@gmail.com",
                subject: "rateus.support.email_subject".localizable,
                body: body
            )
        )

        isRateUsSupportMailPresented = true
    }

    // MARK: - Preview

    @MainActor
    private func loadPreviewIfNeeded() async {
        guard let pickedItem else {
            pickedPreview = nil
            return
        }

        do {
            if let data = try await pickedItem.loadTransferable(type: Data.self),
               let ui = UIImage(data: data) {
                pickedPreview = ui
            } else {
                pickedPreview = nil
            }
        } catch {
            pickedPreview = nil
        }
    }
}

import Foundation

@MainActor
final class PhotoEditChatViewModel: ObservableObject {

    // MARK: - Thread

    let threadId: UUID
    let kind: GenerationKind = .photoEdit
    let shouldLoadHistory: Bool

    // MARK: - UI

    @Published var messages: [GenerationMessage] = [
        GenerationMessage(
            role: .system,
            createdAt: Date(),
            payload: .system("photo_edit_vm.system.hint_pick_photo_prompt_send".localizable)
        )
    ]

    @Published var promptText: String = ""

    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    @Published var shouldShowPaywall: Bool = false
    @Published var limitAlert: LimitAlertItem? = nil

    @Published var shouldShowRateUs: Bool = false

    // MARK: - Dependencies

    let useCase: PromptEditPhotoUseCase
    let historyRepo: GenerationHistoryRepository
    let mediaCache: MediaCacheService

    let inputAttachmentStore: GenerationInputAttachmentStore
    let internalNotifications: InternalNotificationCenter
    let requestCounter: RequestCounterService

    let pendingService: PendingGenerationService
    let updateCenter: GenerationThreadUpdateCenter

    // MARK: - Viewer

    var historyRepoForViewer: GenerationHistoryRepository { historyRepo }
    var mediaCacheForViewer: MediaCacheService { mediaCache }

    func applyMessageUpdateFromViewer(_ updated: GenerationMessage) {
        replaceGenerationMessageInUI(updated: updated)
    }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    var activeGenerationMessageId: UUID?
    var activeGenerationId: UUID?

    var updatesTask: Task<Void, Never>?
    var lastReloadAt: Date = .distantPast

    private var didFireRateUsSignal: Bool = false

    // MARK: - Subscription snapshot provider (injected from factory)

    typealias SubscriptionSnapshot = (isSubscribed: Bool, kind: SubscriptionKind)

    private var subscriptionSnapshotProvider: (@Sendable () -> SubscriptionSnapshot)?

    func setSubscriptionSnapshotProvider(_ provider: @escaping @Sendable () -> SubscriptionSnapshot) {
        self.subscriptionSnapshotProvider = provider
    }

    func consumePaywallSignal() {
        shouldShowPaywall = false
    }

    func consumeRateUsSignal() {
        shouldShowRateUs = false
    }

    // MARK: - Alert item

    struct LimitAlertItem: Identifiable, Equatable {
        let id: String = UUID().uuidString
        let title: String
        let message: String
        let buttonTitle: String
    }

    // MARK: - Anti-spam (DEBUG)

    static let debugImageParallelLimit: Int = .max

    // MARK: - Init

    init(
        threadId: UUID? = nil,
        useCase: PromptEditPhotoUseCase,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService,
        inputAttachmentStore: GenerationInputAttachmentStore,
        internalNotifications: InternalNotificationCenter,
        requestCounter: RequestCounterService,
        pendingService: PendingGenerationService,
        updateCenter: GenerationThreadUpdateCenter
    ) {
        self.threadId = threadId ?? UUID()
        self.shouldLoadHistory = (threadId != nil)

        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache
        self.inputAttachmentStore = inputAttachmentStore
        self.internalNotifications = internalNotifications
        self.requestCounter = requestCounter
        self.pendingService = pendingService
        self.updateCenter = updateCenter

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
            subscribeToThreadUpdates()
        }
    }

    deinit {
        task?.cancel()
        historyTask?.cancel()
        eagerRecoveryTask?.cancel()
        updatesTask?.cancel()
    }

    // MARK: - Public

    func showUserError(_ text: String) {
        errorText = text
    }

    func cancel() {
        task?.cancel()
        task = nil

        historyTask?.cancel()
        historyTask = nil

        eagerRecoveryTask?.cancel()
        eagerRecoveryTask = nil

        if let gid = activeGenerationId {
            Task { [pendingService] in
                await pendingService.remove(generationId: gid)
            }
        }

        isLoading = false
        activeGenerationMessageId = nil
        activeGenerationId = nil
    }

    // MARK: - Gate (shared helper)

    func evaluateGate(snapshot: RequestCounterSnapshot) -> GenerationGateDecision {
        let sub = subscriptionSnapshotProvider?() ?? (false, .unknown)
        return GenerationLimitGate.evaluate(
            isSubscribed: sub.isSubscribed,
            kind: sub.kind,
            snapshot: snapshot
        )
    }

    // MARK: - RateUs helper (call when image is реально готова)

    func tryTriggerRateUsAfterSuccessfulResult() {
        guard didFireRateUsSignal == false else { return }
        guard RateAskScheduler.shared.shouldShowRateAsk() else { return }

        didFireRateUsSignal = true
        shouldShowRateUs = true
    }

    // MARK: - Updates subscription

    func subscribeToThreadUpdates() {
        updatesTask?.cancel()

        updatesTask = Task { [weak self] in
            guard let self else { return }

            let stream = await self.updateCenter.stream()

            for await updatedThreadId in stream {
                if Task.isCancelled { break }
                guard updatedThreadId == self.threadId else { continue }

                let now = Date()
                if now.timeIntervalSince(self.lastReloadAt) < 0.35 { continue }
                self.lastReloadAt = now

                #if DEBUG
                print("🧾 GEN(PhotoEdit): updateCenter -> reload threadId=\(self.threadId)")
                #endif

                await MainActor.run {
                    self.loadHistoryIfExists()
                    self.startEagerRecoveryIfNeeded()
                }
            }
        }
    }

    // MARK: - Notifications

    func markNotificationsAsReadForThisThreadIfNeeded() {
        guard shouldLoadHistory else { return }

        Task {
            let list = await internalNotifications.list()

            let targets = list.filter { n in
                n.isRead == false
                && n.payload.threadId == self.threadId
                && n.payload.generationKind == self.kind
            }

            guard targets.isEmpty == false else { return }

            for n in targets {
                await internalNotifications.markRead(id: n.id)
            }

            #if DEBUG
            print("🔔 GEN(PhotoEdit): marked read \(targets.count) notifications for threadId=\(self.threadId)")
            #endif
        }
    }

    // MARK: - Debug limit helper

    static func validateParallelLimit(snapshot: RequestCounterSnapshot) -> Bool {
        snapshot.image < debugImageParallelLimit
    }
}

import Foundation
import UIKit

extension PhotoEditChatViewModel {

    // MARK: - Actions

    func onApplyTap(photoData: Data) {

        let promptRaw = self.promptText
        let prompt = promptRaw.trimmingCharacters(in: .whitespacesAndNewlines)

        guard prompt.isEmpty == false else {
            self.showUserError("Введите описание редактирования.")
            return
        }

        Task { @MainActor [weak self] in
            guard let self else { return }

            let snap = await self.requestCounter.snapshot()

            let gate = self.evaluateGate(snapshot: snap)
            switch gate {
            case .allow:
                break
            case .showPaywall:
                self.shouldShowPaywall = true
                return
            case .showAlert(let title, let message):
                self.limitAlert = LimitAlertItem(
                    title: title,
                    message: message,
                    buttonTitle: "OK"
                )
                return
            }


            self.errorText = nil
            self.isLoading = true

            let userText = "\(prompt)."
            let now = Date()

            let attachment: GenerationAttachment? = {
                let prepared = Self.prepareImageForAttachment(originalData: photoData)
                guard let localURL = self.saveInputAttachmentToTemp(
                    data: prepared.data,
                    filenameHint: prepared.filename
                ) else {
                    return nil
                }

                return GenerationAttachment(
                    kind: .image,
                    localURL: localURL,
                    previewLocalURL: localURL,
                    filename: prepared.filename,
                    mimeType: prepared.mimeType
                )
            }()

            let userMessage = GenerationMessage(
                role: .user,
                createdAt: now,
                payload: .text(userText),
                attachments: attachment.map { [$0] } ?? []
            )

            self.messages.append(userMessage)
            self.persistUserMessageToHistory(userMessage, previewSeed: prompt)

            let placeholderId = UUID()
            self.activeGenerationMessageId = placeholderId

            let placeholderRemote = URL(string: "https://example.com/image.jpg")!
            let placeholderMsg = GenerationMessage(
                id: placeholderId,
                role: .assistant,
                createdAt: Date(),
                payload: .image(remoteURL: placeholderRemote, localURL: nil)
            )

            self.replaceGenerationMessageInUI(updated: placeholderMsg)
            self.persistAssistantMediaPlaceholderToHistory(message: placeholderMsg, preview: "🖼️ Editing")

            self.task?.cancel()
            self.task = Task { [weak self] in
                guard let self else { return }

                defer {
                    Task { @MainActor in
                        self.isLoading = false
                    }
                }

                do {
                    let (mime, filename) = Self.detectMimeAndFilename(from: photoData)
                    let binary = PhotoBinary(data: photoData, filename: filename, mimeType: mime)

                    let request = PhotoPromptEditRequest(
                        prompt: prompt,
                        images: [binary],
                        aspectRatio: nil
                    )

                    let started = try await self.useCase.start(request)
                    try Task.checkCancellation()

                    let generationId = started.id
                    let jobId = generationId.uuidString

                    await MainActor.run {
                        self.activeGenerationId = generationId
                    }

                    let record = PendingGenerationRecord(
                        generationId: generationId,
                        counterJobId: jobId,
                        threadId: self.threadId,
                        placeholderMessageId: placeholderId,
                        generationKind: self.kind,
                        category: .image,
                        createdAt: Date()
                    )

                    await self.pendingService.upsert(record)

                    #if DEBUG
                    print("🧾 GEN(PhotoEdit): pending upsert ok generationId=\(generationId) placeholderId=\(placeholderId) threadId=\(self.threadId)")
                    #endif

                    let terminal = try await self.useCase.poll(started: started)
                    try Task.checkCancellation()

                    let stillActive = await MainActor.run { self.activeGenerationMessageId == placeholderId }
                    if stillActive == false {
                        #if DEBUG
                        print("🧾 GEN(PhotoEdit): ignored late terminal generationId=\(generationId) placeholderId=\(placeholderId)")
                        #endif
                        return
                    }

                    guard let url = GenerationResultNormalizer.extractUsableResultURL(from: terminal) else {
                        let reasonRaw = (terminal.error ?? "no_valid_result_url").trimmingCharacters(in: .whitespacesAndNewlines)
                        let reason = reasonRaw.isEmpty ? "no_valid_result_url" : reasonRaw

                        await self.requestCounter.releaseOnTerminalFail(
                            category: .image,
                            jobId: jobId,
                            reason: reason
                        )

                        let assistantText = "✅ Done, но result не похож на URL:\n\(terminal.result?.trimmingCharacters(in: .whitespacesAndNewlines) ?? "—")"

                        let final = GenerationMessage(
                            id: placeholderId,
                            role: .assistant,
                            createdAt: placeholderMsg.createdAt,
                            payload: .text(assistantText)
                        )

                        await MainActor.run {
                            self.lastResultURL = nil
                            self.replaceGenerationMessageInUI(updated: final)
                        }

                        try? await self.historyRepo.replaceMessage(
                            threadId: self.threadId,
                            messageId: placeholderId,
                            newMessage: final
                        )

                        _ = await self.internalNotifications.postGenerationFailed(
                            generationKind: .photoEdit,
                            threadId: self.threadId,
                            generationId: placeholderId,
                            title: "Ошибка генерации: \(reason)",
                            createdAt: Date()
                        )

                        await self.pendingService.remove(generationId: generationId)
                        await self.updateCenter.post(threadId: self.threadId)

                        #if DEBUG
                        print("🧾 GEN(PhotoEdit): terminal without url -> release & remove pending generationId=\(generationId) reason=\(reason)")
                        #endif
                        return
                    }

                    await self.requestCounter.commitSuccess(category: .image, jobId: jobId)

                    await MainActor.run {
                        self.lastResultURL = url
                    }

                    _ = await self.internalNotifications.postGenerationReady(
                        generationKind: .photoEdit,
                        threadId: self.threadId,
                        generationId: placeholderId,
                        title: "Редактирование фото",
                        createdAt: Date()
                    )

                    await self.updateImagePlaceholderRemote(messageId: placeholderId, remoteURL: url)
                    await self.downloadAndFinalizeImage(messageId: placeholderId, remoteURL: url)

                    await self.pendingService.remove(generationId: generationId)
                    await self.updateCenter.post(threadId: self.threadId)

                    #if DEBUG
                    print("🧾 GEN(PhotoEdit): success -> commit & remove pending generationId=\(generationId) url=\(url.absoluteString)")
                    #endif

                } catch is CancellationError {
                    await MainActor.run {
                        self.lastResultURL = nil
                        self.errorText = nil

                        let cancelled = GenerationMessage(
                            id: placeholderId,
                            role: .system,
                            createdAt: Date(),
                            payload: .system("⛔️ Cancelled")
                        )
                        self.replaceGenerationMessageInUI(updated: cancelled)
                    }

                    if let gid = await MainActor.run { self.activeGenerationId } {
                        await self.pendingService.remove(generationId: gid)

                        #if DEBUG
                        print("🧾 GEN(PhotoEdit): cancelled -> removed pending generationId=\(gid)")
                        #endif
                    }

                    Task {
                        try? await self.historyRepo.replaceMessage(
                            threadId: self.threadId,
                            messageId: placeholderId,
                            newMessage: GenerationMessage(
                                id: placeholderId,
                                role: .system,
                                createdAt: Date(),
                                payload: .system("⛔️ Cancelled")
                            )
                        )
                    }

                } catch let e as GenerationPollingError {
                    switch e {
                    case .timeout:
                        await MainActor.run {
                            self.errorText = "⏳ Обработка ещё идёт. Я продолжу в фоне и пришлю результат, как будет готово."
                        }
                        #if DEBUG
                        print("🧾 GEN(PhotoEdit): timeout -> keep pending (no release, no remove)")
                        #endif

                    case .failed(let reason):
                        let r = reason.trimmingCharacters(in: .whitespacesAndNewlines)
                        let safe = r.isEmpty ? "failed" : r

                        if let gid = await MainActor.run { self.activeGenerationId } {
                            await self.requestCounter.releaseOnTerminalFail(
                                category: .image,
                                jobId: gid.uuidString,
                                reason: safe
                            )

                            await self.pendingService.remove(generationId: gid)
                            await self.updateCenter.post(threadId: self.threadId)

                            #if DEBUG
                            print("🧾 GEN(PhotoEdit): failed -> release & remove pending generationId=\(gid) reason=\(safe)")
                            #endif
                        }

                        let assistantText = "❌ Error:\n\(safe)"
                        await MainActor.run {
                            self.lastResultURL = nil
                            self.errorText = safe

                            let failed = GenerationMessage(
                                id: placeholderId,
                                role: .assistant,
                                createdAt: Date(),
                                payload: .error(assistantText)
                            )
                            self.replaceGenerationMessageInUI(updated: failed)
                            self.persistAssistantErrorToHistory(assistantText)
                        }

                        Task {
                            try? await self.historyRepo.replaceMessage(
                                threadId: self.threadId,
                                messageId: placeholderId,
                                newMessage: GenerationMessage(
                                    id: placeholderId,
                                    role: .assistant,
                                    createdAt: Date(),
                                    payload: .error(assistantText)
                                )
                            )
                        }

                        _ = await self.internalNotifications.postGenerationFailed(
                            generationKind: .photoEdit,
                            threadId: self.threadId,
                            generationId: placeholderId,
                            title: "Ошибка генерации: \(safe)",
                            createdAt: Date()
                        )

                    case .invalidState:
                        await MainActor.run { self.errorText = "invalid_state" }
                    }

                } catch {
                    let pretty = String(describing: error)

                    if let gid = await MainActor.run { self.activeGenerationId } {
                        await self.requestCounter.releaseOnTerminalFail(
                            category: .image,
                            jobId: gid.uuidString,
                            reason: pretty
                        )
                        await self.pendingService.remove(generationId: gid)
                        await self.updateCenter.post(threadId: self.threadId)

                        #if DEBUG
                        print("🧾 GEN(PhotoEdit): unexpected error -> release & remove pending generationId=\(gid) error=\(pretty)")
                        #endif
                    }

                    await MainActor.run {
                        self.lastResultURL = nil
                        self.errorText = pretty

                        let assistantText = "❌ Error:\n\(pretty)"
                        let failed = GenerationMessage(
                            id: placeholderId,
                            role: .assistant,
                            createdAt: Date(),
                            payload: .error(assistantText)
                        )
                        self.replaceGenerationMessageInUI(updated: failed)
                        self.persistAssistantErrorToHistory(assistantText)
                    }

                    Task {
                        try? await self.historyRepo.replaceMessage(
                            threadId: self.threadId,
                            messageId: placeholderId,
                            newMessage: GenerationMessage(
                                id: placeholderId,
                                role: .assistant,
                                createdAt: Date(),
                                payload: .error("❌ Error:\n\(pretty)")
                            )
                        )
                    }
                }
            }
        }
    }

    // MARK: - Input attachment helpers (local)

    private static func prepareImageForAttachment(originalData: Data) -> (data: Data, mimeType: String, filename: String) {
        if let image = UIImage(data: originalData),
           let jpeg = image.jpegData(compressionQuality: 0.92),
           !jpeg.isEmpty {
            return (jpeg, "image/jpeg", "input.jpg")
        }

        let detected = detectMimeAndFilename(from: originalData)
        return (originalData, detected.mime, detected.filename)
    }

    private func saveInputAttachmentToTemp(data: Data, filenameHint: String) -> URL? {
        let clean = filenameHint.trimmingCharacters(in: .whitespacesAndNewlines)
        let ext = URL(fileURLWithPath: clean).pathExtension.isEmpty
        ? "jpg"
        : URL(fileURLWithPath: clean).pathExtension

        let dir = FileManager.default.temporaryDirectory
            .appendingPathComponent("gen_input_attachments", isDirectory: true)

        do {
            try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        } catch {
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEdit): create dir failed error=\(error)")
            #endif
            return nil
        }

        let fileURL = dir
            .appendingPathComponent("input_\(UUID().uuidString)")
            .appendingPathExtension(ext)

        do {
            try data.write(to: fileURL, options: [.atomic])
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEdit): saved \(fileURL.lastPathComponent) bytes=\(data.count)")
            #endif
            return fileURL
        } catch {
            #if DEBUG
            print("🧾 INPUT_ATTACH(PhotoEdit): save failed error=\(error)")
            #endif
            return nil
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {

    // MARK: - Binary helpers

    static func detectMimeAndFilename(from data: Data) -> (mime: String, filename: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "photo.png") }

        let jpeg: [UInt8] = [0xFF, 0xD8]
        if data.starts(with: jpeg) { return ("image/jpeg", "photo.jpg") }

        return ("application/octet-stream", "photo.bin")
    }
}

import Foundation

extension PhotoEditChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open PhotoEdit): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open PhotoEdit): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEdit): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open PhotoEdit): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open PhotoEdit): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

import Foundation

extension PhotoEditChatViewModel {


    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}
