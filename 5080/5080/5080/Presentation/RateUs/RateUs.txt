import Foundation

final class RateAskScheduler {

    static let shared = RateAskScheduler()

    private init() {}

    // MARK: - Storage

    private let storage = UserDefaults.standard

    private enum Keys {
        static let hasRated = "rateus.hasRated"
        static let lastAskDate = "rateus.lastAskDate"
        static let askCount = "rateus.askCount"
    }

    // MARK: - Config (можно потом вынести)

    /// Минимум дней между показами
    private let minDaysBetweenAsks: Int = 7

    /// Максимальное число показов (пока не нажмут Yes)
    private let maxAskCount: Int = 3

    // MARK: - Public API

    /// 👉 Единственный метод, который ты вызываешь
    func shouldShowRateAsk() -> Bool {
        if hasRated {
            return false
        }

        if askCount >= maxAskCount {
            return false
        }

        if let lastDate = lastAskDate {
            let days = Calendar.current.dateComponents(
                [.day],
                from: lastDate,
                to: Date()
            ).day ?? 0

            if days < minDaysBetweenAsks {
                return false
            }
        }

        return true
    }

    /// Вызываем когда показали RateUsView
    func markAskShown() {
        storage.set(Date(), forKey: Keys.lastAskDate)
        storage.set(askCount + 1, forKey: Keys.askCount)
    }

    /// Пользователь нажал YES
    func markRated() {
        storage.set(true, forKey: Keys.hasRated)
    }

    // MARK: - Private

    private var hasRated: Bool {
        storage.bool(forKey: Keys.hasRated)
    }

    private var askCount: Int {
        storage.integer(forKey: Keys.askCount)
    }

    private var lastAskDate: Date? {
        storage.object(forKey: Keys.lastAskDate) as? Date
    }
}

import SwiftUI

struct RateUsHeaderView: View {

    let title: String
    let isSmallStatusBar: Bool
    let isNotch: Bool
    let isDynamicIsland: Bool

    var body: some View {

        let baseHeight: CGFloat
        if isSmallStatusBar {
            baseHeight = 80.scale
        } else if isDynamicIsland {
            baseHeight = 100.scale
        } else if isNotch {
            baseHeight = 100.scale
        } else {
            baseHeight = 90.scale
        }

        return ZStack {
            Color.clear

            Text(title)
                .font(Tokens.Font.regular17)
                .foregroundStyle(.black)
                .padding(.bottom, 0.scale)
                .offset(y: isSmallStatusBar ? 8.scale : isNotch ? 18.scale : 23.scale)
        }
        .frame(maxWidth: .infinity)
        .frame(height: baseHeight, alignment: .bottom)
        .background(.clear)
        .overlay(
            Rectangle()
                .fill(Color.black.opacity(0.1))
                .frame(height: 0.scale),
            alignment: .bottom
        )
    }
}

import SwiftUI
import UIKit

struct RateUsHighlightedText: View {

    let text: String
    let highlights: [String]
    let baseColor: SwiftUI.Color
    let highlightColor: SwiftUI.Color

    var body: some View {
        Text(makeAttributedText())
    }

    private func makeAttributedText() -> AttributedString {
        let baseUIColor = UIColor(baseColor)
        let highlightUIColor = UIColor(highlightColor)

        let attributed = NSMutableAttributedString(
            string: text,
            attributes: [
                .foregroundColor: baseUIColor
            ]
        )

        let nsText = text as NSString

        for needle in highlights {
            let range = nsText.range(of: needle, options: [.caseInsensitive])
            if range.location != NSNotFound, range.length > 0 {
                attributed.addAttribute(.foregroundColor, value: highlightUIColor, range: range)
                break
            }
        }

        return AttributedString(attributed)
    }
}

import SwiftUI
import StoreKit
import UIKit

struct RateUsView: View {

    let imageName: String
    let onClose: (_ liked: Bool) -> Void

    @State private var didTrackAppear = false
    @Environment(\.dismiss) private var dismiss

    init(
        imageName: String = "rateus_hand",
        onClose: @escaping (_ liked: Bool) -> Void
    ) {
        self.imageName = imageName
        self.onClose = onClose
    }

    var body: some View {

        let isSmallStatusBar = DeviceLayout.type == .smallStatusBar
        let isNotch = DeviceLayout.type == .notch
        let isDynamicIsland = DeviceLayout.type == .dynamicIsland

        ZStack {
            Color.white.ignoresSafeArea()

            VStack(spacing: 0) {

                RateUsHeaderView(
                    title: "rateus.title".localizable,
                    isSmallStatusBar: isSmallStatusBar,
                    isNotch: isNotch,
                    isDynamicIsland: isDynamicIsland
                )

                Spacer(minLength: 24.scale)

                Image(imageName)
                    .resizable()
                    .scaledToFit()
                    .frame(width: 322.scale, height: 330.scale)
                    .frame(maxWidth: .infinity, maxHeight: .infinity)
                    .scaleEffect(isSmallStatusBar ? 1 : 1.0)
                    .offset(y: isSmallStatusBar ? -20.scale : (isNotch ? -20.scale : -20.scale))

                VStack(spacing: 8.scale) {

                    RateUsHighlightedText(
                        text: "rateus.question".localizable,
                        highlights: ["наше приложение?", "our app?"],
                        baseColor: Tokens.Color.textPrimary,
                        highlightColor: Tokens.Color.accent
                    )
                    .font(Tokens.Font.semibold20)
                    .multilineTextAlignment(.center)
                    .padding(.horizontal, 24.scale)

                    Text("rateus.description".localizable)
                        .font(Tokens.Font.regular16)
                        .foregroundStyle(Tokens.Color.textPrimary)
                        .kerning(-0.4)
                        .multilineTextAlignment(.center)
                        .padding(.horizontal, 0.scale)
                        .opacity(0.7)
                }
                .padding(.top, isSmallStatusBar ? -165.scale : (isNotch ? -210.scale : -210.scale))
                .padding(.horizontal, 24.scale)
                .padding(.bottom, 24.scale)

                Spacer()

                HStack(spacing: 8.scale) {

                    PrimaryButton(
                        title: "rateus.cta.now".localizable,
                        backgroundColor: Tokens.Color.accent,
                        foregroundColor: .white
                    ) {
                        requestReviewInCurrentScene()
                        Analytics.shared.track("rateus_custom_now")
                        onClose(true)
                    }
                    .buttonStyle(OpacityTapButtonStyle())

                    PrimaryButton(
                        title: "rateus.cta.later".localizable,
                        backgroundColor: Tokens.Color.accent.opacity(0.18),
                        foregroundColor: Tokens.Color.accent
                    ) {
                        Analytics.shared.track("rateus_custom_later")
                        onClose(false)
                    }
                    .buttonStyle(OpacityTapButtonStyle())
                }
                .padding(.horizontal, 16.scale)
                .padding(.bottom, isSmallStatusBar ? 16.scale : DeviceLayout.isNotch ? 40.scale : 40.scale)
            }
        }
        .accessibilityIdentifier("rateus.screen")
        .onAppear {
            guard !didTrackAppear else { return }
            didTrackAppear = true
            Analytics.shared.track("rateus_custom_shown")
        }
    }

    private func requestReviewInCurrentScene() {
        DispatchQueue.main.async {
            guard let scene = UIApplication.shared.connectedScenes
                .compactMap({ $0 as? UIWindowScene })
                .first(where: { $0.activationState == .foregroundActive }) else {
                return
            }
            SKStoreReviewController.requestReview(in: scene)
        }
    }
}
