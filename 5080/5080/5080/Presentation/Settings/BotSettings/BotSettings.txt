import Foundation

public enum BotInstructionsBuilder {

    public static func build(profile: BotProfile, maxLength: Int = 1800) -> String? {

        func clean(_ s: String) -> String {
            let trimmed = s.trimmingCharacters(in: .whitespacesAndNewlines)
            guard trimmed.isEmpty == false else { return "" }

            let parts = trimmed.split(whereSeparator: { $0.isWhitespace })
            return parts.joined(separator: " ")
        }

        let userName = clean(profile.userName)
        let botName = clean(profile.botName)
        let style = clean(profile.styleText)
        let context = clean(profile.contextText)

        let defaultBotName = clean(BotProfile().botName)
        let isCustomBotName: Bool = botName.isEmpty == false && botName != defaultBotName

        var lines: [String] = []

        if userName.isEmpty == false {
            lines.append(
                String(
                    format: "bot_instructions.address_me".localizable,
                    userName
                )
            )
        }

        if isCustomBotName {
            lines.append(
                String(
                    format: "bot_instructions.bot_name".localizable,
                    botName
                )
            )
        }

        if style.isEmpty == false {
            lines.append(
                String(
                    format: "bot_instructions.style".localizable,
                    style
                )
            )
        }

        if context.isEmpty == false {
            lines.append(
                String(
                    format: "bot_instructions.context".localizable,
                    context
                )
            )
        }

        guard lines.isEmpty == false else { return nil }

        var result = lines.joined(separator: "\n")
        result = result.trimmingCharacters(in: .whitespacesAndNewlines)
        guard result.isEmpty == false else { return nil }

        guard result.count > maxLength else { return result }

        let safeMax = max(0, maxLength)
        let endIdx = result.index(result.startIndex, offsetBy: min(result.count, safeMax))
        var cut = String(result[..<endIdx]).trimmingCharacters(in: .whitespacesAndNewlines)

        if cut.isEmpty == false && cut.count < maxLength {
            cut += "…"
        }

        return cut.isEmpty ? nil : cut
    }
}

import Foundation

public struct BotProfile: Equatable, Codable, Sendable {

    public var userName: String
    public var botName: String
    public var styleText: String
    public var contextText: String

    public init(
        userName: String = "",
        botName: String = "Sonnet",
        styleText: String = "",
        contextText: String = ""
    ) {
        self.userName = userName
        self.botName = botName
        self.styleText = styleText
        self.contextText = contextText
    }
}

import Foundation
import Combine

@MainActor
final class BotSettingsViewModel: ObservableObject {

    // MARK: - Output (bind to UI)

    @Published var userName: String = ""
    @Published var botName: String = "Sonnet"
    @Published var styleText: String = ""
    @Published var contextText: String = ""

    // MARK: - Dependencies

    private let store: BotProfileStore

    // MARK: - Private

    private var cancellables: Set<AnyCancellable> = []

    init(store: BotProfileStore) {
        self.store = store

        let profile = store.load()
        self.userName = profile.userName
        self.botName = profile.botName
        self.styleText = profile.styleText
        self.contextText = profile.contextText

        bindAutoSave()
    }

    private func bindAutoSave() {
        $userName
            .dropFirst()
            .sink { [weak self] _ in self?.persist() }
            .store(in: &cancellables)

        $botName
            .dropFirst()
            .sink { [weak self] _ in self?.persist() }
            .store(in: &cancellables)

        $styleText
            .dropFirst()
            .sink { [weak self] _ in self?.persist() }
            .store(in: &cancellables)

        $contextText
            .dropFirst()
            .sink { [weak self] _ in self?.persist() }
            .store(in: &cancellables)
    }

    private func persist() {
        let profile = BotProfile(
            userName: userName,
            botName: botName,
            styleText: styleText,
            contextText: contextText
        )
        store.save(profile)
    }
}

import Foundation

final class UserDefaultsBotProfileStore: BotProfileStore {

    private enum Keys {
        static let userName = "bot.profile.userName"
        static let botName = "bot.profile.botName"
        static let style = "bot.profile.style"
        static let context = "bot.profile.context"
    }

    private let defaults: UserDefaults

    init(defaults: UserDefaults = .standard) {
        self.defaults = defaults
    }

    func load() -> BotProfile {
        let userName = defaults.string(forKey: Keys.userName) ?? ""
        let botNameRaw = defaults.string(forKey: Keys.botName)
        let botName = (botNameRaw?.isEmpty == false) ? (botNameRaw ?? "Sonnet") : "Sonnet"

        let styleText = defaults.string(forKey: Keys.style) ?? ""
        let contextText = defaults.string(forKey: Keys.context) ?? ""

        return BotProfile(
            userName: userName,
            botName: botName,
            styleText: styleText,
            contextText: contextText
        )
    }

    func save(_ profile: BotProfile) {
        defaults.set(profile.userName, forKey: Keys.userName)
        defaults.set(profile.botName, forKey: Keys.botName)
        defaults.set(profile.styleText, forKey: Keys.style)
        defaults.set(profile.contextText, forKey: Keys.context)
    }
}
