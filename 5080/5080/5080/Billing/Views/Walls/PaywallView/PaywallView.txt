import ApphudSDK
import Foundation

extension Array where Element == ApphudProduct {

    func paywallSortedProducts() -> [ApphudProduct] {
        // Порядок: weekly сверху, yearly снизу.
        self.sorted { lhs, rhs in
            let lt = lhs.timeString.lowercased()
            let rt = rhs.timeString.lowercased()

            if lt.contains("week"), rt.contains("year") { return true }
            if lt.contains("year"), rt.contains("week") { return false }

            return lhs.price < rhs.price
        }
    }
}

struct PaywallPlanOptionsView: View {

    let products: [ApphudProduct]

    @Binding var pickedProd: ApphudProduct?

    let planPriceText: (ApphudProduct) -> String

    var body: some View {

                            product: product,
                            isPicked: pickedProd?.productId == product.productId,
                            planTitle: planTitle(product),
                            planSubtitle: planSubtitle(product),
                            planPriceText: planPriceText(product)
                        )
                    }
                    .disabled(isLoading)
                }
            }
        }
    }
}

import ApphudSDK
import Foundation

enum PaywallProductText {

    static func planTitle(for product: ApphudProduct, isEnglishUI: Bool) -> String {
        let t = product.timeString.lowercased()
        if isEnglishUI {
            if t.contains("week") { return "Weekly" }
            if t.contains("year") { return "Yearly" }
            if t.contains("month") { return "Monthly" }
            return product.timeString.capitalizingFirstLetter()
        } else {
            if t.contains("week") { return "Еженедельно" }
            if t.contains("year") { return "Ежегодно" }
            if t.contains("month") { return "Ежемесячно" }
            return product.timeString.capitalizingFirstLetter()
        }
    }

    static func planSubtitle(for product: ApphudProduct, isEnglishUI: Bool) -> String {
        let t = product.timeString.lowercased()
        if isEnglishUI {
            if t.contains("week") { return "1 week" }
            if t.contains("year") { return "12 month" }
            if t.contains("month") { return "1 month" }
            return ""
        } else {
            if t.contains("week") { return "1 неделя" }
            if t.contains("year") { return "12 месяц" }
            if t.contains("month") { return "1 месяц" }
            return ""
        }
    }

    static func planPriceText(for product: ApphudProduct, isEnglishUI: Bool) -> String {
        if product.timeString.lowercased().contains("year") {
            return isEnglishUI
            ? "\(product.localizedPriceWeek) / week"
            : "\(product.localizedPriceWeek) / неделя"
        } else {
            return product.localizedPrice
        }
    }
}

import ApphudSDK
import SwiftUI

struct PaywallView: View {

    @EnvironmentObject private var purchaseManager: PurchaseManager
    @Environment(\.dismiss) private var dismiss
    @Environment(\.openURL) private var openURL

    // MARK: - External callbacks
    private let onClose: (() -> Void)?

    // MARK: - State
    @State private var pickedProd: ApphudProduct?
    @State private var showErrorAlert: Bool = false

    // MARK: - Init
    init(onClose: (() -> Void)? = nil) {
        self.onClose = onClose
    }

    // MARK: - Derived

    private var isEnglishUI: Bool {
        let lang = Locale.current.languageCode ?? "en"
        return lang != "ru"
    }

    private var isYearlyPicked: Bool {
        guard let pickedProd else { return false }
        return pickedProd.timeString.lowercased().contains("year")
    }

    private var creationsCountText: String {
        let n = isYearlyPicked ? 50 : 5
        return isEnglishUI ? "\(n) AI media creations" : "\(n) Медиа-творений с ИИ"
    }

    private var subtitle1: String {
        isEnglishUI
        ? "Create more images, videos and effects"
        : "Создавайте больше изображений,\nвидео и эффектов"
    }

    private var subtitle2: String {
        isEnglishUI
        ? "Full customization of your AI assistant"
        : "Полная кастомизация вашего\nAI-ассистента"
    }

    private var subtitle3: String {
        isEnglishUI
        ? "Unlimited AI chat anytime"
        : "Безлимитное общение с AI в любое время"
    }

    private var titleText: String {
        isEnglishUI ? "Get full access" : "Получите полный доступ"
    }

    private var continueText: String {
        isEnglishUI ? "Continue" : "Продолжить"
    }

    private var cancelText: String {
        isEnglishUI ? "Cancel anytime" : "Отменить в любое время"
    }

    private var footerTerms: String { isEnglishUI ? "Terms" : "Условия" }
    private var footerRestore: String { isEnglishUI ? "Restore" : "Восстановить" }
    private var footerPrivacy: String { isEnglishUI ? "Privacy" : "Конфиденциальность" }

    // TODO: потом заменишь на реальные
    private var termsURL: URL? { nil }
    private var privacyURL: URL? { nil }

    private var sortedProductsForUI: [ApphudProduct] {
        purchaseManager.products.paywallSortedProducts()
    }

    // MARK: - Body

    var body: some View {
        ZStack {
            PaywallBackgroundView(isEnglishUI: isEnglishUI)

            VStack(spacing: 0) {
                PaywallTopContentView(
                    isEnglishUI: isEnglishUI,
                    creationsCountText: creationsCountText,
                    subtitle1: subtitle1,
                    subtitle2: subtitle2,
                    subtitle3: subtitle3
                )

                PaywallBottomCardView(
                    titleText: titleText,
                    cancelText: cancelText,
                    continueText: continueText,
                    footerTerms: footerTerms,
                    footerRestore: footerRestore,
                    footerPrivacy: footerPrivacy,
                    termsURL: termsURL,
                    privacyURL: privacyURL,
                    openURL: openURL,
                    products: purchaseManager.products,
                    sortedProductsForUI: sortedProductsForUI,
                    purchaseState: purchaseManager.purchaseState,
                    isReady: purchaseManager.isReady,
                    isLoading: purchaseManager.isLoading,
                    pickedProd: $pickedProd,
                    onPick: { pickedProd = $0 },
                    onRestore: { purchaseManager.restorePurchase { _ in } },
                    onContinue: { product in
                        purchaseManager.makePurchase(product: product) { success, _ in
                            if success { close() }
                        }
                    },
                    planTitle: { product in
                        PaywallProductText.planTitle(for: product, isEnglishUI: isEnglishUI)
                    },
                    planSubtitle: { product in
                        PaywallProductText.planSubtitle(for: product, isEnglishUI: isEnglishUI)
                    },
                    planPriceText: { product in
                        PaywallProductText.planPriceText(for: product, isEnglishUI: isEnglishUI)
                    }
                )
            }
            .frame(maxWidth: .infinity, maxHeight: .infinity, alignment: .top)
        }
        .ignoresSafeArea()
        .alert("Error", isPresented: $showErrorAlert) {
            Button("Ok", role: .cancel) {
                purchaseManager.failRestoreText = nil
                purchaseManager.purchaseError = nil
            }
        } message: {
            Text(purchaseManager.purchaseError ?? purchaseManager.failRestoreText ?? "")
        }
        .onAppear {
            pickDefaultProductIfNeeded()
            if purchaseManager.purchaseError != nil || purchaseManager.failRestoreText != nil {
                showErrorAlert = true
            }
        }
        .onChange(of: purchaseManager.failRestoreText != nil) { _, newValue in
            if newValue { showErrorAlert = true }
        }
        .onChange(of: purchaseManager.purchaseError != nil) { _, newValue in
            if newValue { showErrorAlert = true }
        }
        .overlay(alignment: .topTrailing) {
            PaywallCloseButton(
                isDisabled: purchaseManager.isLoading,
                isHidden: purchaseManager.isLoading,
                onTap: { close() }
            )
        }
        .overlay {
            if purchaseManager.isLoading {
                PaywallLoadingOverlayView()
            }
        }
        .animation(.easeInOut(duration: 0.2), value: purchaseManager.isLoading)
        .onChange(of: purchaseManager.isSubscribed) { _, newValue in
            guard newValue else { return }
            close()
        }
    }

    // MARK: - Helpers

    private func pickDefaultProductIfNeeded() {
        guard pickedProd == nil else { return }
        guard !purchaseManager.products.isEmpty else { return }

        if let yearly = purchaseManager.products.first(where: { $0.timeString.lowercased().contains("year") }) {
            pickedProd = yearly
        } else {
            pickedProd = purchaseManager.products.min(by: { $0.price < $1.price })
        }
    }

    private func close() {
        onClose?()
        dismiss()
    }
}
