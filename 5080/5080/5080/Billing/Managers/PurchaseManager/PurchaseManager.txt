import SwiftUI
import Combine
import ApphudSDK
import AdServices
import OSLog
import UIKit

@MainActor
final class PurchaseManager: ObservableObject {

    let logger = Logger(
        subsystem: Bundle.main.bundleIdentifier ?? "demo",
        category: "PurchaseManager"
    )

    enum PurchaseState: Equatable {
        case idle
        case loading
        case ready
        case purchasing
        case error(String)

        static func == (lhs: PurchaseState, rhs: PurchaseState) -> Bool {
            switch (lhs, rhs) {
            case (.idle, .idle), (.loading, .loading), (.ready, .ready), (.purchasing, .purchasing):
                return true
            case (.error(let lhsMsg), .error(let rhsMsg)):
                return lhsMsg == rhsMsg
            default:
                return false
            }
        }
    }

    @Published var paywall: ApphudPaywall?
    @Published var paywallTokens: ApphudPaywall?
    @Published var paywallAvatars: ApphudPaywall?

    @Published var products: [ApphudProduct] = []
    @Published var tokenProducts: [ApphudProduct] = []
    @Published var avatarsProducts: [ApphudProduct] = []

    @Published var isSubscribed: Bool = false
    @Published var isRus: Bool = false
    @Published var purchaseState: PurchaseState = .idle
    @Published var purchaseError: String? = nil
    @Published var isLoad: Bool = false
    @Published var failRestoreText: String? = nil

    @AppStorage("OnBoardEnd") var isOnboardingFinished: Bool = false
    @Published var isShowedPaywall: Bool = false

    // NOTE: в твоём текущем продукте tokens/avatars не используются,
    // но поля пока оставляем, чтобы не ломать компиляцию/ссылки.
    @Published var tokens: Int = 0
    @Published var avatars: Int = 0

    @Published var currentSubKind: SubscriptionKind = .unknown

    // MARK: - API subscription polling (нужно из polling-файла)
    var subscriptionPollingTask: Task<Void, Never>?
    var paymentScreenClosedAt: Date?

    public var userId: String {
        Apphud.userID()
    }

    var isReady: Bool {
        purchaseState == .ready && !products.isEmpty && paywall != nil
    }

    var isLoading: Bool {
        purchaseState == .loading || purchaseState == .purchasing
    }

    static let shared = PurchaseManager()

    private var didBecomeActiveObserver: NSObjectProtocol?

    private init() {
        logger.info("PurchaseManager: Initializing Apphud SDK")

        Apphud.setPaywallsCacheTimeout(3600)
        Apphud.start(apiKey: "app_hkCJUYmAnjVfueB588ny1PWTaKh5h1")

        logger.info("PurchaseManager: Apphud.start() called")

        if let storedTokens = KeychainManager.shared.loadInt(forKey: "tokens") {
            self.tokens = storedTokens
            logger.info("PurchaseManager: Loaded tokens from Keychain: \(storedTokens)")
        } else {
            logger.info("PurchaseManager: No tokens value in Keychain, starting from 0")
        }

        if let storedAvatars = KeychainManager.shared.loadInt(forKey: "avatars") {
            self.avatars = storedAvatars
            logger.info("PurchaseManager: Loaded avatars from Keychain: \(storedAvatars)")
        } else {
            logger.info("PurchaseManager: No avatars value in Keychain, starting from 0")
        }

        purchaseState = .loading

        Task { await loadPaywalls() }

        #if DEBUG
        // В DEBUG часто косячит "моментальная" синхронизация Apphud после покупки,
        // поэтому полагаемся на универсальный refresh (Apphud + StoreKit2 entitlement).
        self.isRus = false
        self.isSubscribed = false
        Task { await self.refreshSubscriptionState(reason: "init_debug") }
        #else

self.isSubscribed = Apphud.hasPremiumAccess()
Task { @MainActor in
    self.isRus = await isRussianUserFromApphud()
    await self.refreshSubscriptionFromAPI() // или твой будущий refreshSubscriptionState(...)
}


        #endif

        isShowedPaywall = !isSubscribed && isOnboardingFinished

        // Критично: после закрытия purchase sheet/возврата из фона
        // Apphud/StoreKit2 могут обновиться позже — обновляем статус без перезапуска.
        didBecomeActiveObserver = NotificationCenter.default.addObserver(
            forName: UIApplication.didBecomeActiveNotification,
            object: nil,
            queue: .main
        ) { [weak self] _ in
            guard let self else { return }
            Task { @MainActor in
                await self.refreshSubscriptionState(reason: "didBecomeActive")
            }
        }
    }

    deinit {
        if let didBecomeActiveObserver {
            NotificationCenter.default.removeObserver(didBecomeActiveObserver)
        }
    }
}

extension PurchaseManager {

    /// Единая точка синхронизации премиума.
    /// - Non-RU: Apphud.hasPremiumAccess OR StoreKit2 entitlement
    /// - RU: API subscription status OR Apphud
    @MainActor
    func refreshSubscriptionState(reason: String) async {
        logger.info("PurchaseManager: refreshSubscriptionState start (reason=\(reason)) isRus=\(self.isRus)")

        if isRus {
            // RU: уже включает Apphud + API внутри
            await refreshSubscriptionFromAPI()
            logger.info("PurchaseManager: refreshSubscriptionState done (RU) isSubscribed=\(self.isSubscribed) reason=\(reason)")
            return
        }

        // Non-RU: не доверяем только Apphud в момент покупки (особенно на симуляторе)
        let apphudHas = Apphud.hasPremiumAccess()
        let skProductId = await activeAutoRenewableEntitlementProductId()
        let skHas = (skProductId != nil)

        let newValue = apphudHas || skHas
        if newValue != isSubscribed {
            logger.info("PurchaseManager: refreshSubscriptionState Non-RU isSubscribed \(self.isSubscribed) -> \(newValue) (Apphud=\(apphudHas), SK2=\(skHas)) reason=\(reason)")
        } else {
            logger.info("PurchaseManager: refreshSubscriptionState Non-RU isSubscribed unchanged=\(newValue) (Apphud=\(apphudHas), SK2=\(skHas)) reason=\(reason)")
        }

        isSubscribed = newValue
        isShowedPaywall = !isSubscribed && isOnboardingFinished

        await updateSubscriptionKindCache(reason: "refreshSubscriptionState Non-RU reason=\(reason) apphud=\(apphudHas) sk2=\(skHas) productId=\(skProductId ?? "nil")")
    }
}

import Foundation
import StoreKit
import ApphudSDK

extension PurchaseManager {

    @MainActor
    func bootstrapNonRUSubscriptionStateOnLaunch() async {
        guard isRus == false else { return }

        let apphudHas = Apphud.hasPremiumAccess()
        if apphudHas {
            if isSubscribed == false {
                logger.info("🧾 DEBUG_BOOT: Apphud.hasPremiumAccess=true -> set isSubscribed=true")
            }
            isSubscribed = true
            isShowedPaywall = !isSubscribed && isOnboardingFinished

            await updateSubscriptionKindCache(reason: "debug_bootstrap_apphud_hasPremium")
            return
        }

        let productId = await activeAutoRenewableEntitlementProductId()
        let hasEntitlement = (productId != nil)

        if hasEntitlement {
            logger.info("🧾 DEBUG_BOOT: StoreKit2 entitlement active -> set isSubscribed=true productId=\(productId ?? "nil")")
            isSubscribed = true
        } else {
            logger.info("🧾 DEBUG_BOOT: No Apphud premium and no StoreKit2 entitlement -> keep isSubscribed=false")
            isSubscribed = false
        }

        isShowedPaywall = !isSubscribed && isOnboardingFinished
        await updateSubscriptionKindCache(reason: "debug_bootstrap_apphud_false_sk2=\(hasEntitlement) productId=\(productId ?? "nil")")
    }

    // MARK: - StoreKit2 entitlement check

    /// Возвращает productID активной auto-renewable подписки, если есть.
    /// Мы используем это не только в DEBUG boot, но и в runtime refresh.
    @MainActor
    func activeAutoRenewableEntitlementProductId() async -> String? {
        for await result in StoreKit.Transaction.currentEntitlements {
            guard case .verified(let tx) = result else { continue }
            guard tx.productType == .autoRenewable else { continue }

            if tx.revocationDate != nil { continue }
            if let exp = tx.expirationDate, exp <= Date() { continue }

            return tx.productID
        }
        return nil
    }
}

import Foundation
import ApphudSDK

extension PurchaseManager {
     func loadPaywalls() async {
        logger.info("PurchaseManager: Starting paywall fetch")
        let paywalls = await Apphud.fetchPaywallsWithFallback()
        logger.info("PurchaseManager: Fetched \(paywalls.count) paywalls")
        self.configure(with: paywalls)
    }

     func configure(with paywalls: [ApphudPaywall]) {
        logger.info("PurchaseManager: Configuring paywalls, received \(paywalls.count) paywalls")
        logger.info("PurchaseManager: Available paywall identifiers: \(paywalls.map { $0.identifier }.joined(separator: ", "))")

        if let main = paywalls.first(where: { $0.identifier == "main" }) ?? paywalls.first {
            self.paywall = main
            self.products = main.products
            logger.info("PurchaseManager: Configured main paywall '\(main.identifier)' with \(main.products.count) products")

            if main.products.isEmpty {
                logger.warning("PurchaseManager: Main paywall has no products!")
            } else {
                let productIds = main.products.map { $0.productId }.joined(separator: ", ")
                logger.info("PurchaseManager: Main paywall products: \(productIds)")
            }
        } else {
            logger.error("PurchaseManager: No main paywall found")
            self.paywall = nil
            self.products = []
        }

        if let tokens = paywalls.first(where: { $0.identifier == "tokens" }) {
            self.paywallTokens = tokens
            self.tokenProducts = tokens.products
            logger.info("PurchaseManager: Configured tokens paywall '\(tokens.identifier)' with \(tokens.products.count) products")

            if !tokens.products.isEmpty {
                let productIds = tokens.products.map { $0.productId }.joined(separator: ", ")
                logger.info("PurchaseManager: Tokens paywall products: \(productIds)")
            }
        } else {
            logger.warning("PurchaseManager: Tokens paywall not found")
            self.paywallTokens = nil
            self.tokenProducts = []
        }

        if let avatars = paywalls.first(where: { $0.identifier == "avatars" }) {
            self.paywallAvatars = avatars
            self.avatarsProducts = avatars.products
            logger.info("PurchaseManager: Configured avatars paywall '\(avatars.identifier)' with \(avatars.products.count) products")

            if !avatars.products.isEmpty {
                let productIds = avatars.products.map { $0.productId }.joined(separator: ", ")
                logger.info("PurchaseManager: Avatars paywall products: \(productIds)")
            }
        } else {
            logger.warning("PurchaseManager: Avatars paywall not found")
            self.paywallAvatars = nil
            self.avatarsProducts = []
        }

         if self.paywall != nil && !self.products.isEmpty {
             purchaseState = .ready
             purchaseError = nil
             logger.info("PurchaseManager: Configuration completed successfully, state set to ready")

             Task { @MainActor in
                 await self.updateSubscriptionKindCache(reason: "app_start_ready")
             }

         } else {
             purchaseState = .error("Subscription options not available. Please check your connection and try again.")
             purchaseError = "Subscription options not available. Please check your connection and try again."
             logger.error("PurchaseManager: Configuration failed - no main paywall or products available")
         }


        #if DEBUG
        print("[Paywalls] main=\(paywall?.identifier ?? "nil") products=\(products.count)")
        print("[Paywalls] tokens=\(paywallTokens?.identifier ?? "nil") tokenProducts=\(tokenProducts.count)")
        print("[Paywalls] avatars=\(paywallAvatars?.identifier ?? "nil") avatarsProducts=\(avatarsProducts.count)")
        #endif
    }
}

import Foundation

extension PurchaseManager {

    func startPurchasePolling(productType: ProductType, productId: String?) {
        guard subscriptionPollingTask == nil else { return }
        paymentScreenClosedAt = nil

        subscriptionPollingTask = Task { [weak self] in
            guard let self else { return }

            let pollingInterval: UInt64 = 5 * 1_000_000_000
            let extraTimeAfterClose: TimeInterval = 30
            var lastCheckedTransactionIds: Set<String> = []

            while !Task.isCancelled {
                do {
                    let transactions = try await APIPaymentManager.shared.fetchTransactions(userId: self.userId)

                    let newTransactions = transactions.filter { !lastCheckedTransactionIds.contains($0.id) }
                    lastCheckedTransactionIds = Set(transactions.map { $0.id })

                    for transaction in newTransactions {
                        await MainActor.run {
                            self.processTransaction(transaction, productType: productType, expectedProductId: productId)
                        }
                    }

                    if productType == .subscription {
                        let status = try await APIPaymentManager.shared.fetchSubscriptionStatus(userId: self.userId)
                        if status.isActive {
                            await MainActor.run {
                                self.isSubscribed = true
                            }

                            await MainActor.run {
                                Task { @MainActor in
                                    await self.updateSubscriptionKindCache(reason: "polling_subscription_active")
                                }
                            }

                            await MainActor.run {
                                self.subscriptionPollingTask?.cancel()
                                self.subscriptionPollingTask = nil
                            }
                            break
                        }
                    }
                } catch {
                }

                if let closedAt = paymentScreenClosedAt {
                    let elapsed = Date().timeIntervalSince(closedAt)
                    if elapsed > extraTimeAfterClose {
                        await MainActor.run {
                            self.subscriptionPollingTask?.cancel()
                            self.subscriptionPollingTask = nil
                        }
                        break
                    }
                }

                try? await Task.sleep(nanoseconds: pollingInterval)
            }
        }
    }

    fileprivate func processTransaction(_ transaction: Transaction, productType: ProductType, expectedProductId: String?) {
        if let expectedId = expectedProductId, transaction.productId != expectedId {
            return
        }

        switch productType {
        case .subscription:
            if transaction.isSubscription {
                self.isSubscribed = true
                logger.info("PurchaseManager: Subscription activated from transaction \(transaction.id)")

                Task { @MainActor in
                    await self.updateSubscriptionKindCache(reason: "polling_new_subscription_tx id=\(transaction.id) productId=\(transaction.productId)")
                }
            }

        case .tokens:
            let tokensToAdd = extractTokensCount(from: transaction.productName)
            if tokensToAdd > 0 {
                self.tokens += tokensToAdd
                logger.info("PurchaseManager: Added \(tokensToAdd) tokens from transaction \(transaction.id). Total tokens: \(self.tokens)")

                Task.detached { [tokens = self.tokens] in
                    _ = await KeychainManager.shared.save(tokens, forKey: "tokens")
                }
            }

        case .avatars:
            let avatarsToAdd = extractAvatarsCount(from: transaction.productName)
            if avatarsToAdd > 0 {
                self.avatars += avatarsToAdd
                logger.info("PurchaseManager: Added \(avatarsToAdd) avatars from transaction \(transaction.id). Total avatars: \(self.avatars)")

                Task.detached { [avatars = self.avatars] in
                    _ = await KeychainManager.shared.save(avatars, forKey: "avatars")
                }
            }
        }
    }

    fileprivate func extractAvatarsCount(from productName: String) -> Int {
        let lowercased = productName.lowercased()
        let regex = try? NSRegularExpression(pattern: "([0-9]+)\\s*avatar", options: [.caseInsensitive])
        let range = NSRange(location: 0, length: lowercased.utf16.count)
        if let match = regex?.firstMatch(in: lowercased, options: [], range: range),
           let matchRange = Range(match.range(at: 1), in: lowercased),
           let count = Int(String(lowercased[matchRange])) {
            return count
        }
        return 0
    }

    func notifyPaymentScreenClosed() {
        if paymentScreenClosedAt == nil {
            paymentScreenClosedAt = Date()
        }
    }

    func startSubscriptionPolling() {
        startPurchasePolling(productType: .subscription, productId: nil)
    }
}

import Foundation
import ApphudSDK

extension PurchaseManager {

    func makePurchase(product: ApphudProduct, completion: @escaping (Bool, String?) -> Void) {
        guard purchaseState != .purchasing else {
            logger.warning("PurchaseManager: Purchase already in progress, ignoring duplicate request")
            completion(false, "Purchase already in progress")
            return
        }

        // Сейчас tokens/avatars не используются, но оставляем проверку на "из loaded paywalls".
        let allProducts = products + tokenProducts + avatarsProducts
        guard !allProducts.isEmpty else {
            logger.error("PurchaseManager: Cannot purchase - no products loaded from any paywall")
            let errorMsg = "Products are not loaded. Please try again."
            purchaseError = errorMsg
            completion(false, errorMsg)
            return
        }

        guard allProducts.contains(where: { $0.productId == product.productId }) else {
            logger.error("PurchaseManager: Product \(product.productId) not found in available products")
            let errorMsg = "Selected product is not available"
            purchaseError = errorMsg
            completion(false, errorMsg)
            return
        }

        logger.info("PurchaseManager: Starting purchase for product \(product.productId)")
        purchaseState = .purchasing
        purchaseError = nil

        Task { @MainActor in
            let result = await Apphud.fallbackPurchase(product: product)

            if result {
                logger.info("PurchaseManager: Purchase successful for product \(product.productId)")

                // КЛЮЧЕВОЕ ИЗМЕНЕНИЕ:
                // Не ждём Apphud.hasPremiumAccess() (на симуляторе может быть false ещё какое-то время).
                // Если покупка успешна — UX должен сразу получить премиум.
                self.isSubscribed = true
                self.isShowedPaywall = !self.isSubscribed && self.isOnboardingFinished

                self.purchaseState = .ready
                self.purchaseError = nil

                // Догоняющий refresh: Apphud + StoreKit2 entitlement (+ RU API если включен RU).
                await self.refreshSubscriptionState(reason: "purchase_success productId=\(product.productId)")

                completion(true, nil)
            } else {
                logger.error("PurchaseManager: Purchase failed for product \(product.productId)")
                let errorMsg = "Purchase was not completed. Please try again."
                self.purchaseState = .ready
                self.purchaseError = errorMsg

                // На всякий: подтянем актуальный статус после неуспеха тоже
                await self.refreshSubscriptionState(reason: "purchase_failed productId=\(product.productId)")

                completion(false, errorMsg)
            }
        }
    }

    func extractTokensCount(from productId: String) -> Int {
        let components = productId.components(separatedBy: "_")
        if let firstComponent = components.first, let count = Int(firstComponent) {
            return count
        }

        let regex = try? NSRegularExpression(pattern: "^([0-9]+)", options: [])
        let range = NSRange(location: 0, length: productId.utf16.count)
        if let match = regex?.firstMatch(in: productId, options: [], range: range),
           let matchRange = Range(match.range, in: productId),
           let count = Int(String(productId[matchRange])) {
            return count
        }

        return 0
    }
}

import Foundation
import ApphudSDK

extension PurchaseManager {

    func restorePurchase(completion: @escaping (Bool) -> Void) {
        logger.info("PurchaseManager: Starting restore purchases")
        Apphud.restorePurchases { [weak self] subscriptions, purchases, error in
            DispatchQueue.main.async {
                guard let self = self else { return }

                if let error = error {
                    self.logger.error("PurchaseManager: Restore failed - \(error.localizedDescription)")
                    self.failRestoreText = "Restore purchases failed: \(error.localizedDescription)"
                    completion(false)
                    return
                }

                if let subscriptions = subscriptions, subscriptions.contains(where: { $0.isActive() }) {
                    self.logger.info("PurchaseManager: Restore successful - active subscription found")

                    // Сразу включаем premium (не ждём hasPremiumAccess)
                    self.isSubscribed = true
                    self.isShowedPaywall = !self.isSubscribed && self.isOnboardingFinished
                    self.failRestoreText = nil

                    Task { @MainActor in
                        await self.refreshSubscriptionState(reason: "restore_success_apphud_subscription")
                    }

                    completion(true)
                    return
                }

                if let purchases = purchases, purchases.contains(where: { $0.isActive() }) {
                    self.logger.info("PurchaseManager: Restore successful - active purchase found")

                    self.isSubscribed = true
                    self.isShowedPaywall = !self.isSubscribed && self.isOnboardingFinished
                    self.failRestoreText = nil

                    Task { @MainActor in
                        await self.refreshSubscriptionState(reason: "restore_success_apphud_nonrenewing")
                    }

                    completion(true)
                    return
                }

                self.logger.warning("PurchaseManager: Nothing to restore")
                self.failRestoreText = "Nothing to restore"

                Task { @MainActor in
                    await self.updateSubscriptionKindCache(reason: "restore_nothing_apphud")
                }

                completion(false)
            }
        }
    }

    @MainActor
    fileprivate func restoreApphudPurchase() async -> Bool {
        await withCheckedContinuation { (cont: CheckedContinuation<Bool, Never>) in
            restorePurchase { ok in
                cont.resume(returning: ok)
            }
        }
    }

    @MainActor
    func restoreAny() async -> Bool {
        let ok: Bool
        if isRus {
            ok = await restoreRussianSubscription()
        } else {
            ok = await restoreApphudPurchase()
        }

        await updateSubscriptionKindCache(reason: "restoreAny ok=\(ok)")

        // Дополнительно синхронизируем состояние после restore любого типа
        await refreshSubscriptionState(reason: "restoreAny_final ok=\(ok)")

        return ok
    }
}

import Foundation
import ApphudSDK

extension PurchaseManager {

    @MainActor
    func restoreRussianSubscription() async -> Bool {
        logger.info("PurchaseManager: Starting RU restore via API")
        failRestoreText = nil
        purchaseError = nil

        do {
            let status = try await APIPaymentManager.shared.fetchSubscriptionStatus(userId: self.userId)
            if status.isActive {
                logger.info("PurchaseManager: RU restore success - API subscription is active")
                isSubscribed = true
                isShowedPaywall = !isSubscribed && isOnboardingFinished
                failRestoreText = nil

                await updateSubscriptionKindCache(reason: "restore_success_ru_api")

                return true
            } else {
                logger.info("PurchaseManager: RU restore - API subscription is NOT active")
                isShowedPaywall = !isSubscribed && isOnboardingFinished
                failRestoreText = "Nothing to restore"

                await updateSubscriptionKindCache(reason: "restore_nothing_ru_api")

                return false
            }
        } catch {
            let msg = "Restore purchases failed: \(error.localizedDescription)"
            logger.error("PurchaseManager: RU restore failed - \(msg)")
            isShowedPaywall = !isSubscribed && isOnboardingFinished
            failRestoreText = msg

            await updateSubscriptionKindCache(reason: "restore_error_ru_api")

            return false
        }
    }

    func restoreRussianPurchase(completion: @escaping (Bool) -> Void) {
        Task { @MainActor in
            let ok = await restoreRussianSubscription()
            completion(ok)
        }
    }

     func isRussianUserFromApphud() async -> Bool {
        print("🔍 [ApphudManager] Начинаем проверку через Apphud плейсменты...")

        if let mainPlacement = await Apphud.placement("main") {
            print("🔍 [ApphudManager] Плейсмент 'main' найден")

            if let mainPaywall = mainPlacement.paywall {
                let paywallId = mainPaywall.identifier
                let isMainRussian = paywallId == "mainRus"

                print("🔍 [ApphudManager] Paywall identifier: '\(paywallId)'")
                print("🔍 [ApphudManager] isRussian: \(isMainRussian)")

                return isMainRussian
            } else {
                print("[ApphudManager] Плейсмент 'main' найден, но paywall отсутствует")
            }
        } else {
            print("[ApphudManager] Плейсмент 'main' не найден")
        }

        print("[ApphudManager] Считаем пользователя НЕ русским (плейсмент не найден)")
        return false
    }

     func refreshSubscriptionFromAPI() async {
        do {
            let status = try await APIPaymentManager.shared.fetchSubscriptionStatus(userId: self.userId)
            let apphudHasAccess = Apphud.hasPremiumAccess()
            let apiHasAccess = status.isActive
            self.isSubscribed = apphudHasAccess || apiHasAccess
            logger.info("PurchaseManager: Refreshed subscription from API. Apphud=\(apphudHasAccess), API=\(apiHasAccess)")
        } catch {
            logger.error("PurchaseManager: Failed to refresh subscription from API - \(error.localizedDescription)")
        }

        self.isShowedPaywall = !self.isSubscribed && self.isOnboardingFinished

        await updateSubscriptionKindCache(reason: "refreshSubscriptionFromAPI")
    }
}

import Foundation

extension PurchaseManager {

    @MainActor
    func currentSubscriptionKind() async -> SubscriptionKind {

        guard isSubscribed else {
            logger.info("🧾 SUB_KIND: currentSubscriptionKind -> unknown (not subscribed)")
            return .unknown
        }

        if isRus {
            let kind = await resolveRussianSubscriptionKind()
            logger.info("🧾 SUB_KIND: resolved RU kind=\(kind.debugName)")
            return kind
        } else {
            let kind = await resolveNonRussianSubscriptionKind()
            logger.info("🧾 SUB_KIND: resolved Non-RU kind=\(kind.debugName)")
            return kind
        }
    }
}

import Foundation

extension PurchaseManager {

    @MainActor
    func updateSubscriptionKindCache(reason: String) async {

        guard isSubscribed else {
            if currentSubKind != .unknown {
                logger.info("🧾 SUB_KIND: cache reset to unknown (reason=\(reason))")
            } else {
                logger.info("🧾 SUB_KIND: keep unknown (not subscribed) (reason=\(reason))")
            }
            currentSubKind = .unknown
            return
        }

        let resolved = await currentSubscriptionKind()

        if resolved != currentSubKind {
            logger.info("🧾 SUB_KIND: cache updated \(self.currentSubKind.debugName) -> \(resolved.debugName) (reason=\(reason)) source=\(self.isRus ? "RU_API" : "NON_RU_SK2")")
            currentSubKind = resolved
        } else {
            logger.info("🧾 SUB_KIND: cache unchanged \(resolved.debugName) (reason=\(reason)) source=\(self.isRus ? "RU_API" : "NON_RU_SK2")")
        }
    }
}

import Foundation
import StoreKit
import ApphudSDK

extension PurchaseManager {

    @MainActor
    func resolveNonRussianSubscriptionKind() async -> SubscriptionKind {

        // РАНЬШЕ было: guard Apphud.hasPremiumAccess() else unknown
        // Это ломает кейс, когда StoreKit2 entitlement уже активен, а Apphud ещё не успел обновить premium.
        // Теперь: если есть entitlement -> считаем premium и резолвим по нему.
        let apphudPremium = Apphud.hasPremiumAccess()

        var activeProductId: String? = nil
        for await result in StoreKit.Transaction.currentEntitlements {
            guard case .verified(let transaction) = result else { continue }
            guard transaction.productType == .autoRenewable else { continue }

            if transaction.revocationDate != nil { continue }
            if let exp = transaction.expirationDate, exp <= Date() { continue }

            activeProductId = transaction.productID
            break
        }

        // Если нет entitlement и Apphud тоже не premium — unknown
        if activeProductId == nil && apphudPremium == false {
            logSubKind(
                source: "NON_RU_SK2",
                message: "no entitlement and hasPremiumAccess=false -> unknown",
                productId: nil,
                kind: .unknown
            )
            return .unknown
        }

        guard let productId = activeProductId else {
            // Apphud premium = true, но entitlements не нашлись — оставим unknown (как и раньше), но лог понятнее
            logSubKind(
                source: "NON_RU_SK2",
                message: "hasPremiumAccess=true but no active autoRenewable entitlement found -> unknown",
                productId: nil,
                kind: .unknown
            )
            return .unknown
        }

        guard let product = self.products.first(where: { $0.productId == productId }) else {
            logSubKind(
                source: "NON_RU_SK2",
                message: "active entitlement productId found, but not in self.products",
                productId: productId,
                kind: .unknown
            )
            return .unknown
        }

        let kind = mapToSubscriptionKind(period: product.subscriptionPeriod)

        logSubKind(
            source: "NON_RU_SK2",
            message: "resolved from StoreKit2 entitlement (ApphudPremium=\(apphudPremium)) + ApphudProduct.subscriptionPeriod",
            productId: productId,
            kind: kind
        )

        return kind
    }

    // MARK: - Mapping

     func mapToSubscriptionKind(period: ProductDataPeriod?) -> SubscriptionKind {
        guard let period else { return .unknown }

        switch period.unit {
        case .weekOfMonth:
            return .weekly

        case .day:
            return period.numberOfUnits == 7 ? .weekly : .unknown

        case .month:
            if period.numberOfUnits >= 12 { return .yearly }
            return .monthly

        case .year:
            return .yearly

        default:
            return .unknown
        }
    }

    // MARK: - Logging

     func logSubKind(source: String, message: String, productId: String?, kind: SubscriptionKind) {
        #if DEBUG
        logger.info("SubKind[\(source)]: \(message) productId=\(productId ?? "nil") kind=\(kind.debugName)")
        #endif
    }
}

import Foundation

extension PurchaseManager {

    @MainActor
    func resolveRussianSubscriptionKind() async -> SubscriptionKind {

        let status: SubscriptionStatus
        do {
            status = try await APIPaymentManager.shared.fetchSubscriptionStatus(userId: userId)
        } catch {
            logSubKind(
                source: "RU_API",
                message: "fetchSubscriptionStatus failed: \(error.localizedDescription) -> unknown",
                productId: nil,
                kind: .unknown
            )
            return .unknown
        }

        guard status.isActive else {
            logSubKind(
                source: "RU_API",
                message: "subscription inactive -> unknown",
                productId: nil,
                kind: .unknown
            )
            return .unknown
        }

        let txs: [Transaction]
        do {
            txs = try await APIPaymentManager.shared.fetchTransactions(userId: userId)
        } catch {
            logSubKind(
                source: "RU_API",
                message: "fetchTransactions failed: \(error.localizedDescription) -> unknown",
                productId: nil,
                kind: .unknown
            )
            return .unknown
        }

        let pickedTx: Transaction? = {
            if let lastId = status.lastTransactionId, lastId.isEmpty == false {
                if let exact = txs.first(where: { $0.id == lastId }) {
                    return exact
                }
            }

            let subs = txs.filter { $0.isSubscription }
            return subs.sorted(by: { parseBackendDate($0.createdAt) > parseBackendDate($1.createdAt) }).first
        }()

        guard let tx = pickedTx else {
            logSubKind(
                source: "RU_API",
                message: "premium=true but no subscription transaction found -> unknown",
                productId: nil,
                kind: .unknown
            )
            return .unknown
        }

        let products: [PaymentProduct]
        do {
            let bundle = Bundle.main.bundleIdentifier ?? ""
            products = try await APIPaymentManager.shared.fetchProducts(appBundle: bundle)
        } catch {
            logSubKind(
                source: "RU_API",
                message: "fetchProducts failed: \(error.localizedDescription) -> unknown",
                productId: tx.productId,
                kind: .unknown
            )
            return .unknown
        }

        let product = products.first(where: { $0.apphudId == tx.productId })
            ?? products.first(where: { String($0.id) == tx.productId })

        guard let product else {
            logSubKind(
                source: "RU_API",
                message: "active tx found but product not matched. tx.productId=\(tx.productId) -> unknown",
                productId: tx.productId,
                kind: .unknown
            )
            return .unknown
        }

        let kind = mapRUProductToKind(product)

        logSubKind(
            source: "RU_API",
            message: "resolved from product. interval=\(product.subscriptionInterval ?? "nil") period=\(product.subscriptionPeriod.map(String.init) ?? "nil")",
            productId: product.apphudId,
            kind: kind
        )

        return kind
    }
}

// MARK: - Helpers

private extension PurchaseManager {

    func mapRUProductToKind(_ product: PaymentProduct) -> SubscriptionKind {
        let intervalRaw = (product.subscriptionInterval ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
        let interval = intervalRaw.lowercased()
        let period = product.subscriptionPeriod ?? 0

        if interval.contains("week") {
            return .weekly
        }

        if interval.contains("year") {
            return .yearly
        }

        if interval.contains("month") {
            if period >= 12 {
                return .yearly
            }
            return .monthly
        }

        if interval.contains("day") {
            if period == 7 { return .weekly }
            return .unknown
        }

        return .unknown
    }

    func parseBackendDate(_ string: String) -> Date {
        let iso = ISO8601DateFormatter()
        if let d = iso.date(from: string) { return d }

        let df = DateFormatter()
        df.locale = Locale(identifier: "en_US_POSIX")

        df.dateFormat = "yyyy-MM-dd'T'HH:mm:ss.SSSZ"
        if let d = df.date(from: string) { return d }

        df.dateFormat = "yyyy-MM-dd'T'HH:mm:ssZ"
        if let d = df.date(from: string) { return d }

        df.dateFormat = "yyyy-MM-dd HH:mm:ss"
        if let d = df.date(from: string) { return d }

        return .distantPast
    }
}
