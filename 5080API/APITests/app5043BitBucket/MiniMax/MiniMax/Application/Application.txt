import UIKit
import ApphudSDK
import AppTrackingTransparency
import AdSupport
import AmplitudeUnified

final class AppDelegate: NSObject, UIApplicationDelegate {

    // MARK: - UIApplicationDelegate

    func application(
        _ application: UIApplication,
        didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]? = nil
    ) -> Bool {

        Analytics.shared.track("app_open")

        // Важно: Apphud.start(...) теперь выполняется в Billing-компоненте (PurchaseManager),
        // чтобы не было двойного старта SDK.

        // IDFV можно прокинуть сразу (IDFA появится после ATT, если разрешено).
        Apphud.setDeviceIdentifiers(
            idfa: nil,
            idfv: UIDevice.current.identifierForVendor?.uuidString
        )

        requestATT()

        return true
    }

    // MARK: - App lifecycle (Amplitude)

    func applicationDidEnterBackground(_ application: UIApplication) {
        Analytics.shared.track("app_background")
    }

    func applicationWillTerminate(_ application: UIApplication) {
        Analytics.shared.track("app_terminate")
    }

    // MARK: - ATT / IDFA

    private func requestATT() {
        DispatchQueue.main.asyncAfter(deadline: .now() + 1.5) { [weak self] in
            guard self != nil else { return }

            ATTrackingManager.requestTrackingAuthorization { status in
                guard status == .authorized else { return }

                let idfa = ASIdentifierManager.shared().advertisingIdentifier.uuidString

                DispatchQueue.main.async {
                    Apphud.setDeviceIdentifiers(
                        idfa: idfa,
                        idfv: UIDevice.current.identifierForVendor?.uuidString
                    )
                }
            }
        }
    }
}

import SwiftUI
import Swinject

private enum AppStep {
    case deciding
    case onboarding
    case root
}

struct AppFlowView: View {
    @Environment(\.resolver) private var resolver
    @EnvironmentObject private var purchaseManager: PurchaseManager

    @State private var step: AppStep = .deciding
    @State private var showLoader: Bool = true

    // 🔥 ТЕСТОВЫЙ флаг показа paywall
    @State private var showPaywall: Bool = false

    var body: some View {
        ZStack {
            if showLoader {
                LoaderView {
                    showLoader = false
                    decideInitialStep()
                }
                .transition(.opacity)
            } else {
                switch step {
                case .deciding:
                    Color.clear
                        .onAppear {
                            decideInitialStep()
                        }

                case .onboarding:
                    OnboardingView(
                        vm: OnboardingViewModel(),
                        onFinish: {
                            Analytics.shared.track("onboarding_complete")

                            Task {
                                let onboardingStore = resolver.resolve(OnboardingStore.self)!
                                await onboardingStore.setOnboardingCompleted(true)

                                await MainActor.run {
                                    step = .root
                                }
                            }
                        }
                    )

                case .root:
                    RootTabView()
                        .onAppear {
                            // ТЕСТОВЫЙ ПОКАЗ
                            DispatchQueue.main.asyncAfter(deadline: .now() + 1.0) {
                                showPaywall = false
                            }
                        }
                        .fullScreenCover(isPresented: $showPaywall) {
                            paywallView
                        }
                }
            }
        }
        .animation(.easeInOut(duration: 0.2), value: showLoader)
    }
}

// MARK: - Private helpers

extension AppFlowView {

    private func decideInitialStep() {
        let onboardingStore = resolver.resolve(OnboardingStore.self)!
        Task {
            async let onboardingDone = onboardingStore.isOnboardingCompleted()
            let isOnboardingDone = await onboardingDone

            await MainActor.run {
                step = isOnboardingDone ? .root : .onboarding
            }
        }
    }

    @ViewBuilder
    private var paywallView: some View {
        if purchaseManager.isRus {
            APIPaywallView()
                .environmentObject(purchaseManager)
                .ignoresSafeArea()
        } else {
            PaywallView()
                .environmentObject(purchaseManager)
                .ignoresSafeArea()
        }
    }
}

import SwiftUI
import Swinject

@main
struct ClaudeApp: App {

    @UIApplicationDelegateAdaptor(AppDelegate.self) private var appDelegate

    private let assembler = AppAssembler.make()
    private var resolver: Resolver { assembler.resolver }

    @Environment(\.scenePhase) private var scenePhase

    // Новый billing-компонент (из демки)
    @StateObject private var purchaseManager = PurchaseManager.shared

#if DEBUG
    @State private var showDebugRateUs: Bool = false
    @State private var showDebugSettings: Bool = false
#endif

    init() {

        ScreenScale.configure()

        UIWindow.appearance().overrideUserInterfaceStyle = .light
        UILabel.appearance().adjustsFontForContentSizeCategory = false
        UITextView.appearance().adjustsFontForContentSizeCategory = false
        UITextField.appearance().adjustsFontForContentSizeCategory = false

        let resolver = self.resolver

        // Прогрев не биллинга, а твоей "генерации" (pending jobs)
        Task {
            if let runner = resolver.resolve(PendingGenerationRecoveryRunner.self) {
                await runner.runOnce()
            }

            if let scheduler = resolver.resolve(PendingGenerationPollingScheduler.self) {
                await scheduler.startIfNeeded()
            }

#if DEBUG
            if let counter = resolver.resolve(RequestCounterService.self) {
                await counter.debugLogCurrentState(tag: "app_start")
            }
#endif
        }
    }

    var body: some Scene {
        WindowGroup {
            AppFlowView()
                .environment(\.resolver, resolver)
                .environmentObject(purchaseManager)
                .preferredColorScheme(.light)

                .onChange(of: scenePhase) { phase in
                    Task {
                        guard let scheduler = resolver.resolve(PendingGenerationPollingScheduler.self) else { return }

                        switch phase {
                        case .active:
                            await scheduler.startIfNeeded()

                        case .background:
                            await scheduler.stop()

                        case .inactive:
                            break

                        @unknown default:
                            break
                        }
                    }
                }

#if DEBUG
                .fullScreenCover(isPresented: $showDebugRateUs) {
                    RateUsView() { _ in }
                        .ignoresSafeArea()
                }
                .fullScreenCover(isPresented: $showDebugSettings) {
                    SettingsView()
                        .environment(\.resolver, resolver)
                        .ignoresSafeArea()
                }
#endif
        }
    }
}
