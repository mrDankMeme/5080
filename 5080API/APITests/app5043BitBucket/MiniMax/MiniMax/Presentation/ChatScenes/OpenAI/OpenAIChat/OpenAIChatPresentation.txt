import SwiftUI

public struct OpenAITextChatView: View {

    @StateObject private var vm: OpenAITextChatViewModel

    public init(vm: OpenAITextChatViewModel) {
        _vm = StateObject(wrappedValue: vm)
    }

    public var body: some View {
        ChatContainerView(
            title: "Chat",
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText
        ) {
            inputBar
        }
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .topBarTrailing) {
                Button {
                    vm.cancel()
                } label: {
                    Image(systemName: "xmark.circle.fill")
                        .font(.system(size: 18, weight: .semibold))
                        .foregroundStyle(Tokens.Color.textSecondary)
                }
                .opacity(vm.isLoading ? 1 : 0)
                .disabled(!vm.isLoading)
            }
        }
    }

    private var inputBar: some View {
        HStack(spacing: 10.scale) {

            TextField("Message", text: $vm.inputText, axis: .vertical)
                .textFieldStyle(.plain)
                .font(Tokens.Font.medium17)
                .foregroundStyle(Tokens.Color.textPrimary)
                .padding(.vertical, 10.scale)
                .padding(.horizontal, 12.scale)
                .background(
                    RoundedRectangle(cornerRadius: Tokens.Radius.medium)
                        .fill(Tokens.Color.componentsBackground)
                )

            Button {
                vm.send()
            } label: {
                Image(systemName: "arrow.up.circle.fill")
                    .font(.system(size: 28, weight: .semibold))
                    .foregroundStyle(
                        vm.inputText.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty
                        ? Tokens.Color.textSecondary
                        : Tokens.Color.accent
                    )
            }
            .disabled(vm.inputText.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty || vm.isLoading)
        }
        .padding(.horizontal, 16.scale)
        .padding(.vertical, 12.scale)
        .background(Tokens.Color.componentsBackground)
    }
}

import Foundation

@MainActor
public final class OpenAITextChatViewModel: ObservableObject {

    @Published public private(set) var messages: [ChatMessage] = []
    @Published public var inputText: String = ""
    @Published public private(set) var isLoading: Bool = false
    @Published public private(set) var errorText: String? = nil

    /// Для MVP держим model константой, позже можно вынести в UI/Settings.
    public var model: String = "gpt-4o-mini"

    private let useCase: SendOpenAIChatMessageUseCase
    private var conversationId: String? = nil
    private var task: Task<Void, Never>? = nil

    public init(useCase: SendOpenAIChatMessageUseCase) {
        self.useCase = useCase
        self.messages = [
            ChatMessage(role: .system, text: "Напиши сообщение — я отправлю его в OpenAI-чат и покажу ответ.")
        ]
    }

    public func cancel() {
        task?.cancel()
        task = nil
        isLoading = false
    }

    public func send() {
        let text = inputText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !text.isEmpty else { return }

        errorText = nil
        inputText = ""

        // 1) сразу показываем сообщение пользователя
        messages.append(ChatMessage(role: .user, text: text))

        // 2) запускаем запрос
        isLoading = true

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }

            do {
                let result = try await useCase.execute(
                    OpenAIChatMessageRequest(
                        conversationId: conversationId,
                        model: model,
                        userText: text
                    )
                )

                // продолжаем тот же разговор на следующих send()
                self.conversationId = result.conversationId

                self.messages.append(
                    ChatMessage(role: .assistant, text: result.assistantText)
                )

                self.isLoading = false

            } catch is CancellationError {
                self.isLoading = false

            } catch {
                self.isLoading = false
                self.errorText = String(describing: error)
            }
        }
    }
}
