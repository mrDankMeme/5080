import SwiftUI
import PhotosUI

struct TwoPhotosToVideoChatView: View {

    @EnvironmentObject private var purchaseManager: PurchaseManager

    @StateObject private var vm: TwoPhotosToVideoChatViewModel

    @State private var startItem: PhotosPickerItem?
    @State private var endItem: PhotosPickerItem?

    @State private var startPreview: UIImage?
    @State private var endPreview: UIImage?

    @State private var isStartPickerPresented: Bool = false
    @State private var isEndPickerPresented: Bool = false

    init(vm: TwoPhotosToVideoChatViewModel) {
        _vm = StateObject(wrappedValue: vm)
    }

    private var isReadyToSend: Bool {
        startItem != nil && endItem != nil
    }

    var body: some View {
        GenerationChatContainerView(
            title: "two_photos_to_video.title".localizable,
            modeTitle: "two_photos_to_video.title".localizable,
            messages: vm.messages,
            isLoading: vm.isLoading,
            errorText: vm.errorText,
            threadId: vm.threadId,
            historyRepo: vm.historyRepoForViewer,
            mediaCache: vm.mediaCacheForViewer,
            onMessageUpdated: { updated in
                vm.applyMessageUpdateFromViewer(updated)
            }
        ) {
            GenerationChatComposerView(
                showsPlus: true,
                isLoading: vm.isLoading,
                text: $vm.prompt, // optional prompt
                attachmentImage: startPreview,
                onPickPhotoTap: {
                    guard !vm.isLoading else { return }
                    isStartPickerPresented = true
                },
                onMicTap: { },
                onSendTap: { onGeneratePressed() },
                isSendEnabled: isReadyToSend,
                secondAttachmentImage: endPreview,
                showsSecondAttachmentSlot: true,
                onSecondPickPhotoTap: {
                    guard !vm.isLoading else { return }
                    isEndPickerPresented = true
                }
            )
            .photosPicker(isPresented: $isStartPickerPresented, selection: $startItem, matching: .images)
            .photosPicker(isPresented: $isEndPickerPresented, selection: $endItem, matching: .images)
            .onChange(of: startItem) { _ in
                Task { @MainActor in
                    await loadStartPreviewIfNeeded()
                }
            }
            .onChange(of: endItem) { _ in
                Task { @MainActor in
                    await loadEndPreviewIfNeeded()
                }
            }
        }
        .onChange(of: vm.shouldShowPaywall) { should in
            guard should else { return }
            purchaseManager.isShowedPaywall = true
            vm.consumePaywallSignal()

            #if DEBUG
            print("💎 GEN(TwoPhotosToVideo): show paywall from gate (kind=\(purchaseManager.currentSubKind))")
            #endif
        }
        .alert(item: $vm.limitAlert) { item in
            Alert(
                title: Text(item.title),
                message: Text(item.message),
                dismissButton: .default(Text(item.buttonTitle))
            )
        }
    }

    private func onGeneratePressed() {
        guard let startItem, let endItem else {
            vm.showUserError("two_photos_to_video.error.pick_two".localizable)
            return
        }

        Task { @MainActor in
            do {
                guard let startData = try await startItem.loadTransferable(type: Data.self) else {
                    vm.showUserError("two_photos_to_video.error.start_nil".localizable)
                    return
                }

                guard let endData = try await endItem.loadTransferable(type: Data.self) else {
                    vm.showUserError("two_photos_to_video.error.end_nil".localizable)
                    return
                }

                vm.onGenerateTap(startPhotoData: startData, endPhotoData: endData)

                self.startItem = nil
                self.endItem = nil
                self.startPreview = nil
                self.endPreview = nil

                vm.prompt = ""

            } catch {
                vm.showUserError(
                    String(
                        format: "two_photos_to_video.error.read_failed_format".localizable,
                        String(describing: error)
                    )
                )
            }
        }
    }

    @MainActor
    private func loadStartPreviewIfNeeded() async {
        guard let startItem else {
            startPreview = nil
            return
        }

        do {
            if let data = try await startItem.loadTransferable(type: Data.self),
               let ui = UIImage(data: data) {
                startPreview = ui
            } else {
                startPreview = nil
            }
        } catch {
            startPreview = nil
        }
    }

    @MainActor
    private func loadEndPreviewIfNeeded() async {
        guard let endItem else {
            endPreview = nil
            return
        }

        do {
            if let data = try await endItem.loadTransferable(type: Data.self),
               let ui = UIImage(data: data) {
                endPreview = ui
            } else {
                endPreview = nil
            }
        } catch {
            endPreview = nil
        }
    }
}

import Foundation

@MainActor
final class TwoPhotosToVideoChatViewModel: ObservableObject {

    // MARK: - Thread

    let threadId: UUID
    let kind: GenerationKind = .twoPhotosToVideo
    let shouldLoadHistory: Bool

    // MARK: - UI

    @Published var messages: [GenerationMessage] = [
        GenerationMessage(
            role: .system,
            createdAt: Date(),
            payload: .system("two_photos_vm.system.hint".localizable)
        )
    ]

    @Published var prompt: String = ""

    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    // ✅ Gate outputs (UI reacts)
    @Published var shouldShowPaywall: Bool = false
    @Published var limitAlert: LimitAlertItem? = nil

    // MARK: - Dependencies

    let useCase: GenerateVideoFromTwoPhotosUseCase
    let historyRepo: GenerationHistoryRepository
    let mediaCache: MediaCacheService
    let inputAttachmentStore: GenerationInputAttachmentStore

    let internalNotifications: InternalNotificationCenter
    let requestCounter: RequestCounterService

    let pendingService: PendingGenerationService
    let updateCenter: GenerationThreadUpdateCenter

    var historyRepoForViewer: GenerationHistoryRepository { historyRepo }
    var mediaCacheForViewer: MediaCacheService { mediaCache }

    func applyMessageUpdateFromViewer(_ updated: GenerationMessage) {
        replaceGenerationMessageInUI(updated: updated)
    }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    var activeStatusMessageId: UUID?
    var activeGenerationMessageId: UUID?

    var activeGenerationId: UUID?

    var updatesTask: Task<Void, Never>?
    var lastReloadAt: Date = .distantPast

    // MARK: - Anti-spam (DEBUG)
    static let debugVideoParallelLimit: Int = .max

    // MARK: - Subscription snapshot provider (injected from factory)

    typealias SubscriptionSnapshot = (isSubscribed: Bool, kind: SubscriptionKind)
    private var subscriptionSnapshotProvider: (@Sendable () -> SubscriptionSnapshot)?

    func setSubscriptionSnapshotProvider(_ provider: @escaping @Sendable () -> SubscriptionSnapshot) {
        self.subscriptionSnapshotProvider = provider
    }

    func consumePaywallSignal() {
        shouldShowPaywall = false
    }

    struct LimitAlertItem: Identifiable, Equatable {
        let id: String = UUID().uuidString
        let title: String
        let message: String
        let buttonTitle: String
    }

    func evaluateGate(snapshot: RequestCounterSnapshot) -> GenerationGateDecision {
        let sub = subscriptionSnapshotProvider?() ?? (false, .unknown)
        return GenerationLimitGate.evaluate(
            isSubscribed: sub.isSubscribed,
            kind: sub.kind,
            snapshot: snapshot
        )
    }

    // MARK: - Init

    init(
        threadId: UUID? = nil,
        useCase: GenerateVideoFromTwoPhotosUseCase,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService,
        inputAttachmentStore: GenerationInputAttachmentStore,
        internalNotifications: InternalNotificationCenter,
        requestCounter: RequestCounterService,
        pendingService: PendingGenerationService,
        updateCenter: GenerationThreadUpdateCenter
    ) {
        self.threadId = threadId ?? UUID()
        self.shouldLoadHistory = (threadId != nil)
        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache
        self.inputAttachmentStore = inputAttachmentStore
        self.internalNotifications = internalNotifications
        self.requestCounter = requestCounter
        self.pendingService = pendingService
        self.updateCenter = updateCenter

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
            subscribeToThreadUpdates()
        }
    }

    deinit {
        task?.cancel()
        historyTask?.cancel()
        eagerRecoveryTask?.cancel()
        updatesTask?.cancel()
    }

    // MARK: - Public

    func showUserError(_ text: String) {
        errorText = text
    }

    func cancel() {
        task?.cancel()
        task = nil

        isLoading = false
        activeGenerationMessageId = nil

        if let gid = activeGenerationId {
            Task { [pendingService] in
                await pendingService.remove(generationId: gid)
            }
        }
        activeGenerationId = nil

        if let messageId = activeStatusMessageId {
            replaceGenerationMessage(
                id: messageId,
                payload: .system("two_photos_vm.vm.cancelled".localizable)
            )
            activeStatusMessageId = nil
        } else {
            messages.append(
                GenerationMessage(
                    role: .system,
                    createdAt: Date(),
                    payload: .system("two_photos_vm.vm.cancelled".localizable)
                )
            )
        }
    }

    // MARK: - Updates subscription

    func subscribeToThreadUpdates() {
        updatesTask?.cancel()

        updatesTask = Task { [weak self] in
            guard let self else { return }

            let stream = await self.updateCenter.stream()

            for await updatedThreadId in stream {
                if Task.isCancelled { break }
                guard updatedThreadId == self.threadId else { continue }

                let now = Date()
                if now.timeIntervalSince(self.lastReloadAt) < 0.35 { continue }
                self.lastReloadAt = now

                #if DEBUG
                print("🧾 GEN(TwoPhotosToVideo): updateCenter -> reload threadId=\(self.threadId)")
                #endif

                await MainActor.run {
                    self.loadHistoryIfExists()
                    self.startEagerRecoveryIfNeeded()
                }
            }
        }
    }

    // MARK: - Notifications (Variant A)

    func markNotificationsAsReadForThisThreadIfNeeded() {
        guard shouldLoadHistory else { return }

        Task {
            let list = await internalNotifications.list()

            let targets = list.filter { n in
                n.isRead == false
                && n.payload.threadId == self.threadId
                && n.payload.generationKind == self.kind
            }

            guard targets.isEmpty == false else { return }

            for n in targets {
                await internalNotifications.markRead(id: n.id)
            }

            #if DEBUG
            print("🔔 GEN(TwoPhotosToVideo): marked read \(targets.count) notifications for threadId=\(self.threadId)")
            #endif
        }
    }
}

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func onGenerateTap(startPhotoData: Data, endPhotoData: Data) {

        let promptTrimmed = self.prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let normalizedPrompt: String? = promptTrimmed.isEmpty ? nil : promptTrimmed

        Task { @MainActor [weak self] in
            guard let self else { return }

            let snap = await self.requestCounter.snapshot()

            // ✅ Gate before doing anything expensive
            let gate = self.evaluateGate(snapshot: snap)
            switch gate {
            case .allow:
                break
            case .showPaywall:
                self.shouldShowPaywall = true
                return
            case .showAlert(let title, let message):
                self.limitAlert = LimitAlertItem(title: title, message: message, buttonTitle: "OK")
                return
            }

//            if snap.video >= Self.debugVideoParallelLimit {
//                self.showUserError("two_photos_vm.generate.error.too_many_parallel".localizable)
//                return
//            }

            self.errorText = nil
            self.lastResultURL = nil

            self.task?.cancel()
            self.task = nil

            self.isLoading = true
            self.activeGenerationMessageId = nil
            self.activeStatusMessageId = nil
            self.activeGenerationId = nil

            let now = Date()

            var attachments: [GenerationAttachment] = []
            do {
                let startAtt = try self.inputAttachmentStore.saveImage(threadId: self.threadId, data: startPhotoData)
                let endAtt = try self.inputAttachmentStore.saveImage(threadId: self.threadId, data: endPhotoData)
                attachments = [startAtt, endAtt]
            } catch {
                #if DEBUG
                print("🧾 INPUT_ATTACHMENTS save failed threadId=\(self.threadId) error=\(error)")
                #endif
            }

            let userText = normalizedPrompt ?? "two_photos_vm.user.fallback_two_photos".localizable

            let userMessage = GenerationMessage(
                role: .user,
                createdAt: now,
                payload: .text(userText),
                attachments: attachments
            )

            self.messages.append(userMessage)

            let previewSeed: String? = {
                if let normalizedPrompt, !normalizedPrompt.isEmpty { return normalizedPrompt }
                if !attachments.isEmpty { return "two_photos_vm.preview.two_photos".localizable }
                return nil
            }()
            self.persistUserMessageToHistory(userMessage, previewSeed: previewSeed)

            let placeholderId = UUID()
            self.activeStatusMessageId = placeholderId

            let placeholderCreatedAt = Date()
            let placeholder = GenerationMessage(
                id: placeholderId,
                role: .assistant,
                createdAt: placeholderCreatedAt,
                payload: .video(remoteURL: Self.placeholderRemoteURL, localURL: nil)
            )

            self.replaceGenerationMessageInUI(updated: placeholder)
            self.persistAssistantPlaceholderToHistory(placeholder)

            self.task = Task { @MainActor [weak self] in
                guard let self else { return }
                defer { self.isLoading = false }

                do {
                    let startBinary = Self.makePhotoBinary(from: startPhotoData, baseName: "start_frame")
                    let endBinary = Self.makePhotoBinary(from: endPhotoData, baseName: "end_frame")

                    let request = TwoPhotosToVideoRequest(
                        startFrame: startBinary,
                        endFrame: endBinary,
                        prompt: normalizedPrompt
                    )

                    let started = try await self.useCase.start(request)
                    try Task.checkCancellation()

                    let generationId = started.id
                    self.activeGenerationId = generationId

                    let jobId = generationId.uuidString

                    let record = PendingGenerationRecord(
                        generationId: generationId,
                        counterJobId: jobId,
                        threadId: self.threadId,
                        placeholderMessageId: placeholderId,
                        generationKind: self.kind,
                        category: .video,
                        createdAt: Date()
                    )

                    await self.pendingService.upsert(record)

                    #if DEBUG
                    print("🧾 GEN(TwoPhotosToVideo): pending upsert ok generationId=\(generationId) placeholderId=\(placeholderId) threadId=\(self.threadId)")
                    #endif

                    let terminal = try await self.useCase.poll(started: started)
                    try Task.checkCancellation()

                    let stillActive = (self.activeStatusMessageId == placeholderId)
                    if stillActive == false {
                        #if DEBUG
                        print("🧾 GEN(TwoPhotosToVideo): ignored late terminal generationId=\(generationId) placeholderId=\(placeholderId)")
                        #endif
                        return
                    }

                    guard let url = GenerationResultNormalizer.extractUsableResultURL(from: terminal) else {
                        let reasonRaw = (terminal.error ?? "no_valid_result_url").trimmingCharacters(in: .whitespacesAndNewlines)
                        let reason = reasonRaw.isEmpty ? "no_valid_result_url" : reasonRaw

                        await self.requestCounter.releaseOnTerminalFail(
                            category: .video,
                            jobId: jobId,
                            reason: reason
                        )

                        let resultText = terminal.result?.trimmingCharacters(in: .whitespacesAndNewlines)

                        let assistantText = String(
                            format: "two_photos_vm.generate.assistant.invalid_url_format".localizable,
                            (resultText?.isEmpty == false ? resultText! : "common.dash".localizable)
                        )

                        let err = GenerationMessage(
                            id: placeholderId,
                            role: .assistant,
                            createdAt: placeholderCreatedAt,
                            payload: .text(assistantText)
                        )

                        self.replaceGenerationMessageInUI(updated: err)

                        Task { [historyRepo = self.historyRepo, threadId = self.threadId] in
                            try? await historyRepo.replaceMessage(
                                threadId: threadId,
                                messageId: placeholderId,
                                newMessage: err
                            )
                        }

                        await self.pendingService.remove(generationId: generationId)
                        await self.updateCenter.post(threadId: self.threadId)

                        #if DEBUG
                        print("🧾 GEN(TwoPhotosToVideo): terminal without url -> release & remove pending generationId=\(generationId) reason=\(reason)")
                        #endif
                        return
                    }

                    // ✅ IMPORTANT: commit success
                    await self.requestCounter.commitSuccess(category: .video, jobId: jobId)

                    self.lastResultURL = url

                    await self.updatePlaceholderRemoteAndDownload(
                        messageId: placeholderId,
                        originalCreatedAt: placeholderCreatedAt,
                        remoteURL: url
                    )

                    await self.pendingService.remove(generationId: generationId)
                    await self.updateCenter.post(threadId: self.threadId)

                    #if DEBUG
                    print("🧾 GEN(TwoPhotosToVideo): success -> commit & remove pending generationId=\(generationId) url=\(url.absoluteString)")
                    #endif

                } catch is CancellationError {
                    self.lastResultURL = nil
                    self.errorText = nil

                    let cancelled = GenerationMessage(
                        id: placeholderId,
                        role: .assistant,
                        createdAt: Date(),
                        payload: .system("two_photos_vm.generate.system.cancelled".localizable)
                    )
                    self.replaceGenerationMessageInUI(updated: cancelled)

                    Task { [historyRepo = self.historyRepo, threadId = self.threadId] in
                        try? await historyRepo.replaceMessage(
                            threadId: threadId,
                            messageId: placeholderId,
                            newMessage: cancelled
                        )
                    }

                    if let gid = self.activeGenerationId {
                        await self.pendingService.remove(generationId: gid)
                        #if DEBUG
                        print("🧾 GEN(TwoPhotosToVideo): cancelled -> removed pending generationId=\(gid)")
                        #endif
                    }
                    self.activeGenerationId = nil

                } catch let e as GenerationPollingError {
                    switch e {
                    case .timeout:
                        self.errorText = "two_photos_vm.generate.error.timeout_background".localizable
                        #if DEBUG
                        print("🧾 GEN(TwoPhotosToVideo): timeout -> keep pending (no release, no remove)")
                        #endif

                    case .failed(let reason):
                        let r = reason.trimmingCharacters(in: .whitespacesAndNewlines)
                        let safe = r.isEmpty ? "failed" : r

                        if let gid = self.activeGenerationId {
                            await self.requestCounter.releaseOnTerminalFail(
                                category: .video,
                                jobId: gid.uuidString,
                                reason: safe
                            )
                            await self.pendingService.remove(generationId: gid)
                            await self.updateCenter.post(threadId: self.threadId)

                            #if DEBUG
                            print("🧾 GEN(TwoPhotosToVideo): failed -> release & remove pending generationId=\(gid) reason=\(safe)")
                            #endif
                        }
                        self.activeGenerationId = nil

                        let assistantText = String(
                            format: "two_photos_vm.generate.assistant.failed_format".localizable,
                            safe
                        )
                        self.errorText = safe

                        let err = GenerationMessage(
                            id: placeholderId,
                            role: .assistant,
                            createdAt: Date(),
                            payload: .error(assistantText)
                        )
                        self.replaceGenerationMessageInUI(updated: err)

                        Task { [historyRepo = self.historyRepo, threadId = self.threadId] in
                            try? await historyRepo.replaceMessage(
                                threadId: threadId,
                                messageId: placeholderId,
                                newMessage: err
                            )
                        }

                        self.persistAssistantErrorToHistory(assistantText)

                    case .invalidState:
                        self.errorText = "two_photos_vm.generate.error.invalid_state".localizable
                    }

                } catch {
                    let pretty = String(describing: error)

                    if let gid = self.activeGenerationId {
                        await self.requestCounter.releaseOnTerminalFail(
                            category: .video,
                            jobId: gid.uuidString,
                            reason: pretty
                        )
                        await self.pendingService.remove(generationId: gid)
                        await self.updateCenter.post(threadId: self.threadId)

                        #if DEBUG
                        print("🧾 GEN(TwoPhotosToVideo): unexpected error -> release & remove pending generationId=\(gid) error=\(pretty)")
                        #endif
                    }
                    self.activeGenerationId = nil

                    self.lastResultURL = nil
                    self.errorText = pretty

                    let assistantText = String(
                        format: "two_photos_vm.generate.assistant.failed_format".localizable,
                        pretty
                    )

                    let err = GenerationMessage(
                        id: placeholderId,
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                    self.replaceGenerationMessageInUI(updated: err)

                    Task { [historyRepo = self.historyRepo, threadId = self.threadId] in
                        try? await historyRepo.replaceMessage(
                            threadId: threadId,
                            messageId: placeholderId,
                            newMessage: err
                        )
                    }

                    self.persistAssistantErrorToHistory(assistantText)
                }
            }
        }
    }

    fileprivate static var placeholderRemoteURL: URL {
        URL(string: "https://example.com/video.mp4")!
    }
}

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            defer { self.eagerRecoveryTask = nil }
            await self.recoverMissingMediaIfNeeded()
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open TwoPhotosToVideo): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open TwoPhotosToVideo): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func replaceGenerationMessage(id: UUID, payload: GenerationPayload) {
        if let idx = messages.firstIndex(where: { $0.id == id }) {
            let old = messages[idx]
            messages[idx] = GenerationMessage(
                id: old.id,
                role: old.role,
                createdAt: old.createdAt,
                payload: payload
            )
        } else {
            messages.append(GenerationMessage(id: id, role: .assistant, createdAt: Date(), payload: payload))
        }
    }

    func replaceGenerationMessageInUI(updated: GenerationMessage) {
        if let idx = messages.firstIndex(where: { $0.id == updated.id }) {
            messages[idx] = updated
        } else {
            messages.append(updated)
        }
    }

    func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }

    func prettyErrorText(_ error: Error) -> String {
        if let e = error as? GenerationPollingError {
            switch e {
            case .timeout:
                return "two_photos_vm.error.timeout".localizable
            case .invalidState:
                return "two_photos_vm.error.invalid_state".localizable
            case .failed(let msg):
                return msg
            }
        }

        let raw = String(describing: error)
        if raw.lowercased().contains("cancel") {
            return "two_photos_vm.error.cancelled".localizable
        }
        return raw
    }

    static func makePhotoBinary(from data: Data, baseName: String) -> PhotoBinary {
        let (mime, ext) = detectMimeAndExt(from: data)
        let filename = "\(baseName).\(ext)"
        return PhotoBinary(data: data, filename: filename, mimeType: mime)
    }

    static func detectMimeAndExt(from data: Data) -> (mime: String, ext: String) {
        let png: [UInt8] = [0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A]
        if data.starts(with: png) { return ("image/png", "png") }

        let jpg: [UInt8] = [0xFF, 0xD8, 0xFF]
        if data.starts(with: jpg) { return ("image/jpeg", "jpg") }

        return ("application/octet-stream", "bin")
    }
}

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func loadHistoryIfExists() {
        let repo = historyRepo
        let id = threadId

        historyTask?.cancel()
        historyTask = Task { [weak self] in
            guard let self else { return }

            do {
                let stored = try await repo.loadMessages(threadId: id)
                guard stored.isEmpty == false else { return }

                self.messages = Self.stableSortMessages(stored)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: loaded \(self.messages.count) messages for thread id=\(id)")
                #endif

                self.markNotificationsAsReadForThisThreadIfNeeded()

                self.startEagerRecoveryIfNeeded()

            } catch is CancellationError {
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: load history failed: \(error)")
                #endif
            }
        }
    }

    // MARK: - Stable sorting

    private static func stableSortMessages(_ items: [GenerationMessage]) -> [GenerationMessage] {
        items.sorted { lhs, rhs in
            if lhs.createdAt != rhs.createdAt {
                return lhs.createdAt < rhs.createdAt
            }

            let lp = rolePriority(lhs.role)
            let rp = rolePriority(rhs.role)
            if lp != rp { return lp < rp }

            return lhs.id.uuidString < rhs.id.uuidString
        }
    }

    private static func rolePriority(_ role: GenerationMessageRole) -> Int {
        switch role {
        case .system: return 0
        case .user: return 1
        case .assistant: return 2
        }
    }
}

    func persistUserMessageToHistory(_ message: GenerationMessage, previewSeed: String?) {
        let repo = historyRepo
        let id = threadId
        let now = message.createdAt

        let preview: String? = {
            let seed = (previewSeed ?? "").trimmingCharacters(in: .whitespacesAndNewlines)
            if !seed.isEmpty { return Self.makePreviewText(seed) }

            if !message.attachments.isEmpty { return "📎 Attachments" }

            switch message.payload {
            case .text(let t):
                let s = t.trimmingCharacters(in: .whitespacesAndNewlines)
                return s.isEmpty ? nil : Self.makePreviewText(s)
            default:
                return nil
            }
        }()

        Task {
            do {
                await self.ensureThreadRegisteredAsync()
                try await repo.appendMessage(threadId: id, message: message)

                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY(TwoPhotosToVideo): saved user message id=\(message.id) attachments=\(message.attachments.count) threadId=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY(TwoPhotosToVideo): persist user message failed: \(error)")
                #endif
            }
        }
    }

    func persistAssistantErrorToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .error(text), previewSeed: text)
    }

    func persistAssistantPayloadToHistory(payload: GenerationPayload, previewSeed: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let preview = Self.makePreviewText(previewSeed)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .assistant,

                )

                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: persist assistant message failed: \(error)")
                #endif
            }
        }
    }

    static func makePreviewText(_ text: String, limit: Int = 80) -> String {
        let s = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard s.count > limit else { return s }
        let idx = s.index(s.startIndex, offsetBy: limit)
        return String(s[..<idx]) + "…"
    }

import Foundation

extension TwoPhotosToVideoChatViewModel {

    func updatePlaceholderRemoteAndDownload(
        messageId: UUID,
        originalCreatedAt: Date,
        remoteURL: URL
    ) async {
        let repo = historyRepo
        let id = threadId

        activeGenerationMessageId = messageId

        do {
            await ensureThreadRegisteredAsync()

            let updatedRemote = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: originalCreatedAt,
                payload: .video(remoteURL: remoteURL, localURL: nil)
            )

            replaceGenerationMessageInUI(updated: updatedRemote)

            do {
                try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updatedRemote)
            } catch {
                try await repo.appendMessage(threadId: id, message: updatedRemote)
            }

            try await touchThread(preview: "🎞️ Video", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA(TwoPhotosToVideo): placeholder updated to real remote threadId=\(id) msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "video"
            )
            try Task.checkCancellation()

            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA(TwoPhotosToVideo): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            let updatedLocal = GenerationMessage(
                id: messageId,
                role: updatedRemote.role,
                createdAt: updatedRemote.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updatedLocal)
            replaceGenerationMessageInUI(updated: updatedLocal)
            try await touchThread(preview: "✅ Video", at: Date())

            let didCreate = await internalNotifications.postGenerationReady(
                generationKind: .twoPhotosToVideo,
                threadId: id,
                generationId: messageId, // ✅ стабильный id = placeholderId
                title: "Два фото в видео",
                createdAt: Date()
            )

            if self.activeStatusMessageId == messageId {
                self.activeStatusMessageId = nil
            }

            #if DEBUG
            print("🔔 GEN(TwoPhotosToVideo): post ready didCreate=\(didCreate) threadId=\(id) generationId=\(messageId)")
            print("🧾 GEN_MEDIA(TwoPhotosToVideo): replaced with local threadId=\(id) msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: Date(),
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

        } catch {
            let pretty = String(describing: error)

            let warning = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: originalCreatedAt,
                payload: .error("🎞️ Video\n\n⚠️ Не удалось скачать: \(pretty)")
            )
            replaceGenerationMessageInUI(updated: warning)

            #if DEBUG
            print("🧾 GEN_MEDIA(TwoPhotosToVideo): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

import Foundation

extension TwoPhotosToVideoChatViewModel {


    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}
