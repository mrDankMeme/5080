import Foundation

@MainActor
final class TextToImageChatViewModel: ObservableObject {

    // MARK: - Thread

    let threadId: UUID
    let kind: GenerationKind = .textToImage
    let shouldLoadHistory: Bool

    // MARK: - UI

    @Published var messages: [GenerationMessage] = [
        GenerationMessage(
            role: .system,
            createdAt: Date(),
            payload: .system("Напиши промпт и нажми отправить.")
        )
    ]

    @Published var prompt: String = ""
    @Published var templateIdText: String = "0"

    @Published var isLoading: Bool = false
    @Published var errorText: String?
    @Published var lastResultURL: URL?

    // MARK: - Dependencies

    let useCase: GenerateImageFromTextUseCase
    let historyRepo: GenerationHistoryRepository
    let mediaCache: MediaCacheService

    var historyRepoForViewer: GenerationHistoryRepository { historyRepo }
    var mediaCacheForViewer: MediaCacheService { mediaCache }

    // MARK: - Tasks

    var task: Task<Void, Never>?
    var historyTask: Task<Void, Never>?

    var activeGenerationMessageId: UUID?

    var eagerRecoveryTask: Task<Void, Never>?
    var eagerRecoveryInFlight: Set<UUID> = []

    // MARK: - Init

    init(
        threadId: UUID? = nil,
        useCase: GenerateImageFromTextUseCase,
        historyRepo: GenerationHistoryRepository,
        mediaCache: MediaCacheService
    ) {
        self.threadId = threadId ?? UUID()
        self.shouldLoadHistory = (threadId != nil)
        self.useCase = useCase
        self.historyRepo = historyRepo
        self.mediaCache = mediaCache

        if shouldLoadHistory {
            ensureThreadRegistered()
            loadHistoryIfExists()
        }
    }

    deinit {
        task?.cancel()
        historyTask?.cancel()
        eagerRecoveryTask?.cancel()
    }

    func applyMessageUpdateFromViewer(_ updated: GenerationMessage) {
        replaceGenerationMessageInUI(updated: updated)
    }

    func cancel() {
        task?.cancel()
        task = nil
        isLoading = false

        activeGenerationMessageId = nil
    }
}

import Foundation

extension TextToImageChatViewModel {

    // MARK: - Actions

    func onGenerateTap() {
        let prompt = self.prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        let templateId = Int(self.templateIdText.trimmingCharacters(in: .whitespacesAndNewlines)) ?? 0

        guard !prompt.isEmpty else {
            errorText = "Prompt пустой."
            return
        }

        errorText = nil
        isLoading = true

        activeGenerationMessageId = nil

        messages.append(
            GenerationMessage(
                role: .user,
                createdAt: Date(),
                payload: .text(prompt)
            )
        )

        persistUserPromptToHistory(prompt)

        task?.cancel()
        task = Task { [weak self] in
            guard let self else { return }
            defer { self.isLoading = false }

            do {
                let dto = try await useCase.execute(prompt: prompt, templateId: templateId)
                let raw = (dto.result ?? "").trimmingCharacters(in: .whitespacesAndNewlines)

                if let url = URL(string: raw), url.scheme != nil {
                    self.lastResultURL = url

                    await self.appendImagePlaceholderAndDownload(remoteURL: url)

                } else {
                    self.lastResultURL = nil

                    let printable = raw.isEmpty ? "—" : raw
                    let assistantText = "✅ Done, но result не похож на URL:\n\(printable)"

                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .text(assistantText)
                        )
                    )

                    self.persistAssistantTextToHistory(assistantText)
                }

                if dto.status == .failed, let e = dto.error, !e.isEmpty {
                    let warnText = "⚠️ Backend status=failed, error:\n\(e)"
                    self.messages.append(
                        GenerationMessage(
                            role: .assistant,
                            createdAt: Date(),
                            payload: .error(warnText)
                        )
                    )
                    self.persistAssistantErrorToHistory(warnText)
                }

            } catch is CancellationError {
                self.lastResultURL = nil
                self.errorText = nil
                self.messages.append(
                    GenerationMessage(
                        role: .system,
                        createdAt: Date(),
                        payload: .system("⛔️ Cancelled")
                    )
                )

            } catch {
                self.lastResultURL = nil
                let pretty = String(describing: error)
                self.errorText = pretty

                let assistantText = "❌ Error:\n\(pretty)"
                self.messages.append(
                    GenerationMessage(
                        role: .assistant,
                        createdAt: Date(),
                        payload: .error(assistantText)
                    )
                )

                self.persistAssistantErrorToHistory(assistantText)
            }
        }
    }
}

// FILE: TextToImageChatViewModel+EagerRecovery.swift
////////////////////////////////////////////////////////////



import Foundation

extension TextToImageChatViewModel {


    func startEagerRecoveryIfNeeded() {
        guard eagerRecoveryTask == nil else { return }

        eagerRecoveryTask = Task { [weak self] in
            guard let self else { return }
            await self.recoverMissingMediaIfNeeded()
            self.eagerRecoveryTask = nil
        }
    }

    func recoverMissingMediaIfNeeded() async {
        let id = threadId
        let repo = historyRepo

        let snapshot = messages

        let candidates: [(msg: GenerationMessage, kind: String, remote: URL)] = snapshot.compactMap { msg in
            switch msg.payload {
            case .image(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "image", remote)

            case .video(let remote, let local):
                if let local, fileExists(local) { return nil }
                return (msg, "video", remote)

            default:
                return nil
            }
        }

        guard candidates.isEmpty == false else {
            #if DEBUG
            print("🧾 MEDIA_RECOVERY(open): nothing to recover threadId=\(id)")
            #endif
            return
        }

        #if DEBUG
        print("🧾 MEDIA_RECOVERY(open): start threadId=\(id) items=\(candidates.count)")
        #endif

        for c in candidates {
            if Task.isCancelled { break }

            if eagerRecoveryInFlight.contains(c.msg.id) { continue }
            eagerRecoveryInFlight.insert(c.msg.id)
            defer { eagerRecoveryInFlight.remove(c.msg.id) }

            do {
                let local = try await mediaCache.fetch(
                    remoteURL: c.remote,
                    threadId: id,
                    kind: c.kind
                )
                try Task.checkCancellation()

                let updated = makeUpdatedMediaMessage(
                    original: c.msg,
                    kind: c.kind,
                    remoteURL: c.remote,
                    localURL: local
                )

                try await repo.replaceMessage(threadId: id, messageId: updated.id, newMessage: updated)
                replaceGenerationMessageInUI(updated: updated)

                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open): success threadId=\(id) msgId=\(updated.id) kind=\(c.kind) local=\(local.path)")
                #endif

            } catch is CancellationError {
                #if DEBUG
                print("🧾 MEDIA_RECOVERY(open): cancelled threadId=\(id)")
                #endif
                break

            } catch {
                #if DEBUG
                print("🚨 MEDIA_RECOVERY(open): failed threadId=\(id) msgId=\(c.msg.id) kind=\(c.kind) error=\(error)")
                #endif
            }
        }
    }

    func makeUpdatedMediaMessage(
        original: GenerationMessage,
        kind: String,
        remoteURL: URL,
        localURL: URL
    ) -> GenerationMessage {
        switch kind {
        case "image":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )
        case "video":
            return GenerationMessage(
                id: original.id,
                role: original.role,
                createdAt: original.createdAt,
                payload: .video(remoteURL: remoteURL, localURL: localURL)
            )
        default:
            return original
        }
    }
}

import Foundation

extension TextToImageChatViewModel {


    func loadHistoryIfExists() {
        let repo = historyRepo
        let id = threadId

        historyTask?.cancel()
        historyTask = Task { [weak self] in
            guard let self else { return }

            do {
                let stored = try await repo.loadMessages(threadId: id)
                guard stored.isEmpty == false else { return }

                self.messages = stored.sorted(by: { $0.createdAt < $1.createdAt })

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: loaded \(self.messages.count) messages for thread id=\(id)")
                #endif

                self.startEagerRecoveryIfNeeded()

            } catch is CancellationError {
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.4: load history failed: \(error)")
                #endif
            }
        }
    }
}

import Foundation

extension TextToImageChatViewModel {


    func persistUserPromptToHistory(_ prompt: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let clean = prompt.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !clean.isEmpty else { return }

        let preview = Self.makePreviewText(clean)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .user,
                    createdAt: now,
                    payload: .text(clean)
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: saved user prompt + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.2: persist user prompt failed: \(error)")
                #endif
            }
        }
    }


    func persistAssistantTextToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .text(text), previewSeed: text)
    }

    func persistAssistantErrorToHistory(_ text: String) {
        persistAssistantPayloadToHistory(payload: .error(text), previewSeed: text)
    }

    func persistAssistantPayloadToHistory(payload: GenerationPayload, previewSeed: String) {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let preview = Self.makePreviewText(previewSeed)

        Task {
            do {
                await self.ensureThreadRegisteredAsync()

                let msg = GenerationMessage(
                    role: .assistant,
                    createdAt: now,
                    payload: payload
                )
                try await repo.appendMessage(threadId: id, message: msg)
                try await self.touchThread(preview: preview, at: now)

                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: saved assistant message + updated thread preview id=\(id)")
                #endif
            } catch {
                #if DEBUG
                print("🧾 GEN_HISTORY Step4.3: persist assistant message failed: \(error)")
                #endif
            }
        }
    }

    static func makePreviewText(_ text: String, limit: Int = 80) -> String {
        let clean = text.trimmingCharacters(in: .whitespacesAndNewlines)
        guard clean.count > limit else { return clean }
        let idx = clean.index(clean.startIndex, offsetBy: limit)
        return String(clean[..<idx]) + "…"
    }
}

import Foundation

extension TextToImageChatViewModel {


    func appendImagePlaceholderAndDownload(remoteURL: URL) async {
        let repo = historyRepo
        let id = threadId
        let now = Date()

        let messageId = UUID()
        activeGenerationMessageId = messageId

        let placeholder = GenerationMessage(
            id: messageId,
            role: .assistant,
            createdAt: now,
            payload: .image(remoteURL: remoteURL, localURL: nil)
        )

        do {
            await ensureThreadRegisteredAsync()

            try await repo.appendMessage(threadId: id, message: placeholder)
            try await touchThread(preview: "🖼️ Image", at: now)

            replaceGenerationMessageInUI(updated: placeholder)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): appended placeholder msgId=\(messageId) remote=\(remoteURL.absoluteString)")
            #endif

            let localURL = try await mediaCache.fetch(
                remoteURL: remoteURL,
                threadId: id,
                kind: "image"
            )
            try Task.checkCancellation()

            guard self.activeGenerationMessageId == messageId else {
                #if DEBUG
                print("🧾 GEN_MEDIA Step5D(TextToImage): ignored late download msgId=\(messageId)")
                #endif
                return
            }

            let updated = GenerationMessage(
                id: messageId,
                role: placeholder.role,
                createdAt: placeholder.createdAt,
                payload: .image(remoteURL: remoteURL, localURL: localURL)
            )

            try await repo.replaceMessage(threadId: id, messageId: messageId, newMessage: updated)

            replaceGenerationMessageInUI(updated: updated)

            try await touchThread(preview: "✅ Image", at: Date())

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): replaced with local msgId=\(messageId) local=\(localURL.path)")
            #endif

        } catch is CancellationError {
            let cancelled = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .system("⛔️ Cancelled")
            )
            replaceGenerationMessageInUI(updated: cancelled)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): cancelled msgId=\(messageId)")
            #endif

        } catch {
            let pretty = String(describing: error)

            let failed = GenerationMessage(
                id: messageId,
                role: .assistant,
                createdAt: now,
                payload: .error("🖼️ Image (remote)\n\(remoteURL.absoluteString)\n\n⚠️ Download failed: \(pretty)")
            )
            replaceGenerationMessageInUI(updated: failed)

            #if DEBUG
            print("🧾 GEN_MEDIA Step5B(TextToImage): failed msgId=\(messageId) error=\(pretty)")
            #endif
        }
    }
}

import Foundation

extension TextToImageChatViewModel {

    // MARK: - Thread helpers

    func ensureThreadRegistered() {
        Task { [weak self] in
            await self?.ensureThreadRegisteredAsync()
        }
    }

    func ensureThreadRegisteredAsync() async {
        let repo = historyRepo
        let id = threadId

        do {
            let threads = try await repo.fetchThreads()
            if threads.contains(where: { $0.id == id }) { return }

            try await repo.upsert(GenerationThread(id: id, kind: kind))

            #if DEBUG
            print("🧾 GEN_HISTORY thread created: id=\(id) kind=\(kind.rawValue)")
            #endif
        } catch {
            #if DEBUG
            print("🧾 GEN_HISTORY thread create failed: \(error)")
            #endif
        }
    }

    func touchThread(preview: String?, at date: Date) async throws {
        let repo = historyRepo
        let id = threadId

        var threads = try await repo.fetchThreads()
        if let idx = threads.firstIndex(where: { $0.id == id }) {
            var t = threads[idx]
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        } else {
            var t = GenerationThread(id: id, kind: kind)
            t.touch(preview: preview, at: date)
            try await repo.upsert(t)
        }
    }
}

import Foundation

extension TextToImageChatViewModel {

    // MARK: - UI helpers

    func replaceGenerationMessageInUI(updated: GenerationMessage) {
        if let idx = messages.firstIndex(where: { $0.id == updated.id }) {
            messages[idx] = updated
        } else {
            messages.append(updated)
        }
    }

    func fileExists(_ url: URL) -> Bool {
        FileManager.default.fileExists(atPath: url.path)
    }
}
