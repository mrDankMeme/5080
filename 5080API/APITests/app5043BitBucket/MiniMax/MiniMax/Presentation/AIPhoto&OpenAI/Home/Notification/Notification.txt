import SwiftUI

struct NotificationsListView: View {

    @StateObject private var vm: NotificationsListViewModel

    private let onSelect: (GenerationKind, UUID) -> Void

    private let onSelectOpenAIChat: ((String) -> Void)?

    private let onClose: () -> Void

    init(
        vm: NotificationsListViewModel,
        onSelect: @escaping (GenerationKind, UUID) -> Void,
        onSelectOpenAIChat: ((String) -> Void)? = nil,
        onClose: @escaping () -> Void
    ) {
        _vm = StateObject(wrappedValue: vm)
        self.onSelect = onSelect
        self.onSelectOpenAIChat = onSelectOpenAIChat
        self.onClose = onClose
    }

    var body: some View {
        VStack(spacing: 0) {
            header
            content
            bottomBar
        }
        .background(
            RoundedRectangle(cornerRadius: Tokens.Radius.medium, style: .continuous)
                .fill(Color(hex:"111111")!)
        )
        .clipShape(RoundedRectangle(cornerRadius: Tokens.Radius.medium, style: .continuous))
        .task { await vm.refresh() }
    }

    // MARK: - Header (72.5)

    private var header: some View {
        VStack(spacing: 0) {
            HStack(alignment: .center) {

                Text("Уведомления")
                    .font(Tokens.Font.semibold16)
                    .foregroundStyle(Color.white)

                Spacer(minLength: 0)

                Button {
                    onClose()
                } label: {
                    Image(Assets.close)
                        .renderingMode(.template)
                        .resizable()
                        .scaledToFit()
                        .frame(width: 32.scale, height: 32.scale)
                        .foregroundStyle(Color.white)
                }
                .buttonStyle(.plain)
            }
            .padding(.horizontal, Tokens.Spacing.x16)
            .frame(height: 72.5.scale)

            divider
        }
    }

    private var divider: some View {
        Rectangle()
            .fill(Tokens.Color.accent.opacity(0.20))
            .frame(height: 1.scale)
    }

    // MARK: - Content

    private var content: some View {
        Group {
            if vm.items.isEmpty && vm.isLoading == false {
                emptyState
            } else {
                list
            }
        }
        .frame(maxWidth: .infinity, maxHeight: .infinity)
    }

    private var list: some View {
        ScrollView(showsIndicators: false) {
            VStack(spacing: Tokens.Spacing.x16) {
                ForEach(vm.items) { item in
                    notificationCard(item)
                }
            }
            .padding(.horizontal, Tokens.Spacing.x16)
            .padding(.top, Tokens.Spacing.x16)
            .padding(.bottom, Tokens.Spacing.x16)
        }
    }

    private func notificationCard(_ item: InternalNotification) -> some View {
        ZStack(alignment: .topTrailing) {
            Button {
                Task {
                    let genKind = item.payload.generationKind
                    let genThreadId = item.payload.threadId
                    let openAIConversationId = item.payload.openAIConversationId

                    await vm.markRead(id: item.id)

                    if let kind = genKind, let threadId = genThreadId {
                        onSelect(kind, threadId)
                        return
                    }

                    if let cid = openAIConversationId, cid.isEmpty == false {
                        onSelectOpenAIChat?(cid)
                        return
                    }
                }
            } label: {
                cardBody(item)
            }
            .buttonStyle(.plain)

            Button {
                Task {
                    await vm.remove(id: item.id)
                }
            } label: {
                Image(systemName: "xmark")
                    .font(Tokens.Font.regular12)
                    .foregroundStyle(Color.white.opacity(0.5))
                    .frame(width: 28.scale, height: 28.scale)
            }
            .buttonStyle(.plain)
            .contentShape(Rectangle())
            .padding(.top, 6.scale)
            .padding(.trailing, 6.scale)
            .zIndex(1)
        }
    }

    private func cardBody(_ item: InternalNotification) -> some View {
        let isNew = (item.isRead == false)

        return HStack(alignment: .center, spacing: Tokens.Spacing.x12) {

            Image(systemName: sfSymbolName(for: item))
                .font(.system(size: 20.scale, weight: .semibold))
                .frame(width: 32.scale, height: 32.scale, alignment: .center)
                .foregroundStyle(Color.white)

            VStack(alignment: .leading, spacing: 10.scale) {

                Text(item.title)
                    .font(Tokens.Font.regular16)
                    .foregroundStyle(Color.white)
                    .lineLimit(3)

                HStack(spacing: Tokens.Spacing.x8) {
                    HStack(spacing: 6.scale) {
                        Image(systemName: "clock")
                            .resizable()
                            .scaledToFit()
                            .frame(width: 16.scale, height: 16.scale)
                            .foregroundStyle(Color.white.opacity(0.65))

                        Text(relativeTime(item.createdAt))
                            .font(Tokens.Font.regular13)
                            .foregroundStyle(Color.white.opacity(0.65))
                            .lineLimit(1)
                    }

                    Spacer(minLength: 0)
                }

                Text(
                    (
                        isNew
                        ? "notifications.item.state.new"
                        : "notifications.item.state.viewed"
                    ).localizable
                )
                .font(Tokens.Font.semibold13)
                .foregroundStyle(isNew ? Color.white : Color.white.opacity(0.75))
            }

            Spacer(minLength: 0)

            Image(systemName: "chevron.right")
                .font(.system(size: 16.scale, weight: .semibold))
                .foregroundStyle(Color.white.opacity(0.85))
        }
        .padding(Tokens.Spacing.x16)
        .background(cardBackground(isNew: isNew))
        .clipShape(
            RoundedRectangle(
                cornerRadius: Tokens.Radius.medium,
                style: .continuous
            )
        )
    }


    private func cardBackground(isNew: Bool) -> some View {
        Group {
            if isNew {
                Tokens.Color.accent
            } else {
                Color(hex:"292929")!
            }
        }
    }

    // MARK: - Bottom bar (80)

    private var bottomBar: some View {
        VStack(spacing: 0) {
            divider

            HStack {
                Button {
                    onClose()
                } label: {
                    Text("Закрыть")
                        .font(Tokens.Font.medium16)
                        .foregroundStyle(Color.white)
                        .frame(maxWidth: .infinity)
                        .frame(height: 48.scale)
                        .background(
                            RoundedRectangle(cornerRadius: Tokens.Radius.small, style: .continuous)
                                .fill(Tokens.Color.accent)
                        )
                }
                .buttonStyle(.plain)
            }
            .padding(.horizontal, Tokens.Spacing.x24)
            .padding(.vertical, Tokens.Spacing.x16)
            .frame(height: 80.scale)
        }
    }

    // MARK: - Empty state

    private var emptyState: some View {
        VStack(spacing: 10.scale) {
            Text("Пока пусто")
                .font(Tokens.Font.medium16)
                .foregroundStyle(Color.white)

            Text("Когда что-то будет готово — здесь появится уведомление.")
                .font(Tokens.Font.regular13)
                .foregroundStyle(Color.white.opacity(0.65))
                .multilineTextAlignment(.center)
        }
        .padding(Tokens.Spacing.x24)
    }

    // MARK: - Helpers

    private func relativeTime(_ date: Date) -> String {
        let f = RelativeDateTimeFormatter()
        f.unitsStyle = .short
        return f.localizedString(for: date, relativeTo: Date())
    }

    private func sfSymbolName(for item: InternalNotification) -> String {
        switch item.kind {
        case .openAIChatReplyReady:
            return "message.fill"

        case .generationReady:
            if let kind = item.payload.generationKind {
                switch kind {
                case .textToImage:      return "photo"
                case .photoEdit:        return "wand.and.stars"
                case .photoEnhance:     return "sparkles"
                case .textToVideo:      return "video"
                case .photoToVideo:     return "video"
                case .twoPhotosToVideo: return "rectangle.stack"
                }
            }
            return "sparkles"

        case .generationFailed:
            if let kind = item.payload.generationKind {
                switch kind {
                case .textToImage:      return "exclamationmark.triangle"
                case .photoEdit:        return "exclamationmark.triangle"
                case .photoEnhance:     return "exclamationmark.triangle"
                case .textToVideo:      return "exclamationmark.triangle"
                case .photoToVideo:     return "exclamationmark.triangle"
                case .twoPhotosToVideo: return "exclamationmark.triangle"
                }
            }
            return "exclamationmark.triangle"
        }
    }

    // MARK: - Assets (только close)

    private enum Assets {
        static let close = "ic_notifications_close_32"   // 32x32 (template/white)
    }
}

import Foundation

@MainActor
final class NotificationsListViewModel: ObservableObject {

    @Published private(set) var items: [InternalNotification] = []
    @Published private(set) var isLoading: Bool = false

    private let center: InternalNotificationCenter

    init(center: InternalNotificationCenter) {
        self.center = center
    }

    func refresh() async {
        isLoading = true
        defer { isLoading = false }

        let list = await center.list()
        items = list
    }

    func markRead(id: UUID) async {
        await center.markRead(id: id)
        await refresh()
    }

    func remove(id: UUID) async {
        await center.remove(id: id)
        await refresh()
    }
}
